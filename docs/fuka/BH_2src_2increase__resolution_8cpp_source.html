23 #include "kadath_adapted_bh.hpp"
24 #include "bco_utilities.hpp"
25 #include "Configurator/config_bco.hpp"
26 #include <math.h>
27 #include <sstream>
28 #include <cmath>
29 #include "mpi.h"
30 #include <memory>
31 
32 using namespace Kadath;
33 using namespace Kadath::FUKA_Config;
34 
35 int main(int argc, char **argv) {
36  int rc = MPI_Init(&argc, &argv) ;
37 
38  if (rc!=MPI_SUCCESS) {
39  cerr << "Error starting MPI" << endl ;
40  MPI_Abort(MPI_COMM_WORLD, rc) ;
41  }
42  int rank = 0 ;
43  MPI_Comm_rank(MPI_COMM_WORLD, &rank) ;
44 
45  if (argc < 2) {
46  std::cerr << "Usage: ./increase_resol /<path>/<ID base name>.info "\
47  "./<new_path>/<new ID base name> <Resolution>" << endl;
48  std::cerr << "Ex: ./increase_resol converged.9.info initbh 11" << endl;
49  std::_Exit(EXIT_FAILURE);
50  }
51  else if (argc < 3) {
52  std::cerr << "Output file base name missing...(e.g. initbh)" << endl;
53  std::_Exit(EXIT_FAILURE);
54  }
55  else if (argc < 4) {
56  std::cerr << "Missing new resolution...(e.g. 11)" << std::endl;
57  std::_Exit(EXIT_FAILURE);
58  }
59  std::string outputfile{argv[2]};
60 
61  std::string in_filename = argv[1];
62  kadath_config_boost<BCO_BH_INFO> bconfig(in_filename);
63 
64  std::string in_spacefile = bconfig.space_filename();
65  FILE *ff1 = fopen(in_spacefile.c_str(), "r");
66 
67  Space_adapted_bh old_space(ff1) ;
68  Scalar old_conf (old_space, ff1) ;
69  Scalar old_lapse(old_space, ff1) ;
70  Vector old_shift(old_space, ff1) ;
71 
72  fclose(ff1) ;
73 
74  std::cout << "Resolution of old space: "
75  << old_space.get_domain(0)->get_nbr_points()(0) << " (r), "
76  << old_space.get_domain(0)->get_nbr_points()(1) << " (theta), "
77  << old_space.get_domain(0)->get_nbr_points()(2) << " (phi)" << std::endl;
78 
79  int ndim = 3;
80  int old_ndom = old_space.get_nbr_domains() ;
81 
82  bconfig.set(BCO_RES) = std::atoi(argv[3]);
87  if(!bconfig.control(USE_CONFIG_VARS)) {
88  bconfig.set(RMID) = bco_utils::get_radius(old_space.get_domain(1), OUTER_BC) ;
89 
90  //estimate how small the inner radius should be based on relation
91  //between conformal factor and numerical radius.
92  //see https://arxiv.org/pdf/0805.4192, eq(64)
93  double conf_inner = bco_utils::get_boundary_val(2, old_conf, INNER_BC);
94  double conf_i_sq = conf_inner * conf_inner;
95  double est_r_div2 = bconfig(MCH) / conf_i_sq;
96  bconfig.set(RIN) = est_r_div2;
97  bconfig.set(ROUT) = 4. * est_r_div2 * 2. ;
98  }
99 
100  int ndom = 4 + bconfig(NSHELLS);
101  std::cout << "Number of Domains: " << ndom << std::endl;
102  std::vector<double> bounds(ndom - 1);
103  bco_utils::set_isolated_BH_bounds(bounds, bconfig);
104 
105  // FIXME not sure if it's only about oddness...
106  if(((int)bconfig(BCO_RES) % 2) == 0){
107  std::cout << "New Resolution is invalid. Must be odd (9,11,13,etc)" << std::endl;
108  std::_Exit(EXIT_FAILURE);
109  }
110 
111  std::cout << "Resolution of new space: "
112  << bconfig(BCO_RES) << " (r), "
113  << bconfig(BCO_RES) << " (theta), "
114  << bconfig(BCO_RES) - 1 << " (phi)" << std::endl;
115  Dim_array res(ndim);
116  res.set(0) = bconfig(BCO_RES);
117  res.set(1) = bconfig(BCO_RES);
118  res.set(2) = bconfig(BCO_RES)-1;
119 
120  int type_coloc = old_space.get_type_base();
121 
122  Point center(old_space.get_domain(0)->get_center());
123 
124  Space_adapted_bh space(type_coloc, center, res, bounds);
125 
126  Base_tensor basis (space, CARTESIAN_BASIS);
127 
128  std::cout << "old bounds:" << std::endl;
129  for(int i = 0; i < old_ndom; ++i)
130  std::cout << bco_utils::get_radius(old_space.get_domain(i), OUTER_BC) << " ";
131  std::cout << std::endl;
132  std::cout << "New bounds:" << std::endl;
133  for(int i = 0; i < ndom; ++i)
134  std::cout << bco_utils::get_radius(space.get_domain(i), OUTER_BC) << " ";
135  std::cout << std::endl;
136 
137  //setup new fields
138  Scalar conf(space);
139  conf = 1.;
140  conf.std_base();
141 
142  Scalar lapse(space);
143  lapse = 1.;
144  lapse.std_base();
145 
146  Vector shift(space, CON, basis);
147  shift.annule_hard();
148  shift.std_base();
149  //end setup new fields
150 
151  // needed in some cases, since there is no data in domains [0,1] and the interpolation can go crazy
152  const Domain_shell_outer_homothetic* old_outer_homothetic = dynamic_cast<const Domain_shell_outer_homothetic*>(old_space.get_domain(1));
153 
154  //import fields
155  bco_utils::update_adapted_field(old_conf , 2, 1, old_outer_homothetic, OUTER_BC);
156  bco_utils::update_adapted_field(old_lapse , 2, 1, old_outer_homothetic, OUTER_BC);
157  for(int i = 1; i <= 3; ++i)
158  bco_utils::update_adapted_field(old_shift.set(i), 2, 1, old_outer_homothetic, OUTER_BC);
159 
160  conf.import(old_conf);
161  lapse.import(old_lapse);
162 
163  shift.set(1).import(old_shift.set(1));
164  shift.set(2).import(old_shift.set(2));
165  shift.set(3).import(old_shift.set(3));
166  //end import fields
167 
168  //reset fields to zero inside excised region
169  for(auto& d : {0,1}) {
170  lapse.set_domain(d).annule_hard();
171  conf.set_domain(d).annule_hard();
172  for(int i = 1; i <= 3; ++i) {
173  shift.set(i).set_domain(d).annule_hard();
174  }
175  }
176  lapse.std_base();
177  conf.std_base();
178  shift.std_base();
179 
180  bconfig.set_filename(outputfile);
181  if(rank == 0)
182  bco_utils::save_to_file(space, bconfig, conf, lapse, shift);
183  MPI_Finalize();
184  return EXIT_SUCCESS;
185 }
