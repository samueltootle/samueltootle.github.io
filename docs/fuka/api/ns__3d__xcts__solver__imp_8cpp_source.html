80  int rank = 0;
81  MPI_Comm_rank(MPI_COMM_WORLD, &rank);
82  int exit_status = EXIT_SUCCESS;
83 
84  std::array<bool, NUM_STAGES>& stage_enabled = bconfig.return_stages();
85  auto [last_stage, last_stage_idx] =
86  get_last_enabled_no_throw(MSTAGE, stage_enabled);
87  if (rank == 0)
88  std::cout << "Last stage: " << last_stage << "\n";
89 
90  double const final_chi = bconfig.seq_setting(SEQ_SETTINGS::FINAL_CHI);
91  double const initial_chi = bconfig(BCO_PARAMS::CHI);
92 
93  if (stage_enabled[STAGES::NOROT_BC]) {
94  bconfig(BCO_PARAMS::CHI) = 0.;
95  this->solver_stage = STAGES::NOROT_BC;
96  exit_status = norot_stage(false);
97  bconfig(BCO_PARAMS::CHI) = initial_chi;
98  }
99 
100  if (stage_enabled[STAGES::TOTAL_BC] && exit_status != RELOAD_FILE) {
101  this->solver_stage = STAGES::TOTAL_BC;
102  if (bconfig.control(CONTROLS::ITERATIVE_CHI)) {
103  exit_status = uniform_rot_stage();
104  bconfig.control(CONTROLS::ITERATIVE_CHI) = false;
105  stage_enabled[STAGES::TOTAL_BC] = true;
106  }
107 
108  if (exit_status != RELOAD_FILE) {
109  bconfig(BCO_PARAMS::CHI) = final_chi;
110  exit_status = uniform_rot_stage();
111  }
112  }
113 
114  if (stage_enabled[STAGES::DIFF_ROT] && exit_status != RELOAD_FILE) {
115  this->solver_stage = STAGES::DIFF_ROT;
116  auto law = str_tolower(
117  bconfig.template diffrot<std::string>(DIFFROT_PARAMS::DIFF_LAW));
118  if (law == "keh")
119  exit_status = keh_stage();
120  }
121 
122  // Barrier needed in case we need to read from the previous output
123  MPI_Barrier(MPI_COMM_WORLD);
124  return exit_status;
125 }

136 template <class eos_t, typename config_t, typename space_t>
137 void ns_3d_xcts_solver<eos_t, config_t, space_t>::syst_init(
138  System_of_eqs& syst) {
139  using namespace ::Kadath::Margherita;
140 
141  auto& space = syst.get_space();
142  const int ndom = space.get_nbr_domains();
143  // call the (flat) conformal metric "f"
144  fmet.set_system(syst, "f");
145 
146  // define numerical constants
147  syst.add_cst("4piG", bconfig(BCO_QPIG));
148 
149  // include the coordinate fields
150  syst.add_cst("mg", *coord_vectors[GLOBAL_ROT]);
151  syst.add_cst("sm", *coord_vectors[S_BCO1]);
152  syst.add_cst("einf", *coord_vectors[S_INF]);
153 
154  // the basic fields, conformal factor, lapse and (log) enthalpy
155  syst.add_var("P", conf);
156  syst.add_var("N", lapse);
157 
158  // define common combinations of conformal factor and lapse
159  syst.add_def("NP = P*N");
160  syst.add_def("Ntilde = N / P^6");
161 
162  // define quantity to be integrated at infinity
163  // two (in this case) equivalent definitions of ADM mass
164  // as well as the Komar mass
165  syst.add_def(ndom - 1, "intMadm = - einf^i * D_i P / 4piG * 2");
166  syst.add_def(ndom - 1, "intMk = einf^i * D_i N / 4piG");
167  syst.add_def(ndom - 1, "intMadmalt = -dr(P) * 2 / 4piG");
168 
169  // enthalpy from the logarithmic enthalpy, the latter is the actual variable
170  // in this system
171  syst.add_def("h = exp(H)");
172 
173  // define the EOS operators
174  Param p;
175  syst.add_ope("eps", &EOS<eos_t, EPSILON>::action, &p);
176  syst.add_ope("press", &EOS<eos_t, PRESSURE>::action, &p);
177  syst.add_ope("rho", &EOS<eos_t, DENSITY>::action, &p);
178  syst.add_ope("dHdlnrho", &EOS<eos_t, DHDRHO>::action, &p);
179 
180  // define rest-mass density, internal energy and pressure through the enthalpy
181  syst.add_def("rho = rho(h)");
182  syst.add_def("eps = eps(h)");
183  syst.add_def("press = press(h)");
184  syst.add_def("dHdlnrho = dHdlnrho(h)");
185 
186  // definition to rescale the equations
187  // delta = p / rho
188  syst.add_def("delta = h - eps - 1.");
189 }

194  const int ite,
195  const double conv) const {
196  // total number of domains
197  int ndom = space.get_nbr_domains();
198

201  syst.give_val_def("intMb")()(1).integ_volume();
202 
203  // compute the ADM mass as surface integral at infinity
204  Val_domain integMadm(syst.give_val_def("intMadm")()(ndom - 1));
205  double Madm = space.get_domain(ndom - 1)->integ(integMadm, OUTER_BC);

209  double Mk = space.get_domain(ndom - 1)->integ(integMk, OUTER_BC);

211  // get the maximum and minimum coordinate radius along the surface,
212  // i.e. the adapted domain boundary
213  auto rs = Kadath::bco_utils::get_rmin_rmax(space, 1);
214 
215  // alternative, equivalent ADM mass integral
216  Val_domain integMadmalt(syst.give_val_def("intMadmalt")()(ndom - 1));
217  double Madmalt = space.get_domain(ndom - 1)->integ(integMadmalt, OUTER_BC);
218 
219  // output to standard output
220  std::ios_base::fmtflags f(std::cout.flags());
221  std::cout << "=======================================" << std::endl
222  << FORMAT << "Iter: " << ite << std::endl

225  << FORMAT << "Madm: " << Madm << std::endl

228  << FORMAT << "Mk: " << Mk << " [" << std::abs(Madm - Mk) / Madm

231  std::cout.flags(f);

264 template <class eos_t, typename config_t, typename space_t>
265 void ns_3d_xcts_solver<eos_t, config_t, space_t>::update_config_quantities(
266  System_of_eqs& syst) {
267  auto rs = bco_utils::get_rmin_rmax(space, 1);
268  bconfig.set(BCO_PARAMS::RMID) = rs[0];
269 
270  // compute the ADM mass as surface integral at infinity
271  Val_domain integMadm(syst.give_val_def("intMadm")()(ndom - 1));
272  double Madm = space.get_domain(ndom - 1)->integ(integMadm, OUTER_BC);
273  bconfig.set(BCO_PARAMS::MADM) = Madm;
274 
275  // compute the baryonic mass at volume integral from the given integrant
276  double baryonic_mass = syst.give_val_def("intMb")()(0).integ_volume() +
277  syst.give_val_def("intMb")()(1).integ_volume();
278  bconfig.set(BCO_PARAMS::MB) = baryonic_mass;
279 
280  auto loghc = bco_utils::get_boundary_val(0, logh, INNER_BC);
281 
282  // compute the ADM Angular Momentum as surface integral at infinity
283  double chi{0};
284  if (solver_stage != STAGES::NOROT_BC) {
285  Val_domain integJ(syst.give_val_def("intJ")()(ndom - 1));
286  double Jadm = space.get_domain(ndom - 1)->integ(integJ, OUTER_BC);
287  chi = Jadm / Madm / Madm;
288  }
289 
290  if (seq) {
291  auto idx{seq->mass_idx()};
292  switch (idx) {
293  case BCO_PARAMS::HC:
294  bconfig.set(BCO_PARAMS::NC) =
295  EOS<eos_t, DENSITY>::get(bconfig.set(BCO_PARAMS::HC));
296  bconfig.set(BCO_PARAMS::MADM) = Madm;
297  bconfig.set(BCO_PARAMS::MB) = baryonic_mass;
298  break;
299  case BCO_PARAMS::NC:
300  bconfig.set(BCO_PARAMS::HC) = std::exp(loghc);
301  bconfig.set(BCO_PARAMS::MADM) = Madm;
302  bconfig.set(BCO_PARAMS::MB) = baryonic_mass;
303  break;
304  case BCO_PARAMS::MB:
305  bconfig.set(BCO_PARAMS::HC) = std::exp(loghc);
306  bconfig.set(BCO_PARAMS::NC) =
307  EOS<eos_t, DENSITY>::get(bconfig.set(BCO_PARAMS::HC));
308  bconfig.set(BCO_PARAMS::MADM) = Madm;
309  break;
310  default:
311  bconfig.set(BCO_PARAMS::HC) = std::exp(loghc);
312  bconfig.set(BCO_PARAMS::NC) =
313  EOS<eos_t, DENSITY>::get(bconfig.set(BCO_PARAMS::HC));
314  bconfig.set(BCO_PARAMS::MB) = baryonic_mass;
315  break;
316  }
317 
318  idx = seq->spin_idx();
319  switch (idx) {
320  case BCO_PARAMS::CHI:
321  break;
322  default:
323  bconfig.set(BCO_PARAMS::CHI) = chi;
324  break;
325  }
326  } else {
327  bconfig.set(BCO_PARAMS::NC) =
328  EOS<eos_t, DENSITY>::get(bconfig.set(BCO_PARAMS::HC));
329  bconfig.set(BCO_PARAMS::CHI) = chi;
330  }
331  bconfig.set(QLMADM) = bconfig(MADM);
332 }
