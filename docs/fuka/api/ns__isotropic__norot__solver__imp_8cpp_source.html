62 int ns_isotropic_norot_solver<eos_t, config_t, space_t>::solve() {
63  int rank = 0;
64  MPI_Comm_rank(MPI_COMM_WORLD, &rank);
65  int exit_status = EXIT_SUCCESS;
66 
67  this->solver_stage = STAGES::NOROT_BC;
68  // If we start from scratch, we currently need
69  // to fix the stellar surface otherwise the solver
70  // more often than not diverges.
71  if (bconfig.control(CONTROLS::SEQUENCES)) {
72  bconfig.control(CONTROLS::USE_FIXED_R) = true;
73  exit_status = norot_stage(true);
74  // Rerun with fixed central enthalpy, but with
75  // the surface no longer fixed. This helps with
76  // stability before running with, for instance,
77  // Fixed Madm.
78  bconfig.control(CONTROLS::USE_FIXED_R) = false;
79  exit_status = norot_stage(true);
80  }
81  exit_status = norot_stage(false);
82 
83  // Barrier needed in case we need to read from the previous output
84  MPI_Barrier(MPI_COMM_WORLD);
85  return exit_status;
86 }

98 void ns_isotropic_norot_solver<eos_t, config_t, space_t>::syst_init(
99  System_of_eqs& syst) {
100  using namespace ::Kadath::Margherita;
101 
102  // define numerical constants
103  syst.add_cst("4piG", bconfig(BCO_PARAMS::BCO_QPIG));
104 
105  // the basic fields, conformal factor, lapse and (log) enthalpy
106  syst.add_var("H", logh);
107  syst.add_var("nu", nu);
108  syst.add_var("lapAterm", lap_Aterm);
109  syst.add_var("lapBterm", lap_Bterm);
110 
111  // Useful definitions
112  syst.add_def("N = exp(nu)");
113  syst.add_def("A = exp(lapAterm - nu)");
114  syst.add_def("B = (divrsint(lapBterm) + 1) / N");
115  syst.add_def("Brsint = multrsint(B)");
116 
117  // define quantity to be integrated at infinity
118  // two (in this case) equivalent definitions of ADM mass
119  // as well as the Komar mass
120  // syst.add_def(ndom - 1, "intMadm = - (dr(A^2 + B^2) + divr(B^2 - A^2)) / 4
121  // / 4piG "); syst.add_def(ndom - 1, "intMadm = -dr(A) / 4piG ");
122  syst.add_def(ndom - 1, "intMadm = - (dr(B)) / 4piG ");
123  syst.add_def(ndom - 1, "intMk = dr(N) / 4piG");
124 
125  // enthalpy from the logarithmic enthalpy, the latter is the actual variable
126  // in this system
127  syst.add_def("h = exp(H)");
128 
129  // define the EOS operators
130  Param p;
131  syst.add_ope("eps", &EOS<eos_t, EPSILON>::action, &p);
132  syst.add_ope("press", &EOS<eos_t, PRESSURE>::action, &p);
133  syst.add_ope("rho", &EOS<eos_t, DENSITY>::action, &p);
134 
135  // define rest-mass density, internal energy and pressure through the enthalpy
136  syst.add_def("rho = rho(h)");
137  syst.add_def("eps = eps(h)");
138  syst.add_def("press = press(h)");
139 
140  // definition to rescale the equations
141  // delta = p / rho
142  syst.add_def("delta = h - eps - 1.");
143 }
