42 template <typename config_t, typename space_t>
43 int bbh_xcts_solver<config_t, space_t>::solve_stage(std::string stage_text) {
44  int exit_status = EXIT_SUCCESS;
45  int rank = 0;
46  MPI_Comm_rank(MPI_COMM_WORLD, &rank);
47  if (solver_stage != ECC_RED && bconfig.control(FIXED_GOMEGA))
48  stage_text += "_FIXED_OMEGA";
49 
50  std::string const lapse_cond = (bconfig.control(USE_FIXED_LAPSE))
51  ? "Total system with fixed Lapse BC"
52  : "Total system with v.Neumann BC";
53  if (rank == 0) {
54  std::cout << std::string(42, '#') << '\n';
55  std::cout << lapse_cond << '\n';
56 
57  if (solver_stage == ECC_RED)
58  std::cout << "Ecc. Reduction - fixed Omega & Adot" << '\n';
59  if (bconfig.control(FIXED_GOMEGA))
60  std::cout << "using fixed orbital velocity" << '\n';
61  if (bconfig.control(COROT_BIN))
62  std::cout << "and co-rotation conditions" << '\n';
63  }
64 
65  if (solver_stage == ECC_RED) {
66  // determine whether to use PN estimates of the orbital frequency and adot
67  // in case of the eccentricity stage
68  if (std::isnan(bconfig.set(ADOT)) || std::isnan(bconfig.set(ECC_OMEGA)) ||
69  bconfig.control(USE_PN)) {
70  bco_u::KadathPNOrbitalParams(bconfig, bconfig(MCH, BCO1),
71  bconfig(MCH, BCO2));
72 
73  if (rank == 0)
74  std::cout << "### Using PN estimate for adot and omega! ###"
75  << std::endl;
76  bconfig.set(ECC_OMEGA) = bconfig(GOMEGA);
77  } else {
78  bconfig.set(GOMEGA) = bconfig(ECC_OMEGA);
79  }
80  }
81 
82  // setup background position vector field - only needed for ECC_RED stage
83  Vector CART(space, CON, basis);
84  CART = cfields.cart();
85 
86  System_of_eqs syst(space, 0, ndom - 1);
87  // allow for a shift of the "center of mass" on the x-axis
88  // enforced by a vanishing ADM linear momentum
89  // determine whether orbital frequency is fixed
90  if (bconfig.control(FIXED_GOMEGA) || solver_stage == ECC_RED) {
91  syst.add_cst("ome", bconfig(GOMEGA));
92  } else {
93  syst.add_var("ome", bconfig(GOMEGA));
94  }
95 
96  syst.add_var("xaxis", bconfig(COM));
97  syst.add_var("yaxis", bconfig(COMY));
98 
99  syst_init(syst);
100 
101  std::string bigB{"B^i = bet^i + ome * Morb^i"};
102  if (solver_stage == ECC_RED) {
103  syst.add_cst("adot", bconfig(ADOT));
104  syst.add_cst("r", CART);
105  syst.add_def("comr^i = r^i - xaxis * ex^i + yaxis * ey^i");
106  // add contribution of ADOT to the total shift definition
107  bigB += " + adot * comr^i";
108  }
109 
110  // total shift = inertial + orbital contributions
111  syst.add_def(bigB.c_str());
112 
113  // vacuum XCTS constraint equations
114  syst.add_def("eqP = D^i D_i P + A_ij * A^ij / P^7 / 8");
115  syst.add_def("eqNP = D^i D_i NP - 7. / 8. * NP / P^8 * A_ij * A^ij");
116  syst.add_def(
117  "eqbet^i = D_j D^j bet^i + D^i D_j bet^j / 3. - 2. * A^ij * D_j Ntilde");
118  // end definitions
119 
120  // system of equations - see space for details regarding add_eq
121  space.add_eq(syst, "eqNP = 0", "N", "dn(N)");
122  space.add_eq(syst, "eqP = 0", "P", "dn(P)");
123  space.add_eq(syst, "eqbet^i = 0", "bet^i", "dn(bet^i)");
124 
125  // boundary conditions on variable fields at infinity
126  syst.add_eq_bc(ndom - 1, OUTER_BC, "N = 1");
127  syst.add_eq_bc(ndom - 1, OUTER_BC, "P = 1");
128  syst.add_eq_bc(ndom - 1, OUTER_BC, "bet^i = 0");
129 
130  // quasi-equallibrium condition if binary orbital frequency is not fixed
131  if (!bconfig.control(FIXED_GOMEGA) && solver_stage != ECC_RED) {
132  space.add_eq_int_inf(syst, "integ(dn(N) + 2 * dn(P)) = 0");
133  }
134 
135  space.add_eq_int_inf(syst, "integ(intPx) = 0");
136  space.add_eq_int_inf(syst, "integ(intPy) = 0");
137 
138  // Lapse condition BCs
139  if (bconfig.control(USE_FIXED_LAPSE)) {
140  syst.add_cst("nm", bconfig(FIXED_LAPSE, BCO1));
141  syst.add_cst("np", bconfig(FIXED_LAPSE, BCO2));
142 
143  space.add_bc_sphere_one(syst, "N = nm");
144  space.add_bc_sphere_two(syst, "N = np");
145  } else {
146  space.add_bc_sphere_one(syst, "dn(NP) = 0");
147  space.add_bc_sphere_two(syst, "dn(NP) = 0");
148  }
149 
150  // Apparent Horizon condition BCs
151  space.add_bc_sphere_one(
152  syst,
153  "sm^j * D_j P + P / 4 * D^j sm_j + A_ij * sm^i * sm^j / P^3 / 4 = 0");
154  space.add_bc_sphere_two(
155  syst,
156  "sp^j * D_j P + P / 4 * D^j sp_j + A_ij * sp^i * sp^j / P^3 / 4 = 0");
157 
158  // Irreducible Mass BCs
159  space.add_eq_int_sphere_one(syst, "integ(intMsq) - Mm * Mm = 0 ");
160  space.add_eq_int_sphere_two(syst, "integ(intMsq) - Mp * Mp = 0 ");
161 
162  // in the case of corotation, the tangential term is zero by definition
163  if (bconfig.control(COROT_BIN)) {
164  space.add_bc_sphere_one(syst, "B^i = N / P^2 * sm^i");
165  space.add_bc_sphere_two(syst, "B^i = N / P^2 * sp^i");
166  } else {
167  // otherwise we include equations on the boundary with the addition of the
168  // tangential spin component note that these assume a conformally flat
169  // background metric
170  space.add_bc_sphere_one(syst, "B^i = N / P^2 * sm^i + ExOme^i");
171  if (std::isnan(bconfig.set(FIXED_BCOMEGA, BCO1)))
172  space.add_eq_int_sphere_one(syst, "integ(intSm) - chim * CMm * CMm = 0 ");
173 
174  space.add_bc_sphere_two(syst, "B^i = N / P^2 * sp^i + ExOme^i");
175  if (std::isnan(bconfig.set(FIXED_BCOMEGA, BCO2)))
176  space.add_eq_int_sphere_two(syst, "integ(intSp) - chip * CMp * CMp = 0 ");
177  }
178 
179  // exclude the excised regions
180  for (int i : excluded_doms) {
181  syst.add_eq_full(i, "N = 0");
182  syst.add_eq_full(i, "P = 0");
183  syst.add_eq_full(i, "bet^i = 0");
184  }
185 
186  if (rank == 0)
187  print_diagnostics(syst, 0, 0);
188 
189  // begin iterative solver
190  bool endloop = false;
191  int ite = 1;
192  double conv;
193  while (!endloop) {
194  endloop = syst.do_newton(bconfig.seq_setting(PREC), conv);
195 
196  std::stringstream ss;
197  ss << "bbh_" + stage_text << ite - 1;
198  bconfig.set_filename(ss.str());
199  if (rank == 0) {
200  print_diagnostics(syst, ite, conv);
201 
202  if (bconfig.control(CHECKPOINT))
203  checkpoint();
204  }
205  update_fields(cfields, coord_vectors, {}, xo, xc1, xc2, &syst);
206  ite++;
207  check_max_iter_exceeded(rank, ite, conv);
208  }
209 
210  // in the case of corotation or fixed local spin frequency, we calculate MCH
211  // accordingly
212  if (!std::isnan(bconfig.set(FIXED_BCOMEGA, BCO1)) ||
213  bconfig.control(COROT_BIN))
214  bconfig(MCH, BCO1) = bco_u::syst_mch(syst, space, "intSm", space.BH1 + 2);
215  if (!std::isnan(bconfig.set(FIXED_BCOMEGA, BCO2)) ||
216  bconfig.control(COROT_BIN))
217  bconfig(MCH, BCO2) = bco_u::syst_mch(syst, space, "intSp", space.BH2 + 2);
218 
219  bconfig.set_filename(converged_filename(stage_text));
220  if (rank == 0)
221  checkpoint();
222 
223  return exit_status;
224 }
