36 double func_afake (double aa, const Param& par) {
37  double r1 = par.get_double(0) ;
38  double r2 = par.get_double(1) ;
39  double d = par.get_double(2) ;
40  return (sqrt(aa*aa+r1*r1)+sqrt(aa*aa+r2*r2)-d) ;
41 }

43 Space_bin_fake::Space_bin_fake (int ttype, double dist, double r1, double r2, double rbi, double rext, int nr) {
44 
45  ndim = 3 ;
46 
47  nbr_domains = 9;
48  type_base = ttype ;
49  domains = new Domain* [nbr_domains] ;
50 
51  Dim_array res (ndim) ;
52  res.set(0) = nr ; res.set(1) = nr ; res.set(2) = nr-1 ;
53  Dim_array res_bi(ndim) ;
54  res_bi.set(0) = nr ; res_bi.set(1) = nr ; res_bi.set(2) = nr ;
55 
56  // Bispheric :
57  // Computation of aa
58  Param par_a ;
59  par_a.add_double(r1,0) ;
60  par_a.add_double(r2,1) ;
61  par_a.add_double(dist,2) ;
62  double a_min = 0 ;
63  double a_max = dist/2. ;
64  double precis = PRECISION ;
65  int nitermax = 500 ;
66  int niter ;
67  double aa = zerosec(func_afake, par_a, a_min, a_max, precis, nitermax, niter) ;
68  double eta_plus = asinh(aa/r2) ;
69  double eta_minus = -asinh(aa/r1) ;
70 
71  double chi_c = 2*atan(aa/rbi) ;
72  double eta_c = log((1+rbi/aa)/(rbi/aa-1)) ;
73  double eta_lim = eta_c/2. ;
74  double chi_lim = chi_lim_eta (eta_lim, rbi, aa, chi_c) ;
75 
76  // Star 1
77  Point center_minus (ndim) ;
78  center_minus.set(1) = aa*cosh(eta_minus)/sinh(eta_minus) ;
79  domains[0] = new Domain_nucleus (0, ttype, r1, center_minus, res) ;
80 
81  // Star 2
82  Point center_plus (ndim) ;
83  center_plus.set(1) = aa*cosh(eta_plus)/sinh(eta_plus) ;
84  domains[1] = new Domain_nucleus (1, ttype, r2, center_plus, res) ;
85 
86  // Bispheric part
87  domains[2] = new Domain_bispheric_chi_first(2, ttype, aa, eta_minus, rbi, chi_lim, res_bi) ;
88  domains[3] = new Domain_bispheric_rect(3, ttype, aa, rbi, eta_minus, -eta_lim, chi_lim, res_bi) ;
89  domains[4] = new Domain_bispheric_eta_first(4, ttype, aa, rbi, -eta_lim, eta_lim, res_bi) ;
90  domains[5] = new Domain_bispheric_rect(5, ttype, aa, rbi, eta_plus, eta_lim, chi_lim, res_bi) ;
91  domains[6] = new Domain_bispheric_chi_first(6, ttype, aa, eta_plus, rbi, chi_lim, res_bi) ;
92 
93  Point center(3) ;
94  // Shell
95  domains[7] = new Domain_shell_surr (7, ttype, rbi, rext, center, res) ;
96 
97  // Compactified domain
98  domains[8] = new Domain_compact (8, ttype, rext, center, res) ;
99 }
