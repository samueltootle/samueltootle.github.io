29 double Domain_spheric_periodic_shell::give_harmonique_nonflat (const Val_domain& so, double masse, int nn, double phase, int dim, double factor) const {
30  so.coef() ;
31  Index pos (nbr_coefs) ;
32  pos.set(1) = nn ;
33  int expo = 0;
34  switch ((*so.base.bases_1d[1])(0)) {
35  case COS :
36  expo = nn ;
37  break ;
38  case COS_EVEN :
39  expo = 2*nn ;
40  break ;
41  case COS_ODD :
42  expo = 2*nn+1 ;
43  break ;
44  default :
45  cerr << "Unknown case in Domain_spheric_periodic_shell::fitwaves" << endl ;
46  break ;
47  }
48 
49 
50 
51  // Get the value at rmax of this harmonic
52  double AA = val_boundary (OUTER_BC, so, pos) ;
53  double lambda2 = factor - expo*expo*ome*ome ;
54  double cor = 2*expo*expo*ome*ome - 1 ;
55  double res ;
56  double rmax = alpha + beta ;
57 
58  if (lambda2>0)
59  res = AA/exp(-sqrt(lambda2)*rmax)*pow(rmax, double(dim-1)/2.) ;
60  else
61  res = AA/cos(sqrt(-lambda2)*rmax+masse/sqrt(-lambda2)*cor*log(rmax)+phase)* pow(rmax, double(dim-1)/2.) ;
62  return res ;
63 }

66 Term_eq Domain_spheric_periodic_shell::fitwaves_nonflat (const Term_eq& so, const Term_eq& field, const Array<double>& phases, int dim, double factor) const {
67 
68 
69  // Verifications
70  if ((field.get_type_data()!=TERM_T) || (so.get_type_data()!=TERM_T)) {
71  cerr << "Domain_spheric_periodic_shell::fitwaves_nonflat only defined with respect to tensors" << endl ;
72  abort() ;
73  }
74 
75  if ((field.get_val_t().get_valence()!=0) || (so.get_val_t().get_valence()!=0)) {
76  cerr << "Domain_spheric_periodic_shell::fitwaves_nonflat only defined with respect to scalars" << endl ;
77  abort() ;
78  }
79 
80  bool doder = ((so.get_p_der_t()==0x0) || (field.get_p_der_t()==0x0)) ? false : true ;
81 
82  int dom = so.get_dom() ;
83 
84  Index ppp(nbr_coefs) ;
85  double masse_val = val_boundary (OUTER_BC, mult_r(field.get_val_t()()(dom)-1)/2., ppp) ;
86  double masse_der = (doder) ? val_boundary (OUTER_BC, mult_r(field.get_der_t()()(dom)-1)/2., ppp) : 0 ;
87 
88  double rmax = alpha + beta ;
89 
90  int max = 0;
91  int baset = (*so.get_val_t()()(dom).get_base().bases_1d[1])(0) ;
92  switch (baset) {
93  case COS :
94  max = nbr_coefs(1) ;
95  break ;
96  case COS_EVEN :
97  max = nbr_coefs(1) ;
98  break ;
99  case COS_ODD :
100  max = nbr_coefs(1)-1 ;
101  break ;
102  default :
103  cerr << "Unknown case in Domain_spheric_periodic_shell::fitwaves" << endl ;
104  break ;
105  }
106 
107  Index pos (nbr_coefs) ;
108 
109  Val_domain resval (this) ;
110  resval.allocate_coef() ;
111  Val_domain resder (this) ;
112  resder.allocate_coef() ;
113  do {
114  resval.set_coef(pos) = 0 ;
115  resder.set_coef(pos) = 0 ;
116  }
117  while (pos.inc()) ;
118  pos.set_start() ;
119 
120  for (int nn=0 ; nn<max ; nn++) {
121 
122  pos.set(1) = nn ;
123 
124  // Correspond to what ?
125  int expo = 0;
126  switch (baset) {
127  case COS :
128  expo = nn ;
129  break ;
130  case COS_EVEN :
131  expo = 2*nn ;
132  break ;
133  case COS_ODD :
134  expo = 2*nn+1 ;
135  break ;
136  default :
137  cerr << "Unknown case in Domain_spheric_periodic_shell::fitwaves" << endl ;
138  break ;
139  }
140  double lambda2 = factor - expo*expo*ome*ome ;
141  double cor = 2*expo*expo*ome*ome - 1 ;
142 
143  Term_eq* sh ;
144  Term_eq* dsh ;
145  double la = sqrt(fabs(lambda2)) ;
146  double arg = (lambda2>0) ? la*rmax : la*rmax+masse_val*cor*log(rmax)/la+phases(nn) ;
147  double shval = (lambda2>0) ? exp(-arg)/rmax : cos(arg)/rmax ;
148  double dshval = (lambda2>0) ? -la*exp(-arg)/rmax - exp(-arg)/rmax/rmax : -sin(arg)/rmax*(la+masse_val*cor/la/rmax) - cos(arg)/rmax/rmax ;
149 
150  if (doder) {
151  double shder = (lambda2>0) ? 0 : -sin(arg)/rmax*log(rmax)/la*cor*masse_der ;
152  sh = new Term_eq (dom, shval, shder) ;
153  double dshder = (lambda2>0) ? 0 : (-cos(arg)/rmax*(la+masse_val*cor/la/rmax)*cor*log(rmax)/la + sin(arg)/rmax/rmax*(cor*log(rmax)/la-cor/la))*masse_der ;
154  dsh = new Term_eq (dom, dshval, dshder) ;
155  }
156  else {
157  sh = new Term_eq (dom, shval) ;
158  dsh = new Term_eq (dom, dshval) ;
159  }
160 
161  Term_eq* ff ;
162  Term_eq* dff ;
163 
164  double ffval = val_boundary (OUTER_BC, so.get_val_t()()(dom), pos) ;
165  double dffval = val_boundary (OUTER_BC, so.get_val_t()()(dom).der_r(), pos) ;
166  if (doder) {
167  double ffder = val_boundary (OUTER_BC, so.get_der_t()()(dom), pos) ;
168  double dffder = val_boundary (OUTER_BC, so.get_der_t()()(dom).der_r(), pos) ;
169  ff = new Term_eq (dom, ffval, ffder) ;
170  dff = new Term_eq (dom, dffval, dffder) ;
171  }
172  else {
173  ff = new Term_eq (dom, ffval) ;
174  dff = new Term_eq (dom, dffval) ;
175  }
176 
177  Term_eq auxi (rmax*rmax*((*sh)*(*dff)-(*dsh)*(*ff))) ;
178  resval.set_coef(pos) = auxi.get_val_d() ;
179  if (doder)
180  resder.set_coef(pos) = auxi.get_der_d() ;
181 
182 
183  delete sh ;
184  delete dsh ;
185  delete ff ;
186  delete dff ;
187  }
188 
189 
190 
191  // Put the basis and the result in a term_eq
192  if (so.get_val_t()()(dom).check_if_zero()) {
193  resval.set_zero() ;
194  }
195  else {
196  resval.set_base() = so.get_val_t()()(dom).get_base() ;
197  }
198  if (doder) {
199  if (so.get_der_t()()(dom).check_if_zero()) {
200  resder.set_zero() ;
201  }
202  else {
203  resder.set_base() = so.get_der_t()()(dom).get_base() ;
204  }
205  }
206  Scalar res (so.get_val_t()(), false) ;
207  res.set_domain(dom) = resval ;
208 
209 
210 
211  if (doder) {
212  Scalar der (so.get_der_t()(), false) ;
213  der.set_domain(dom) = resder ;
214  return Term_eq (dom, res, der) ;
215  }
216  else
217  return Term_eq (dom, res) ;
218 }
