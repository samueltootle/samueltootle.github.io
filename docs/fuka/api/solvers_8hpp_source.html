104  void checkpoint(bool termination_chkpt = false) const {
105  // Backup activated stages
106  auto const final_stages{bconfig.return_stages()};
107 
108  // Clear stages and only set the current stage as active
109  auto& stages = bconfig.return_stages();
110  stages.fill(false);
111  stages[this->solver_stage] = true;
112 
113  // Save to file
114  save_to_file();
115 
116  // Reset to original stages
117  stages = final_stages;
118 
119  if (termination_chkpt) {
120  std::cerr << "***Writing early termination chkpt "
121  << bconfig.config_filename() << "***\n";
122  std::_Exit(EXIT_FAILURE);
123  }
124  }

133  void check_max_iter_exceeded(const int& rank,
134  const int& ite,
135  const double& conv) const {
136  bool exceeded = (ite > bconfig.seq_setting(MAX_ITER)) &&
137  conv >= bconfig.seq_setting(PREC);
138  if (exceeded && conv < 10. * bconfig.seq_setting(PREC)) {
139  if (rank == 0)
140  std::cout << "Max iterations exceeded at precison, " << conv
141  << "\nFinishing since precision < 10. * PREC....\n"
142  << "Running at higher resolution may help.\n";
143  } else if (exceeded) {
144  if (rank == 0)
145  std::cout << "Max iterations exceeded at precison, " << conv;
146  } else {
147  return;
148  }
149  auto s = converged_filename("termination_chkpt");
150  bconfig.set_filename(s);
151  if (rank == 0)
152  checkpoint(true);
153 
154  // FIXME this should be handled better than hard termination
155  std::_Exit(EXIT_FAILURE);
156  }

164  bool solution_exists(std::string last_stage = "") {
165  bool exists = false;
166  std::string prev_name{converged_filename(last_stage)};
167  std::string prev_abs{bconfig.config_outputdir() + prev_name};
168 
169  const std::string home_kadath{std::getenv("HOME_KADATH")};
170  std::string central_abs{home_kadath + "/COs/" + prev_name};
171 
172  auto check_and_update_config = [&](auto p) {
173  exists = true;
174  base_config_t old_solution(p + ".info");
175  if (bconfig.set(BCO_PARAMS::NSHELLS) !=
176  old_solution.set(BCO_PARAMS::NSHELLS))
177  return false;
178 
179  // make sure we copy stages, controls, and settings over
180  for (auto idx = 0; idx < STAGES::NUM_STAGES; ++idx)
181  old_solution.set_stage(idx) = bconfig.set_stage(idx);
182  for (auto idx = 0; idx < CONTROLS::NUM_CONTROLS; ++idx)
183  old_solution.control(idx) = bconfig.control(idx);
184  for (auto idx = 0; idx < SEQ_SETTINGS::NUM_SEQ_SETTINGS; ++idx)
185  old_solution.seq_setting(idx) = bconfig.seq_setting(idx);
186 
187  auto& stages = bconfig.return_stages();
188  auto [last_stage_name, last_stage_idx] = get_last_enabled(MSTAGE, stages);
189  if (solver_stage != last_stage_idx)
190  // Deactivate current stage since we found solution
191  old_solution.set_stage(solver_stage) = false;
192  bconfig = old_solution;
193  return exists;
194  };
195 
197  if (fs::exists(prev_abs + ".info") && fs::exists(prev_abs + ".dat")) {
198  exists = check_and_update_config(prev_abs);
199  }
201  else if (bconfig.control(SAVE_COS) && fs::exists(central_abs + ".info") &&
202  fs::exists(central_abs + ".dat")) {
203  exists = check_and_update_config(central_abs);
204  }
205 
206  return exists;
207  }
