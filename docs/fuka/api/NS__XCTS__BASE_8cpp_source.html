93  if (rank == 0) {

95  << space->get_domain(0)->get_nbr_points()(0) << " (r), "
96  << space->get_domain(0)->get_nbr_points()(1) << " (theta), "

102  // call its member functions
103  const Domain_shell_outer_adapted* old_outer_adapted =
104  dynamic_cast<const Domain_shell_outer_adapted*>(space->get_domain(1));
105 
106  // setup a scalar field representing the old radius
107  Scalar old_space_radius(*space);
108  old_space_radius = 0.;
109 
110  // get the radius from each domain
111  for (int i = 0; i < space->get_nbr_domains(); ++i)
112  old_space_radius.set_domain(i) = space->get_domain(i)->get_radius();
113  // get the adapted radius of the adapted domain
114  old_space_radius.set_domain(1) = old_outer_adapted->get_outer_radius();
115 
116  // define a standard decomposition, compatible with the parity of this
117  // field
118  old_space_radius.std_base();
119  // end setup old radius field
120 
121  // get the minimal and maximal radius from the adapted domain
122  auto [r_min, r_max] = Kadath::bco_utils::get_rmin_rmax(*space, 1);
123 
124  std::cout << "Rmin/max: " << r_min << " " << r_max << std::endl;
125 
126  // set new resolutions in each spatial dimension
127  Dim_array res(ndim);
128  res.set(0) = (*bconfig)(BCO_RES);
129  res.set(1) = res(0);
130  res.set(2) = res(0) - 1;
131 
132  // FIXME not sure if it's only about oddness...
133  if (res(0) % 2 == 0 || res(2) % 2 != 0) {
134  std::cout << "New Resolution is invalid. Must be odd (9,11,13,etc)"
135  << std::endl;
136  std::_Exit(EXIT_FAILURE);
137  }
138 
139  std::cout << "Resolution of new space: " << res(0) << " (r), " << res(1)
140  << " (theta), " << res(2) << " (phi)" << std::endl;
141 
142  // get the type of the colocation points
143  int type_coloc = space->get_type_base();
144 
145  // Update config
146  bconfig->set(BCO_PARAMS::RIN) = 0.5 * r_min;
147  bconfig->set(BCO_PARAMS::ROUT) = bco_utils::gold_ratio * r_max;
148  bconfig->set(BCO_PARAMS::RMID) = r_max;
149  // end update config
150 
151  // setup radius bounds of the domains
152 
153  int ndom = 4 + (*bconfig)(BCO_PARAMS::NSHELLS);
154  std::vector<double> bounds(ndom - 1);
155  Kadath::bco_utils::set_NS_bounds(bounds, *bconfig);
156 
157  Kadath::bco_utils::print_bounds("New bounds: ", bounds);
158 
159  // get origin of nucleus domain
160  Point center = space->get_domain(0)->get_center();
161 
162  // initialize space with new resolution and domain decomposition
163  Space_spheric_adapted new_space(type_coloc, center, res, bounds);
164  Base_tensor basis(new_space, CARTESIAN_BASIS);
165 
166  // get adapted domains to update the radius
167  const Domain_shell_outer_adapted* new_outer_adapted =
168  dynamic_cast<const Domain_shell_outer_adapted*>(
169  new_space.get_domain(1));
170  const Domain_shell_inner_adapted* new_inner_adapted =
171  dynamic_cast<const Domain_shell_inner_adapted*>(
172  new_space.get_domain(2));
173 
174  // update adapted domain mapping
175  Kadath::bco_utils::interp_adapted_mapping(new_outer_adapted, 1,
176  old_space_radius);
177  Kadath::bco_utils::interp_adapted_mapping(new_inner_adapted, 1,
178  old_space_radius);
179

181  // initialize to one or zero first
182  Scalar new_conf(new_space);
183  new_conf = 1.;
184  new_conf.std_base();
185

188  new_lapse.std_base();

193  new_shift.std_base();
194 
195  Scalar new_logh(new_space);
196  new_logh.annule_hard();
197  new_logh.std_base();
198  // end setup new fields
199

201  new_conf.import(*conformal_factor);
202  new_lapse.import(*lapse);
203  new_logh.import(*logh);
204 
205  new_shift.set(1).import(shift->set(1));
206  new_shift.set(2).import(shift->set(2));
207  new_shift.set(3).import(shift->set(3));
208 
209  // end import old fields
210 
211  // enforce spectral decomposition compatible with the parities
212  new_lapse.std_base();
213  new_conf.std_base();
214  new_logh.std_base();
215  new_shift.std_base();
216 
217  // output data
218  bconfig->set_filename(outputfile);
219

272 bool NS_XCTS_BASE::increment_resolution() {
273  if (!(resolution->final() > resolution->init()) ||
274  last_stage_idx != solver_stage)
275  return false;
276 
277  auto resolution_indices = resolution->get_indices();
278  auto const& final_res = resolution->final();
279  if ((*bconfig)(resolution_indices) >= final_res)
280  return false;
281 
282  int next_res = bco_utils::next_resolution((*bconfig)(resolution_indices));
283 
284  // Greater would mean that the desired resolution may not be possible
285  // (see bco_utils::next_resolution) so we go to the next available
286  // resolution
287  if (next_res >= final_res) {
288  bconfig->set(resolution_indices) = next_res;
289  } else {
290  bconfig->set(resolution_indices) = next_res;
291  }
292  return true;
293 }

295 bool NS_XCTS_BASE::increment_seq() {
296  if (!seq->is_set() || last_stage_idx != solver_stage)
297  return false;
298 
299  auto sequence_var_indices = seq->get_indices();
300  auto const& dx = seq->step_size();
301  auto x = bconfig->set(sequence_var_indices) + dx;
302  x = (std::fabs(1. - x / seq->final()) < 1e-4) ? seq->final() : x;
303  if (seq->loop_condition(x)) {
304  bconfig->set(sequence_var_indices) = x;
305  return true;
306  }
307  return false;
308 }

310 int NS_XCTS_BASE::do_newton() {
311  int exit_status = EXIT_SUCCESS;
312  // parameters for the solver loop
313  bool endloop = false;
314  int ite = 1;
315  double conv;
316  if (rank == 0) {
317  print_diagnostics(ite - 1, std::nan("1"));
318  }
319  // solve until convergence is achieved
320  while (!endloop) {
321  // do exactly one newton step, given the system above
322  endloop = syst->do_newton(bconfig->seq_setting(SEQ_SETTINGS::PREC), conv);
323 
324  update_config_quantities();
325  // output files at this iteration and print diagnostics
326  std::stringstream ss;
327  ss << stagename << "_ckpt_" << ite - 1;
328 
329  bconfig->set_filename(ss.str());
330  if (rank == 0) {
331  print_diagnostics(ite, conv);
332  std::cout << std::endl;
333  if (bconfig->control(CHECKPOINT))
334  checkpoint();
335  }
336 
337  // update all coordinate fields, in case the domain extents have changed
338  update_fields_co(*cfields, *coord_vectors, {}, 0., syst);
339 
340  ite++;
341  check_max_iter_exceeded(*this, ite, conv);
342  }
343 
344  bconfig->set_filename(converged_filename(stagename));
345  if (rank == 0) {
346  checkpoint();
347  }
348  MPI_Barrier(MPI_COMM_WORLD);
349  return exit_status;
