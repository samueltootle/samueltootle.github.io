31 Val_domain Domain_shell_inner_adapted::mult_cos_phi (const Val_domain& so) const {
32  if (so.check_if_zero())
33  return so ;
34  else {
35  so.coef() ;
36  Val_domain res(this) ;
37 
38  res.base.allocate (nbr_coefs) ;
39  *res.base.bases_1d[2] = *so.base.bases_1d[2] ;
40 
41  Index index_t (res.base.bases_1d[1]->get_dimensions()) ;
42  // Inversion in theta :
43  do {
44  switch ((*so.base.bases_1d[1])(index_t)) {
45  case COS_EVEN :
46  res.base.bases_1d[1]->set(index_t) = SIN_ODD ;
47  break ;
48  case COS_ODD :
49  res.base.bases_1d[1]->set(index_t) = SIN_EVEN ;
50  break ;
51  case SIN_EVEN :
52  res.base.bases_1d[1]->set(index_t) = COS_ODD ;
53  break ;
54  case SIN_ODD :
55  res.base.bases_1d[1]->set(index_t) = COS_EVEN ;
56  break ;
57  default :
58  cout << "Unknown case in Domain_shell_inner_adapted::mult_cos_phi" << endl ;
59  abort() ;
60  }
61  }
62  while (index_t.inc()) ;
63  *res.base.bases_1d[0] = *so.base.bases_1d[0] ;
64 
65  res.base.def = true ;
66 
67  res.cf = new Array<double> (so.base.ope_1d(mult_cos_1d, 2, *so.cf, res.base)) ;
68  res.in_coef = true ;
69  return res ; }
70 }

193 Val_domain Domain_shell_inner_adapted::laplacian (const Val_domain& so, int m) const {
194  Val_domain derr (so.der_r()) ;
195  Val_domain dert (so.der_var(2)) ;
196  Val_domain res (derr.der_r() + div_r(2*derr + div_r(dert.der_var(2) + dert.mult_cos_theta().div_sin_theta()))) ;
197  if (m!=0)
198  res -= m * m * div_r(div_r(so.div_sin_theta().div_sin_theta())) ;
199  return res ;
200 }

212 double Domain_shell_inner_adapted::integ_volume (const Val_domain& so) const {
213 
214  if (so.check_if_zero())
215  return 0 ;
216  else {
217  Val_domain r2 (get_radius()*get_radius()) ;
218  r2.std_base() ;
219  Val_domain integrant (r2*mult_sin_theta(so)*(*der_rad_term_eq->val_t)()(num_dom)) ;
220  integrant.coef() ;
221 
222  double val = 0 ;
223  // Only k = 0
224  int baset = (*integrant.get_base().bases_1d[1]) (0) ;
225  assert(baset==SIN_ODD) ;
226  Index pos (nbr_coefs) ;
227  for (int j=0 ; j<nbr_coefs(1) ; j++) {
228  pos.set(1) = j ;
229  int baser = (*integrant.get_base().bases_1d[0]) (j, 0) ;
230  assert (baser==CHEB) ;
231 
232  Array<double> cf (nbr_coefs(0)) ;
233  for (int i=0 ; i<nbr_coefs(0) ; i++) {
234  pos.set(0) = i ;
235  cf.set(i) = integrant.get_coef(pos) ;
236  }
237  val += 2./(2.*j+1) * integral_1d(CHEB, cf) ;
238  }
239 
240  return val * 2*M_PI ;
241 }
