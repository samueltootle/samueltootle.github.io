112  std::string eccstr{};

235 template <class config_t>
236 inline std::string set_ns_mass_fixing(
237  System_of_eqs& syst,
238  config_t& bconfig,
239  std::unique_ptr<Kadath::FUKA_Solvers::ns_sequence const>& seq) {
240  std::string central_fixing_definition{"h - hc"};
241  auto idx{seq->mass_idx()};
242  switch (idx) {
243  case BCO_PARAMS::HC:
244  syst.add_cst("hc", bconfig(BCO_PARAMS::HC));
245  break;
246  case BCO_PARAMS::NC:
247  syst.add_cst("Nc", bconfig(BCO_PARAMS::NC));
248  central_fixing_definition = "rho - Nc";
249  break;
250  case BCO_PARAMS::MADM:
251  syst.add_var("hc", bconfig(BCO_PARAMS::HC));
252  syst.add_cst("Madm", bconfig(BCO_PARAMS::MADM));
253  break;
254  case BCO_PARAMS::MB:
255  syst.add_var("hc", bconfig(BCO_PARAMS::HC));
256  syst.add_cst("Mb", bconfig(BCO_PARAMS::MB));
257  break;
258  default:
259  std::string msg{
260  "Sequence initialized, but not implemented for Mass index = " +
261  std::to_string(int(idx))};
262  throw std::runtime_error(msg.c_str());
263  break;
264  }
265  return central_fixing_definition;
266 }
