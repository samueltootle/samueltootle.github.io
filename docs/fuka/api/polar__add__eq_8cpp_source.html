59 void Space_polar::add_eq_int_volume (System_of_eqs& sys, const char* nom) {
60  sys.eq_int_list.push_back(std::make_tuple(nom, nbr_domains-1, -1));
61 
62  // Get the lhs and rhs
63  char p1[LMAX] ;
64  char p2[LMAX] ;
65  bool indic = sys.is_ope_bin(nom, p1, p2, '=') ;
66  if (!indic) {
67  cerr << "= needed for equations" << endl ;
68  abort() ;
69  }
70  else {
71  // Construction of the equation
72  sys.eq_int[sys.neq_int] = new Eq_int(nbr_domains+1) ;
73 
74  // Affectation of the intregrale parts
75  for (int d=0 ; d<nbr_domains ; d++)
76  sys.eq_int[sys.neq_int]->set_part(d, sys.give_ope(d, p1)) ;
77  // Affectation of the second member (constant value)
78  sys.eq_int[sys.neq_int]->set_part(nbr_domains, new Ope_minus(&sys, sys.give_ope(0, p2))) ;
79  sys.neq_int ++ ;
80  }
81  sys.nbr_conditions = -1 ;
82 }

85 void Space_polar::add_eq_int_inf (System_of_eqs& sys, const char* nom) {
86  sys.eq_int_list.push_back(std::make_tuple(nom, nbr_domains-1, OUTER_BC));
87 
88  // Check the last domain is of the right type :
89  const Domain_polar_compact* pcomp = dynamic_cast <const Domain_polar_compact*> (domains[nbr_domains-1]) ;
90  if (pcomp==0x0) {
91  cerr << "add_eq_int_inf requires a compactified domain" << endl ;
92  abort() ;
93  }
94  int dom = nbr_domains-1 ;
95 
96  // Get the lhs and rhs
97  char p1[LMAX] ;
98  char p2[LMAX] ;
99  bool indic = sys.is_ope_bin(nom, p1, p2, '=') ;
100  if (!indic) {
101  cerr << "= needed for equations" << endl ;
102  abort() ;
103  }
104  else {
105  // Verif lhs = 0 ?
106  indic = ((p2[0]=='0') && (p2[1]==' ') && (p2[2]=='\0')) ?
107  true : false ;
108 
109  // Construction of the equation
110  sys.eq_int[sys.neq_int] = new Eq_int(1) ;
111 
112  // Affectation :
113  // no lhs :
114  if (indic)
115  sys.eq_int[sys.neq_int]->set_part(0, sys.give_ope(dom, p1, OUTER_BC)) ;
116 
117  else
118  sys.eq_int[sys.neq_int]->set_part(0, new Ope_sub(&sys, sys.give_ope(dom, p1, OUTER_BC), sys.give_ope(dom, p2, OUTER_BC))) ;
119  sys.neq_int ++ ;
120  }
121  sys.nbr_conditions = -1 ;
122 }

183 void Space_polar::add_eq_point (System_of_eqs& sys, const Point& MM, const char* name) {
184 
185  // Get the domain and num coordinates :
186  int ld = -1;
187  bool* inside = new bool[nbr_domains] ;
188  for (int l=0 ; l<nbr_domains; l++)
189  inside[l] = get_domain(l)->is_in(MM) ;
190  for (int l=0 ; l<nbr_domains ; l++)
191  if ((ld==-1) && (inside[l]))
192  ld = l ;
193  Point num (get_domain(ld)->absol_to_num(MM)) ;
194 
195  char auxi[LMAX] ;
196  trim_spaces (auxi, name) ;
197  sys.add_eq_point (ld, auxi, num) ;
