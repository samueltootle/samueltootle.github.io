35 
36  assert (residus[conte]->get_type_data()==TERM_T) ;
37  assert (residus[conte+1]->get_type_data()==TERM_T) ;
38 
39 
40  int start = pos_res ;
41  dom->export_tau_boundary_array (residus[conte]->get_val_t(), ndom, bound, order, sec, pos_res, *n_cond, n_cmp_used, p_cmp_used) ;
42 
43  Array<double> auxi (pos_res - start) ;
44  auxi = 0. ;
45  int zero = 0 ;
46  residus[conte+1]->get_val_t().get_space().get_domain(other_dom)->export_tau_boundary_array
