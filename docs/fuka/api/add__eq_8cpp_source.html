27  Ope_eq* System_of_eqs::parse_eq (int dom, const char* nom, int boundary) const {
28  // Is the eq written like "... = 0" ?
29  char p1[LMAX] ;
30  char p2[LMAX] ;
31  bool indic = is_ope_bin(nom, p1, p2, '=') ;
32  if (!indic) {
33  cerr << "= needed for equations" << endl ;
34  abort() ;
35  }
36  else {
37  // lhs == 0 ?
38  indic = ((p2[0]=='0') && (p2[1]==' ') && (p2[2]=='\0'));
39 
40  // no lhs
41  if (indic)
42  return give_ope(dom, p1, boundary);
43  else
44  return new Ope_sub(this, give_ope(dom, p1, boundary), give_ope(dom, p2, boundary));
45  }
46 }

48  Ope_eq* System_of_eqs::parse_eq_trim (int dom, const char* nom, int boundary, bool first) const {
49  // Is the eq written like "... = ..." ?
50  char p1[LMAX] ;
51  char p2[LMAX] ;
52  bool indic = is_ope_bin(nom, p1, p2, '=');
53 
54  if (!indic) {
55  char auxi[LMAX] ;
56  trim_spaces(auxi, nom);
57 
58  // Version without =
59  return give_ope(dom, auxi, boundary);
60  }
61  else {
62  return first ? give_ope(dom, p1, boundary) : give_ope(dom, p2, boundary);
63  }
64 }

98 void System_of_eqs::add_eq_bc_exception (int dom, int bound, const char* nom, const char* const_part) {
99  // Is it written like =0 ?
100  char p1[LMAX] ;
101  char p2[LMAX] ;
102  bool indic1 = is_ope_bin(nom, p1, p2, '=') ;
103 
104  char p3[LMAX] ;
105  char p4[LMAX] ;
106  bool indic2 = is_ope_bin(const_part, p3, p4, '=') ;
107 
108 
109  if ((!indic1) || (!indic2)) {
110  cerr << "= needed for equations" << endl ;
111  abort() ;
112  }
113  else {
114  // Verif lhs1 = 0 ?
115  indic1 = ((p2[0]=='0') && (p2[1]==' ') && (p2[2]=='\0')) ?
116  true : false ;
117 
118  indic2 = ((p4[0]=='0') && (p4[1]==' ') && (p4[2]=='\0')) ?
119  true : false ;
120 
121  // no lhs :
122  if ((indic1) && (indic2))
123  eq[neq] = new Eq_bc_exception (espace.get_domain(dom), dom, bound, give_ope(dom, p1), give_ope(dom,p3)) ;
124  // lhs in 1
125  if ((!indic1) && (indic2))
126  eq[neq] = new Eq_bc_exception(espace.get_domain(dom), dom, bound, new Ope_sub(this, give_ope(dom, p1), give_ope(dom, p2)), give_ope(dom,p3)) ;
127  // lhs in 2
128  if ((indic1) && (!indic2))
129  eq[neq] = new Eq_bc_exception(espace.get_domain(dom), dom, bound, give_ope(dom,p1), new Ope_sub(this, give_ope(dom, p3), give_ope(dom, p4))) ;
130  // both lhs
131  if ((!indic1) && (!indic2))
132  eq[neq] = new Eq_bc_exception(espace.get_domain(dom), dom, bound, new Ope_sub(this, give_ope(dom, p1), give_ope(dom, p2)), new Ope_sub(this, give_ope(dom, p3), give_ope(dom, p4))) ;
133 
134  neq ++ ;
135  }
136  nbr_conditions = -1 ;
137 }

216 void System_of_eqs::add_eq_matching_non_std (int dom, int bound, const char* nom, int n_cmp, Array<int>** p_cmp) {
217 
218  // First get the number, the indices and associated boundaries of the other domains (member of espace) :
219  Array<int> other_props (espace.get_indices_matching_non_std (dom, bound)) ;
220 
221  eq_list.push_back(std::make_tuple(nom, dom, bound));
222 
223  // The equation
224  eq[neq] = new Eq_matching_non_std(espace.get_domain(dom), dom, bound, other_props, n_cmp, p_cmp) ;
225 
226  // Affectation of the operator in each concerned domain :
227  // Current one :bool indic = is_ope_bin(nom, p1, p2, '=') ;
228  // Is it written with = ?
229  char p1[LMAX] ;
230  char p2[LMAX] ;
231  bool indic = is_ope_bin(nom, p1, p2, '=') ;
232 
233  if (!indic) {
234  char auxi[LMAX] ;
235  trim_spaces(auxi, nom) ;
236  // Version without =
237  eq[neq]->parts[0] = give_ope (dom, auxi, bound) ;
238  // The associated ones :
239  for (int i=0 ; i<eq[neq]->n_ope-1 ; i++)
240  eq[neq]->parts[i+1] = give_ope (other_props(0,i), auxi, other_props(1,i)) ;
241  neq++ ;
242  }
243  else {
244  // Version without =
245  eq[neq]->parts[0] = give_ope (dom, p1, bound) ;
246  // The associated ones :
247  for (int i=0 ; i<eq[neq]->n_ope-1 ; i++)
248  eq[neq]->parts[i+1] = give_ope (other_props(0,i), p2, other_props(1,i)) ;
249  neq++ ;
250  }
251 
252  nbr_conditions = -1 ;
253 }
