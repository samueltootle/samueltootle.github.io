34 void Eq_matching::export_val(int& conte, Term_eq** residus, Array<double>& sec, int& pos_res) const {
35 
36  assert (residus[conte]->get_type_data()==TERM_T) ;
37  assert (residus[conte+1]->get_type_data()==TERM_T) ;
38 
39  int start = pos_res ;
40  dom->export_tau_boundary (residus[conte]->get_val_t(), ndom, bound, sec, pos_res, *n_cond, n_cmp_used, p_cmp_used) ;
41  Array<double> auxi (pos_res - start) ;
42  auxi = 0. ;
43  int zero = 0 ;
44  residus[conte+1]->get_val_t().get_space().get_domain(other_dom)->export_tau_boundary
45  (residus[conte+1]->get_val_t(), other_dom, other_bound, auxi, zero, *n_cond, n_cmp_used, p_cmp_used) ;
46  for (int i =start ; i<pos_res ; i++)
47  sec.set(i) -= auxi(i-start) ;
48  conte +=2 ;
49 }
