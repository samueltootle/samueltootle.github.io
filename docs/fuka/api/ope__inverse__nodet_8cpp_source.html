30 Term_eq Ope_inverse_nodet::action() const {
31 
32  Term_eq target (parts[0]->action()) ;
33 
34  // Various checks
35  if (target.type_data != TERM_T) {
36  cerr << "Ope_inverse only defined with respect for a tensor" << endl ;
37  abort() ;
38  }
39  if (target.val_t->get_valence()!=2) {
40  cerr << "Ope_inverse only defined with respect to a second order tensor" << endl ;
41  abort() ;
42  }
43  if (target.val_t->get_index_type(0)!=target.val_t->get_index_type(1)) {
44  cerr << "Ope_inverse only defined with respect to a tensor which indices are of the same type" << endl ;
45  abort() ;
46  }
47  int typeresult = - target.val_t->get_index_type(0) ;
48  int dim = target.val_t->get_space().get_ndim() ;
49 
50  bool m_doder = (target.der_t==0x0) ? false : true ;
51 
52  Term_eq** res(new Term_eq* [dim*(dim + 1)/2]); // 6 components to compute...
53  Scalar val(target.val_t->get_space()); // val and cmpval are more or less the same intermediate, but with different types
54  Scalar der(target.val_t->get_space());
55  Val_domain cmpval(target.val_t->get_space().get_domain(dom));
56  Val_domain cmpder(target.val_t->get_space().get_domain(dom));
57 
58 
59 
60  // compute the transposed comatrix
61  double signature;
62  gsl_permutation* p(gsl_permutation_calloc(dim-1)); // initialise to identity 0, 1
63  for (int i(1) ; i <= dim ; ++i) // sum over components, only the upper triangular part
64  {
65  for (int j(i) ; j <= dim ; ++j)
66  {
67  cmpval = 0.0; // initialise to 0 before each computation
68  cmpder = 0.0; // initialise to 0 before each computation
69  do
70  {
71  signature = sign(p);
72  int p1(static_cast<int>(gsl_permutation_get(p, 0)) + 1); // always add one for tensor indices, now we have a permutation of 1, 2
73  int p2(static_cast<int>(gsl_permutation_get(p, 1)) + 1);
74  vector<int> index1, index2; // manage the indices of the transposed comatrix
75  index1 = ind_com(i, j, p1, 1); // we need to play a bit with indices to get transpose(COM(metric))
76  index2 = ind_com(i, j, p2, 2);
77  cmpval += pow(-1.0, i + j)*signature*(*target.val_t)(index1[0], index1[1])(dom)*(*target.val_t)(index2[0],index2[1])(dom); // sum on permutations
78  if (m_doder)
79  cmpder += pow(-1.0, i + j)*signature*(*target.der_t)(index1[0], index1[1])(dom)*(*target.val_t)(index2[0], index2[1])(dom)
80  + pow(-1.0, i + j)*signature*(*target.val_t)(index1[0], index1[1])(dom)*(*target.der_t)(index2[0], index2[1])(dom);
81  }while(gsl_permutation_next(p) == GSL_SUCCESS);
82  gsl_permutation_free(p) ;
83  p = gsl_permutation_calloc(dim - 1); // reinitialise to identity 0, 1, ready to be used for next component
84  val.set_domain(dom) = cmpval ; // now we have the component of the inverse matrix
85  if (m_doder)
86  {
87  der.set_domain(dom) = cmpder ;
88  res[j - 1 + dim*(i - 1) - i*(i - 1)/2] = new Term_eq (dom, val, der); // just save upper triangular components in a 1D array of Term_eq (the indice will run throug 0, 1, 2, 3, 4, 5 with i, j)
89  }
90  else
91  res[j - 1 + dim*(i - 1) - i*(i - 1)/2] = new Term_eq (dom, val);
92  }
93  }
94  gsl_permutation_free(p);
95 
96  // Value field :
97  Metric_tensor resval(target.val_t->get_space() , typeresult, target.val_t->get_basis());
98  Metric_tensor resder(target.val_t->get_space() , typeresult, target.val_t->get_basis());
99  for (int i(1) ; i <= dim ; ++i)
100  for (int j(i) ; j <= dim ; ++j)
101  resval.set(i, j) = res[j - 1 + dim*(i - 1) - i*(i - 1)/2]->get_val_t();
102 
103  // Der field
104  if (m_doder) {
105  for (int i(1) ; i <= dim ; ++i)
106  for (int j(i) ; j <= dim ; ++j)
107  resder.set(i, j) = res[j - 1 + dim*(i - 1) - i*(i - 1)/2]->get_der_t();
108  }
109 
110  for (int i(0) ; i < dim*(dim + 1)/2 ; ++i)
111  delete res[i];
112  delete[] res;
113 
114  if (!m_doder)
115  return Term_eq (dom, resval) ;
116  else
117  return Term_eq (dom, resval, resder) ;
118 }
