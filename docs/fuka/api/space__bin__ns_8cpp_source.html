36 double func_abns (double aa, const Param& par) {
37  double r1 = par.get_double(0) ;
38  double r2 = par.get_double(1) ;
39  double d = par.get_double(2) ;
40  return (sqrt(aa*aa+r1*r1)+sqrt(aa*aa+r2*r2)-d) ;
41 }

43 Space_bin_ns::Space_bin_ns (int ttype, double dist, double rinstar1, double rstar1, double routstar1,
44  double rinstar2, double rstar2, double routstar2, double rext, int nr, int nshells1, int nshells2) {
45 
46  ndim = 3 ;
47 
48  n_shells1 = nshells1;
49  n_shells2 = nshells2;
50 
51  NS1 = 0;
52  NS2 = 3 + n_shells1;
53  ADAPTED1 = NS1 + n_shells1 + 1;
54  ADAPTED2 = NS2 + n_shells2 + 1;
55  OUTER = 6 + n_shells1 + n_shells2;
56 
57  n_shells_outer = 0;
58  nbr_domains = 12 + n_shells1 + n_shells2;
59  type_base = ttype ;
60  domains = new Domain* [nbr_domains] ;
61 
62  Dim_array res (ndim) ;
63  res.set(0) = nr ; res.set(1) = nr ; res.set(2) = nr-1 ;
64  Dim_array res_bi(ndim) ;
65  res_bi.set(0) = nr ; res_bi.set(1) = nr ; res_bi.set(2) = nr ;
66 
67  // Bispheric :
68  // Computation of aa
69  Param par_a ;
70  par_a.add_double(routstar1,0) ;
71  par_a.add_double(routstar2,1) ;
72  par_a.add_double(dist,2) ;
73  double a_min = 0 ;
74  double a_max = dist/2. ;
75  double precis = PRECISION ;
76  int nitermax = 500 ;
77  int niter ;
78  double aa = zerosec(func_abns, par_a, a_min, a_max, precis, nitermax, niter) ;
79  double eta_plus = asinh(aa/routstar2) ;
80  double eta_minus = -asinh(aa/routstar1) ;
81 
82  double chi_c = 2*atan(aa/rext) ;
83  double eta_c = log((1+rext/aa)/(rext/aa-1)) ;
84  double eta_lim = eta_c/2. ;
85  double chi_lim = chi_lim_eta (eta_lim, rext, aa, chi_c) ;
86  // First NS
87  Point center_minus (ndim) ;
88  center_minus.set(1) = aa*cosh(eta_minus)/sinh(eta_minus) ;
89  domains[NS1] = new Domain_nucleus (NS1, ttype, rinstar1, center_minus, res) ;
90 
91  double delta_r1 = (rstar1 - rinstar1);
92  std::vector<double> r_shells1(n_shells1+1);
93  //r_shells1[0] = rinstar1;
94  for(int i = 0; i <= n_shells1; ++i) {
95  //r_shells1[i] = (delta_r1/(M_PI/2))*atan(i+1)+rstar1; //puts shells closer to router
96  r_shells1[i] = rinstar1 + delta_r1 / (nshells1 + 1) * i;
97  //std::cout << r_shells1[i] << std::endl;
98  }
99 
100  for(int i = 0; i < n_shells1; ++i) {
101  domains[NS1+1+i] = new Domain_shell(NS1+1+i, ttype, r_shells1[i], r_shells1[i+1], center_minus, res);
102  }
103 
104  domains[ADAPTED1] = new Domain_shell_outer_adapted (*this, ADAPTED1, ttype, r_shells1[n_shells1], rstar1, center_minus, res) ;
105  domains[ADAPTED1+1] = new Domain_shell_inner_adapted (*this, ADAPTED1+1, ttype, rstar1, routstar1, center_minus, res) ;
106 
107  // Second NS
108  Point center_plus (ndim) ;
109  center_plus.set(1) = aa*cosh(eta_plus)/sinh(eta_plus) ;
110  domains[NS2] = new Domain_nucleus (NS2, ttype, rinstar2, center_plus, res) ;
111 
112  double delta_r2 = (rstar2 - rinstar2);
113  std::vector<double> r_shells2(n_shells2+1);
114  //r_shells2[0] = rinstar2;
115  for(int i = 0; i <= n_shells2; ++i) {
116  //r_shells2[i] = (delta_r1/(M_PI/2))*atan(i+1)+rstar2; //puts shells closer to router
117  r_shells2[i] = rinstar2 + delta_r2 / (nshells2 + 1) * i;
118  //std::cout << r_shells2[i] << std::endl;
119  }
120 
121  for(int i = 0; i < n_shells2; ++i) {
122  domains[NS2+1+i] = new Domain_shell(NS2+1+i, ttype, r_shells2[i], r_shells2[i+1], center_plus, res);
123  }
124 
125  domains[ADAPTED2] = new Domain_shell_outer_adapted (*this, ADAPTED2, ttype, r_shells2[n_shells2], rstar2, center_plus, res) ;
126  domains[ADAPTED2+1] = new Domain_shell_inner_adapted (*this, ADAPTED2+1, ttype, rstar2, routstar2, center_plus, res) ;
127 
128  // Bispheric part
129  domains[OUTER] = new Domain_bispheric_chi_first(OUTER, ttype, aa, eta_minus, rext, chi_lim, res_bi) ;
130  domains[OUTER+1] = new Domain_bispheric_rect(OUTER+1, ttype, aa, rext, eta_minus, -eta_lim, chi_lim, res_bi) ;
131  domains[OUTER+2] = new Domain_bispheric_eta_first(OUTER+2, ttype, aa, rext, -eta_lim, eta_lim, res_bi) ;
132  domains[OUTER+3] = new Domain_bispheric_rect(OUTER+3, ttype, aa, rext, eta_plus, eta_lim, chi_lim, res_bi) ;
133  domains[OUTER+4] = new Domain_bispheric_chi_first(OUTER+4, ttype, aa, eta_plus, rext, chi_lim, res_bi) ;
134 
135  // Compactified domain
136  Point center(3) ;
137  domains[OUTER+5] = new Domain_compact (OUTER+5, ttype, rext, center, res) ;
138 
139  const Domain_shell_outer_adapted* pouter_1 = dynamic_cast<const Domain_shell_outer_adapted*> (domains[ADAPTED1]) ;
140  pouter_1->vars_to_terms() ;
141  pouter_1->update() ;
142  const Domain_shell_inner_adapted* pinner_1 = dynamic_cast<const Domain_shell_inner_adapted*> (domains[ADAPTED1+1]) ;
143  pinner_1->vars_to_terms() ;
144  pinner_1->update() ;
145  const Domain_shell_outer_adapted* pouter_2 = dynamic_cast<const Domain_shell_outer_adapted*> (domains[ADAPTED2]) ;
146  pouter_2->vars_to_terms() ;
147  pouter_2->update() ;
148  const Domain_shell_inner_adapted* pinner_2 = dynamic_cast<const Domain_shell_inner_adapted*> (domains[ADAPTED2+1]) ;
149  pinner_2->vars_to_terms() ;
150  pinner_2->update() ;
151 }

153 Space_bin_ns::Space_bin_ns (int ttype, double dist, double rinstar1, double rstar1, double routstar1,
154  double rinstar2, double rstar2, double routstar2, double rext, double rshell, int nr) {
155 
156  ndim = 3 ;
157 
158  n_shells1 = 0;
159  n_shells2 = 0;
160 
161  NS1 = 0;
162  NS2 = 3;
163  ADAPTED1 = NS1 + n_shells1 + 1;
164  ADAPTED2 = NS2 + n_shells2 + 1;
165  OUTER = 6;
166 
167  n_shells_outer = 1 ;
168  nbr_domains = 13 ;
169  type_base = ttype ;
170  domains = new Domain* [nbr_domains] ;
171 
172  Dim_array res (ndim) ;
173  res.set(0) = nr ; res.set(1) = nr ; res.set(2) = nr-1 ;
174  Dim_array res_bi(ndim) ;
175  res_bi.set(0) = nr ; res_bi.set(1) = nr ; res_bi.set(2) = nr ;
176 
177  // Bispheric :
178  // Computation of aa
179  Param par_a ;
180  par_a.add_double(routstar1,0) ;
181  par_a.add_double(routstar2,1) ;
182  par_a.add_double(dist,2) ;
183  double a_min = 0 ;
184  double a_max = dist/2. ;
185  double precis = PRECISION ;
186  int nitermax = 500 ;
187  int niter ;
188  double aa = zerosec(func_abns, par_a, a_min, a_max, precis, nitermax, niter) ;
189  double eta_plus = asinh(aa/routstar2) ;
190  double eta_minus = -asinh(aa/routstar1) ;
191 
192  double chi_c = 2*atan(aa/rext) ;
193  double eta_c = log((1+rext/aa)/(rext/aa-1)) ;
194  double eta_lim = eta_c/2. ;
195  double chi_lim = chi_lim_eta (eta_lim, rext, aa, chi_c) ;
196 
197  // First NS
198  Point center_minus (ndim) ;
199  center_minus.set(1) = aa*cosh(eta_minus)/sinh(eta_minus) ;
200  domains[0] = new Domain_nucleus (0, ttype, rinstar1, center_minus, res) ;
201  domains[1] = new Domain_shell_outer_adapted (*this, 1, ttype, rinstar1, rstar1, center_minus, res) ;
202  domains[2] = new Domain_shell_inner_adapted (*this, 2, ttype, rstar1, routstar1, center_minus, res) ;
203 
204  // Second NS
205  Point center_plus (ndim) ;
206  center_plus.set(1) = aa*cosh(eta_plus)/sinh(eta_plus) ;
207  domains[3] = new Domain_nucleus (3, ttype, rinstar2, center_plus, res) ;
208  domains[4] = new Domain_shell_outer_adapted (*this, 4, ttype, rinstar2, rstar2, center_plus, res) ;
209  domains[5] = new Domain_shell_inner_adapted (*this, 5, ttype, rstar2, routstar2, center_plus, res) ;
210 
211  // Bispheric part
212  domains[6] = new Domain_bispheric_chi_first(6, ttype, aa, eta_minus, rext, chi_lim, res_bi) ;
213  domains[7] = new Domain_bispheric_rect(7, ttype, aa, rext, eta_minus, -eta_lim, chi_lim, res_bi) ;
214  domains[8] = new Domain_bispheric_eta_first(8, ttype, aa, rext, -eta_lim, eta_lim, res_bi) ;
215  domains[9] = new Domain_bispheric_rect(9, ttype, aa, rext, eta_plus, eta_lim, chi_lim, res_bi) ;
216  domains[10] = new Domain_bispheric_chi_first(10, ttype, aa, eta_plus, rext, chi_lim, res_bi) ;
217 
218  // Shell in 1/R
219  Point center(3) ;
220  domains[11] = new Domain_shell (11, ttype, rext, rshell, center, res) ;
221 
222  // Compactified domain
223  domains[12] = new Domain_compact (12, ttype, rshell, center, res) ;
224 
225  const Domain_shell_outer_adapted* pouter_1 = dynamic_cast<const Domain_shell_outer_adapted*> (domains[1]) ;
226  pouter_1->vars_to_terms() ;
227  pouter_1->update() ;
228  const Domain_shell_inner_adapted* pinner_1 = dynamic_cast<const Domain_shell_inner_adapted*> (domains[2]) ;
229  pinner_1->vars_to_terms() ;
230  pinner_1->update() ;
231  const Domain_shell_outer_adapted* pouter_2 = dynamic_cast<const Domain_shell_outer_adapted*> (domains[4]) ;
232  pouter_2->vars_to_terms() ;
233  pouter_2->update() ;
234  const Domain_shell_inner_adapted* pinner_2 = dynamic_cast<const Domain_shell_inner_adapted*> (domains[5]) ;
235  pinner_2->vars_to_terms() ;
236  pinner_2->update() ;
237 }

239 Space_bin_ns::Space_bin_ns (Space_bin_ns const & sp) {
240 
241  nbr_domains = sp.nbr_domains;
242  ndim = sp.ndim;
243  type_base = sp.type_base;
244 
245  n_shells1 = sp.n_shells1;
246  n_shells2 = sp.n_shells2;
247 
248  n_inner_shells1 = sp.n_inner_shells1;
249  n_inner_shells2 = sp.n_inner_shells2;
250 
251  // We calculate domain indicies and ensure they match
252  // the imported space - should be consistent!
253  this->NS1 = 0;
254  assert(NS1 == sp.NS1);
255 
256  this->NS2 = 3 + n_inner_shells1 + n_shells1;
257  assert(NS2 == sp.NS2);
258 
259  this->ADAPTED1 = NS1 + n_inner_shells1 + 1;
260  assert(ADAPTED1 == sp.ADAPTED1);
261 
262  this->ADAPTED2 = NS2 + n_inner_shells2 + 1;
263  assert(ADAPTED2 == sp.ADAPTED2);
264 
265  this->OUTER = 6 + n_shells1 + n_inner_shells1 + n_shells2 + n_inner_shells2;
266  assert(OUTER == sp.OUTER);
267 
268  this->n_shells_outer = nbr_domains - 12 - n_inner_shells1 - n_shells1 - n_inner_shells2 - n_shells2;
269  assert(n_shells_outer >= 0 && n_shells_outer == sp.n_shells_outer);
270 
271  domains = new Domain* [nbr_domains] ;
272 
273  auto add_spherical_shells = [&](auto start_idx, auto nshells) {
274  for(int i = 0; i < nshells; ++i) {
275  const Domain_shell* d_shell = dynamic_cast<const Domain_shell*> (sp.get_domain(start_idx+i)) ;
276  domains[start_idx+i] = new Domain_shell(*d_shell, true);
277  }
278  };
279 
280  //First NS :
281  const Domain_nucleus* d_nuc1 = dynamic_cast<const Domain_nucleus*> (sp.get_domain(NS1)) ;
282  domains[NS1] = new Domain_nucleus(*d_nuc1, true);
283 
284  add_spherical_shells(NS1+1, n_inner_shells1);
285 
286  const Domain_shell_outer_adapted* sp_pouter_1 = dynamic_cast<const Domain_shell_outer_adapted*> (sp.get_domain(ADAPTED1)) ;
287  domains[ADAPTED1] = new Domain_shell_outer_adapted(*this, *sp_pouter_1) ;
288 
289  const Domain_shell_inner_adapted* sp_pinner_1 = dynamic_cast<const Domain_shell_inner_adapted*> (sp.get_domain(ADAPTED1+1)) ;
290  domains[ADAPTED1+1] = new Domain_shell_inner_adapted(*this, *sp_pinner_1) ;
291 
292  add_spherical_shells(ADAPTED1+2, n_shells1);
293 
294  //second NS :
295  const Domain_nucleus* d_nuc2 = dynamic_cast<const Domain_nucleus*> (sp.get_domain(NS2)) ;
296  domains[NS2] = new Domain_nucleus(*d_nuc2, true);
297 
298  add_spherical_shells(NS2+1, n_inner_shells2);
299 
300  const Domain_shell_outer_adapted* sp_pouter_2 = dynamic_cast<const Domain_shell_outer_adapted*> (sp.get_domain(ADAPTED2)) ;
301  domains[ADAPTED2] = new Domain_shell_outer_adapted(*this, *sp_pouter_2) ;
302 
303  const Domain_shell_inner_adapted* sp_pinner_2 = dynamic_cast<const Domain_shell_inner_adapted*> (sp.get_domain(ADAPTED2+1)) ;
304  domains[ADAPTED2+1] = new Domain_shell_inner_adapted(*this, *sp_pinner_2) ;
305 
306  add_spherical_shells(ADAPTED2+2, n_shells2);
307 
308  // Bispheric
309  const Domain_bispheric_chi_first* sp_chi_1 = dynamic_cast<const Domain_bispheric_chi_first*> (sp.get_domain(OUTER)) ;
310  domains[OUTER] = new Domain_bispheric_chi_first(*this, *sp_chi_1) ;
311 
312  const Domain_bispheric_rect* sp_rec_1 = dynamic_cast<const Domain_bispheric_rect*> (sp.get_domain(OUTER+1)) ;
313  domains[OUTER+1] = new Domain_bispheric_rect(*this, *sp_rec_1) ;
314 
315  const Domain_bispheric_eta_first* sp_eta = dynamic_cast<const Domain_bispheric_eta_first*> (sp.get_domain(OUTER+2)) ;
316  domains[OUTER+2] = new Domain_bispheric_eta_first(*this, *sp_eta) ;
317 
318  const Domain_bispheric_rect* sp_rec_2 = dynamic_cast<const Domain_bispheric_rect*> (sp.get_domain(OUTER+3)) ;
319  domains[OUTER+3] = new Domain_bispheric_rect(*this, *sp_rec_2) ;
320 
321  const Domain_bispheric_chi_first* sp_chi_2 = dynamic_cast<const Domain_bispheric_chi_first*> (sp.get_domain(OUTER+4)) ;
322  domains[OUTER+4] = new Domain_bispheric_chi_first(*this, *sp_chi_2) ;
323 
324  add_spherical_shells(OUTER+5, n_shells_outer);
325 
326  // Compactified
327  const Domain_compact* d_compact = dynamic_cast<const Domain_compact*> (sp.get_domain(nbr_domains-1)) ;
328  domains[nbr_domains-1] = new Domain_compact(*d_compact, true) ;
329 
330  const Domain_shell_outer_adapted* pouter_1 = dynamic_cast<const Domain_shell_outer_adapted*> (domains[ADAPTED1]) ;
331  pouter_1->vars_to_terms() ;
332  pouter_1->update() ;
333  const Domain_shell_inner_adapted* pinner_1 = dynamic_cast<const Domain_shell_inner_adapted*> (domains[ADAPTED1+1]) ;
334  pinner_1->vars_to_terms() ;
335  pinner_1->update() ;
336  const Domain_shell_outer_adapted* pouter_2 = dynamic_cast<const Domain_shell_outer_adapted*> (domains[ADAPTED2]) ;
337  pouter_2->vars_to_terms() ;
338  pouter_2->update() ;
339  const Domain_shell_inner_adapted* pinner_2 = dynamic_cast<const Domain_shell_inner_adapted*> (domains[ADAPTED2+1]) ;
340  pinner_2->vars_to_terms() ;
341  pinner_2->update() ;
342 }

344 Space_bin_ns::Space_bin_ns (FILE* fd, bool old) {
345  fread_be (&nbr_domains, sizeof(int), 1, fd) ;
346  fread_be (&n_inner_shells1, sizeof(int), 1, fd) ;
347  fread_be (&n_inner_shells2, sizeof(int), 1, fd) ;
348 
349  // in the original BNS, n_shells referred to interior shells
350  // whereas it now refers to exterior shells like for the BBH and BHNS
351  if(!old) {
352  fread_be (&n_shells1, sizeof(int), 1, fd) ;
353  fread_be (&n_shells2, sizeof(int), 1, fd) ;
354  }
355  fread_be (&ndim, sizeof(int), 1, fd) ;
356  fread_be (&type_base, sizeof(int), 1, fd) ;
357 
358  n_shells_outer = nbr_domains - 12 - n_inner_shells1 - n_shells1 - n_shells2 - n_inner_shells2;
359  assert(n_shells_outer >= 0);
360 
361  NS1 = 0;
362  NS2 = 3 + n_inner_shells1 + n_shells1;
363  ADAPTED1 = NS1 + n_inner_shells1 + 1;
364  ADAPTED2 = NS2 + n_inner_shells2 + 1;
365  OUTER = 6 + n_shells1 + n_inner_shells1 + n_shells2 + n_inner_shells2;
366 
367  domains = new Domain* [nbr_domains] ;
368 
369  //First NS :
370  domains[NS1] = new Domain_nucleus (NS1, fd) ;
371 
372  for(int i = 0; i < n_inner_shells1; ++i)
373  domains[NS1+1+i] = new Domain_shell(NS1+1+i, fd);
374 
375  domains[ADAPTED1] = new Domain_shell_outer_adapted (*this, ADAPTED1, fd) ;
376  domains[ADAPTED1+1] = new Domain_shell_inner_adapted (*this, ADAPTED1+1, fd) ;
377 
378  for(int i = 0; i < n_shells1; ++i)
379  domains[ADAPTED1+2+i] = new Domain_shell(ADAPTED1+2+i, fd);
380 
381  //second NS :
382  domains[NS2] = new Domain_nucleus (NS2, fd) ;
383 
384  for(int i = 0; i < n_inner_shells2; ++i)
385  domains[NS2+1+i] = new Domain_shell(NS2+1+i, fd);
386 
387  domains[ADAPTED2] = new Domain_shell_outer_adapted (*this, ADAPTED2, fd) ;
388  domains[ADAPTED2+1] = new Domain_shell_inner_adapted (*this, ADAPTED2+1, fd) ;
389 
390  for(int i = 0; i < n_shells2; ++i)
391  domains[ADAPTED2+2+i] = new Domain_shell(ADAPTED2+2+i, fd);
392 
393  // Bispheric
394  domains[OUTER] = new Domain_bispheric_chi_first(OUTER, fd) ;
395  domains[OUTER+1] = new Domain_bispheric_rect(OUTER+1, fd) ;
396  domains[OUTER+2] = new Domain_bispheric_eta_first(OUTER+2, fd) ;
397  domains[OUTER+3] = new Domain_bispheric_rect(OUTER+3, fd) ;
398  domains[OUTER+4] = new Domain_bispheric_chi_first(OUTER+4, fd) ;
399 
400  for (int i=0 ; i<n_shells_outer ; i++)
401  domains[OUTER+5+i] = new Domain_shell (OUTER+5+i, fd) ;
402 
403  // Compactified
404  domains[OUTER+5+n_shells_outer] = new Domain_compact(OUTER+5+n_shells_outer, fd) ;
405 
406  const Domain_shell_outer_adapted* pouter_1 = dynamic_cast<const Domain_shell_outer_adapted*> (domains[ADAPTED1]) ;
407  pouter_1->vars_to_terms() ;
408  pouter_1->update() ;
409  const Domain_shell_inner_adapted* pinner_1 = dynamic_cast<const Domain_shell_inner_adapted*> (domains[ADAPTED1+1]) ;
410  pinner_1->vars_to_terms() ;
411  pinner_1->update() ;
412  const Domain_shell_outer_adapted* pouter_2 = dynamic_cast<const Domain_shell_outer_adapted*> (domains[ADAPTED2]) ;
413  pouter_2->vars_to_terms() ;
414  pouter_2->update() ;
415  const Domain_shell_inner_adapted* pinner_2 = dynamic_cast<const Domain_shell_inner_adapted*> (domains[ADAPTED2+1]) ;
416  pinner_2->vars_to_terms() ;
417  pinner_2->update() ;
418 
419 }

754  double eta_minus = -asinh(aa/NS1_bounds.back()) ;
