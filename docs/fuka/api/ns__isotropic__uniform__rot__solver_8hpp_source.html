98  // Update bconfig(HC) and bconfig(NC)
99  void update_config_quantities(System_of_eqs& syst) {
100  auto rs = bco_utils::get_rmin_rmax(space, 1);
101  bconfig.set(BCO_PARAMS::RMID) = rs[0];
102 
103  // compute the ADM mass as surface integral at infinity
104  Val_domain integMadm(syst.give_val_def("intMadm")()(ndom - 1));
105  double Madm = space.get_domain(ndom - 1)->integ(integMadm, OUTER_BC);
106  bconfig.set(BCO_PARAMS::MADM) = Madm;
107 
108  // compute the baryonic mass at volume integral from the given integrant
109  double baryonic_mass = syst.give_val_def("intMb")()(0).integ_volume() +
110  syst.give_val_def("intMb")()(1).integ_volume();
111  bconfig.set(BCO_PARAMS::MB) = baryonic_mass;
112 
113  auto loghc = bco_utils::get_boundary_val(0, logh, INNER_BC);
114 
115  // compute the ADM Angular Momentum as surface integral at infinity
116  Val_domain integJ(syst.give_val_def("intJ")()(ndom - 1));
117  double Jadm = space.get_domain(ndom - 1)->integ(integJ, OUTER_BC);
118  double chi = Jadm / Madm / Madm;
119 
120  if (seq) {
121  auto idx{seq->mass_idx()};
122  switch (idx) {
123  case BCO_PARAMS::HC:
124  bconfig.set(BCO_PARAMS::NC) =
125  EOS<eos_t, DENSITY>::get(bconfig(BCO_PARAMS::HC));
126  break;
127  case BCO_PARAMS::NC:
128  bconfig.set(BCO_PARAMS::HC) = std::exp(loghc);
129  break;
130  default:
131  bconfig.set(BCO_PARAMS::HC) = std::exp(loghc);
132  bconfig.set(BCO_PARAMS::NC) =
133  EOS<eos_t, DENSITY>::get(bconfig(BCO_PARAMS::HC));
134  break;
135  }
136 
137  idx = seq->spin_idx();
138  switch (idx) {
139  case BCO_PARAMS::CHI:
140  break;
141  default:
142  bconfig.set(BCO_PARAMS::CHI) = chi;
143  break;
144  }
145  } else {
146  bconfig.set(BCO_PARAMS::NC) =
147  EOS<eos_t, DENSITY>::get(bconfig(BCO_PARAMS::HC));
148  bconfig.set(BCO_PARAMS::CHI) = chi;
149  }
150  }
151 };
