202 template <class tree_t>
203 auto find_leaf(tree_t const& tree, std::string key) {
204 
205  std::string branch_name{};
206 
207  auto search_branch =
208  [&](auto& branch) -> std::tuple<std::string, std::string> {
209  for (auto const& node : branch) {
210  if (node.second.empty())
211  if (node.first == key) {
212  return std::make_tuple(node.first, node.second.data());
213  }
214  }
215  return {};
216  };
217  std::function<std::tuple<std::string, std::string, std::string>(
218  tree_t const&)>
219  recursive_search;
220  recursive_search =
221  [&](auto& branch) -> std::tuple<std::string, std::string, std::string> {
222  auto tmp = search_branch(branch);
223 
224  if (!std::get<0>(tmp).empty()) {
225 #ifdef DEBUG
226  std::cout << branch_name << ", " << std::get<0>(tmp) << ","
227  << std::get<1>(tmp) << std::endl;
228 #endif
229  return std::make_tuple(branch_name, std::get<0>(tmp), std::get<1>(tmp));
230  }
231  for (auto const& node : branch) {
232  if (!node.second.empty()) {
233  branch_name = node.first;
234  tree_t t_branch = read_branch(branch, branch_name);
235  auto tmp_tuple = recursive_search(t_branch);
236  if (!std::get<0>(tmp_tuple).empty()) {
237  return tmp_tuple;
238  }
239  }
240  }
241  return {};
242  };
243  auto res = recursive_search(tree);
244 
245  return res;
246 }
