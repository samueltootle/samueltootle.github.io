192 template <class tree_t>
193 auto find_leaf(tree_t const & tree, std::string key) {
194 
195  std::string branch_name{};
196 
197  auto search_branch = [&](auto& branch) -> std::tuple<std::string, std::string> {
198  for(auto const & node : branch) {
199  if(node.second.empty())
200  if(node.first == key) {
201  return std::make_tuple(node.first, node.second.data());
202  }
203  }
204  return {};
205  };
206  std::function<std::tuple<std::string,std::string,std::string>(tree_t const &)> recursive_search;
207  recursive_search = [&](auto& branch) -> std::tuple<std::string,std::string,std::string> {
208  auto tmp = search_branch(branch);
209 
210  if(!std::get<0>(tmp).empty()) {
211  #ifdef DEBUG
212  std::cout << branch_name << ", " << std::get<0>(tmp) << "," << std::get<1>(tmp) << std::endl;
213  #endif
214  return std::make_tuple(branch_name, std::get<0>(tmp), std::get<1>(tmp));
215  }
216  for(auto const & node : branch) {
217  if(!node.second.empty()) {
218  branch_name = node.first;
219  tree_t t_branch = read_branch(branch, branch_name);
220  auto tmp_tuple = recursive_search(t_branch);
221  if(!std::get<0>(tmp_tuple).empty()) {
222  return tmp_tuple;
223  }
224  }
225  }
226  return {};
227  };
228  auto res = recursive_search(tree);
229 
230  return res;
231 }
