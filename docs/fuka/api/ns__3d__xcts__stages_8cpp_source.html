22  auto const current = bconfig.config_filename();
23  if (!bconfig.control(RESOLVE) && solution_exists(stagename)) {
24  if (rank == 0)
25  std::cout << "Solved previously: " << bconfig.config_filename_abs()
26  << std::endl;
27  return (current == bconfig.config_filename()) ? EXIT_SUCCESS : RELOAD_FILE;
28  }
29 
30  update_fields_co(cfields, coord_vectors, {}, 0.);

33  System_of_eqs syst(space, 0, ndom - 1);
34  syst.add_var("H", logh);
35  syst_init(syst);
36 
37  for (int d = 0; d < ndom; d++) {
38  switch (d) {
39  // in the star the constraint equations are sourced by the matter
40  case 0:
41  case 1:
42  // sources
43  syst.add_def(d, "Etilde = press * h - press * delta");
44  syst.add_def(d, "Stilde = 3 * press * delta");
45 
46  // constraint equations
47  syst.add_def(d,
48  "eqP = delta * D^i D_i P + 4piG / 2. * P^5 * Etilde");
49  syst.add_def(d,
50  "eqNP = delta * D^i D_i NP - 4piG / 2. * N * P^5 * "

55  // first integral of the euler equation for a static, non-rotating star,
56  // i.e. a TOV
57  syst.add_def(d, "firstint = H + log(N)");
58 
59  break;
60  // outside the matter is absent and the sources are zero
61  default:
62  syst.add_eq_full(d, "H = 0");
63 
64  syst.add_def(d, "eqP = D^i D_i P");
65  syst.add_def(d, "eqNP = D^i D_i NP");

77  } else {

82  output_str = output.str();

103  space.add_eq(syst, "eqP = 0", "P", "dn(P)");
104

106  syst.add_eq_bc(ndom - 1, OUTER_BC, "N=1");
107  syst.add_eq_bc(ndom - 1, OUTER_BC, "P=1");

111

113  // and condition on the central value, either fixed directly or by the
114  // integral below
115  syst.add_eq_first_integral(0, 1, "firstint",
116  central_fixing_definition.c_str());
117 
118  // constrain stellar mass by these integrals and the central log enthalpy
119  if (seq) {
120  auto idx{seq->mass_idx()};
121  switch (idx) {
122  case BCO_PARAMS::MADM:
123  space.add_eq_int_inf(syst, "integ(intMadm) = Madm");

125  case BCO_PARAMS::MB:
126  space.add_eq_int_volume(syst, 2, "integvolume(intMb) = Mb");

128  default:
129  break;
130  }
131  } else {

133  space.add_eq_int_inf(syst, "integ(intMadm) = Madm");
134  }
135 
136  // print the variation of the surface radius over the whole star
137  if (rank == 0) {
138  auto rs = Kadath::bco_utils::get_rmin_rmax(space, 1);
139  std::cout << "[Rmin, Rmax] : [" << rs[0] << ", " << rs[1] << "]\n";
140  }

144  bool endloop = false;
145  int ite = 1;
146  double conv;
147 
148  // solve until convergence is achieved

153  update_config_quantities(syst);
154  // output files at this iteration and print diagnostics
155  std::stringstream ss;
156  ss << "norot_3d_";
157  if (fixed) {
158  ss << "fixed";
159  } else {
160  ss << "norot_bc";
161  }
162  ss << "_" << ite - 1;
163 
164  bconfig.set_filename(ss.str());
165  if (rank == 0) {
166  print_diagnostics_norot(syst, ite, conv);
167  std::cout << std::endl;
168  if (bconfig.control(CHECKPOINT))
169  checkpoint();
170  }
171 
172  // update all coordinate fields, in case the domain extents have changed
173  update_fields_co(cfields, coord_vectors, {}, 0.);
174 
175  ite++;
176  check_max_iter_exceeded(rank, ite, conv);
177  }
178 
179  bconfig.set_filename(converged_filename(stagename));
180  if (rank == 0) {
181  checkpoint();
182  }
183  return exit_status;
