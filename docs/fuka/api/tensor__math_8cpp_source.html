27 
28  assert (valence == t.valence) ;
29  assert (&espace==&t.espace) ;
30 
31  basis = t.basis ;
32 
33  // Case when one does not care about the indices :
34  if ((!t.name_affected) || (!name_affected)) {
35 
36  for (int i=0 ; i<valence ; i++)
37  assert(t.type_indice(i) == type_indice(i)) ;
38 
39  for (int i=0 ; i<n_comp ; i++) {

41  *cmp[i] = *t.cmp[place_t] ;
42  }
43  }
44  else {
45  // Find the permutation :

48  if (!same_ind) {

54  Array<int> ind (indices(i)) ;
55  Array<int> ind_t (valence) ;
56  for (int j=0 ; j<valence ; j++)
57  ind_t.set(perm(j)) = ind(j) ;
58  set(ind) = t(ind_t) ;
59  }

313 Tensor operator* (const Tensor& t1, const Tensor& t2) {
314 
315  assert (&t1.espace==&t2.espace) ;
316 
317  if (t1.valence==0) {
318  Tensor res (Scalar(t1)*t2) ;
319  return res ;
320  }
321  else if (t2.valence==0) {
322  Tensor res (t1*Scalar(t2)) ;
323  return res ;
324  }
325  else {
326  assert (t1.basis==t2.basis) ;
327 
328  if ((!t1.name_affected) || (!t2.name_affected)) {
329  // Standard product
330  int val_res = t1.valence+t2.valence ;
331  Array<int> ind_res (val_res) ;
332  for (int i=0 ; i<t1.valence ; i++)
333  ind_res.set(i) = t1.type_indice(i) ;
334  for (int i=0 ; i<t2.valence ; i++)
335  ind_res.set(i+t1.valence) = t2.type_indice(i) ;
336 
337  const Base_tensor& base = (t1.valence!=0) ? t1.basis : t2.basis ;
338  Tensor res (t1.espace, val_res, ind_res, base) ;
339 
340  Array<int> ind1 (t1.valence) ;
341  Array<int> ind2 (t2.valence) ;
342  for (int i=0 ; i<res.n_comp ; i++) {
343  ind_res = res.indices(i) ;
344  for (int j=0 ; j<t1.valence ; j++)
345  ind1.set(j) = ind_res(j) ;
346  for (int j=0 ; j<t2.valence ; j++)
347  ind2.set(j) = ind_res(j+t1.valence) ;
348  res.set(ind_res) = t1(ind1) * t2(ind2) ;
349  }
350 
351  return res ;
352  }
353 
354  else {
355  // Case with indices name
356 
357  // Check if one needs to sum on some indices :
358  Array<int> sum_1 (t1.valence) ;
359  Array<int> sum_2 (t2.valence) ;
360  for (int i=0 ; i<t2.valence ; i++)
361  sum_2.set(i) = -1 ;
362 
363  for (int i=0 ; i<t1.valence ; i++) {
364  sum_1.set(i) = -1 ;
365  for (int j=0 ; j<t2.valence ; j++)
366  if (t1.name_indice[i]==t2.name_indice[j]){
367  sum_1.set(i) = j ;
368  sum_2.set(j) = i ;
369  }
370  if ((sum_1(i)!=-1) && (t1.type_indice(i)==t2.type_indice(sum_1(i)))) {
371  cerr << "Can not sum on indices of the same type in operator*" << endl ;
372  abort() ;
373  }
374  }
375 
376  // Valence of the result
377  int val_res = 0 ;
378  for (int i=0 ; i<t1.valence ; i++)
379  if (sum_1(i)==-1)
380  val_res ++ ;
381  for (int i=0 ; i<t2.valence ; i++)
382  if (sum_2(i)==-1)
383  val_res ++ ;
384 
385  if (val_res>0) {
386  // Type on indices
387  Array<int> ind_res (val_res) ;
388  int conte = 0 ;
389  for (int i=0 ; i<t1.valence ; i++)
390  if (sum_1(i)==-1) {
391  ind_res.set(conte)=t1.type_indice(i) ;
392  conte ++ ;
393  }
394  for (int i=0 ; i<t2.valence ; i++)
395  if (sum_2(i)==-1) {
396  ind_res.set(conte) = t2.type_indice(i) ;
397  conte ++ ;
398  }
399 
400  // Le tenseur
401  const Base_tensor& base = (t1.valence!=0) ? t1.basis : t2.basis ;
402  Tensor res (t1.espace, val_res, ind_res, base) ;
403 
404  // Name of the remaining variables :
405  res.name_affected = true ;
406  conte = 0 ;
407  for (int i=0 ; i<t1.valence ; i++)
408  if (sum_1(i)==-1) {
409  res.name_indice[conte] = t1.name_indice[i] ;
410  conte ++ ;
411  }
412  for (int i=0 ; i<t2.valence ; i++)
413  if (sum_2(i)==-1) {
414  res.name_indice[conte] = t2.name_indice[i] ;
415  conte ++ ;
416  }
417 
418  Array<bool> first (res.get_n_comp()) ;
419  first = true ;
420 
421  Index pos_t1 (t1) ;
422  Index pos_t2 (t2) ;
423  Index pos_res (res) ;
424 
425  do {
426  pos_t2.set_start() ;
427  do {
428  // Check if the summation indices are the same
429  bool same = true ;
430  for (int i=0 ; i<t1.valence ; i++)
431  if (sum_1(i)!=-1)
432  if (pos_t1(i)!=pos_t2(sum_1(i)))
433  same = false ;
434  // if the same :
435  if (same) {
436  conte = 0 ;
437  for (int i=0 ; i<t1.valence ; i++)
438  if (sum_1(i)==-1) {
439  pos_res.set(conte) = pos_t1(i) ;
440  conte ++ ;
441  }
442  for (int i=0 ; i<t2.valence ; i++)
443  if (sum_2(i)==-1) {
444  pos_res.set(conte) = pos_t2(i) ;
445  conte ++ ;
446  }
447 
448  int ind (res.position(pos_res)) ;
449  if (first(ind)) {
450  res.set(pos_res) = t1(pos_t1)*t2(pos_t2) ;
451  first.set(ind) = false ;
452  }
453  else
454  res.set(pos_res) += t1(pos_t1)*t2(pos_t2) ;
455  }
456  }
457  while (pos_t2.inc()) ;
458  }
459  while (pos_t1.inc()) ;
460 
461  return res ;
462  }
463  else {
464  // Result is a scalar :
465  Scalar res (t1.espace) ;
466  Index pos_t1 (t1) ;
467  Index pos_t2 (t2) ;
468  int first = true ;
469 
470  do {
471  pos_t2.set_start() ;
472  do {
473  // Check if the summation indices are the same
474  bool same = true ;
475  for (int i=0 ; i<t1.valence ; i++)
476  if (sum_1(i)!=-1)
477  if (pos_t1(i)!=pos_t2(sum_1(i)))
478  same = false ;
479  // if the same :
480  if (same) {
481  if (first) {
482  res = t1(pos_t1)*t2(pos_t2) ;
483  first = false ;
484  }
485  else
486  res += t1(pos_t1)*t2(pos_t2) ;
487  }
488  }
489  while (pos_t2.inc()) ;
490  }
491  while (pos_t1.inc()) ;
492  return res ;
493  }
494  }
495  }
496 }
