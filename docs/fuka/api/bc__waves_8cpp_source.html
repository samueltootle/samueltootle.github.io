32 Term_eq Domain_shell::bc_waves (const Term_eq& gamma, const Term_eq& omega) const {
33 
34  // Free parameters for checking
35  //int kkmax = nbr_coefs(2)-1 ;
36  //int jjmax = nbr_coefs(1) ;
37  int kkmax = 10 ;
38  int jjmax = 5 ;
39  int dom = gamma.get_dom() ;
40 
41  // Check if gamma is a metric
42  if (gamma.type_data != TERM_T) {
43  cerr << "Domain_shell::bc_waves only defined for a tensor" << endl ;
44  abort() ;
45  }
46  int valence = gamma.val_t->get_valence() ;
47  if (valence!=2) {
48  cerr << "Domain_shell::bc_waves only defined with respect to second order tensor (i.e. valence must be 2)" << endl ;
49  abort() ;
50  }
51  if (gamma.val_t->get_basis().get_basis(dom) != CARTESIAN_BASIS) {
52  cerr << "Domain_shell::bc_waves only defined with respect to Cartesian tensorial basis" << endl ;
53  abort() ;
54  }
55 
56 
57  bool doder = ((gamma.der_t==0x0) || (omega.der_d==0x0)) ? false : true ;
58 
59  // Put rmax in a Term_eq
60  Index posr (nbr_points) ;
61  posr.set(0) = nbr_points(1)-1 ;
62  double rmax = get_radius()(posr) ;
63 
64  // Compute the term_eq for the BC
65  int nbrm = int(nbr_coefs(2)/2) + 2 ;
66  int nbrl = 2*nbr_coefs(1)-1 ;
67  // Real parts and Imaginar parts
68  Term_eq** Rsh = new Term_eq* [nbrm*nbrl] ;
69  Term_eq** Ish = new Term_eq* [nbrm*nbrl] ;
70  Term_eq** dRsh = new Term_eq* [nbrm*nbrl] ;
71  Term_eq** dIsh = new Term_eq* [nbrm*nbrl] ;
72 
73  for (int m=0 ; m<nbrm ; m++)
74  for (int l=0 ; l<nbrl ; l++) {
75  Rsh[m*nbrl+l] = (m==0) ? new Term_eq (dom, pow(rmax, -(l+1)), 0) : new Term_eq (bessel_jl(m*omega*rmax, l)) ;
76  Ish[m*nbrl+l] = (m==0) ? new Term_eq (dom, pow(rmax, -(l+1)), 0) : new Term_eq (bessel_yl(m*omega*rmax, l)) ;
77  dRsh[m*nbrl+l] = (m==0) ? new Term_eq (dom, -(l+1)*pow(rmax, -(l+2)), 0) :
78  new Term_eq (m*omega*bessel_djl(m*omega*rmax, l)) ;
79  dIsh[m*nbrl+l] = (m==0) ? new Term_eq (dom, -(l+1)*pow(rmax, -(l+2)), 0) :
80  new Term_eq (m*omega*bessel_dyl(m*omega*rmax, l)) ;
81  }
82 
83  // To store the result
84  Tensor valres (*gamma.val_t, false) ;
85  valres.std_base() ;
86  for (int i=0 ; i<valres.get_n_comp() ; i++) {
87  valres.set(valres.indices(i)).set_domain(dom).allocate_coef() ;
88  Index pcf (nbr_coefs) ;
89  do {
90  valres.set(valres.indices(i)).set_domain(dom).set_coef(pcf) = 0 ;
91  }
92  while (pcf.inc()) ;
93  }
94  Tensor derres (valres) ;
95 
96  // Spherical harmonics -> Standard basis
97  // Passage matrices :
98  Array<double> inv_even (mat_inv_leg_even(nbr_coefs(1), nbr_coefs(2))) ;
99  Array<double> inv_odd (mat_inv_leg_odd(nbr_coefs(1), nbr_coefs(2))) ;
100  Array<double> passage_even (mat_leg_even(nbr_coefs(1), nbr_coefs(2))) ;
101  Array<double> passage_odd (mat_leg_odd(nbr_coefs(1), nbr_coefs(2))) ;
102 
103  // Symetric part :
104  // Loop on phi :
105  Index posylm (nbr_coefs) ;
106  Index posradial (nbr_coefs) ;
107 
108  for (int k=0 ; k<kkmax ; k+=2) {
109 
110  posylm.set(2) = k ;
111 
112  int mm = int(k/2) ;
113 
114  // loop on theta
115  int jmin = (mm%2==0) ? int(mm/2) : int((mm-1)/2) ;
116  int jmax = (mm%2==0) ? jjmax : jjmax-1 ;
117  for (int j=jmin ; j<jmax ; j++) {
118 
119  int ll = (mm%2==0) ? 2*j : 2*j+1 ;
120 
121  // Radial derivatives
122  double Rvaldrxx = multipoles_sym (k, j, OUTER_BC, (*gamma.val_t)(1,1)(dom).der_r(), passage_even) ;
123  double Rvaldrxy = multipoles_sym (k, j, OUTER_BC, (*gamma.val_t)(1,2)(dom).der_r(), passage_even) ;
124  double Rvaldryy = multipoles_sym (k, j, OUTER_BC, (*gamma.val_t)(2,2)(dom).der_r(), passage_even) ;
125  double Rvaldrzz = multipoles_sym (k, j, OUTER_BC, (*gamma.val_t)(3,3)(dom).der_r(), passage_even) ;
126 
127  double Ivaldrxx = ((k==kkmax-1) || (k==0)) ? 0 : multipoles_sym (k+1, j, OUTER_BC, (*gamma.val_t)(1,1)(dom).der_r(), passage_even) ;
128  double Ivaldrxy = ((k==kkmax-1) || (k==0))? 0 : multipoles_sym (k+1, j, OUTER_BC, (*gamma.val_t)(1,2)(dom).der_r(), passage_even) ;
129  double Ivaldryy = ((k==kkmax-1) || (k==0)) ? 0 : multipoles_sym (k+1, j, OUTER_BC, (*gamma.val_t)(2,2)(dom).der_r(), passage_even) ;
130  double Ivaldrzz = ((k==kkmax-1) || (k==0)) ? 0 : multipoles_sym (k+1, j, OUTER_BC, (*gamma.val_t)(3,3)(dom).der_r(), passage_even) ;
131 
132  Term_eq* RdV1 ;
133  Term_eq* RdV2 ;
134  Term_eq* RdV3 ;
135  Term_eq* RdV6 ;
136 
137  Term_eq* IdV1 ;
138  Term_eq* IdV2 ;
139  Term_eq* IdV3 ;
140  Term_eq* IdV6 ;
141 
142  if (doder) {
143 
144  double Rderdrxx = multipoles_sym (k, j, OUTER_BC, (*gamma.der_t)(1,1)(dom).der_r(), passage_even) ;
145  double Rderdrxy = multipoles_sym (k, j, OUTER_BC, (*gamma.der_t)(1,2)(dom).der_r(), passage_even) ;
146  double Rderdryy = multipoles_sym (k, j, OUTER_BC, (*gamma.der_t)(2,2)(dom).der_r(), passage_even) ;
147  double Rderdrzz = multipoles_sym (k, j, OUTER_BC, (*gamma.der_t)(3,3)(dom).der_r(), passage_even) ;
148 
149  double Iderdrxx = ((k==kkmax-1) || (k==0)) ? 0 :multipoles_sym (k+1, j, OUTER_BC, (*gamma.der_t)(1,1)(dom).der_r(), passage_even) ;
150  double Iderdrxy = ((k==kkmax-1) || (k==0)) ? 0 :multipoles_sym (k+1, j, OUTER_BC, (*gamma.der_t)(1,2)(dom).der_r(), passage_even) ;
151  double Iderdryy = ((k==kkmax-1) || (k==0)) ? 0 :multipoles_sym (k+1, j, OUTER_BC, (*gamma.der_t)(2,2)(dom).der_r(), passage_even) ;
152  double Iderdrzz = ((k==kkmax-1) || (k==0)) ? 0 :multipoles_sym (k+1, j, OUTER_BC, (*gamma.der_t)(3,3)(dom).der_r(), passage_even) ;
153 
154  RdV1 = new Term_eq (dom, Rvaldrxx-2*Ivaldrxy-Rvaldryy , Rderdrxx-2*Iderdrxy-Rderdryy) ;
155  RdV2 = new Term_eq (dom, Rvaldrxx+2*Ivaldrxy-Rvaldryy , Rderdrxx+2*Iderdrxy-Rderdryy) ;
156  RdV3 = new Term_eq (dom, Rvaldrxx+Rvaldryy, Rderdrxx+Rderdryy) ;
157  RdV6 = new Term_eq (dom, Rvaldrzz, Rderdrzz) ;
158 
159  IdV1 = new Term_eq (dom, Ivaldrxx+2*Rvaldrxy-Ivaldryy , Iderdrxx+2*Rderdrxy-Iderdryy) ;
160  IdV2 = new Term_eq (dom, Ivaldrxx-2*Rvaldrxy-Ivaldryy , Iderdrxx-2*Rderdrxy-Iderdryy) ;
161  IdV3 = new Term_eq (dom, Ivaldrxx+Ivaldryy, Iderdrxx+Iderdryy) ;
162  IdV6 = new Term_eq (dom, Ivaldrzz, Iderdrzz) ;
163 
164 
165  }
166  else {
167 
168 
169  RdV1 = new Term_eq (dom, Rvaldrxx-2*Ivaldrxy-Rvaldryy) ;
170  RdV2 = new Term_eq (dom, Rvaldrxx+2*Ivaldrxy-Rvaldryy) ;
171  RdV3 = new Term_eq (dom, Rvaldrxx+Rvaldryy) ;
172  RdV6 = new Term_eq (dom, Rvaldrzz) ;
173 
174  IdV1 = new Term_eq (dom, Ivaldrxx+2*Rvaldrxy-Ivaldryy) ;
175  IdV2 = new Term_eq (dom, Ivaldrxx-2*Rvaldrxy-Ivaldryy) ;
176  IdV3 = new Term_eq (dom, Ivaldrxx+Ivaldryy) ;
177  IdV6 = new Term_eq (dom, Ivaldrzz) ;
178  }
179 
180 
181  Term_eq RA1 ( *RdV1 / (*dRsh[(mm+2)*nbrl+ll])) ;
182  Term_eq RA2 ( *RdV2 / (*dRsh[abs(mm-2)*nbrl+ll])) ;
183  Term_eq RA3 ( *RdV3 / (*dRsh[mm*nbrl+ll])) ;
184  Term_eq RA6 ( *RdV6 / (*dRsh[mm*nbrl+ll])) ;
185 
186  Term_eq IA1 ( *IdV1 / (*dIsh[(mm+2)*nbrl+ll])) ;
187  Term_eq IA2 ( *IdV2 / (*dIsh[abs(mm-2)*nbrl+ll])) ;
188  Term_eq IA3 ( *IdV3 / (*dIsh[mm*nbrl+ll])) ;
189  Term_eq IA6 ( *IdV6 / (*dIsh[mm*nbrl+ll])) ;
190 
191  // Imaginary and real part of the eighenvectors
192  Term_eq Rxx (0.25*RA1*(*Rsh[(mm+2)*nbrl+ll]) + 0.25*RA2*(*Rsh[abs(mm-2)*nbrl+ll])
193  + 0.5*RA3*(*Rsh[mm*nbrl+ll])) ;
194 
195  Term_eq Ixx (0.25*IA1*(*Ish[(mm+2)*nbrl+ll]) + 0.25*IA2*(*Ish[abs(mm-2)*nbrl+ll])
196  + 0.5*IA3*(*Ish[mm*nbrl+ll])) ;
197 
198  Term_eq Rxy (0.25*IA1*(*Ish[(mm+2)*nbrl+ll]) - 0.25*IA2*(*Ish[abs(mm-2)*nbrl+ll])) ;
199 
200  Term_eq Ixy (-0.25*RA1*(*Rsh[(mm+2)*nbrl+ll]) + 0.25*RA2*(*Rsh[abs(mm-2)*nbrl+ll])) ;
201 
202  Term_eq Ryy (-0.25*RA1*(*Rsh[(mm+2)*nbrl+ll]) - 0.25*RA2*(*Rsh[abs(mm-2)*nbrl+ll])
203  + 0.5*RA3*(*Rsh[mm*nbrl+ll])) ;
204 
205  Term_eq Iyy (-0.25*IA1*(*Ish[(mm+2)*nbrl+ll]) - 0.25*IA2*(*Ish[abs(mm-2)*nbrl+ll])
206  + 0.5*IA3*(*Ish[mm*nbrl+ll])) ;
207 
208  Term_eq Rzz (RA6*(*Rsh[mm*nbrl+ll])) ;
209 
210  Term_eq Izz (IA6*(*Ish[mm*nbrl+ll])) ;
211 
212 
213 
214  // Put in the right component :
215  posylm.set(0) = 0 ;
216 
217  // Loop on theta for the matrix part :
218  for (int inds=0 ; inds<nbr_coefs(1) ; inds++) {
219 
220  posylm.set(1) = inds ;
221 
222  // Parts in cosines
223  valres.set(1,1).set_domain(dom).set_coef(posylm) += inv_even(mm, j, inds) * (*Rxx.val_d) ;
224  valres.set(1,2).set_domain(dom).set_coef(posylm) += inv_even(mm, j, inds) * (*Rxy.val_d) ;
225  valres.set(2,2).set_domain(dom).set_coef(posylm) += inv_even(mm, j, inds) * (*Ryy.val_d) ;
226  valres.set(3,3).set_domain(dom).set_coef(posylm) += inv_even(mm, j, inds) * (*Rzz.val_d) ;
227 
228  if (doder) {
229  derres.set(1,1).set_domain(dom).set_coef(posylm) += inv_even(mm, j, inds) * (*Rxx.der_d) ;
230  derres.set(1,2).set_domain(dom).set_coef(posylm) += inv_even(mm, j, inds) * (*Rxy.der_d) ;
231  derres.set(2,2).set_domain(dom).set_coef(posylm) += inv_even(mm, j, inds) * (*Ryy.der_d) ;
232  derres.set(3,3).set_domain(dom).set_coef(posylm) += inv_even(mm, j, inds) * (*Rzz.der_d) ;
233  }
234 
235  // Parts in sines
236  if ((k!=kkmax-1) && (k!=0)) {
237  posylm.set(2) ++ ;
238  valres.set(1,1).set_domain(dom).set_coef(posylm) += inv_even(mm, j, inds) * (*Ixx.val_d) ;
239  valres.set(1,2).set_domain(dom).set_coef(posylm) += inv_even(mm, j, inds) * (*Ixy.val_d) ;
240  valres.set(2,2).set_domain(dom).set_coef(posylm) += inv_even(mm, j, inds) * (*Iyy.val_d) ;
241  valres.set(3,3).set_domain(dom).set_coef(posylm) += inv_even(mm, j, inds) * (*Izz.val_d) ;
242 
243 
244  if (doder) {
245  derres.set(1,1).set_domain(dom).set_coef(posylm) += inv_even(mm, j, inds) * (*Ixx.der_d) ;
246  derres.set(1,2).set_domain(dom).set_coef(posylm) += inv_even(mm, j, inds) * (*Ixy.der_d) ;
247  derres.set(2,2).set_domain(dom).set_coef(posylm) += inv_even(mm, j, inds) * (*Iyy.der_d) ;
248  derres.set(3,3).set_domain(dom).set_coef(posylm) += inv_even(mm, j, inds) * (*Izz.der_d) ;
249  }
250  posylm.set(2) -- ;
251  }
252  }
253

255  delete RdV2 ;
256  delete RdV3 ;
257  delete RdV6 ;
258 
259  delete IdV1 ;
260  delete IdV2 ;
261  delete IdV3 ;
262  delete IdV6 ;
263  }
264  }

267  // Loop on phi :
268  for (int k=0 ; k<kkmax ; k+=2)
269  if (k!=1) {
270 
271  posylm.set(2) = k ;
272 
273  int mm = int(k/2) ;
274 
275  // loop on theta
276  int jmin = (mm%2==0) ? int(mm/2) : int((mm+1)/2) ;
277  int jmax = jjmax-1 ;
278  for (int j=jmin ; j<jmax ; j++) {
279 
280  int ll = (mm%2==0) ? 2*j+1 : 2*j ;
281 
282  // Radial derivatives
283  double Rvaldrxz = multipoles_asym (k, j, OUTER_BC, (*gamma.val_t)(1,3)(dom).der_r(), passage_odd) ;
284  double Rvaldryz = multipoles_asym (k, j, OUTER_BC, (*gamma.val_t)(2,3)(dom).der_r(), passage_odd) ;
285 
286  double Ivaldrxz = ((k==kkmax-1) || (k==0)) ? 0 : multipoles_asym (k+1, j, OUTER_BC, (*gamma.val_t)(1,3)(dom).der_r(), passage_odd) ;
287  double Ivaldryz = ((k==kkmax-1) || (k==0)) ? 0 : multipoles_asym (k+1, j, OUTER_BC, (*gamma.val_t)(2,3)(dom).der_r(), passage_odd) ;
288 
289  Term_eq* RdV4 ;
290  Term_eq* RdV5 ;
291  Term_eq* IdV4 ;
292  Term_eq* IdV5 ;
293 
294  if (doder) {
295 
296  double Rderdrxz = multipoles_asym (k, j, OUTER_BC, (*gamma.der_t)(1,3)(dom).der_r(), passage_odd) ;
297  double Rderdryz = multipoles_asym (k, j, OUTER_BC, (*gamma.der_t)(2,3)(dom).der_r(), passage_odd) ;
298 
299  double Iderdrxz = ((k==kkmax-1) || (k==0)) ? 0 : multipoles_asym (k+1, j, OUTER_BC, (*gamma.der_t)(1,3)(dom).der_r(), passage_odd) ;
300  double Iderdryz = ((k==kkmax-1) || (k==0)) ? 0 : multipoles_asym (k+1, j, OUTER_BC, (*gamma.der_t)(2,3)(dom).der_r(), passage_odd) ;
301 
302  RdV4 = new Term_eq (dom, Rvaldrxz - Ivaldryz, Rderdrxz - Iderdryz) ;
303  RdV5 = new Term_eq (dom, Rvaldrxz + Ivaldryz, Rderdrxz + Iderdryz) ;
304 
305  IdV4 = new Term_eq (dom, Ivaldrxz + Rvaldryz, Iderdrxz + Rderdryz) ;
306  IdV5 = new Term_eq (dom, Ivaldrxz - Rvaldryz, Iderdrxz - Rderdryz) ;
307  }
308  else {
309  RdV4 = new Term_eq (dom, Rvaldrxz - Ivaldryz) ;
310  RdV5 = new Term_eq (dom, Rvaldrxz + Ivaldryz) ;
311 
312  IdV4 = new Term_eq (dom, Ivaldrxz + Rvaldryz) ;
313  IdV5 = new Term_eq (dom, Ivaldrxz - Rvaldryz) ;
314  }
315 
316  Term_eq RA4 ( *RdV4 / (*dRsh[(mm+1)*nbrl+ll])) ;
317  Term_eq RA5 ( *RdV5 / (*dRsh[abs(mm-1)*nbrl+ll])) ;
318 
319  Term_eq IA4 ( *IdV4 / (*dIsh[(mm+1)*nbrl+ll])) ;
320  Term_eq IA5 ( *IdV5 / (*dIsh[abs(mm-1)*nbrl+ll])) ;
321 
322  // Case m==0 is different
323  // int signe = (mm==0) ? -1 : 1 ;
324 
325  // Imaginary and real part of the eighenvectors
326  Term_eq Rxz (0.5*RA4*(*Rsh[(mm+1)*nbrl+ll]) + 0.5*RA5*(*Rsh[abs(mm-1)*nbrl+ll])) ;
327  Term_eq Ixz (0.5*IA4*(*Ish[(mm+1)*nbrl+ll]) + 0.5*IA5*(*Ish[abs(mm-1)*nbrl+ll])) ;
328 
329  Term_eq Ryz (0.5*IA4*(*Ish[(mm+1)*nbrl+ll]) - 0.5*IA5*(*Ish[abs(mm-1)*nbrl+ll])) ;
330  Term_eq Iyz (-0.5*RA4*(*Rsh[(mm+1)*nbrl+ll]) + 0.5*RA5*(*Rsh[abs(mm-1)*nbrl+ll])) ;
331 
332  // Put in the right component :
333  posylm.set(0) = 0 ;
334 
335  // Loop on theta for the matrix part :
336 
337  for (int inds=0 ; inds<nbr_coefs(1) ; inds++) {
338 
339  posylm.set(1) = inds ;
340 
341  // Parts in cosines
342  valres.set(1,3).set_domain(dom).set_coef(posylm) += inv_odd(mm, j, inds) * (*Rxz.val_d) ;
343  valres.set(2,3).set_domain(dom).set_coef(posylm) += inv_odd(mm, j, inds) * (*Ryz.val_d) ;
344 
345  if (doder) {
346  derres.set(1,3).set_domain(dom).set_coef(posylm) += inv_odd(mm, j, inds) * (*Rxz.der_d) ;
347  derres.set(2,3).set_domain(dom).set_coef(posylm) += inv_odd(mm, j, inds) * (*Ryz.der_d) ;
348  }
349 
350  // Parts in sines
351  if ((k!=kkmax-1) && (k!=0)) {
352  posylm.set(2) ++ ;
353  valres.set(1,3).set_domain(dom).set_coef(posylm) += inv_odd(mm, j, inds) * (*Ixz.val_d) ;
354  valres.set(2,3).set_domain(dom).set_coef(posylm) += inv_odd(mm, j, inds) * (*Iyz.val_d) ;
355 
356  if (doder) {
357  derres.set(1,3).set_domain(dom).set_coef(posylm) += inv_odd(mm, j, inds) * (*Ixz.der_d) ;
358  derres.set(2,3).set_domain(dom).set_coef(posylm) += inv_odd(mm, j, inds) * (*Iyz.der_d) ;
359  }
360  posylm.set(2) -- ;
361  }
362  }
363 
364  delete RdV4 ;
365  delete RdV5 ;
366  delete IdV4 ;
367  delete IdV5 ;
368  }
369  }
370 
371  // Delete the shs
372  for (int i=0 ; i<nbrm*nbrl ; i++) {
373  delete Rsh[i] ;
374  delete Ish[i] ;
375  delete dRsh[i] ;
376  delete dIsh[i] ;
377  }
378  delete [] Rsh ;
379  delete [] Ish ;
380  delete [] dRsh ;
381  delete [] dIsh ;
382 
383  if (doder)
384  return Term_eq (dom, valres, derres) ;
385  else
386  return Term_eq (dom, valres) ;
