105  };
106 
107  int err = bbh_xcts_solution_driver(bconfig, outputdir);
108  while (res_inc) {
109  // iterative res increase
110  if (bconfig(resolution_indices) + 2 >= final_res) {
111  bconfig.set(resolution_indices) = final_res;
112  res_inc = false;
113  } else {
114  bconfig.set(resolution_indices) += 2;

116  regrid();
