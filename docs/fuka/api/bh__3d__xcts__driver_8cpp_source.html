78 template <typename config_t>
79 int bh_3d_xcts_stationary_driver(config_t& bconfig, std::string outputdir) {
80  int exit_status = RELOAD_FILE;
81  int rank = 0;
82  MPI_Comm_rank(MPI_COMM_WORLD, &rank);
83 
84  if (std::abs(bconfig(BCO_PARAMS::CHI)) > 0.84) {
85  std::cerr << "Warning: Solutions above |chi| > 0.84 are not necessarily "
86  "stable solutions\n";
87  }
88 
89  // In the event we wish to solve for a highly spinning solution
90  // we need to do an initial slow rotating solution before going to
91  // faster rotations otherwise the solution will diverge.
92  bconfig.control(CONTROLS::ITERATIVE_CHI) =
93  bconfig.control(CONTROLS::SEQUENCES) &&
94  std::abs(bconfig(BCO_PARAMS::CHI)) > 0.5;
95 
96  // We need to store the final desired CHI in case of iterative chi
97  bconfig.seq_setting(SEQ_SETTINGS::FINAL_CHI) = bconfig(BCO_PARAMS::CHI);
98  // Lower chi in case of iterative chi
99  bconfig(BCO_PARAMS::CHI) =
100  (bconfig.control(CONTROLS::ITERATIVE_CHI))
101  ? std::copysign(0.5, bconfig.seq_setting(SEQ_SETTINGS::FINAL_CHI))
102  : bconfig(BCO_PARAMS::CHI);
103 
104  // make sure BH directory exists for outputs
105  if (outputdir == "./") {
106  fs::path cwd = fs::current_path();
107  outputdir = cwd.string();
108  }
109  if (rank == 0)
110  std::cout << "Solutions will be stored in: " << outputdir << "\n"
111  << "Directory will be created if it doesn't exist.\n";
112  fs::create_directory(outputdir);
113 
114  std::string spacein = bconfig.space_filename();
115  if (!fs::exists(spacein)) {
116  // mainly for debugging MPI bugs
117  if (rank == 0) {
118  std::cerr << "File: " << spacein << " not found.\n\n";
119  } else {
120  std::cerr << "File: " << spacein << " not found for another rank.\n\n";
121  }
122  std::_Exit(EXIT_FAILURE);
123  }
124 
125  while (exit_status == RELOAD_FILE) {
126  spacein = bconfig.space_filename();
127  // just so you really know
128  if (rank == 0) {
129  std::cout << "Config File: " << bconfig.config_filename_abs() << std::endl
130  << "Fields File: " << spacein << std::endl
131  << bconfig << std::endl;
132  }
133  FILE* ff1 = fopen(spacein.c_str(), "r");
134  if (ff1 == NULL) {
135  // mainly for debugging MPI bugs
136  std::cerr << spacein.c_str() << " failed to open for rank " << rank
137  << "\n";
138  std::_Exit(EXIT_FAILURE);
139  }
140  Space_adapted_bh space(ff1);
141  Scalar conf(space, ff1);
142  Scalar lapse(space, ff1);
143  Vector shift(space, ff1);
144  fclose(ff1);
145  Base_tensor basis(space, CARTESIAN_BASIS);
146 
147  if (outputdir != "")
148  bconfig.set_outputdir(outputdir);
149 
150  bh_3d_xcts_solver<decltype(bconfig), decltype(space)> bh_solver(
151  bconfig, space, basis, conf, lapse, shift);
152 
153  exit_status = bh_solver.solve();
154 
155  MPI_Barrier(MPI_COMM_WORLD);
156  }
157  return exit_status;
158 }

212 template <class config_t, class Res_t>
213 inline int bh_3d_xcts_driver(config_t& bconfig,
214  Res_t& resolution,
215  std::string outputdir) {
216  int exit_status = RELOAD_FILE;
217  int rank = 0;
218  MPI_Comm_rank(MPI_COMM_WORLD, &rank);
219 
220  std::string spacein = bconfig.space_filename();
221  if (!fs::exists(spacein)) {
222  // mainly for debugging MPI bugs
223  if (rank == 0) {
224  std::cerr << "File: " << spacein << " not found.\n\n";
225  } else {
226  std::cerr << "File: " << spacein << " not found for another rank.\n\n";
227  }
228  std::_Exit(EXIT_FAILURE);
229  }
230 
231  double const xboost = bconfig.set(BCO_PARAMS::BVELX);
232  double const yboost = bconfig.set(BCO_PARAMS::BVELY);
233  bool res_inc = (resolution.final() != resolution.init());
234  auto resolution_indices = resolution.get_indices();
235  auto const& final_res = resolution.final();
236 
237  std::array<bool, NUM_STAGES>& stage_enabled = bconfig.return_stages();
238  auto [last_stage, last_stage_idx] = get_last_enabled(MSTAGE, stage_enabled);
239 
240  // These need to be zero prior to computing
241  // a stationary solution
242  bconfig.set(BCO_PARAMS::BVELX) = 0.;
243  bconfig.set(BCO_PARAMS::BVELY) = 0.;
244 
245  exit_status = bh_3d_xcts_stationary_driver(bconfig, outputdir);
246 
247  if (stage_enabled[STAGES::LINBOOST]) {
248  bconfig.set(BCO_PARAMS::BVELX) = xboost;
249  bconfig.set(BCO_PARAMS::BVELY) = yboost;
250  exit_status = bh_3d_xcts_linear_boost_driver(bconfig, outputdir);
251  }
252 
253  auto regrid = [&]() {
254  std::string fname{"bh_regrid"};
255 
256  if (rank == 0)
257  exit_status = bh_3d_xcts_regrid(bconfig, fname);
258  MPI_Barrier(MPI_COMM_WORLD);
259  bconfig.set_filename(fname);
260  bconfig.open_config();
261 
262  // Deactivate to avoid iterative CHI
263  bconfig.control(CONTROLS::SEQUENCES) = false;
264 
265  // Ensure last stage is activated to resolve
266  stage_enabled[last_stage_idx] = true;
267  };
268 
269  while (res_inc) {
270  // iterative res increase
271  if (bconfig(BCO_PARAMS::BCO_RES) + 2 >= final_res) {
272  bconfig.set(BCO_PARAMS::BCO_RES) = final_res;
273  res_inc = false;
274  } else {
275  bconfig.set(BCO_PARAMS::BCO_RES) += 2;
276  }
277  regrid();
278 
279  // Rerun with new grid
280  if (stage_enabled[STAGES::LINBOOST]) {
281  exit_status = bh_3d_xcts_linear_boost_driver(bconfig, outputdir);
282  } else {
283  exit_status = bh_3d_xcts_stationary_driver(bconfig, outputdir);
284  }
285  }
286  return exit_status;
287 }
