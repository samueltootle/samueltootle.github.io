33 Term_eq Domain_shell_outer_adapted::derive_r (const Term_eq& so) const {
34 
35  assert (so.get_dom() == num_dom) ;
36 
37  Tensor val_res(*so.val_t, false) ;
38 
39  for (int cmp = 0 ; cmp<so.val_t->get_n_comp() ; cmp++) {
40  Array<int>ind (val_res.indices(cmp)) ;
41  val_res.set(ind).set_domain(num_dom) = (*so.val_t)(ind)(num_dom).der_var(1) / (*der_rad_term_eq->val_t)()(num_dom) ;
42  }
43 
44  bool doder = ((so.der_t==0x0) || (der_rad_term_eq->der_t==0x0)) ? false : true ;
45 
46  if (doder) {
47  Tensor der_res (*so.der_t, false) ;
48  for (int cmp = 0 ; cmp<so.val_t->get_n_comp() ; cmp++) {
49  Array<int>ind (val_res.indices(cmp)) ;
50  der_res.set(ind).set_domain(num_dom) = ((*so.der_t)(ind)(num_dom).der_var(1)*(*der_rad_term_eq->val_t)()(num_dom)
51  - (*so.val_t)(ind)(num_dom).der_var(1)*(*der_rad_term_eq->der_t)()(num_dom)) /
52  ((*der_rad_term_eq->val_t)()(num_dom) * (*der_rad_term_eq->val_t)()(num_dom)) ;
53  }
54  return Term_eq (num_dom, val_res, der_res) ;
55  }
56  else
57  return Term_eq (num_dom, val_res) ;
58 }

60 Term_eq Domain_shell_outer_adapted::derive_t (const Term_eq& so) const {
61 
62  assert (so.get_dom() == num_dom) ;
63  Term_eq* dtprime ;
64  Tensor val_res(*so.val_t, false) ;
65 
66  for (int cmp = 0 ; cmp<so.val_t->get_n_comp() ; cmp++) {
67  Array<int>ind (val_res.indices(cmp)) ;
68  val_res.set(ind).set_domain(num_dom) = (*so.val_t)(ind)(num_dom).der_var(2) ;
69  }
70  bool doder = (so.der_t==0x0) ? false : true ;
71 
72  if (doder) {
73  Tensor der_res (*so.der_t, false) ;
74  for (int cmp = 0 ; cmp<so.val_t->get_n_comp() ; cmp++) {
75  Array<int>ind (val_res.indices(cmp)) ;
76  der_res.set(ind).set_domain(num_dom) = (*so.der_t)(ind)(num_dom).der_var(2) ;
77  }
78  dtprime = new Term_eq (num_dom, val_res, der_res) ;
79  }
80  else
81  dtprime = new Term_eq (num_dom, val_res) ;
82 
83 
84  Term_eq res (*dtprime - (*dt_rad_term_eq) * derive_r(so)) ;
85  delete dtprime ;
86  return res ;

120 Term_eq Domain_shell_outer_adapted::flat_grad_spher (const Term_eq& so) const {
121 
122  assert (so.get_dom() == num_dom) ;
123 
124  int valso = so.get_val_t().get_valence() ;
125  Array<int> type_ind (valso+1) ;
126  type_ind.set(0) = COV ;
127  for (int i=0 ; i<valso ; i++)
128  type_ind.set(i+1) = so.get_val_t().get_index_type(i) ;
129 
130  Base_tensor basis (sp) ;
131  basis.set_basis(num_dom) = SPHERICAL_BASIS ;
132  Tensor res (sp, valso+1 , type_ind, basis) ;
133 
134  Term_eq comp_r (derive_r(so)) ;
135  Term_eq comp_t (derive_t(so)/(*rad_term_eq)) ;
136  Term_eq comp_p (do_comp_by_comp(derive_p(so), &Domain::div_sin_theta) / (*rad_term_eq)) ;
137 
138 
139  // Loop on cmp :
140  for (int nc=0 ; nc<so.get_val_t().get_n_comp() ; nc++) {
141 
142  Array<int> ind (so.get_val_t().indices(nc)) ;
143  Array<int> indtarget (valso+1) ;
144  for (int i=0 ; i<valso ; i++)
145  indtarget.set(i+1) = ind(i) ;
146 
147  // R comp :
148  indtarget.set(0) = 1 ;
149  res.set(indtarget).set_domain(num_dom) = comp_r.get_val_t()(ind)(num_dom);
150  // theta comp :
151  indtarget.set(0) = 2 ;
152  res.set(indtarget).set_domain(num_dom) = comp_t.get_val_t()(ind)(num_dom) ;
153  // Phi comp :
154  indtarget.set(0) = 3 ;
155  res.set(indtarget).set_domain(num_dom) = comp_p.get_val_t()(ind)(num_dom) ;
156  }
157  bool doder = ((so.der_t==0x0) || (outer_radius_term_eq->der_t==0x0)) ? false : true ;
158 
159  if (!doder) {
160  return Term_eq (num_dom, res) ;
161  }
162  else {
163 
164  Tensor resder (sp, valso+1 , type_ind, basis) ;
165  // Loop on cmp :
166  for (int nc=0 ; nc<so.get_val_t().get_n_comp() ; nc++) {
167 
168  Array<int> ind (so.get_val_t().indices(nc)) ;
169  Array<int> indtarget (valso+1) ;
170  for (int i=0 ; i<valso ; i++)
171  indtarget.set(i+1) = ind(i) ;
172 
173  // R comp :
174  indtarget.set(0) = 1 ;
175  resder.set(indtarget).set_domain(num_dom) = comp_r.get_der_t()(ind)(num_dom);
176  // theta comp :
177  indtarget.set(0) = 2 ;
178  resder.set(indtarget).set_domain(num_dom) = comp_t.get_der_t()(ind)(num_dom) ;
179  // Phi comp :
180  indtarget.set(0) = 3 ;
181  resder.set(indtarget).set_domain(num_dom) = comp_p.get_der_t()(ind)(num_dom) ;
182  }
183  return Term_eq (num_dom, res, resder) ;
184  }

187 void Domain_shell_outer_adapted::do_normal_spher () const {
188 
189  Term_eq grad (flat_grad_spher(*rad_term_eq - *outer_radius_term_eq)) ;
190 
191  Scalar val_norme (sp) ;
192  val_norme.set_domain(num_dom) = sqrt((*grad.val_t)(1)(num_dom)*(*grad.val_t)(1)(num_dom)
193  + (*grad.val_t)(2)(num_dom)*(*grad.val_t)(2)(num_dom)
194  + (*grad.val_t)(3)(num_dom)*(*grad.val_t)(3)(num_dom)) ;
195  bool doder = ((grad.der_t==0x0) || (outer_radius_term_eq->der_t==0x0)) ? false : true ;
196  if (doder) {
197  Scalar der_norme (sp) ;
198  der_norme.set_domain(num_dom) = ((*grad.der_t)(1)(num_dom)*(*grad.val_t)(1)(num_dom)
199  + (*grad.der_t)(2)(num_dom)*(*grad.val_t)(2)(num_dom)
200  + (*grad.der_t)(3)(num_dom)*(*grad.val_t)(3)(num_dom)) / val_norme(num_dom) ;
201  Term_eq norme (num_dom, val_norme, der_norme) ;
202  if (normal_spher==0x0)
203  normal_spher = new Term_eq (grad / norme) ;
204  else
205  *normal_spher = Term_eq (grad/norme) ;
206  }
207  else {
208  Term_eq norme (num_dom, val_norme) ;
209  if (normal_spher==0x0)
210  normal_spher = new Term_eq (grad / norme) ;
211  else
212  *normal_spher = Term_eq (grad/norme) ;
213  }

218 void Domain_shell_outer_adapted::do_normal_cart () const {
219 
220 
221  do_normal_spher() ;
222 
223  Base_tensor basis (sp) ;
224  basis.set_basis(num_dom) = CARTESIAN_BASIS ;
225  Vector val (sp, CON, basis) ;
226 
227  val.set(1).set_domain(num_dom) = mult_cos_phi(mult_sin_theta((*normal_spher->val_t)(1)(num_dom)) + mult_cos_theta((*normal_spher->val_t)(2)(num_dom)))
228  - mult_sin_phi((*normal_spher->val_t)(3)(num_dom)) ;
229  val.set(2).set_domain(num_dom) = mult_sin_phi(mult_sin_theta((*normal_spher->val_t)(1)(num_dom)) + mult_cos_theta((*normal_spher->val_t)(2)(num_dom)))
230  + mult_cos_phi((*normal_spher->val_t)(3)(num_dom)) ;
231  val.set(3).set_domain(num_dom) = mult_cos_theta((*normal_spher->val_t)(1)(num_dom)) - mult_sin_theta((*normal_spher->val_t)(2)(num_dom)) ;
232 
233  bool doder = (normal_spher->der_t==0x0) ? false : true ;
234  if (doder) {
235  Vector der (sp, CON, basis) ;
236 
237  der.set(1).set_domain(num_dom) = mult_cos_phi(mult_sin_theta((*normal_spher->der_t)(1)(num_dom)) + mult_cos_theta((*normal_spher->der_t)(2)(num_dom)))
238  - mult_sin_phi((*normal_spher->der_t)(3)(num_dom)) ;
239  der.set(2).set_domain(num_dom) = mult_sin_phi(mult_sin_theta((*normal_spher->der_t)(1)(num_dom)) + mult_cos_theta((*normal_spher->der_t)(2)(num_dom)))
240  + mult_cos_phi((*normal_spher->der_t)(3)(num_dom)) ;
241  der.set(3).set_domain(num_dom) = mult_cos_theta((*normal_spher->der_t)(1)(num_dom)) - mult_sin_theta((*normal_spher->der_t)(2)(num_dom)) ;
242 
243  if (normal_cart==0x0)
244  normal_cart = new Term_eq (num_dom, val, der) ;
245  else
246  *normal_cart = Term_eq(num_dom, val,der) ;
247  }
248  else {
249  if (normal_cart==0x0)
250  normal_cart = new Term_eq (num_dom, val) ;
251  else
252  *normal_cart = Term_eq(num_dom, val) ;
253  }

256 Term_eq Domain_shell_outer_adapted::der_normal_term_eq (const Term_eq& so, int bound) const {
257 
258 
259  switch (bound) {
260  case OUTER_BC : {
261  // Deformed surface
262  if (normal_spher == 0x0)
263  do_normal_spher() ;
264 
265  int valso = so.get_val_t().get_valence() ;
266 
267  Term_eq grad (flat_grad_spher(so)) ;
268 
269  Tensor res (so.get_val_t(), false) ;
270 
271  // Loop on cmp :
272  for (int nc=0 ; nc<so.get_val_t().get_n_comp() ; nc++) {
273 
274  Array<int> ind (so.get_val_t().indices(nc)) ;
275  Array<int> indgrad (valso+1) ;
276  for (int i=0 ; i<valso ; i++)
277  indgrad.set(i+1) = ind(i) ;
278 
279  indgrad.set(0) = 1 ;
280  res.set(ind).set_domain(num_dom) = (*grad.val_t)(indgrad)(num_dom) * (*normal_spher->val_t)(1)(num_dom) ;
281  indgrad.set(0) = 2 ;
282  res.set(ind).set_domain(num_dom) += (*grad.val_t)(indgrad)(num_dom) * (*normal_spher->val_t)(2)(num_dom) ;
283  indgrad.set(0) = 3 ;
284  res.set(ind).set_domain(num_dom) += (*grad.val_t)(indgrad)(num_dom) * (*normal_spher->val_t)(3)(num_dom) ;
285  }
286 
287  bool doder = ((grad.der_t == 0x0) || (normal_spher->der_t == 0x0)) ? false : true ;
288  if (doder) {
289 
290  Tensor der (so.get_val_t(), false) ;
291 
292  // Loop on cmp :
293  for (int nc=0 ; nc<so.get_val_t().get_n_comp() ; nc++) {
294 
295  Array<int> ind (so.get_val_t().indices(nc)) ;
296  Array<int> indgrad (valso+1) ;
297  for (int i=0 ; i<valso ; i++)
298  indgrad.set(i+1) = ind(i) ;
299 
300  indgrad.set(0) = 1 ;
301  der.set(ind).set_domain(num_dom) = (*grad.der_t)(indgrad)(num_dom) * (*normal_spher->val_t)(1)(num_dom)
302  + (*grad.val_t)(indgrad)(num_dom) * (*normal_spher->der_t)(1)(num_dom) ;
303  indgrad.set(0) = 2 ;
304  der.set(ind).set_domain(num_dom) += (*grad.der_t)(indgrad)(num_dom) * (*normal_spher->val_t)(2)(num_dom)
305  + (*grad.val_t)(indgrad)(num_dom) * (*normal_spher->der_t)(2)(num_dom) ;
306  indgrad.set(0) = 3 ;
307  der.set(ind).set_domain(num_dom) += (*grad.der_t)(indgrad)(num_dom) * (*normal_spher->val_t)(3)(num_dom)
308  + (*grad.val_t)(indgrad)(num_dom) * (*normal_spher->der_t)(3)(num_dom) ;
309  }
310 
311 
312  return Term_eq (num_dom, res, der) ;
313  }
314  else
315  return Term_eq (num_dom, res) ;
316  }
317  case INNER_BC : {
318  return derive_r(so) ;
319  }
320  default :
321  cerr << "Unknown boundary in Domain_shell_outer_adapted::der_normal" << endl ;
322  abort() ;
323  }

326 Term_eq Domain_shell_outer_adapted::lap_term_eq (const Term_eq& so, int mm) const {
327  if (mm!=0) {
328  cerr << "Domain_shell_outer_adapted::lap_term_eq not defined for m != 0 (for now)" << endl ;
329  abort() ;
330  }
331 
332  if (so.get_val_t().get_valence() != 0) {
333  cerr << "Domain_shell_outer_adapted::lap_term_eq only defined for scalars" << endl ;
334  abort() ;
335  }
336 
337  // Angular part :
338  Term_eq dert (derive_t(so));
339  Term_eq p1 (derive_t(dert)) ;
340  Term_eq p2 (do_comp_by_comp(do_comp_by_comp(dert, &Domain::mult_cos_theta), &Domain::div_sin_theta)) ;
341  Term_eq der2p (derive_p(derive_p(so))) ;
342  Term_eq p3 (do_comp_by_comp(do_comp_by_comp(der2p, &Domain::div_sin_theta), &Domain::div_sin_theta)) ;
343 
344  Term_eq dr(derive_r(so)) ;
345 
346  Term_eq res (derive_r(dr) + 2 *dr / (*rad_term_eq) + (p1+p2+p3)/(*rad_term_eq)/(*rad_term_eq));
347 
348 
349  return res ;

358 void Domain_shell_outer_adapted::update_term_eq (Term_eq* so) const {
359 
360  Tensor der (*so->val_t, false) ;
361  for (int cmp=0 ; cmp<so->val_t->get_n_comp() ; cmp ++) {
362 
363 
364 
365  Val_domain derr ((*(*so->val_t).cmp[cmp])(num_dom).der_var(1) / (*der_rad_term_eq->val_t)()(num_dom)) ;
366 
367  if (!derr.check_if_zero()) {
368  Val_domain res(this) ;
369  res.allocate_conf() ;
370  Index pos (nbr_points) ;
371  do {
372  res.set(pos) = (*(*outer_radius_term_eq->der_t).cmp[0])(num_dom)(pos) / 2. * (1+((*coloc[0])(pos(0)))) * derr(pos) ;
373  }
374  while (pos.inc()) ;
375  res.set_base() = (*(*so->val_t).cmp[cmp])(num_dom).get_base() ;
376  der.cmp[cmp]->set_domain(num_dom) = res ;
377  }
378  else
379  der.cmp[cmp]->set_domain(num_dom).set_zero() ;
380  }
381 
382  so->set_der_t (der) ;

385 Term_eq Domain_shell_outer_adapted::partial_spher (const Term_eq& so) const {
386  int dom = so.get_dom() ;
387  assert (dom == num_dom) ;
388 
389  int valence = so.val_t->get_valence() ;
390 
391  Array<int> type_ind (valence+1) ;
392  type_ind.set(0) = COV ;
393  for (int i=1 ; i<valence+1 ; i++)
394  type_ind.set(i) = so.val_t->get_index_type(i-1) ;
395 
396  Base_tensor basis (so.get_val_t().get_space()) ;
397  basis.set_basis(dom) = SPHERICAL_BASIS ;
398 
399  Term_eq comp_r (derive_r(so)) ;
400  Term_eq comp_t (derive_t(so) / (*rad_term_eq)) ;
401  Term_eq comp_p (do_comp_by_comp((derive_p(so)), &Domain::div_sin_theta) / (*rad_term_eq)) ;
402 
403  Tensor val_res (so.get_val_t().get_space(), valence+1, type_ind, basis) ;
404  {
405  Index pos_so (*so.val_t) ;
406  Index pos_res (val_res) ;
407  do {
408  for (int i=1 ; i<valence+1 ; i++)
409  pos_res.set(i) = pos_so(i-1) ;
410  // R part
411  pos_res.set(0) = 0 ;
412  val_res.set(pos_res).set_domain(num_dom) = (*comp_r.val_t)(pos_so)(num_dom) ;
413  // Theta part
414  pos_res.set(0) = 1 ;
415  val_res.set(pos_res).set_domain(num_dom) = (*comp_t.val_t)(pos_so)(num_dom) ;
416  // Phi part
417  pos_res.set(0) = 2 ;
418  val_res.set(pos_res).set_domain(num_dom) = (*comp_p.val_t)(pos_so)(num_dom) ;
419  }
420  while (pos_so.inc()) ;
421 }
422 
423  bool doder = ((so.der_t==0x0) || (outer_radius_term_eq->der_t==0x0)) ? false : true ;
424  if (doder) {
425  Tensor der_res (so.get_val_t().get_space(), valence+1, type_ind, basis) ;
426  Index pos_so (*so.der_t) ;
427  Index pos_res (val_res) ;
428  do {
429  for (int i=1 ; i<valence+1 ; i++)
430  pos_res.set(i) = pos_so(i-1) ;
431  // R part
432  pos_res.set(0) = 0 ;
433  der_res.set(pos_res).set_domain(num_dom) = (*comp_r.der_t)(pos_so)(num_dom) ;
434  // Theta part
435  pos_res.set(0) = 1 ;
436  der_res.set(pos_res).set_domain(num_dom) = (*comp_t.der_t)(pos_so)(num_dom) ;
437  // Phi part
438  pos_res.set(0) = 2 ;
439  der_res.set(pos_res).set_domain(num_dom) = (*comp_p.der_t)(pos_so)(num_dom) ;
440  }
441  while (pos_so.inc()) ;
442 
443  return Term_eq (num_dom, val_res, der_res) ;
444  }
445  else
446  return Term_eq (num_dom, val_res) ;
447 }

450 Term_eq Domain_shell_outer_adapted::connection_spher (const Term_eq& so) const {
451 
452  int dom = so.get_dom() ;
453  assert (dom == num_dom) ;
454 
455  int valence = so.val_t->get_valence() ;
456  int val_res = so.val_t->get_valence() + 1 ;
457 
458  Array<int> type_ind (val_res) ;
459  type_ind.set(0) = COV ;
460  for (int i=1 ; i<val_res ; i++)
461  type_ind.set(i) = so.val_t->get_index_type(i-1) ;
462 
463  Base_tensor basis (so.get_val_t().get_space()) ;
464  basis.set_basis(dom) = SPHERICAL_BASIS ;
465 
466  Tensor auxi_val (so.get_val_t().get_space(), val_res, type_ind, basis) ;
467  for (int cmp=0 ; cmp<auxi_val.get_n_comp() ; cmp ++)
468  auxi_val.set(auxi_val.indices(cmp)) = 0 ;
469 
470  for (int ind_sum=0 ; ind_sum<valence ; ind_sum++) {
471 
472  //Loop on the components :
473  Index pos_auxi(auxi_val) ;
474  Index pos_so (*so.val_t) ;
475 
476  do {
477  for (int i=0 ; i<valence ; i++)
478  pos_so.set(i) = pos_auxi(i+1) ;
479  // Different cases of the derivative index :
480  switch (pos_auxi(0)) {
481  case 0 :
482  // Dr nothing
483  break ;
484  case 1 :
485  // Dtheta
486  // Different cases of the source index
487  switch (pos_auxi(ind_sum+1)) {
488  case 0 :
489  //Dtheta S_r
490  pos_so.set(ind_sum) = 1 ;
491  auxi_val.set(pos_auxi).set_domain(dom) -= (*so.val_t)(pos_so)(dom).div_r() ;
492  break ;
493  case 1 :
494  // Dtheta S_theta
495  pos_so.set(ind_sum) = 0 ;
496  auxi_val.set(pos_auxi).set_domain(dom) += (*so.val_t)(pos_so)(dom).div_r() ;
497  break ;
498  case 2 :
499  //Dtheta S_phi
500  break ;
501  default :
502  cerr << "Bad indice in Domain_shell_outer_adapted::connection_spher" << endl ;
503  abort() ;
504  }
505  break ;
506  case 2 :
507  // Dphi
508  // Different cases of the source index
509  switch (pos_auxi(ind_sum+1)) {
510  case 0 :

634 Term_eq Domain_shell_outer_adapted::partial_cart (const Term_eq& so) const {
635  int dom = so.get_dom() ;
636  assert (dom == num_dom) ;
637 
638  int valence = so.val_t->get_valence() ;
639 
640  Array<int> type_ind (valence+1) ;
641  type_ind.set(0) = COV ;
642  for (int i=1 ; i<valence+1 ; i++)
643  type_ind.set(i) = so.val_t->get_index_type(i-1) ;
644 
645  Base_tensor basis (so.get_val_t().get_space()) ;
646  basis.set_basis(dom) = CARTESIAN_BASIS ;
647 
648  Term_eq comp_r (derive_r(so)) ;
649  Term_eq comp_t (derive_t(so) / (*rad_term_eq)) ;
650  Term_eq comp_p (do_comp_by_comp((derive_p(so)/(*rad_term_eq)), &Domain::div_sin_theta)) ;
651 
652  Tensor val_res (so.get_val_t().get_space(), valence+1, type_ind, basis) ;
653  {
654  Index pos_so (*so.val_t) ;
655  Index pos_res (val_res) ;
656  do {
657  for (int i=1 ; i<valence+1 ; i++)
658  pos_res.set(i) = pos_so(i-1) ;
659 
660  Val_domain auxi (mult_sin_theta((*comp_r.val_t)(pos_so)(num_dom)) + mult_cos_theta((*comp_t.val_t)(pos_so)(num_dom))) ;
661  // X part
662  pos_res.set(0) = 0 ;
663  val_res.set(pos_res).set_domain(num_dom) = mult_cos_phi(auxi) - mult_sin_phi((*comp_p.val_t)(pos_so)(num_dom)) ;
664  // Y part
665  pos_res.set(0) = 1 ;
666  val_res.set(pos_res).set_domain(num_dom) = mult_sin_phi(auxi) + mult_cos_phi((*comp_p.val_t)(pos_so)(num_dom)) ;
667  // Z part
668  pos_res.set(0) = 2 ;
669  val_res.set(pos_res).set_domain(num_dom) =
670  mult_cos_theta((*comp_r.val_t)(pos_so)(num_dom)) - mult_sin_theta((*comp_t.val_t)(pos_so)(num_dom)) ;
671  }
672  while (pos_so.inc()) ;
673 }
674 
675  bool doder = ((so.der_t==0x0) || (rad_term_eq->der_t==0x0)) ? false : true ;
676  if (doder) {
677  Tensor der_res (so.get_val_t().get_space(), valence+1, type_ind, basis) ;
678  Index pos_so (*so.der_t) ;
679  Index pos_res (val_res) ;
680  do {
681  for (int i=1 ; i<valence+1 ; i++)
682  pos_res.set(i) = pos_so(i-1) ;
683 
684  Val_domain auxi (mult_sin_theta((*comp_r.der_t)(pos_so)(num_dom)) + mult_cos_theta((*comp_t.der_t)(pos_so)(num_dom))) ;
685  // X part
686  pos_res.set(0) = 0 ;
687  der_res.set(pos_res).set_domain(num_dom) = mult_cos_phi(auxi) - mult_sin_phi((*comp_p.der_t)(pos_so)(num_dom)) ;
688  // Y part
689  pos_res.set(0) = 1 ;
690  der_res.set(pos_res).set_domain(num_dom) = mult_sin_phi(auxi) + mult_cos_phi((*comp_p.der_t)(pos_so)(num_dom)) ;
691  // Z part
692  pos_res.set(0) = 2 ;
693  der_res.set(pos_res).set_domain(num_dom) =
694  mult_cos_theta((*comp_r.der_t)(pos_so)(num_dom)) - mult_sin_theta((*comp_t.der_t)(pos_so)(num_dom)) ;
695  }
696  while (pos_so.inc()) ;

698  return Term_eq (num_dom, val_res, der_res) ;
699  }
