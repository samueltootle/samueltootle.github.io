44 template <typename config_t>
45 std::string solve_NS_from_binary(config_t& bconfig, const size_t bco) {
46  int rank = 0;
47  MPI_Comm_rank(MPI_COMM_WORLD, &rank);
48  std::string output_path = (bconfig.control(CONTROLS::SAVE_COS))
49  ? get_cos_path()
50  : bconfig.config_outputdir() + "/COs";
51  fs::create_directory(output_path);
52 
53  kadath_config_boost<BCO_NS_INFO> nsconfig;
54  nsconfig.set_defaults();
55 
56  // Tells the NS driver to initialize the numerical space and fields
57  nsconfig.control(CONTROLS::SEQUENCES) = true;
58 
59  // copy parameters from binary configuration
60  for (int i = 0; i < BCO_PARAMS::NUM_BCO_PARAMS; ++i)
61  nsconfig.set(i) = bconfig.set(i, bco);
62 
63  for (int i = 0; i < EOS_PARAMS::NUM_EOS_PARAMS; ++i)
64  nsconfig.set_eos(i) = bconfig.set_eos(i, bco);
65 
66  for (int i = 0; i < CONTROLS::NUM_CONTROLS; ++i)
67  nsconfig.control(i) = bconfig.control(i);
68 
69  for (int i = 0; i < SEQ_SETTINGS::NUM_SEQ_SETTINGS; ++i)
70  nsconfig.seq_setting(i) = bconfig.seq_setting(i);
71 
72  Parameter_sequence resolution("res", BCO_PARAMS::BCO_RES);
73  resolution.set(9, 9, nsconfig(BCO_PARAMS::BCO_RES));
74 
75  nsconfig.set_filename("initns");
76  nsconfig.set_outputdir(output_path);
77 
78  if (bconfig.control(CONTROLS::USE_BOOSTED_CO))
79  nsconfig.set_stage(STAGES::BIN_BOOST) = true;
80 
81  const int ninshells = bconfig(BCO_PARAMS::NINSHELLS, bco);
82  const int nshells = bconfig(BCO_PARAMS::NSHELLS, bco);
83 
84  nsconfig(BCO_PARAMS::NINSHELLS) = 0;
85  nsconfig(BCO_PARAMS::NSHELLS) =
86  (bconfig.control(CONTROLS::CO_USE_SHELLS)) ? nshells : 0;
87 
88  int err = ns_3d_xcts_binary_boost_driver(nsconfig, resolution, output_path,
89  bconfig, bco);
90  MPI_Barrier(MPI_COMM_WORLD);
91 
92  // update binary parameters
93  for (int i = 0; i < BCO_PARAMS::NUM_BCO_PARAMS; ++i)
94  bconfig.set(i, bco) = nsconfig.set(i);
95  // put shells back for the binary
96  bconfig.set(BCO_PARAMS::NINSHELLS, bco) = ninshells;
97  bconfig.set(BCO_PARAMS::NSHELLS, bco) = nshells;
98 
99  return nsconfig.config_filename_abs();
100 }

153 inline void check_dist(double dist,
154  double M1,
155  double M2,
156  double garbage_factor) {
157  auto M = M1 + M2;
158  auto const garbage_dist = garbage_factor * M;
159  auto const recommended_dist = 8. * M;
160  if (dist <= garbage_dist) {
161  std::cerr << "Distance is set to (" << dist
162  << ") which may not give results. \nSet to (" << recommended_dist
163  << ") for something reasonable.\n";
164  std::_Exit(EXIT_FAILURE);
165  }
166 }

168 template <class space_t, class metric_t>
169 Scalar compute_ddrPsi(space_t& space,
170  Scalar& conf,
171  metric_t metric,
172  std::vector<int> excluded_doms,
173  int bound_dom = -1) {
174  std::string const def_drP{"drP = dr(P)"};
175  std::string const def_drdrP{"ddrP = dr(drP)"};
176  auto ndom{space.get_nbr_domains()};
177  System_of_eqs syst(space);
178  metric.set_system(syst, "f");
179  syst.add_cst("P", conf);
180 
181  if (excluded_doms.size() == 0) {
182  syst.add_def(def_drP.c_str());
183  syst.add_def(def_drdrP.c_str());
184  } else {
185  auto last_dom = (bound_dom == -1) ? ndom : bound_dom;
186  for (auto dom = 0; dom < last_dom; ++dom) {
187  auto res = std::find(excluded_doms.begin(), excluded_doms.end(), dom);
188  if (res == std::end(excluded_doms)) {
189  syst.add_def(dom, def_drP.c_str());
190  syst.add_def(dom, def_drdrP.c_str());
191  }
192  }
193  }
194  Scalar field(syst.give_val_def("ddrP"));
195  field.std_base();
196  return field;
197 }

227 template <typename config_t>
228 std::string solve_NS_ISO_from_XCTS_config(config_t& bconfig,
229  ns_sequence const& seq) {
230  int rank = 0;
231  MPI_Comm_rank(MPI_COMM_WORLD, &rank);
232  std::string output_path = (bconfig.control(CONTROLS::SAVE_COS))
233  ? get_cos_path()
234  : bconfig.config_outputdir() + "/COs";
235  fs::create_directory(output_path);
236 
237  kadath_config_boost<BCO_ISO_NS_INFO> nsconfig;
238  nsconfig.set_defaults();
239 
240  // Activate all controls that are active in the seq config
241  for (int i = 0; i < SEQ_SETTINGS::NUM_SEQ_SETTINGS; ++i)
242  nsconfig.seq_setting(i) = bconfig.seq_setting(i);
243 
244  // copy parameters from binary configuration
245  for (int i = 0; i < BCO_PARAMS::NUM_BCO_PARAMS; ++i)
246  nsconfig.set(i) = bconfig.set(i);
247  nsconfig.set(BCO_PARAMS::DIM) = 2;
248 
249  for (int i = 0; i < CONTROLS::NUM_CONTROLS; ++i)
250  nsconfig.control(i) = bconfig.control(i);
251 
252  for (int i = 0; i < STAGES::NUM_STAGES; ++i) {
253  nsconfig.set_stage(i) = bconfig.set_stage(i);
254  }
255 
256  // Tells the NS driver to initialize the numerical space and fields
257  nsconfig.control(CONTROLS::SEQUENCES) = true;
258  update_eos_parameters(bconfig, nsconfig);
259  update_diffrot_parameters(bconfig, nsconfig);
260 
261  Parameter_sequence resolution("res", BCO_PARAMS::BCO_RES);
262  resolution.set(9, 9, nsconfig(BCO_PARAMS::BCO_RES));
263 
264  nsconfig.set_filename("initns");
265  nsconfig.set_outputdir(output_path);
266 
267  auto ns_iso_sol_config =
268  ns_isotropic_sequence(nsconfig, seq, resolution, output_path);
269 
270  const std::string eos_type = bconfig.template eos<std::string>(EOSTYPE);
271  nsconfig.control(CONTROLS::SEQUENCES) = false;
272  if (rank == 0) {
273  EOS_Function_Dispatcher::dispatch<NS_ISO_to_XCTS_convert>(ns_iso_sol_config,
274  eos_type,
275  ns_iso_sol_config,
276  output_path);
277  }
278  ns_iso_sol_config.set_filename("initns_xcts.info");
279  ns_iso_sol_config.set_outputdir(output_path);
280  MPI_Barrier(MPI_COMM_WORLD);
281 
282  return ns_iso_sol_config.config_filename_abs();
283 }
