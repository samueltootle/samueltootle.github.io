41 template<typename config_t>
42 std::string solve_NS_from_binary(config_t& bconfig, const size_t bco) {
43  int rank = 0;
44  MPI_Comm_rank(MPI_COMM_WORLD, &rank);
45  std::string output_path = (bconfig.control(CONTROLS::SAVE_COS)) ?
46  get_cos_path() : bconfig.config_outputdir()+"/COs";
47  fs::create_directory(output_path);
48 
49  kadath_config_boost<BCO_NS_INFO> nsconfig;
50  nsconfig.set_defaults();
51 
52  // Tells the NS driver to initialize the numerical space and fields
53  nsconfig.control(CONTROLS::SEQUENCES) = true;
54 
55  // copy parameters from binary configuration
56  for(int i = 0; i < BCO_PARAMS::NUM_BCO_PARAMS; ++i)
57  nsconfig.set(i) = bconfig.set(i, bco);
58 
59  for(int i = 0; i < EOS_PARAMS::NUM_EOS_PARAMS; ++i)
60  nsconfig.set_eos(i) = bconfig.set_eos(i, bco);
61 
62  for(int i = 0; i < CONTROLS::NUM_CONTROLS; ++i)
63  nsconfig.control(i) = bconfig.control(i);
64 
65  for(int i = 0; i < SEQ_SETTINGS::NUM_SEQ_SETTINGS; ++i)
66  nsconfig.seq_setting(i) = bconfig.seq_setting(i);
67 
68  Parameter_sequence resolution("res", BCO_PARAMS::BCO_RES);
69  resolution.set(9,9,nsconfig(BCO_PARAMS::BCO_RES));
70 
71  nsconfig.set_filename("initns");
72  nsconfig.set_outputdir(output_path);
73 
74  if(bconfig.control(CONTROLS::USE_BOOSTED_CO))
75  nsconfig.set_stage(STAGES::BIN_BOOST) = true;
76 
77  const int ninshells = bconfig(BCO_PARAMS::NINSHELLS, bco);
78  const int nshells = bconfig(BCO_PARAMS::NSHELLS, bco);
79 
80  nsconfig(BCO_PARAMS::NINSHELLS) = 0;
81  nsconfig(BCO_PARAMS::NSHELLS) = (bconfig.control(CONTROLS::CO_USE_SHELLS)) ?
82  nshells : 0;
83 
84  int err = ns_3d_xcts_binary_boost_driver(nsconfig,
85  resolution, output_path, bconfig, bco);
86  MPI_Barrier(MPI_COMM_WORLD);
87 
88  // update binary parameters
89  for(int i = 0; i < BCO_PARAMS::NUM_BCO_PARAMS; ++i)
90  bconfig.set(i, bco) = nsconfig.set(i) ;
91  // put shells back for the binary
92  bconfig.set(BCO_PARAMS::NINSHELLS, bco) = ninshells;
93  bconfig.set(BCO_PARAMS::NSHELLS, bco) = nshells;
94 
95  return nsconfig.config_filename_abs();
96 }

148 inline
149 void check_dist(double dist, double M1, double M2, double garbage_factor) {
150  auto M = M1 + M2;
151  auto const garbage_dist = garbage_factor * M;
152  auto const recommended_dist = 8. * M;
153  if(dist <= garbage_dist) {
154  std::cerr << "Distance is set to (" << dist
155  << ") which will not give results. \nSet to ("
156  << recommended_dist << ") for something reasonable.\n";
157  std::_Exit(EXIT_FAILURE);
158  }
159 }

161 template<class space_t, class metric_t>
162 Scalar compute_ddrPsi(space_t& space, Scalar& conf, metric_t metric,
163  std::vector<int> excluded_doms, int bound_dom=-1) {
164 
165  std::string const def_drP{"drP = dr(P)"};
166  std::string const def_drdrP{"ddrP = dr(drP)"};
167  auto ndom{space.get_nbr_domains()};
168  System_of_eqs syst(space);
169  metric.set_system(syst, "f");
170  syst.add_cst("P", conf);
171 
172  if(excluded_doms.size() == 0) {
173  syst.add_def(def_drP.c_str());
174  syst.add_def(def_drdrP.c_str());
175  }
176  else {
177  auto last_dom = (bound_dom == -1) ? ndom : bound_dom;
178  for(auto dom = 0; dom < last_dom; ++dom) {
179  auto res = std::find(excluded_doms.begin(), excluded_doms.end(), dom);
180  if(res == std::end(excluded_doms)) {
181  syst.add_def(dom, def_drP.c_str());
182  syst.add_def(dom, def_drdrP.c_str());
183  }
184  }
185  }
186  Scalar field(syst.give_val_def("ddrP"));
187  field.std_base();
188  return field;
189 }
