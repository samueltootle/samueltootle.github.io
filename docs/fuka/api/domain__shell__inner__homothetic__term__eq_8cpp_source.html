28 Term_eq Domain_shell_inner_homothetic::integ_term_eq (const Term_eq& so, int bound) const {
29  // Check it is a tensor
30  if (so.get_type_data() != TERM_T) {
31  cerr << "integ_term_eq only defined with respect for a tensor" << endl ;
32  abort() ;
33  }
34 
35  if (so.get_val_t().get_n_comp() != 1) {
36  cerr << "integ_term_eq only defined with respect to a scalar" << endl ;
37  abort() ;
38  }
39 
40  assert (so.get_dom()==num_dom) ;
41 
42 
43  Term_eq rrso (mult_r_term_eq(mult_r_term_eq(so))) ;
44 
45 
46 
47  double resval = 0 ;
48  {// The value
49  Array<int> ind (so.get_val_t().indices(0)) ;
50  Val_domain value (mult_sin_theta(rrso.get_val_t()(ind)(num_dom))) ;
51  if (value.check_if_zero())
52  resval= 0. ;
53  else {
54 
55  int baset = (*value.get_base().bases_1d[1]) (0) ;
56  Index pcf (nbr_coefs) ;
57  switch (baset) {
58  case COS_ODD :
59  break ;
60  case SIN_EVEN :
61  break ;
62  case COS_EVEN : {
63  resval += M_PI*val_boundary(bound, value, pcf) ;
64  break ;
65  }
66  case SIN_ODD : {
67  for (int j=0 ; j<nbr_coefs(1) ; j++) {
68  pcf.set(1) = j ;
69  resval += 2./(2*double(j)+1) * val_boundary(bound, value, pcf) ;
70  }
71  break ;
72  }
73 
74  default :
75  cerr << "Case not yet implemented in Domain_shell_inner_homothetic::integrale" << endl ;
76  abort() ;
77  }
78 
79  }
80  resval *= 2*M_PI ;
81  }
82 
83  if (rrso.get_p_der_t()!=0x0) {
84  // The value
85  Array<int> ind (so.get_val_t().indices(0)) ;
86  Val_domain valueder (mult_sin_theta(rrso.get_der_t()(ind)(num_dom))) ;
87  double resder = 0;
88  if (valueder.check_if_zero())
89  resder= 0. ;
90  else {
91 
92  int baset = (*valueder.get_base().bases_1d[1]) (0) ;
93  Index pcf (nbr_coefs) ;
94  switch (baset) {
95  case COS_ODD :
96  break ;
97  case SIN_EVEN :
98  break ;
99  case COS_EVEN : {
100  resder += M_PI*val_boundary(bound, valueder, pcf) ;
101  break ;
102  }
103  case SIN_ODD : {
104  for (int j=0 ; j<nbr_coefs(1) ; j++) {
105  pcf.set(1) = j ;
106  resder += 2./(2*double(j)+1) * val_boundary(bound, valueder, pcf) ;
107  }
108  break ;
109  }
110  default :
111  cerr << "Case not yet implemented in Domain_shell_inner_homothetic::integrale" << endl ;
112  abort() ;
113  }
114  resder *= 2*M_PI ;
115  }
116  return Term_eq (num_dom, resval, resder) ;
117  }
118  else {
119  return Term_eq (num_dom, resval) ;
120  }
121 }
