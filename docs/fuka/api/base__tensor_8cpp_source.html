66 ostream& operator<< (ostream& flux , const Base_tensor& so) {
67  flux << "Tensorial basis" << endl ;
68  for (int d=0 ; d<so.space.get_nbr_domains() ; d++) {
69  flux << "Domain " << d << " : " ;
70  switch (so.get_basis(d)) {
71  case -1 :
72  flux << "Undefined" << endl ;
73  break ;
74  case CARTESIAN_BASIS :
75  flux << "Cartesian" << endl ;
76  break ;
77  case SPHERICAL_BASIS :
78  flux << "Spherical" << endl ;
79  break ;
80  case MTZ_BASIS :
81  flux << "MTZ" << endl ;
82  break ;
83  default:
84  cerr << "Unknown tensorial basis" << endl ;
85  abort() ;
86  }
87  }
88  return flux ;
89 }
