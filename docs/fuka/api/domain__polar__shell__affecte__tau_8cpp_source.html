27 void Domain_polar_shell::affecte_tau_val_domain (Val_domain& so, int mquant, const Array<double>& values, int& conte) const {
28 
29  so.allocate_coef() ;
30  *so.cf = 0. ;
31 
32  Index pos_cf (nbr_coefs) ;
33  // True values
34  // Loop on theta
35  int baset = (*so.get_base().bases_1d[1]) (0) ;
36  for (int j=0 ; j<nbr_coefs(1) ; j++) {
37  pos_cf.set(1) = j ;
38  bool true_tet = true ;
39  switch (baset) {
40  case COS_EVEN:
41  if ((j==0) && (mquant!=0))
42  true_tet = false ;
43  break ;
44  case COS_ODD:
45  if ((j==nbr_coefs(1)-1) || ((j==0) && (mquant!=0)))
46  true_tet = false ;
47  break ;
48  case SIN_EVEN:
49  if (((j==1) && (mquant>1)) || (j==0) || (j==nbr_coefs(1)-1))
50  true_tet = false ;
51  break ;
52  case SIN_ODD:
53  if (((j==0) && (mquant>1)) || (j==nbr_coefs(1)-1))
54  true_tet = false ;
55  break ;
56  default:
57  cerr << "Unknow theta basis in Domain_polar_shell::affecte_tau_val_domain" << endl ;
58  abort() ;
59  }
60 
61  if (true_tet)
62  for (int i=0 ; i<nbr_coefs(0) ; i++) {
63  pos_cf.set(0) = i ;
64  so.cf->set(pos_cf) += values(conte);
65  conte ++ ;
66  }
67  }
68 
69  // Regularity on the axis :
70  if (mquant>1) {
71  // Loop on r :
72  for (int i=0 ; i<nbr_coefs(0) ; i++) {
73  pos_cf.set(0) = i ;
74  double sum = 0. ;
75  switch (baset) {
76  case COS_EVEN:
77  for (int j=1 ; j<nbr_coefs(1) ; j++) {
78  pos_cf.set(1) = j ;
79  sum += (*so.cf)(pos_cf) ;
80  }
81  pos_cf.set(1) = 0 ;
82  so.cf->set(pos_cf) = -sum ;
83  break ;
84  case COS_ODD:
85  for (int j=1 ; j<nbr_coefs(1) ; j++) {
86  pos_cf.set(1) = j ;
87  sum += (*so.cf)(pos_cf) ;
88  }
89  pos_cf.set(1) = 0 ;
90  so.cf->set(pos_cf) = -sum ;
91  break ;
92  case SIN_EVEN:
93  for (int j=2 ; j<nbr_coefs(1) ; j++) {
94  pos_cf.set(1) = j ;
95  sum += j*(*so.cf)(pos_cf) ;
96  }
97  pos_cf.set(1) = 1 ;
98  so.cf->set(pos_cf) = -sum ;
99  break ;
100  case SIN_ODD:
101  for (int j=1 ; j<nbr_coefs(1) ; j++) {
102  pos_cf.set(1) = j ;
103  sum += (2*j+1)*(*so.cf)(pos_cf) ;
104  }
105  pos_cf.set(1) = 0 ;
106  so.cf->set(pos_cf) = -sum ;
107  break ;
108  default:
109  cerr << "Unknow theta basis in Domain_polar_shell::affecte_tau_val_domain" << endl ;
110  abort() ;
111  }
112  }
113  }
