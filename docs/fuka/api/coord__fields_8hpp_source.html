269 template<int ind_t>
270 Kadath::Vector CoordFields<space_t>::cart(double shift_x, double shift_y, double shift_z) const {
271  Kadath::Vector cart(space, ind_t, basis);
272 
273  int ndom = space.get_nbr_domains();
274 
275  for (int d = 0; d < ndom - 1; d++) {
276  cart.set(1).set_domain(d) = space.get_domain(d)->get_cart(1) - shift_x;
277  cart.set(2).set_domain(d) = space.get_domain(d)->get_cart(2) - shift_y;
278  cart.set(3).set_domain(d) = space.get_domain(d)->get_cart(3) - shift_z;
279  }
280 
281  // set outter boundary to surface coordinates (1/r) for compact domain
282  // FIXME: these are not coordinate shifted
283  cart.set(1).set_domain(ndom - 1) = space.get_domain(ndom - 1)->get_cart_surr(1);
284  cart.set(2).set_domain(ndom - 1) = space.get_domain(ndom - 1)->get_cart_surr(2);
285  cart.set(3).set_domain(ndom - 1) = space.get_domain(ndom - 1)->get_cart_surr(3);
286 
287  cart.set(1).std_base();
288  cart.set(2).std_base();
289  cart.set(3).std_anti_base();
290 
291  return cart;
292 }

294 template<typename space_t>
295 Kadath::Scalar CoordFields<space_t>::radius(double shift_x, double shift_y, double shift_z) const {
296  auto coords = this->cart(shift_x, shift_y, shift_z);
297  int ndom = space.get_nbr_domains();
298 
299  Kadath::Scalar r_sq = coords(1) * coords(1) + coords(2) * coords(2) + coords(3) * coords(3);
300  r_sq.std_base();
301 
302  Kadath::Scalar r = sqrt(r_sq);
303  r.std_base();
304 
305  // set outter boundary to constant large radius
306  Index pos(space.get_domain(ndom-1)->get_nbr_points());
307  const int npts_r = space.get_domain(ndom-1)->get_nbr_points()(0);
308  for(int i = 1; i <= 3; ++i) {
309  const int dom = ndom - 1;
310  do{
311  if(pos(0) == npts_r - 1) {
312  r.set_domain(dom).set(pos) = 1e10;
313  }
314  }while(pos.inc());
315  }
316 
317  return r;
318 }

334 template<typename space_t>
335 template<int ind_t>
336 Kadath::Vector CoordFields<space_t>::e_rad(double shift_x, double shift_y, double shift_z) const {
337  auto coords = this->cart(shift_x, shift_y, shift_z);
338 
339  int ndom = space.get_nbr_domains();
340 
341  // this can be problematic around the given origin
342  Kadath::Vector e_rad(space, ind_t, basis);
343  for(int i : {1,2,3}) {
344  e_rad.set(i) = coords(i) / this->radius(shift_x, shift_y, shift_z);
345 
346  // fix non-finite values likely at (0,0,0)
347  // Note: this fix is sufficient for the FUKA codes as
348  // we are only concerned with e_rad on the surfaces of compact
349  // objects or at infinity. This fix is to remove NaNs and their
350  // impact on the system of equations
351  for(int dom=0; dom < ndom - 1; ++dom) {
352  Index pos(space.get_domain(dom)->get_nbr_points());
353  __attribute__((unused)) const int npts_r = space.get_domain(dom)->get_nbr_points()(0);
354  do{
355  if(!std::isfinite(e_rad(i)(dom)(pos)) && pos(0) == 0) {
356  Index tempos(pos);
357  tempos.set(0) = pos(0) + 1;
358  e_rad.set(i).set_domain(dom).set(pos) = e_rad(i)(dom)(tempos);
359  }
360  }while(pos.inc());
361  }
362 
363  // FIXME this is not correct for shifted cartesian coords
364  e_rad.set(i).set_domain(ndom - 1) = space.get_domain(ndom - 1)->get_cart_surr(i);
365  }
366 
367  e_rad.set(1).std_base();
368  e_rad.set(2).std_base();
369  e_rad.set(3).std_anti_base();
370 
371  return e_rad;
372 }

390 template<typename space_t>
391 void update_fields (CoordFields<space_t> const & cf_generator,
392  vec_ary_t & coord_vectors,
393  scalar_ary_t & coord_scalars,
394  const double xo, const double xc1, const double xc2,
395  std::unique_ptr<System_of_eqs>& syst) {
396 
397  if(coord_vectors[GLOBAL_ROT])
398  *coord_vectors[GLOBAL_ROT] = cf_generator.template rot_z<>(xo);
399  if(coord_vectors[BCO1_ROT])
400  *coord_vectors[BCO1_ROT] = cf_generator.template rot_z<>(xc1) ;
401  if(coord_vectors[BCO2_ROT])
402  *coord_vectors[BCO2_ROT] = cf_generator.template rot_z<>(xc2);
403  if(coord_vectors[EX])
404  *coord_vectors[EX] = cf_generator.template e_cart<COV>(1);
405  if(coord_vectors[EY])
406  *coord_vectors[EY] = cf_generator.template e_cart<COV>(2);
407  if(coord_vectors[EZ])
408  *coord_vectors[EZ] = cf_generator.template e_cart<COV>(3);
409  if(coord_vectors[S_BCO1])
410  *coord_vectors[S_BCO1] = cf_generator.template e_rad<COV>(xc1);
411  if(coord_vectors[S_BCO2])
412  *coord_vectors[S_BCO2] = cf_generator.template e_rad<COV>(xc2);
413  if(coord_vectors[S_INF])
414  *coord_vectors[S_INF] = cf_generator.template e_rad<COV>(xo);
415  if(coord_scalars[R_BCO1])
416  *coord_scalars[R_BCO1] = cf_generator.radius(xc1);
417  if(coord_scalars[R_BCO2])
418  *coord_scalars[R_BCO2] = cf_generator.radius(xc2);
419 
420  int ndom = coord_vectors[GLOBAL_ROT]->get_space().get_nbr_domains();
421 
422  auto update = [&] (auto& name, auto& field) {
423  for(int dom = 0; dom < ndom; ++dom){
424  __attribute__((unused)) bool succ = update_field(*syst, dom, name.c_str(), field);
425  #ifdef DEBUG
426  if(!succ)
427  std::cout << name << " failed\n";
428  #endif
429  }
430  };
431  if(syst) {
432  for(int i = 0; i < NUM_VECTORS; ++i) {
433  if(coord_vectors[i])
434  update(cv_names[i], *coord_vectors[i]);
435  }
436  for(int i = 0; i < NUM_SCALARS; ++i) {
437  if(coord_scalars[i])
438  update(cs_names[i], *coord_scalars[i]);
439  }
440  syst->sec_member();
441  }
442 }

510 inline bool update_field(System_of_eqs& syst, int dom, const char* n, Tensor& so) {
511  char* name;
512  name = new char[LMAX];
513  trim_spaces(name, n);
514  int which = -1 ;
515  int valence;
516  char* name_ind = 0x0 ;
517  Array<int>* type_ind = 0x0 ;
518  bool found = false ;
519 
520  found = syst.iscst (name, which, valence, name_ind, type_ind) ;
521  if(found) {
522  Term_eq* mm = syst.give_cst(which, dom);
523  mm->set_val_t(so);
524  }
525  return found;
526 }

529 vec_ary_t default_binary_vector_ary(space_t& space) {
530  vec_ary_t coord_vectors{};
