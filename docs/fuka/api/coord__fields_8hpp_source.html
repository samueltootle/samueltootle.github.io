261 template<typename space_t>
262 template<int ind_t>
263 Kadath::Vector CoordFields<space_t>::cart(double shift_x, double shift_y, double shift_z) const {
264  Kadath::Vector cart(space, ind_t, basis);
265 
266  int ndom = space.get_nbr_domains();
267 
268  for (int d = 0; d < ndom - 1; d++) {
269  cart.set(1).set_domain(d) = space.get_domain(d)->get_cart(1) - shift_x;
270  cart.set(2).set_domain(d) = space.get_domain(d)->get_cart(2) - shift_y;
271  cart.set(3).set_domain(d) = space.get_domain(d)->get_cart(3) - shift_z;
272  }
273 
274  // set outter boundary to surface coordinates (1/r) for compact domain
275  // FIXME: these are not coordinate shifted
276  cart.set(1).set_domain(ndom - 1) = space.get_domain(ndom - 1)->get_cart_surr(1);
277  cart.set(2).set_domain(ndom - 1) = space.get_domain(ndom - 1)->get_cart_surr(2);
278  cart.set(3).set_domain(ndom - 1) = space.get_domain(ndom - 1)->get_cart_surr(3);
279 
280  cart.set(1).std_base();
281  cart.set(2).std_base();
282  cart.set(3).std_anti_base();
283 
284  return cart;
285 }

287 template<typename space_t>
288 Kadath::Scalar CoordFields<space_t>::radius(double shift_x, double shift_y, double shift_z) const {
289  auto coords = this->cart(shift_x, shift_y, shift_z);
290  int ndom = space.get_nbr_domains();
291 
292  Kadath::Scalar r_sq = coords(1) * coords(1) + coords(2) * coords(2) + coords(3) * coords(3);
293  r_sq.std_base();
294 
295  Kadath::Scalar r = sqrt(r_sq);
296  r.std_base();
297 
298  // set outter boundary to constant large radius
299  Index pos(space.get_domain(ndom-1)->get_nbr_points());
300  const int npts_r = space.get_domain(ndom-1)->get_nbr_points()(0);
301  for(int i = 1; i <= 3; ++i) {
302  const int dom = ndom - 1;
303  do{
304  if(pos(0) == npts_r - 1) {
305  r.set_domain(dom).set(pos) = 1e10;
306  }
307  }while(pos.inc());
308  }
309 
310  return r;
311 }

327 template<typename space_t>
328 template<int ind_t>
329 Kadath::Vector CoordFields<space_t>::e_rad(double shift_x, double shift_y, double shift_z) const {
330  auto coords = this->cart(shift_x, shift_y, shift_z);
331 
332  int ndom = space.get_nbr_domains();
333 
334  // this can be problematic around the given origin
335  Kadath::Vector e_rad(space, ind_t, basis);
336  for(int i : {1,2,3}) {
337  e_rad.set(i) = coords(i) / this->radius(shift_x, shift_y, shift_z);
338 
339  // fix non-finite values likely at (0,0,0)
340  // Note: this fix is sufficient for the FUKA codes as
341  // we are only concerned with e_rad on the surfaces of compact
342  // objects or at infinity. This fix is to remove NaNs and their
343  // impact on the system of equations
344  for(int dom=0; dom < ndom - 1; ++dom) {
345  Index pos(space.get_domain(dom)->get_nbr_points());
346  const int npts_r = space.get_domain(dom)->get_nbr_points()(0);
347  do{
348  if(!std::isfinite(e_rad(i)(dom)(pos)) && pos(0) == 0) {
349  Index tempos(pos);
350  tempos.set(0) = pos(0) + 1;
351  e_rad.set(i).set_domain(dom).set(pos) = e_rad(i)(dom)(tempos);
352  }
353  }while(pos.inc());
354  }
355 
356  // FIXME this is not correct for shifted cartesian coords
357  e_rad.set(i).set_domain(ndom - 1) = space.get_domain(ndom - 1)->get_cart_surr(i);
358  }
359 
360  e_rad.set(1).std_base();
361  e_rad.set(2).std_base();
362  e_rad.set(3).std_anti_base();
363 
364  return e_rad;
365 }

383 template<typename space_t>
384 void update_fields (CoordFields<space_t> const & cf_generator,
385  vec_ary_t & coord_vectors,
386  scalar_ary_t & coord_scalars,
387  const double xo, const double xc1, const double xc2,
388  System_of_eqs* syst) {
389 
390  if(coord_vectors[GLOBAL_ROT])
391  *coord_vectors[GLOBAL_ROT] = cf_generator.template rot_z<>(xo);
392  if(coord_vectors[BCO1_ROT])
393  *coord_vectors[BCO1_ROT] = cf_generator.template rot_z<>(xc1) ;
394  if(coord_vectors[BCO2_ROT])
395  *coord_vectors[BCO2_ROT] = cf_generator.template rot_z<>(xc2);
396  if(coord_vectors[EX])
397  *coord_vectors[EX] = cf_generator.template e_cart<COV>(1);
398  if(coord_vectors[EY])
399  *coord_vectors[EY] = cf_generator.template e_cart<COV>(2);
400  if(coord_vectors[EZ])
401  *coord_vectors[EZ] = cf_generator.template e_cart<COV>(3);
402  if(coord_vectors[S_BCO1])
403  *coord_vectors[S_BCO1] = cf_generator.template e_rad<COV>(xc1);
404  if(coord_vectors[S_BCO2])
405  *coord_vectors[S_BCO2] = cf_generator.template e_rad<COV>(xc2);
406  if(coord_vectors[S_INF])
407  *coord_vectors[S_INF] = cf_generator.template e_rad<COV>(xo);
408  if(coord_scalars[R_BCO1])
409  *coord_scalars[R_BCO1] = cf_generator.radius(xc1);
410  if(coord_scalars[R_BCO2])
411  *coord_scalars[R_BCO2] = cf_generator.radius(xc2);
412 
413  int ndom = coord_vectors[GLOBAL_ROT]->get_space().get_nbr_domains();
414 
415  auto update = [&] (auto& name, auto& field) {
416  for(int dom = 0; dom < ndom; ++dom){
417  bool succ = update_field(*syst, dom, name.c_str(), field);
418  #ifdef DEBUG
419  if(!succ)
420  std::cout << name << " failed\n";
421  #endif
422  }
423  };
424  if(syst != nullptr) {
425  for(int i = 0; i < NUM_VECTORS; ++i) {
426  if(coord_vectors[i])
427  update(cv_names[i], *coord_vectors[i]);
428  }
429  for(int i = 0; i < NUM_SCALARS; ++i) {
430  if(coord_scalars[i])
431  update(cs_names[i], *coord_scalars[i]);
432  }
433  syst->sec_member();
434  }
435 }

462 inline bool update_field(System_of_eqs& syst, int dom, const char* n, Tensor& so) {
463  char* name;
464  name = new char[LMAX];
465  trim_spaces(name, n);
466  int which = -1 ;
467  int valence;
468  char* name_ind = 0x0 ;
469  Array<int>* type_ind = 0x0 ;
470  bool found = false ;
471 
472  found = syst.iscst (name, which, valence, name_ind, type_ind) ;
473  if(found) {
474  Term_eq* mm = syst.give_cst(which, dom);
475  mm->set_val_t(so);
476  }
477  return found;
478 }
