290 double Scalar::val_point(const Point& xx, int sens) const {
291 
292  assert ((sens==+1) || (sens==-1)) ;
293 
294  bool* inside = MemoryMapper::get_memory<bool>(ndom);
295 
296  for (int l=ndom-1 ; l>=0 ; l--)
297  inside[l] = get_domain(l)->is_in(xx) ;
298  // First domain in which the point is :
299  int ld = -1 ;
300  if (sens == -1) {
301  for (int l=ndom-1 ; l>=0 ; l--)
302  if ((ld==-1) && (inside[l]))
303  ld = l ;
304  }
305  else {
306  for (int l=0 ; l<ndom ; l++)
307  if ((ld==-1) && (inside[l]))
308  ld = l ;
309  }
310 
311  if (ld==-1) {
312  cout << "Point " << xx << "not found in the computational space..." << endl ;
313  abort() ;
314  }
315  else {
316  if (val_zones[ld]->check_if_zero())
317  return 0. ;
318  else {
319  Point num(get_domain(ld)->absol_to_num(xx)) ;
320 
321  coef() ;
322  MemoryMapper::release_memory<bool>(inside, ndom);
323  return val_zones[ld]->base.summation(num, *val_zones[ld]->cf) ;
324  }
325  }
326 }

376 void Scalar::filter_phi (int dom, int ncf) {
377  coef() ;
378  Index pcf (get_space().get_domain(dom)->get_nbr_coefs()) ;
379  int np = get_space().get_domain(dom)->get_nbr_coefs()(2) ;
380  do {
381  if (pcf(2)+ncf > np-1)
382  set_domain(dom).set_coef(pcf) = 0 ;
383  }
384  while (pcf.inc()) ;
385 }
