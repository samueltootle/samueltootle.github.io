41 inline void update_bin_config(config_t& bconfig,
42  const Space_bin_bh& space,
43  const Scalar& conf,
44  const Scalar& lapse) {
45  if (std::isnan(bconfig.set(OUTER_SHELLS)))
46  bconfig.set(OUTER_SHELLS) = 0;
47 
48  bconfig.set(Q) = bconfig(MCH, BCO1) / bconfig(MCH, BCO2);
49  bconfig.set(BIN_PARAMS::Q) = (bconfig.set(BIN_PARAMS::Q) > 1.0)
50  ? 1.0 / bconfig.set(BIN_PARAMS::Q)
51  : bconfig.set(BIN_PARAMS::Q);
52 
53  if (!bconfig.control(USE_CONFIG_VARS)) {
54 
55  bconfig.set(REXT) = 2. * bconfig(DIST);
56 
57  const double r_max_tot =
58  (bconfig(BCO_PARAMS::RMID, NODES::BCO1) > bconfig(RMID, NODES::BCO2))
59  ? bconfig(BCO_PARAMS::RMID, NODES::BCO1)
60  : bconfig(BCO_PARAMS::RMID, NODES::BCO2);
61 
62  const double rout_sep_est =
63  (bconfig(BIN_PARAMS::DIST) / 2. - r_max_tot) / 3. + r_max_tot;
64  const double rout_min_est = bco_u::gold_ratio * r_max_tot;
65 
66  bconfig.set(BCO_PARAMS::ROUT, NODES::BCO1) = rout_sep_est;
67  // (rout_sep_est < rout_min_est) ? rout_min_est : rout_sep_est;
68  bconfig.set(BCO_PARAMS::ROUT, NODES::BCO2) =
69  bconfig(BCO_PARAMS::ROUT, NODES::BCO1);
70 
71  const double q = bconfig.set(BIN_PARAMS::Q);
72  bconfig.set(BCO_PARAMS::MIN_SHELL_DR, NODES::BCO1) =
73  1.7 * std::pow(2.0, bco_u::shell_factor / q);
74  bconfig.set(BCO_PARAMS::MIN_SHELL_DR, NODES::BCO2) =
75  bconfig(BCO_PARAMS::MIN_SHELL_DR, NODES::BCO1);
76  }
77 }

79 inline int bbh_xcts_regrid(config_t& bconfig, std::string outputfile) {
80  int exit_status = EXIT_SUCCESS;
81  std::string kadath_filename = bconfig.space_filename();
82 
83  FILE* fin = fopen(kadath_filename.c_str(), "r");
84  Space_bin_bh old_space(fin);
85  Scalar old_conf(old_space, fin);
86  Scalar old_lapse(old_space, fin);
87  Vector old_shift(old_space, fin);
88  fclose(fin);
89 
90  // Update config vars
91  // This control was mainly for testing
92  if (!bconfig.control(USE_CONFIG_VARS)) {
93  bco_u::update_config_BH_radii(old_space, bconfig, old_space.BH1 + 1,
94  old_conf, NODES::BCO1);
95  bco_u::update_config_BH_radii(old_space, bconfig, old_space.BH2 + 1,
96  old_conf, NODES::BCO2);
97  } // end updating config vars
98 
99  update_bin_config(bconfig, old_space, old_conf, old_lapse);
100 
101  int ndom = old_space.get_nbr_domains();
102 
103  int res = bconfig.set(BIN_RES);
104 
105  if ((res % 2) == 0) {
106  std::cout << "New Resolution is invalid. Must be odd (9,11,13,etc)"
107  << std::endl;
108  std::_Exit(EXIT_FAILURE);
109  }
110 
111  int type_coloc = old_space.get_type_base();
112 
113  // setup bounds
114  std::vector<double> out_bounds(1 + bconfig(OUTER_SHELLS));
115  for (int e = 0; e < out_bounds.size(); ++e)
116  out_bounds[e] = bconfig(REXT) * (1. + e * 0.25);
117 
118  std::vector<int> exclusion_doms{old_space.BH1, old_space.BH1 + 1,
119  old_space.BH2, old_space.BH2 + 1};
120  auto drPsi(compute_drPsi(old_space, old_conf,
121  Metric_flat(old_space, old_shift.get_basis()),
122  exclusion_doms, old_space.OUTER));
123  std::vector<double> BH1_bounds;
124  std::vector<double> BH2_bounds;
125  BH1_bounds =
126  bco_u::set_arb_boundsv3(bconfig, drPsi, old_space.BH1 + 2, NODES::BCO1);
127  BH2_bounds =
128  bco_u::set_arb_boundsv3(bconfig, drPsi, old_space.BH2 + 2, NODES::BCO2);
129  // end setup bounds
130 
131  Space_bin_bh space(type_coloc, bconfig(DIST), BH1_bounds, BH2_bounds,
132  out_bounds, bconfig(BIN_RES));
133  Base_tensor basis(space, CARTESIAN_BASIS);
134 
135  std::cout << "Resolution of old space: ";
136  bco_u::print_constant_space_resolution(old_space);
137 
138  std::cout << "Resolution of new space: ";
139  bco_u::print_constant_space_resolution(space);
140 
141  std::cout << "\nold bounds:" << std::endl;
142  bco_u::print_bounds_from_space(old_space);
143 
144  std::cout << "New bounds:" << std::endl;
145  bco_u::print_bounds_from_space(space);
146 
147  // needed in some cases, since there is no data in 0,1 and the interpolation
148  // can go crazy
149  std::array<const int, 2> old_nuc_doms{old_space.BH1, old_space.BH2};
150  std::array<const Domain_shell_outer_homothetic*, 2> old_outer_homothetic{
151  dynamic_cast<const Domain_shell_outer_homothetic*>(
152  old_space.get_domain(old_space.BH1 + 1)),
153  dynamic_cast<const Domain_shell_outer_homothetic*>(
154  old_space.get_domain(old_space.BH2 + 1))};
155  for (auto& i : {0, 1}) {
156  // update BH fields to help with interpolation later
157  bco_u::update_adapted_field(old_conf, old_nuc_doms[i] + 2,
158  old_nuc_doms[i] + 1, old_outer_homothetic[i],
159  OUTER_BC);
160  bco_u::update_adapted_field(old_lapse, old_nuc_doms[i] + 2,
161  old_nuc_doms[i] + 1, old_outer_homothetic[i],
162  OUTER_BC);
163  for (int j = 1; j < 4; ++j)
164  bco_u::update_adapted_field(old_shift.set(j), old_nuc_doms[i] + 2,
165  old_nuc_doms[i] + 1, old_outer_homothetic[i],
166  OUTER_BC);
167  }
168 
169  // setup new fields
170  Scalar conf(space);
171  conf = 1.;
172 
173  Scalar lapse(space);
174  lapse = 1.;
175 
176  Vector shift(space, CON, basis);
177  for (int i = 1; i <= 3; i++)
178  shift.set(i).annule_hard();
179  // end setup
180 
181  // import old fields into new space
182  conf.import(old_conf);
183  lapse.import(old_lapse);
184 
185  for (int c = 1; c <= 3; ++c)
186  shift.set(c).import(old_shift(c));
187  // end import fields
188 
189  // make sure excised domains are set to zero
190  for (auto& dom : {space.BH1, space.BH2}) {
191  lapse.set_domain(dom).annule_hard();
192  lapse.set_domain(dom + 1).annule_hard();
193 
194  conf.set_domain(dom).annule_hard();
195  conf.set_domain(dom + 1).annule_hard();
196 
197  for (int i = 1; i <= 3; ++i) {
198  shift.set(i).set_domain(dom).annule_hard();
199  shift.set(i).set_domain(dom + 1).annule_hard();
200  }
201  }
202 
203  lapse.std_base();
204  conf.std_base();
205  shift.std_base();
206 
207  bconfig.set_filename(outputfile);
208  bco_u::save_to_file(space, bconfig, conf, lapse, shift);
209  return exit_status;
210 }
