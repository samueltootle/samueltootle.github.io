37 template <typename config_t>
38 int ns_isotropic_uniform_rot_regrid(config_t& bconfig, std::string outputfile) {
39  int exit_status = EXIT_SUCCESS;
40  using space_t = Space_polar_adapted;
41 
42  std::string kadath_filename = bconfig.space_filename();
43 
44  if (!fs::exists(kadath_filename)) {
45  std::cerr << "File: " << kadath_filename << " not found.\n\n";
46  std::_Exit(EXIT_FAILURE);
47  }
48 
49  FILE* ff1 = fopen(kadath_filename.c_str(), "r");
50  space_t old_space(ff1);
51  Scalar old_lap_Aterm(old_space, ff1);
52  Scalar old_nu(old_space, ff1);
53  Scalar old_logh(old_space, ff1);
54  Scalar old_lap_Bterm(old_space, ff1);
55  Scalar old_lap_wterm(old_space, ff1);
56  fclose(ff1);
57 
58  std::cout << "Resolution of old space: "
59  << old_space.get_domain(0)->get_nbr_points()(0) << " (r), "
60  << old_space.get_domain(0)->get_nbr_points()(1) << " (theta)\n";
61 
62  int ndim = 2;
63 
64  // get the adapted domain and cast it to its correct type to be able to call
65  // its member functions
66  const Domain_polar_shell_outer_adapted* old_outer_adapted =
67  dynamic_cast<const Domain_polar_shell_outer_adapted*>(
68  old_space.get_domain(old_space.ADAPTED_OUTER));
69 
70  // setup a scalar field representing the old radius
71  Scalar old_space_radius(old_space);
72  old_space_radius = 0.;
73 
74  // get the radius from each domain
75  for (int i = 0; i < old_space.get_nbr_domains(); ++i)
76  old_space_radius.set_domain(i) = old_space.get_domain(i)->get_radius();
77  // get the adapted radius of the adapted domain
78  old_space_radius.set_domain(old_space.ADAPTED_OUTER) =
79  old_outer_adapted->get_outer_radius();
80 
81  // define a standard decomposition, compatible with the parity of this field
82  old_space_radius.std_base();
83  // end setup old radius field
84 
85  // get the minimal and maximal radius from the adapted domain
86  auto [r_min, r_max] = Kadath::bco_utils::get_rmin_rmax(old_space, 1);
87 
88  std::cout << "Rmin/max: " << r_min << " " << r_max << std::endl;
89 
90  // set new resolutions in each spatial dimension
91  Dim_array res(ndim);
92  res.set(0) = bconfig(BCO_RES);
93  res.set(1) = res(0);
94 
95  // FIXME not sure if it's only about oddness...
96  if (res(0) % 2 == 0) {
97  std::cout << "New Resolution is invalid. Must be odd (9,11,13,etc)"
98  << std::endl;
99  std::_Exit(EXIT_FAILURE);
100  }
101 
102  std::cout << "Resolution of new space: " << res(0) << " (r), " << res(1)
103  << " (theta)" << std::endl;
104 
105  // get the type of the colocation points
106  int type_coloc = old_space.get_type_base();
107 
108  // ignore domain radii scaling from config if needed
109  if (!bconfig.control(USE_CONFIG_VARS)) {
110  bconfig.set(RIN) = 0.5 * r_min;
111  bconfig.set(ROUT) = bco_utils::gold_ratio * r_max;
112  bconfig.set(RMID) = r_max;
113  }
114  // end update config
115 
116  // setup radius bounds of the domains
117 
118  int const shells = bconfig(BCO_PARAMS::NSHELLS);
119  int const ndom = 4 + shells;
120  Array<double> bounds(ndom - 1);
121  bounds.set(0) = bconfig(RIN);
122  bounds.set(1) = bconfig(RMID);
123  bounds.set(2) = bconfig(ROUT);
124 
125  for (int shell = 1, b = 3; shell <= shells; ++shell, ++b) {
126  bounds.set(b) = bounds(b - 1) * 2;
127  }
128 
129  // get origin of nucleus domain
130  Point center = old_space.get_domain(0)->get_center();
131 
132  // initialize space with new resolution and domain decomposition
133  Space_polar_adapted space(type_coloc, center, res, bounds);
134 
135  // get adapted domains to update the radius
136  const Domain_polar_shell_outer_adapted* new_outer_adapted =
137  dynamic_cast<const Domain_polar_shell_outer_adapted*>(
138  space.get_domain(space.ADAPTED_OUTER));
139  const Domain_polar_shell_inner_adapted* new_inner_adapted =
140  dynamic_cast<const Domain_polar_shell_inner_adapted*>(
141  space.get_domain(space.ADAPTED_INNER));
142 
143  // update adapted domain mapping
144  Kadath::bco_utils::interp_adapted_mapping(new_outer_adapted, 1,
145  old_space_radius);
146  Kadath::bco_utils::interp_adapted_mapping(new_inner_adapted, 1,
147  old_space_radius);
148 
149  // setup new fields
150  // initialize to one or zero first
151  Scalar lap_Aterm(space);
152  lap_Aterm.annule_hard();
153  lap_Aterm.std_base();
154 
155  Scalar lap_Bterm(lap_Aterm);
156  lap_Bterm.std_base();
157 
158  Scalar lap_wterm(lap_Aterm);
159  lap_wterm.std_base();
160 
161  Scalar nu(space);
162  nu.annule_hard();
163  nu.std_base();
164 
165  Scalar logh(space);
166  logh.annule_hard();
167  logh.std_base();
168  // end setup new fields
169 
170  // import data from fields in the old space
171  lap_Aterm.import(old_lap_Aterm);
172  nu.import(old_nu);
173  logh.import(old_logh);
174  lap_Bterm.import(old_lap_Bterm);
175  lap_wterm.import(old_lap_wterm);
176  // end import old fields
177 
178  Scalar one(space);
179  one = 1.;
180  one.std_base();
181 
182  Scalar rsint = Scalar(one.mult_r().mult_sin_theta());
183 
184  // enforce spectral decomposition compatible with the parities
185  lap_Aterm.std_base();
186  nu.std_base();
187  logh.std_base();
188 
189  for (int d = 0; d < ndom; ++d) {
190  lap_Bterm.set_domain(d).set_base() = rsint(d).get_base();
191  lap_wterm.set_domain(d).set_base() = rsint(d).get_base();
192  }
193  // lap_Bterm.set_base() = rsint.get_base();
194  // lap_wterm.set_base() = rsint.get_base();
195 
196  // output data
197  bconfig.set_filename(outputfile);
198  bco_utils::save_to_file(space, bconfig, lap_Aterm, nu, logh, lap_Bterm,
199  lap_wterm);
200 
201  return exit_status;
202 }
