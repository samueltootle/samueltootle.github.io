128 template<typename dom_t>
129 double get_radius (const dom_t* dom, const int bound) {
130  auto const npts = dom->get_nbr_points();
131  Index pos(npts);
132  switch(bound) {
133  case INNER_BC:
134  break;
135  case OUTER_BC:
136  pos.set(0) = npts(0) - 1;
137  pos.set(1) = npts(1) - 1;
138  pos.set(2) = npts(2) - 1;
139  break;
140  case EQUI:
141  pos.set(0) = npts(0) - 1;
142  pos.set(2) = npts(2) - 1;
143  break;
144  case INNER_EQUI:
145  pos.set(2) = npts(2) - 1;
146  break;
147  default:
148  std::cout << "Unknown bound sent to get_radius: " << bound << std::endl;
149  break;
150  }
151  return dom->get_radius()(pos);
152 }

164 inline std::array<double,2> get_field_min_max(const Scalar& field, const int dom, const int bound = OUTER_BC) {
165  const int npts_r = field.get_domain(dom)->get_nbr_points()(0);
166 
167  // position index to loop over
168  Index pos(field.get_domain(dom)->get_nbr_points());
169  // position index to update with a fixed radial boundary
170  Index bpos(field.get_domain(dom)->get_nbr_points());
171 
172  int r_bound = 0;
173  switch(bound) {
174  case INNER_BC:
175  break;
176  case OUTER_BC:
177  r_bound = npts_r - 1;
178  break;
179  default:
180  std::cout << "Unknown bound sent to get_field_min_max: " << bound << std::endl;
181  break;
182  }
183 
184  double fmax = field(dom)(pos);
185  double fmin = field(dom)(pos);
186 
187  // FIXME this currently loops over theta and phi multiple times, but the syntax is straightforward.
188  do {
189  bpos.set(0) = r_bound;
190  bpos.set(1) = pos(1);
191  bpos.set(2) = pos(2);
192  double f = field(dom)(bpos);
193 
194  if(f > fmax)
195  fmax = f;
196  if(f < fmin)
197  fmin = f;
198  } while(pos.inc());
199  return std::array<double,2>{fmin,fmax};
200 }

274 template<typename adapted_t>
275 void interp_adapted_mapping(const adapted_t* new_shell, const int old_outer_adapted_dom, const Scalar& old_radius_field) {
276  Val_domain new_mapping = new_shell->get_radius();
277 
278  //Need to normalize by the constant radius at the INNER_BC of the outer_adapted shell in the old space
279  auto old_shell = old_radius_field.get_space().get_domain(old_outer_adapted_dom);
280  double rinner = get_radius(old_shell, INNER_BC);
281 
282  Index new_pos(new_shell->get_nbr_points());
283  double xc_new = new_shell->get_center()(1);
284  double xc_old = old_shell->get_center()(1);
285  do {
286  double x = new_shell->get_cart(1)(new_pos) - xc_new;
287  double y = new_shell->get_cart(2)(new_pos);
288  double z = new_shell->get_cart(3)(new_pos);
289 
290  double r = std::sqrt(x*x + y*y + z*z);
291 
292  x /= r / rinner;
293  y /= r / rinner;
294  z /= r / rinner;
295 
296  Kadath::Point absol(3);
297  absol.set(1) = x + xc_old;
298  absol.set(2) = y;
299  absol.set(3) = z;
300 
301  new_mapping.set(new_pos) = old_radius_field.val_point(absol);
302 
303  } while(new_pos.inc());
304 
305  new_mapping.std_base();
306  new_shell->set_mapping(new_mapping);
307 }

384 template<typename ary_t, typename config_t, typename... idx_t>
385 void set_NS_bounds (ary_t& bounds, config_t& bconfig, idx_t... bco) {
386  //sizes
387  const int size = bounds.size();
388  const int ninshells = (!std::isnan(bconfig.set(NINSHELLS,bco...))) ? bconfig(NINSHELLS, bco...) : 0;
389  const int nshells = (!std::isnan(bconfig.set(NSHELLS,bco...))) ? bconfig(NSHELLS, bco...) : 0;
390 
391  //indexes
392  const int rin = 0;
393  const int rout = size - 1;
394  const int r = rin + ninshells + 1;
395 
396  bounds[rin] = bconfig(RIN, bco...);
397  bounds[r] = bconfig(RMID, bco...);
398  bounds[rout] = bconfig(ROUT, bco...);
399 
400  double lower = bounds[rin] + (bounds[r] - bounds[rin]) * 0.8;
401  double delta_r = (bounds[r] - lower) / (ninshells + 1.);
402 
403  for(int shell = rin+1; shell <= ninshells; ++shell){
404  bounds[shell] = delta_r * shell + lower;
405  }
406 
407  delta_r = (bounds[rout] - bounds[r]) / (nshells + 1.);
408  for(int i = 1, shell = r+1; i <= nshells; ++shell, ++i){
409  bounds[shell] = bounds[r] + delta_r * i;
410  }
411 }

459 template<typename ary_t, typename config_t>
460 void set_BH_bounds (ary_t& bounds, config_t& bconfig, const int bco, const bool adapt_shells = false) {
461 
462  // Configurator index of companion compact object
463  const int bco2 = (bco == BCO1) ? BCO2 : BCO1;
464  const int size = bounds.size();
465 
466  //indexes
467  const int rin = 0;
468  const int r = rin + 1;
469  int rout = size - 1;
470 
471  bounds[rin] = bconfig(RIN, bco);
472  bounds[r] = bconfig(RMID, bco);
473  bounds[rout] = bconfig(ROUT, bco);
474 
475  const int shells = bconfig(NSHELLS, bco);
476  double sum = 0.;
477  ary_t new_ary;
478  new_ary.push_back(bounds[rin]);
479  new_ary.push_back(bounds[r]);
480 
481  // fixme - need to recall how this limit was obtained
482  const double limit = bconfig(ROUT, bco) * invpsisq;
483 
484  // needed to determine proper shall spacing to not have problems in kadath import?
485  const double scale_fact = bco_utils::gold_ratio;
486 
487  // y_intercept of new_bound == the location of the first shell
488  const double y_intercept = 2. * scale_fact * bounds[r];
489 
490  // the intercept_fac ensures the y_intercept is just that
491  const double intercept_fac = - scale_fact * std::log(y_intercept);
492 
493  // relation that provides shell locations
494  auto new_bound =[&](int n) { return std::exp((n - intercept_fac) / scale_fact); };
495 
496  // define new shells bounds
497  int N = 0;
498  double next_bound = new_bound(N);
499  while(next_bound < limit && next_bound < bconfig(ROUT, bco)) {
500  new_ary.push_back(next_bound);
501  N++;
502  next_bound = new_bound(N);
503  }
504  // in case NSHELLS are put in by hand - we add additional ones based on a
505  // naive even distribution
506  if(new_ary.size() - 2 < bconfig(NSHELLS, bco)) {
507  // determine how many shells have already been defined
508  auto current_shells = new_ary.size() - 2;
509  // how many shells still need to defined
510  auto remaining_shells = shells - current_shells;
511 
512  // radius of the last defined shell
513  auto last_shell_r = new_ary.back();
514  // radial distance between last shell and ROUT
515  double delrBH = bconfig(ROUT, bco) - last_shell_r;
516  // determine the equal spacing between reminaing shells
517  double delr = delrBH / (remaining_shells + 1);
518 
519  for(int i = 0; i < remaining_shells; ++i)
520  new_ary.push_back(last_shell_r + delr * (i + 1));
521  }
522  // define the bound corresponding to ROUT
523  new_ary.push_back(bounds[rout]);
524  // update the number of shells in the event shells have been automatically
525  // added
526  bconfig(NSHELLS, bco) = new_ary.size() - 3;
527  bounds = std::move(new_ary);
528 }

544 template<class T>
545 T gen_shell_bound_radius(Scalar& field, T r0, T r1,
546  T xc = 0., T threshold = 0.9, T fac = 1.)
547 {
548  auto reldiff = [](auto ref, auto cmp) {
549  return 1. - cmp / ref;
550  };
551 
552  // Setup points relative to coordinate center
553  Kadath::Point pt_r0(3);
554  pt_r0.set(1) = xc + r0;
555 
556  Kadath::Point pt_r1(3);
557  pt_r1.set(1) = xc + r1;
558  // end point setup
559 
560  // relative difference of ddrP at both radii
561  auto ddrP_rel_diff = std::abs(
562  reldiff(
563  field.val_point(pt_r0),
564  field.val_point(pt_r1)
565  )
566  );
567 
568  // relative difference to the desired threshold
569  auto relth = std::abs(
570  reldiff(
571  threshold,
572  ddrP_rel_diff
573  )
574  );
575 
576  #ifdef DEBUG
577  std::cout << "r0: " << r0
578  << "\t ddrP0: " << field.val_point(pt_r0)
579  << "\t P: " << pt_r0 << '\n'
580  << "r1: " << r1
581  << "\t ddrP1: " << field.val_point(pt_r1) << "\t ddrP_rel_diff: " << ddrP_rel_diff
582  << "\t P: " << pt_r1 << '\n'
583  << "\t relth: " << relth << "\t r1 / r0: " << r1/r0 << '\n';
584  std::cout << pt_r0 << "," << pt_r1 << '\n';
585  #endif
586 
587  auto dr = r1 - r0;
588 
589  if(ddrP_rel_diff > threshold) {
590  auto new_fac = 0.5 * fac;
591  r1 = r0 + dr * new_fac;
592  r1 = gen_shell_bound_radius(
593  field,
594  r0,
595  r1,
596  xc,
597  threshold,
598  new_fac
599  );
600  } else {
601  // If the radii are far enough apart we found a good radius
602  if(r1 / r0 > bco_utils::gold_ratio)
603  return r1;
604  else {
605  auto new_fac = fac + fac * 0.1;
606  r1 = r0 + (1. / fac) * dr * new_fac;
607  r1 = gen_shell_bound_radius(
608  field,
609  r0,
610  r1,
611  xc,
612  threshold,
613  new_fac
614  );
615  }
616  }
617  return r1;
618 }

634 template<typename config_t>
635 std::vector<double> set_arb_bounds (config_t& bconfig, const int bco,
636  Scalar& ddrconf_sol, const int adapted_dom_sol,
637  const double threshold = 0.95) {
638 
639  // Generate a new vector for bounds
640  // FIXME: this ignores shells inside of NS'
641  std::vector<double> bounds;
642  bounds.push_back(bconfig(RIN, bco));
643  bounds.push_back(bconfig(RMID, bco));
644 
645  auto adapted_dom(ddrconf_sol.get_space().get_domain(adapted_dom_sol));
646 
647  // Coordinate centered point
648  Kadath::Point pt(adapted_dom->get_center());
649  double xc{pt(1)};
650 
651  // Initial radii starting at the inner adapted radius on the
652  // equitorial plane since this should be the largest radius
653  auto r0{get_radius(adapted_dom, INNER_EQUI)};
654  // We only add shells out to ROUT
655  auto r1{bconfig(ROUT,bco)};
656  auto Rout{r1};
657 
658  // some upper bound that should never be hit!
659  auto max_shells = std::ceil((Rout - r0)/(3.* r0));
660  for(auto i = 0; i < max_shells; ++i) {
661 
662  pt.set(1) = xc + r0;
663 
664  // stop looking for shells once \partial_r^2 \Psi
665  // is roughly flat
666  // FIXME:? this will fail at local extrema!
667  if (ddrconf_sol.val_point(pt) < 0.01)
668  break;
669 
670  // get new shell boundary outer radius
671  r1 = gen_shell_bound_radius(ddrconf_sol, r0, r1, xc, threshold);
672 
673  // check for minimum spacing between last shell boundary
674  // and Rout
675  if (Rout / r1 > bco_utils::gold_ratio) {
676  #ifdef DEBUG
677  cout << "Rout/r1: " << Rout / r1 << ", Success.\n";
678  #endif
679  bounds.push_back(r1);
680  }
681  else {
682  #ifdef DEBUG
683  cout << "Rout/r1: " << Rout / r1 << ", failed.\n";
684  #endif
685  break;
686  }
687 
688  // update variables for finding next shell
689  r0 = r1;
690  r1 = Rout;
691  if(i == max_shells - 1){
692  std::cerr << "Max iterations hit in setting CO bounds. Something went very wrong!\n";
693  std::cerr << "r0: " << r0 << '\n'
694  << "r1: " << r1 << '\n'
695  << "max_shells: " << (Rout - r0)/(3.* r0) << '\n';
696  std::_Exit(EXIT_FAILURE);
697  }
698  }
699  // Add ROUT as the last bound
700  bounds.push_back(Rout);
701  bconfig(NSHELLS, bco) = bounds.size() - 3;
702  return bounds;
703 }
