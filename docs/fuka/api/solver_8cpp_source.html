24 Array<double> System_of_eqs::check_equations() {
25 
26 
27  Array<double> sec (sec_member()) ;
28  Array<double> errors (neq_int + neq) ;
29  errors = 0 ;
30 
31  int pos = 0 ;
32  for (int i=0 ; i<neq_int ; i++) {
33  errors.set(i) = fabs(sec(pos)) ;
34  pos ++ ;
35  }
36 
37  for (int i=0 ; i<neq ; i++) {
38  double max = 0 ;
39  for (int j=0 ; j<eq[i]->get_n_cond_tot() ; j++) {
40  if (fabs(sec(pos)) > max)
41  max = fabs(sec(pos)) ;
42  pos ++ ;
43  }
44  errors.set(neq_int+i) = max ;
45  }
46  return errors ;
47 }

49 Array<double> System_of_eqs::sec_member() {
50 
51  vars_to_terms() ;
52 
53  if (met!=0x0)
54  for (int d=dom_min ; d<=dom_max ; d++)
55  met->update(d) ;
56  for (int i=0 ; i<ndef ; i++)
57  def[i]->compute_res() ;
58 
59  int conte = 0 ;
60  for (int i=0 ; i<neq ; i++)
61  eq[i]->apply(conte, results) ;
62 
63  // Need to assert the size :
64  if (nbr_conditions==-1) {
65  nbr_conditions = 0 ;
66  for (int i=0 ; i<neq_int ; i++)
67  nbr_conditions ++ ;
68  for (int i=0 ; i<neq ; i++)
69  nbr_conditions += eq[i]->get_n_cond_tot() ;
70  }
71 
72  // Computation of the second member itself :
73  Array<double> res (nbr_conditions) ;
74  res = 0 ;
75  conte = 0 ;
76  int pos_res = 0 ;
77  for (int i=0 ; i<neq_int ; i++) {
78  res.set(pos_res) = eq_int[i]->get_val() ;
79  pos_res ++ ;
80  }
81 
82  for (int i=0 ; i<neq ; i++)
83  eq[i]->export_val(conte, results, res, pos_res) ;
84  return res ;

120 Array<double> System_of_eqs::do_col_J (int cc) {
121 
122  assert ((cc>=0) && (cc<nbr_unknowns)) ;
123 
124  // Affecte nterms derivatives :
125  int conte = 0 ;
126  int zedom = -1 ;
127  bool is_var_double = false ;
128  Array<int> zedoms (2) ;
129  zedoms = - 1 ;
130  // Variable Domains :
131  espace.affecte_coef_to_variable_domains(conte, cc, zedoms) ;
132  if (zedoms(0)!=-1)
133  update_terms_from_variable_domains(zedoms) ;
134  else {
135  for (int i=0 ; i<nterm_cst ; i++)
136  cst[i]->set_der_zero() ;
137  for (int i=0 ; i<nterm ; i++)
138  term[i]->set_der_zero() ;
139  }
140 
141  // Double
142  for (int i=0 ; i<nvar_double ; i++) {
143  if (conte==cc) {
144  for (int dd=dom_min ; dd<=dom_max ; dd++)
145  term_double[i*ndom+(dd-dom_min)]->set_der_d(1.) ;
146  is_var_double = true ;
147  }
148  else
149  for (int dd=dom_min ; dd<=dom_max ; dd++)
150  term_double[i*ndom+(dd-dom_min)]->set_der_d(0.) ;
151  conte ++ ;
152  }
153 
154  // Fields
155  for (int i=0 ; i<nterm ; i++) {
156  int dom = term[i]->get_dom() ;
157  Tensor auxi (term[i]->get_val_t(), false) ;
158  espace.get_domain(dom)->affecte_tau_one_coef(auxi, dom, cc, conte) ;
159  for (int j=0 ; j<auxi.get_n_comp() ; j++) {
160  // Si la base n'est pas affectee on la met
161  if ((!auxi(auxi.indices(j))(dom).check_if_zero()) && (!auxi(auxi.indices(j))(dom).get_base().is_def()))
162  auxi.set(auxi.indices(j)).set_domain(dom).set_base() = term[i]->get_val_t()(auxi.indices(j))(dom).get_base() ;
163  if ((zedom==-1) && (!auxi(auxi.indices(j))(dom).check_if_zero()))
164  zedom = dom ;
165  }
166 
167  term[i]->set_der_t(auxi + term[i]->get_der_t()) ;
168  }
169 
170  // Delete the metric derivative terms :
171  if (met!=0x0)
172  for (int d=dom_min ; d<=dom_max ; d++)
173  met->update(d) ;
174 
175  // Delete the definitions
176  for (int i=0 ; i<ndef ; i++)
177  def[i]->compute_res() ;
178 
179  conte = 0 ;
180  for (int i=0 ; i<neq ; i++) {
181  if ((is_var_double) || (eq[i]->take_into_account(zedom)) || (eq[i]->take_into_account(zedoms(0))) || (eq[i]->take_into_account(zedoms(1)))) {
182  eq[i]->apply(conte, results) ;
183  }
184  else
185  conte += eq[i]->n_ope ;
186  }
187 
188  if (nbr_conditions==-1) {
189  cerr << "Number of conditions unknown ; call sec_member first" << endl ;
190  abort() ;
191  }
192 
193  Array<double> res (nbr_conditions) ;
194  conte = 0 ;
195  int pos_res = 0 ;
196  res = 0 ;
197  for (int i=0 ; i<neq_int ; i++) {
198  res.set(pos_res) = eq_int[i]->get_der() ;
199  pos_res ++ ;
200  }
201 
202  for (int i=0 ; i<neq ; i++) {
203  if ((is_var_double) || (eq[i]->take_into_account(zedom))|| (eq[i]->take_into_account(zedoms(0))) || (eq[i]->take_into_account(zedoms(1)))) {
204  eq[i]->export_der(conte, results, res, pos_res) ;
205  }
206  else
207  {
208  pos_res += eq[i]->n_cond_tot ;
209  conte += eq[i]->n_ope ;
210  }
211  }
212 
213  return res ;
214 }

216 void System_of_eqs::update_terms_from_variable_domains(const Array<int>& zedoms) {
217 
218  for (int i=0 ; i<nterm ; i++) {
219  int dom = term[i]->get_dom() ;
220  bool todo = false ;
221  for (int d=0 ; d<zedoms.get_size(0) ; d++)
222  if (zedoms(d)==dom)
223  todo = true ;
224  if (todo)
225  espace.get_domain(dom)->update_term_eq (term[i]) ;
226  else
227  term[i]->set_der_zero() ;
228  }
229 
230  for (int i=0 ; i<nterm_cst ; i++)
231  if (cst[i]->get_type_data() == TERM_T) {
232  int dom = cst[i]->get_dom() ;
233  bool todo = false ;
234  for (int d=0 ; d<zedoms.get_size(0) ; d++)
235  if (zedoms(d)==dom)
236  todo = true ;
237  if (todo)
238  espace.get_domain(dom)->update_term_eq (cst[i]) ;
239  else
240  cst[i]->set_der_zero() ;
241  }
