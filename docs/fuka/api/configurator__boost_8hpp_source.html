279  std::enable_if_t<std::is_arithmetic_v<T>, bool> = true>
280  constexpr const T eos(idx_t... idx) {
281  T var{};
282  auto set_var = [&](auto&& v) mutable {
283  using v_t = std::decay_t<decltype(v)>;
284  if constexpr (std::is_arithmetic_v<v_t>) {
285  if (!std::isnan(v))

288  };
289  std::visit(set_var, this->set_eos(idx...));
290  return var;
