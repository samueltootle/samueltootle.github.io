37 double Domain_shell::multipoles_sym (int k, int j, int bound, const Val_domain& so, const Array<double>& passage) const {
38 
39 
40  assert ((bound==INNER_BC) || (bound==OUTER_BC)) ;
41 
42  Index pos(passage.dimensions) ;
43  Index pos_bound (nbr_coefs) ;
44 
45  double alm = 0 ;
46 
47  int mm = (k%2==0) ? int(k/2) : int((k-1)/2) ;
48  pos.set(0) = mm ;
49  pos_bound.set(2) = k ;
50  if (mm%2==0) {
51  //Case m even
52  pos.set(1) = j ;
53  for (int i=0 ; i<nbr_coefs(1) ; i++) {
54  pos.set(2) = i ;
55  pos_bound.set(1) = i ;
56  alm += passage(pos)*val_boundary(bound, so, pos_bound) ;
57  }
58  }
59  else {
60  //Case m odd
61  pos.set(1) = j ;
62  for (int i=0 ; i<nbr_coefs(1)-1 ; i++) {
63  pos.set(2) = i ;
64  pos_bound.set(1) = i ;
65  alm += passage(pos)*val_boundary(bound, so, pos_bound) ;
66  }
67  }
68  return alm ;
69 }
