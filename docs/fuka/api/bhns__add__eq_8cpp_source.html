42 void Space_bhns::add_eq (System_of_eqs& sys, const char* eq, const char* rac, const char* rac_der, int nused, Array<int>** pused) {
43  // NS
44  sys.add_eq_inside (NS, eq, nused, pused) ;
45  sys.add_eq_matching (NS, OUTER_BC, rac, nused, pused) ;
46  sys.add_eq_matching (NS, OUTER_BC, rac_der, nused, pused) ;
47 
48  for(int i = 0; i < n_inner_shells1; ++i) {
49  sys.add_eq_inside (NS+1+i, eq, nused, pused) ;
50  sys.add_eq_matching (NS+1+i, OUTER_BC, rac, nused, pused) ;
51  sys.add_eq_matching (NS+1+i, OUTER_BC, rac_der, nused, pused) ;
52  }
53 
54  sys.add_eq_inside (ADAPTEDNS , eq, nused, pused) ;
55  sys.add_eq_matching (ADAPTEDNS , OUTER_BC, rac, nused, pused) ;
56  sys.add_eq_matching (ADAPTEDNS , OUTER_BC, rac_der, nused, pused) ;
57  sys.add_eq_inside (ADAPTEDNS+1, eq, nused, pused) ;
58 
59  for(int i = 0; i < n_shells1; ++i) {
60  sys.add_eq_matching (ADAPTEDNS+1+i, OUTER_BC, rac, nused, pused) ;
61  sys.add_eq_matching (ADAPTEDNS+1+i, OUTER_BC, rac_der, nused, pused) ;
62  sys.add_eq_inside (ADAPTEDNS+2+i, eq, nused, pused) ;
63  }
64 
65  // Matching with bispheric :
66  sys.add_eq_matching_import (ADAPTEDNS + 1 + n_shells1, OUTER_BC, rac, nused, pused) ;
67  sys.add_eq_matching_import (OUTER , INNER_BC, rac_der, nused, pused) ;
68  sys.add_eq_matching_import (OUTER+1 , INNER_BC, rac_der, nused, pused) ;
69 
70  // BH :
71  sys.add_eq_inside (ADAPTEDBH+1, eq, nused, pused) ;
72  for(int i = 0; i < n_shells2; ++i) {
73  sys.add_eq_matching (ADAPTEDBH+1+i, OUTER_BC, rac, nused, pused) ;
74  sys.add_eq_matching (ADAPTEDBH+1+i, OUTER_BC, rac_der, nused, pused) ;
75  sys.add_eq_inside (ADAPTEDBH+2+i, eq, nused, pused) ;
76  }
77 
78  // Matching with bispheric :
79  sys.add_eq_matching_import (ADAPTEDBH + 1 + n_shells2, OUTER_BC, rac, nused, pused) ;
80  sys.add_eq_matching_import (OUTER+3, INNER_BC, rac_der, nused, pused) ;
81  sys.add_eq_matching_import (OUTER+4, INNER_BC, rac_der, nused, pused) ;
82 
83  // Chi first
84  sys.add_eq_inside (OUTER, eq, nused, pused) ;
85  sys.add_eq_matching (OUTER, CHI_ONE_BC, rac, nused, pused) ;
86  sys.add_eq_matching (OUTER, CHI_ONE_BC, rac_der, nused, pused) ;
87 
88  // Rect :
89  sys.add_eq_inside (OUTER+1, eq, nused, pused) ;
90  sys.add_eq_matching (OUTER+1, ETA_PLUS_BC, rac, nused, pused) ;
91  sys.add_eq_matching (OUTER+1, ETA_PLUS_BC, rac_der, nused, pused) ;
92 
93  // Eta first
94  sys.add_eq_inside (OUTER+2, eq, nused, pused) ;
95  sys.add_eq_matching (OUTER+2, ETA_PLUS_BC, rac, nused, pused) ;
96  sys.add_eq_matching (OUTER+2, ETA_PLUS_BC, rac_der, nused, pused) ;
97 
98  // Rect
99  sys.add_eq_inside (OUTER+3, eq, nused, pused) ;
100  sys.add_eq_matching (OUTER+3, CHI_ONE_BC, rac, nused, pused) ;
101  sys.add_eq_matching (OUTER+3, CHI_ONE_BC, rac_der, nused, pused) ;
102 
103  // chi first :
104  sys.add_eq_inside (OUTER+4, eq, nused, pused) ;
105 
106  // Matching outer domain :
107  for (int d=OUTER ; d<=OUTER+4 ; d++)
108  sys.add_eq_matching_import (d, OUTER_BC, rac, nused, pused) ;
109 
110  //Matching for first shell or compactified domain
111  sys.add_eq_matching_import (OUTER+5, INNER_BC, rac_der, nused, pused) ;
112 
113  // Optional spherical shells between bi-spherical and compactified domain
114  for (int d=0 ; d<n_shells_outer ; d++) {
115  sys.add_eq_inside (OUTER+5+d, eq, nused, pused) ;
116  sys.add_eq_matching (OUTER+5+d, OUTER_BC, rac, nused, pused) ;
117  sys.add_eq_matching (OUTER+5+d, OUTER_BC, rac_der, nused, pused) ;
118  }
119 
120  //Compactified domain
121  sys.add_eq_inside (OUTER+5+n_shells_outer, eq, nused, pused) ;
122 
123 }

125 void Space_bhns::add_eq_int_inf (System_of_eqs& sys, const char* nom) {
126  sys.eq_int_list.push_back(std::make_tuple(nom, nbr_domains-1, OUTER_BC));
127 
128  // Check the last domain is of the right type :
129  const Domain_compact* pcomp = dynamic_cast <const Domain_compact*> (domains[nbr_domains-1]) ;
130  if (pcomp==0x0) {
131  cerr << "add_eq_int_inf requires a compactified domain" << endl ;
132  abort() ;
133  }
134  int dom = nbr_domains-1 ;
135 
136  // Get the lhs and rhs
137  char p1[LMAX] ;
138  char p2[LMAX] ;
139  bool indic = sys.is_ope_bin(nom, p1, p2, '=') ;
140  if (!indic) {
141  cerr << "= needed for equations" << endl ;
142  abort() ;
143  }
144  else {
145  // Verif lhs = 0 ?
146  indic = ((p2[0]=='0') && (p2[1]==' ') && (p2[2]=='\0')) ?
147  true : false ;
148 
149  // Construction of the equation
150  sys.eq_int[sys.neq_int] = new Eq_int(1) ;
151 
152  // Affectation :
153  // no lhs :
154  if (indic)
155  sys.eq_int[sys.neq_int]->set_part(0, sys.give_ope(dom, p1, OUTER_BC)) ;
156 
157  else
158  sys.eq_int[sys.neq_int]->set_part(0, new Ope_sub(&sys, sys.give_ope(dom, p1, OUTER_BC), sys.give_ope(dom, p2, OUTER_BC))) ;
159  sys.neq_int ++ ;
160  }
161  sys.nbr_conditions = -1 ;
162 }

164 void Space_bhns::add_eq_int_volume (System_of_eqs& sys, int dmin, int dmax, const char* nom) {
165  sys.eq_int_list.push_back(std::make_tuple(nom, dmin, -1));
166 
167  // Get the lhs and rhs
168  char p1[LMAX] ;
169  char p2[LMAX] ;
170 
171  bool indic = sys.is_ope_bin(nom, p1, p2, '=') ;
172  if (!indic) {
173  cerr << "= needed for equations" << endl ;
174  abort() ;
175  }
176  else {
177  // Construction of the equation
178  int nz = dmax - dmin + 1;
179  sys.eq_int[sys.neq_int] = new Eq_int(nz+1) ;
180 
181  // Affectation of the intregrale parts
182  for (int d=0 ; d<nz ; d++)
183  sys.eq_int[sys.neq_int]->set_part(d, sys.give_ope(d+dmin, p1)) ;
184  // Affectation of the second member (constant value)
185  sys.eq_int[sys.neq_int]->set_part(nz, new Ope_minus(&sys, sys.give_ope(dmin, p2))) ;
186  sys.neq_int ++ ;
187  }
188  sys.nbr_conditions = -1 ;
189 }
