61 void Eq_matching_non_std::export_val(int& conte, Term_eq** residus, Array<double>& sec, int& pos_res) const {
62  for (int i=0 ; i<n_ope ; i++)
63  assert (residus[i]->get_type_data()==TERM_T) ;
64  assert (residus[conte+1]->get_type_data()==TERM_T) ;
65 
66  Tensor copie (residus[conte]->get_val_t()) ;
67 
68  // Loop on the components :
69  int start = pos_res ;
70  // Case for all the cmps :
71  if (n_cmp_used==-1) {
72  for (int comp=0 ; comp<n_comp ; comp++) {
73  Array<int> ind (copie.indices(comp)) ;
74  copie.set(ind).set_domain(ndom).coef_i() ; //Configuration space
75 
76  for (int j=0 ; j<(*n_cond)(comp) ; j++) {
77  // Value at the colocation point
78  sec.set(pos_res) = (copie(ind)(ndom).check_if_zero()) ? 0. : (*copie(ind)(ndom).c)(*which_points[pos_res-start]) ;
79 
80  // Get the absolute coordinates :
81  Point absol (dom->get_ndim()) ;
82  for (int i=1 ; i<=dom->get_ndim() ; i++)
83  absol.set(i) = (*dom->get_absol(i).c)(*which_points[pos_res-start]) ;
84 
85  bool loop = true ;
86  // Loop on the other domains :
87  int num_other = 0 ;
88  bool indic= false ;
89  while (loop) {
90  const Domain* zedom = residus[conte+num_other+1]->get_val_t().espace.get_domain(other_doms(num_other)) ;
91  int ozerbound = other_bounds(num_other) ;
92  bool found = zedom->is_in(absol, 1e-5) ;
93  if (found) {
94  Point num(zedom->absol_to_num_bound(absol, ozerbound)) ;
95  Val_domain other_copie
96  (residus[conte+num_other+1]->get_val_t()(ind)(other_doms(num_other))) ;
97  if (!other_copie.check_if_zero()) {
98  other_copie.coef() ;
99  sec.set(pos_res) -= other_copie.base.summation(num, *other_copie.cf) ;
100 
101  }
102  loop = false ;
103  indic = true ;
104  }
105  num_other ++ ;
106  if (num_other>=n_ope-1)
107  loop = false ;
108  }
109 
110  if (!indic) {
111  cerr << absol << endl ;
112  cerr << "not found in Eq_matching_non_std::export_val" << endl ;
113  abort() ;
114  }
115 
116  pos_res ++ ;
117  }
118  }
119  }
120  else {
121  for (int comp=0 ; comp<n_cmp_used ; comp++) {
122  copie.set(*p_cmp_used[comp]).set_domain(ndom).coef_i() ; //Configuration space
123 
124  for (int j=0 ; j<(*n_cond)(comp) ; j++) {
125  // Value at the colocation point
126  sec.set(pos_res) = (copie(*p_cmp_used[comp])(ndom).check_if_zero()) ? 0. :
127  (*copie(*p_cmp_used[comp])(ndom).c)(*which_points[pos_res-start]) ;
128 
129  // Get the absolute coordinates :
130  Point absol (dom->get_ndim()) ;
131  for (int i=1 ; i<=dom->get_ndim() ; i++)
132  absol.set(i) = (*dom->get_absol(i).c)(*which_points[pos_res-start]) ;
133 
134  bool loop = true ;
135  // Loop on the other domains :
136  int num_other = 0 ;
137  bool indic = false ;
138  while (loop) {
139  const Domain* zedom = residus[conte+num_other+1]->get_val_t().espace.get_domain(other_doms(num_other)) ;
140  int ozerbound = other_bounds(num_other) ;
141  bool found = zedom->is_in(absol, 1e-5) ;
142  if (found) {
143  Point num(zedom->absol_to_num_bound(absol, ozerbound)) ;
144  Val_domain other_copie
145  (residus[conte+num_other+1]->get_val_t()(*p_cmp_used[comp])(other_doms(num_other))) ;
146  if (!other_copie.check_if_zero()) {
147  other_copie.coef() ;
148  sec.set(pos_res) -= other_copie.base.summation(num, *other_copie.cf) ;
149  }
150  loop = false ;
151  indic= true ;

153  num_other ++ ;

160  abort() ;
161  }
162 
163  pos_res ++ ;
164  }
165  }
166  }
167  conte +=n_ope ;

288 void Eq_matching_non_std::apply(int& conte, Term_eq** res) {
289 
290  int old_conte = conte ;
291  for (int i=0 ; i<n_ope ; i++) {
292  if (res[conte]!=0x0)
293  *res[conte] = parts[i]->action() ;
294  else
295  res[conte] = new Term_eq (parts[i]->action()) ;
296 
297  assert (res[conte]->get_type_data()==TERM_T) ;
298  conte ++ ;
299  }
300 
301  if (!called) {
302  Tensor copie (res[old_conte]->get_val_t()) ;
303  n_cond = new Array<int> (do_nbr_conditions(copie)) ;
304  n_comp = n_cond->get_size(0) ;
305  n_cond_tot = 0 ;
306  if (n_cmp_used==-1)
307  for (int i=0 ; i<n_cond->get_size(0) ; i++)
308  n_cond_tot += (*n_cond)(i) ;
309  else
310  for (int i=0 ; i<n_cmp_used ; i++)
311  n_cond_tot += (*n_cond)(i) ;
312 
313  which_points = new Index* [n_cond_tot] ;
314  int start = 0 ;
315  if (n_cmp_used ==-1)
316  for (int comp=0 ; comp<n_comp ; comp++) {
317  Array<int> ind (copie.indices(comp)) ;
318  Base_spectral bb (copie(ind)(ndom).get_base()) ;
319  do_which_points (bb, start) ;
320  start += (*n_cond)(comp) ;
321  }
322  else
323  for (int comp=0 ; comp<n_cmp_used ; comp++) {
324  Base_spectral bb (copie(*p_cmp_used[comp])(ndom).get_base()) ;
325  do_which_points (bb, start) ;
326  start += (*n_cond)(comp) ;
327  }
328  called = true ;
329  }
