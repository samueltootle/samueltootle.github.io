34 double func_ab2 (double aa, const Param& par) {
35  double r1 = par.get_double(0) ;
36  double r2 = par.get_double(1) ;
37  double d = par.get_double(2) ;
38  return (sqrt(aa*aa+r1*r1)+sqrt(aa*aa+r2*r2)-d) ;
39 }

43 Space_bbh::Space_bbh (int ttype, double dist, double rbh1, double rbh2, double rbi, double rext, int nr) {
44 
45  ndim = 3 ;
46 
47  nbr_domains = 10 ;
48  type_base = ttype ;
49  domains = new Domain* [nbr_domains] ;
50 
51  Dim_array res (ndim) ;
52  res.set(0) = nr ; res.set(1) = nr ; res.set(2) = nr-1 ;
53  Dim_array res_bi(ndim) ;
54  res_bi.set(0) = nr ; res_bi.set(1) = nr ; res_bi.set(2) = nr ;
55 
56  // Bispheric :
57  // Computation of aa
58  Param par_a ;
59  par_a.add_double(2*rbh1,0) ;
60  par_a.add_double(2*rbh2,1) ;
61  par_a.add_double(dist,2) ;
62  double a_min = 0 ;
63  double a_max = dist/2. ;
64  double precis = PRECISION ;
65  int nitermax = 500 ;
66  int niter ;
67  double aa = zerosec(func_ab2, par_a, a_min, a_max, precis, nitermax, niter) ;
68  double eta_plus = asinh(aa/2/rbh2) ;
69  double eta_minus = -asinh(aa/2/rbh1) ;
70 
71  double chi_c = 2*atan(aa/rbi) ;
72  double eta_c = log((1+rbi/aa)/(rbi/aa-1)) ;
73  double eta_lim = eta_c/2. ;
74  double chi_lim = chi_lim_eta (eta_lim, rbi, aa, chi_c) ;
75 
76  // First BH
77  Point center_minus (ndim) ;
78  center_minus.set(1) = aa*cosh(eta_minus)/sinh(eta_minus) ;
79  domains[0] = new Domain_shell_outer_adapted (*this, 0, ttype, rbh1/2., rbh1, center_minus, res) ;
80  domains[1] = new Domain_shell_inner_adapted (*this, 1, ttype, rbh1, rbh1*2., center_minus, res) ;
81 
82  // Second bh
83  Point center_plus (ndim) ;
84  center_plus.set(1) = aa*cosh(eta_plus)/sinh(eta_plus) ;
85  domains[2] = new Domain_shell_outer_adapted (*this, 2, ttype, rbh2/2., rbh2, center_plus, res) ;
86  domains[3] = new Domain_shell_inner_adapted (*this, 3, ttype, rbh2, rbh2*2., center_plus, res) ;
87 
88  // Bispheric part
89  domains[4] = new Domain_bispheric_chi_first(4, ttype, aa, eta_minus, rbi, chi_lim, res_bi) ;
90  domains[5] = new Domain_bispheric_rect(5, ttype, aa, rbi, eta_minus, -eta_lim, chi_lim, res_bi) ;
91  domains[6] = new Domain_bispheric_eta_first(6, ttype, aa, rbi, -eta_lim, eta_lim, res_bi) ;
92  domains[7] = new Domain_bispheric_rect(7, ttype, aa, rbi, eta_plus, eta_lim, chi_lim, res_bi) ;
93  domains[8] = new Domain_bispheric_chi_first(8, ttype, aa, eta_plus, rbi, chi_lim, res_bi) ;
94 
95  // Compactified domain
96  Point center(3) ;
97  domains[9] = new Domain_shell (9, ttype, rbi, rext, center, res) ;
98 
99  const Domain_shell_outer_adapted* pouter_1 = dynamic_cast<const Domain_shell_outer_adapted*> (domains[0]) ;
100  pouter_1->vars_to_terms() ;
101  pouter_1->update() ;
102  const Domain_shell_inner_adapted* pinner_1 = dynamic_cast<const Domain_shell_inner_adapted*> (domains[1]) ;
103  pinner_1->vars_to_terms() ;
104  pinner_1->update() ;
105  const Domain_shell_outer_adapted* pouter_2 = dynamic_cast<const Domain_shell_outer_adapted*> (domains[2]) ;
106  pouter_2->vars_to_terms() ;
107  pouter_2->update() ;
108  const Domain_shell_inner_adapted* pinner_2 = dynamic_cast<const Domain_shell_inner_adapted*> (domains[3]) ;
109  pinner_2->vars_to_terms() ;
110  pinner_2->update() ;
111 }

175 void Space_bbh::affecte_coef_to_variable_domains (int& pos, int cc, Array<int>& zedoms) const {
176 
177  // In BH 1 ?
178  bool found_outer_1 = false ;
179  int old_pos = pos ;
180  domains[0]->affecte_coef(pos, cc, found_outer_1) ;
181  pos = old_pos ;
182  bool found_inner_1 = false ;
183  domains[1]->affecte_coef(pos, cc, found_inner_1) ;
184  assert (found_outer_1 == found_inner_1) ;
185  if (found_outer_1) {
186  zedoms.set(0) = 0 ;
187  zedoms.set(1) = 1 ;
188  }
189  else {
190  zedoms.set(0) = -1 ;
191  zedoms.set(1) = -1 ;
192  }
193 
194  // In BH 2 ?
195  bool found_outer_2 = false ;
196  old_pos = pos ;
197  domains[2]->affecte_coef(pos, cc, found_outer_2) ;
198  pos = old_pos ;
199  bool found_inner_2 = false ;
200  domains[3]->affecte_coef(pos, cc, found_inner_2) ;
201  assert (found_outer_2 == found_inner_2) ;
202  if (found_outer_2) {
203  zedoms.set(0) = 2 ;
204  zedoms.set(1) = 3 ;
205  }

221 void Space_bbh::xx_to_vars_variable_domains (System_of_eqs* sys, const Array<double>& xx, int& pos) const {
222 
223  // First get the corrections :
224  // BH 1
225  int old_pos = pos ;
226  Val_domain cor_outer_1 (domains[0]) ;
227  domains[0]->xx_to_vars_from_adapted(cor_outer_1, xx, pos) ;
228  pos = old_pos ;
229  Val_domain cor_inner_1 (domains[1]) ;
230  domains[1]->xx_to_vars_from_adapted(cor_inner_1, xx, pos) ;
231 
232  // BH 2
233  old_pos = pos ;
234  Val_domain cor_outer_2 (domains[2]) ;
235  domains[2]->xx_to_vars_from_adapted(cor_outer_2, xx, pos) ;
236  pos = old_pos ;
237  Val_domain cor_inner_2 (domains[3]) ;
238  domains[3]->xx_to_vars_from_adapted(cor_inner_2, xx, pos) ;
239 
240  // Now update the variables :
241  for (int i=0 ; i<sys->nvar ; i++) {
242  for (int n=0 ; n<sys->var[i]->get_n_comp() ; n++) {
243  Scalar res(*this) ;
244  domains[0]->update_variable(cor_outer_1, *sys->var[i]->cmp[n], res) ;
245  domains[1]->update_variable(cor_inner_1, *sys->var[i]->cmp[n], res) ;
246  domains[2]->update_variable(cor_outer_2, *sys->var[i]->cmp[n], res) ;
247  domains[3]->update_variable(cor_inner_2, *sys->var[i]->cmp[n], res) ;
248 
249 
250  sys->var[i]->cmp[n]->set_domain(0) = res(0) ;
251  sys->var[i]->cmp[n]->set_domain(1) = res(1) ;
252  sys->var[i]->cmp[n]->set_domain(2) = res(2) ;
253  sys->var[i]->cmp[n]->set_domain(3) = res(3) ;
254  }
255  }
256 
257  // Now the constants :
258  for (int i=0 ; i<sys->ncst ; i++)
259  if (sys->cst[i*(sys->dom_max-sys->dom_min+1)]->get_type_data()==TERM_T)
260  for (int n=0 ; n<sys->cst[i*(sys->dom_max-sys->dom_min+1)]->val_t->get_n_comp() ; n++) {
261 
262  Scalar so (*this) ;
263  so = 0 ;
264  for (int d=0 ; d<=3 ; d++)
265  so.set_domain(d) = (*sys->cst[i*(sys->dom_max-sys->dom_min+1)+d]->val_t->cmp[n])(d) ;
266 
267  Scalar res(*this) ;
268  res = 0 ;
269  domains[0]->update_constante(cor_outer_1, so, res) ;
270  domains[1]->update_constante(cor_inner_1, so, res) ;
271  domains[2]->update_constante(cor_outer_2, so, res) ;
272  domains[3]->update_constante(cor_inner_2, so, res) ;
273 
274  sys->cst[i*(sys->dom_max-sys->dom_min+1)]->val_t->cmp[n]->set_domain(0) = res(0) ;
275  sys->cst[i*(sys->dom_max-sys->dom_min+1)+1]->val_t->cmp[n]->set_domain(1) = res(1) ;
276  sys->cst[i*(sys->dom_max-sys->dom_min+1)+2]->val_t->cmp[n]->set_domain(2) = res(2) ;
277  sys->cst[i*(sys->dom_max-sys->dom_min+1)+3]->val_t->cmp[n]->set_domain(3) = res(3) ;
278  }
279 
280  // Update the mapping :
281  domains[0]->update_mapping(cor_outer_1) ;
282  domains[1]->update_mapping(cor_inner_1) ;
283  domains[2]->update_mapping(cor_outer_2) ;
284  domains[3]->update_mapping(cor_inner_2) ;
285 
286 }
