27 void Domain_bispheric_rect::export_tau_val_domain_boundary_one_side (const Val_domain& so, int bound, Array<double>& sec, int& pos_sec, int ncond) const {
28 
29  if (so.check_if_zero())
30  pos_sec += ncond ;
31  else {
32  so.coef() ;
33 
34  Index pos (nbr_coefs) ;
35  Index pos_galerkin (nbr_coefs) ;
36  int basep = (*so.get_base().bases_1d[2]) (0) ;
37 
38  if (bound==INNER_BC) {
39 
40  // Loop on phi :
41  for (int k=0 ; k<nbr_coefs(2) ; k++) {
42  pos.set(2) = k ;
43  // Loop on chi :
44  for (int j=0 ; j<nbr_coefs(1) ; j++) {
45  pos.set(1) = j ;
46  switch (basep) {
47  case COS :
48  // Avoid last odd ones
49  if ((k%2!=1) || (j!=nbr_coefs(1)-1)) {
50  if ((k==0) || (k%2==1)) {
51  // The ones without regularity issues
52  sec.set(pos_sec) = val_boundary (bound, so, pos) ;
53  pos_sec ++ ;
54  }
55  else if (j!=0) {
56  // Regularity on the axis thanks to Galerkin
57  // Factor fo Galerkin different for Legendre or Chebyshev:
58  double factor_galerkin ;
59  switch (type_base) {
60  case CHEB_TYPE :
61  factor_galerkin = (j%2==1) ? -2. : 2. ;
62  break ;
63  case LEG_TYPE :
64  factor_galerkin = -double(4*j+1) ;
65  for (int jj=1 ; jj<=j ; jj++)
66  factor_galerkin *= -double(2*jj-1)/double(2*jj) ;
67  break ;
68  default :
69  cerr << "Unknown type of basis in Domain_bispheric_rect::export_tau_boundary_one_side" << endl ;
70  abort() ;
71  }
72  pos_galerkin = pos ;
73  pos_galerkin.set(1) = 0 ;
74  sec.set(pos_sec) = val_boundary (bound, so, pos) +
75  factor_galerkin * val_boundary (bound, so, pos_galerkin) ;
76  pos_sec ++ ;
77  }}
78  break ;
79  case SIN :
80  // Avoid sin(0)
81  if ((k!=0) && (k!=nbr_coefs(2)-1))
82  //Avoid last odd ones
83  if ((k%2!=1) || (j!=nbr_coefs(1)-1)) {
84  if (k%2==1) {
85  // The ones without regularity issues
86  sec.set(pos_sec) = val_boundary (bound, so, pos) ;
87  pos_sec ++ ;
88  }
89  else if (j!=0) {
90  // Regularity on the axis thanks to Galerkin

101  break ;
102  default :
103  cerr << "Unknown type of basis in Domain_bispheric_rect::export_tau_boundary_one_side" << endl ;
104  abort() ;
105  }

107  pos_galerkin.set(1) = 0 ;

109  factor_galerkin * val_boundary (bound, so, pos_galerkin) ;
110  pos_sec ++ ;
111  }}
112  break ;
113  default :
114  cerr << "Unknown base in Domain_bispheric_rect::export_tau_boundary_one_side" << endl ;
115  break ;
116  }
117  }
118  }

123  for (int k=0 ; k<nbr_coefs(2) ; k++) {
124  pos.set(2) = k ;

128  switch (basep) {

203  if (bound==CHI_ONE_BC) {

205  pos.set(2) = k ;
206  if ((basep==COS) || ((k!=0) && (k!=nbr_coefs(2)-1)))

217  pos.set(0) = 0 ;

219  pos.set(2) = k ;
220  if ((basep==COS) || ((k!=0) && (k!=nbr_coefs(2)-1))) {
221  sec.set(pos_sec) = val_boundary (bound, so, pos) ;

230  int n_cmp, Array<int>** p_cmp) const {
231 
232  // Check boundary
233  if ((bound!=INNER_BC) && (bound!=ETA_PLUS_BC) && (bound!=CHI_ONE_BC) && (bound!=OUTER_BC)) {
234  cerr << "Unknown boundary in Domain_bispheric_rect::export_tau_boundary_one_side" << endl ;
235  abort() ;

237 
238  int val = tt.get_valence() ;
239  switch (val) {
240  case 0 :
241  export_tau_val_domain_boundary_one_side (tt()(dom), bound, res, pos_res, ncond(0)) ;
242  break ;
243  case 1 : {
244  bool found = false ;
245  // Cartesian basis
246  if (tt.get_basis().get_basis(dom)==CARTESIAN_BASIS) {
247  if (n_cmp==-1) {
248  export_tau_val_domain_boundary_one_side (tt(1)(dom), bound, res, pos_res, ncond(0)) ;
249  export_tau_val_domain_boundary_one_side (tt(2)(dom), bound, res, pos_res, ncond(1)) ;
250  export_tau_val_domain_boundary_one_side (tt(3)(dom), bound, res, pos_res, ncond(2)) ;
251  }
252  else for (int i=0 ; i<n_cmp ; i++) {
253  if ((*p_cmp[i])(0)==1)
254  export_tau_val_domain_boundary_one_side (tt(1)(dom), bound, res, pos_res, ncond(i)) ;
255  if ((*p_cmp[i])(0)==2)
256  export_tau_val_domain_boundary_one_side (tt(2)(dom), bound, res, pos_res, ncond(i)) ;
257  if ((*p_cmp[i])(0)==3)
258  export_tau_val_domain_boundary_one_side (tt(3)(dom), bound, res, pos_res, ncond(i)) ;
259  }
260  found = true ;
261  }
262  if (!found) {
263  cerr << "Unknown type of vector Domain_bispheric_rect::export_tau_boundary_one_side" << endl ;
264  abort() ;
265  }
266  }
267  break ;
268  case 2 : {
269  bool found = false ;
270  // Cartesian basis and symetric
271  if ((tt.get_basis().get_basis(dom)==CARTESIAN_BASIS) && (tt.get_n_comp()==6)) {

279  }
280  else for (int i=0 ; i<n_cmp ; i++) {
281  if (((*p_cmp[i])(0)==1) && ((*p_cmp[i])(1)==1))
282  export_tau_val_domain_boundary_one_side (tt(1, 1)(dom), bound, res, pos_res, ncond(i)) ;
283  if (((*p_cmp[i])(0)==1) && ((*p_cmp[i])(1)==2))
