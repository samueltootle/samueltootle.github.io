30 Term_eq Ope_ddtime::action() const {
31 
32  Term_eq target (parts[0]->action()) ;
33  int valence = target.val_t->get_valence() ;
34  // Check it is a tensor
35  if (target.type_data != TERM_T) {
36  cerr << "Ope_ddtime only defined with respect for a tensor" << endl ;
37  abort() ;
38  }
39 
40  // The value
41  Tensor resval (*target.val_t, false) ;
42 
43  for (int i=0 ; i<target.val_t->get_n_comp() ; i++) {
44  Array<int> ind (target.val_t->indices(i)) ;
45  Val_domain value ((*target.val_t)(ind)(dom)) ;
46  if (value.check_if_zero())
47  resval.set(ind).set_domain(dom).set_zero() ;
48  else {
49  Val_domain auxi (value.get_domain()->ddtime(value)) ;
50  resval.set(ind).set_domain(dom) = auxi ;
51  }
52  }
53 
54  // Put name indices :
55  if (target.val_t->is_name_affected()) {
56  resval.set_name_affected() ;
57  for (int ncmp = 0 ; ncmp<valence ; ncmp++)
58  resval.set_name_ind (ncmp, target.val_t->get_name_ind()[ncmp]) ;
59  }
60 
61  if (target.der_t!=0x0) {
62  Tensor resder (*target.der_t, false) ;
63  for (int i=0 ; i<target.der_t->get_n_comp() ; i++) {
64  Array<int> ind (target.der_t->indices(i)) ;
65  Val_domain value ((*target.der_t)(ind)(dom)) ;
66  if (value.check_if_zero())
67  resder.set(ind).set_domain(dom).set_zero() ;
68  else {
69  Val_domain auxi (value.get_domain()->ddtime(value)) ;
70  resder.set(ind).set_domain(dom) = auxi ;
71  }
72  }
73  // Put name indices :
74  if (target.der_t->is_name_affected()) {
75  resder.set_name_affected() ;
76  for (int ncmp = 0 ; ncmp<valence ; ncmp++)
77  resder.set_name_ind (ncmp, target.der_t->get_name_ind()[ncmp]) ;
78  }
79  Term_eq res (dom, resval, resder) ;
80  return res ;
81  }
82  else {
83  Term_eq res (dom, resval) ;
84  return res ;
85  }
86 }}
