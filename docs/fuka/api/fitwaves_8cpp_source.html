29 double Domain_spheric_periodic_shell::give_harmonique (const Val_domain& so, int nn, double phase, int dim, double factor) const {
30  so.coef() ;
31  Index pos (nbr_coefs) ;
32  pos.set(1) = nn ;
33  int expo = 0;
34  switch ((*so.base.bases_1d[1])(0)) {
35  case COS :
36  expo = nn ;
37  break ;
38  case COS_EVEN :
39  expo = 2*nn ;
40  break ;
41  case COS_ODD :
42  expo = 2*nn+1 ;
43  break ;
44  default :
45  cerr << "Unknown case in Domain_spheric_periodic_shell::fitwaves" << endl ;
46  break ;
47  }
48 
49 
50 
51  // Get the value at rmax of this harmonic
52  double AA = val_boundary (OUTER_BC, so, pos) ;
53  double lambda2 = factor - expo*expo*ome*ome ;
54  double res ;
55  double rmax = alpha + beta ;
56 
57  if (lambda2>0)
58  res = AA/exp(-sqrt(lambda2)*rmax)*pow(rmax, double(dim-1)/2.) ;
59  else
60  res = AA/sin(sqrt(-lambda2)*rmax+phase)* pow(rmax, double(dim-1)/2.) ;
61  return res ;
62 }

65 Val_domain Domain_spheric_periodic_shell::fitwaves (const Val_domain& so, const Array<double>& phases, int dim, double factor) const {
66 
67  so.coef() ;
68 
69  Val_domain dso (der_normal(so, OUTER_BC)) ;
70 
71  Val_domain res(this) ;
72  res.base = so.base ;
73  res.allocate_coef() ;
74 
75  double rmax = alpha + beta ;
76 
77  Index pos (nbr_coefs) ;
78 
79  int max = 0;
80  switch ((*so.base.bases_1d[1])(0)) {
81  case COS :
82  max = nbr_coefs(1) ;
83  break ;
84  case COS_EVEN :
85  max = nbr_coefs(1) ;
86  break ;
87  case COS_ODD :
88  max = nbr_coefs(1)-1 ;
89  break ;
90  default :
91  cerr << "Unknown case in Domain_spheric_periodic_shell::fitwaves" << endl ;
92  break ;
93  }
94 
95  // Loop on harmonic
96  for (int nn=0 ; nn<max ; nn++) {
97 
98  pos.set(1) = nn ;
99 
100  // Correspond to what ?
101  int expo = 0;
102  switch ((*so.base.bases_1d[1])(0)) {
103  case COS :
104  expo = nn ;
105  break ;
106  case COS_EVEN :
107  expo = 2*nn ;
108  break ;
109  case COS_ODD :
110  expo = 2*nn+1 ;
111  break ;
112  default :
113  cerr << "Unknown case in Domain_spheric_periodic_shell::fitwaves" << endl ;
114  break ;
115  }
116 
117  double lambda2 = factor - expo*expo*ome*ome ;
118  // Various cases :
119  double sh = (lambda2>0) ? exp(-sqrt(lambda2)*rmax)/pow(rmax, double(dim-1)/2.) :
120  sin(sqrt(-lambda2)*rmax+phases(nn))/ pow(rmax, double(dim-1)/2.) ;
121 
122  double dsh = (lambda2>0) ? exp(-sqrt(lambda2)*rmax)/pow(rmax, double(dim-1)/2.)*(-sqrt(lambda2)-double(dim-1)/2./rmax) :
123  sqrt(-lambda2)*cos(sqrt(-lambda2)*rmax+phases(nn))/ pow(rmax, double(dim-1)/2.) - double(dim-1)/2.*sin(sqrt(-lambda2)*rmax+phases(nn))/ pow(rmax, double(dim+1)/2.) ;
124 
125  // Affecte to res
126  for (int i=0 ; i<nbr_coefs(0) ; i++) {
127  pos.set(0) = i ;
128  res.set_coef(pos) = rmax*rmax*(so.get_coef(pos)*dsh - dso.get_coef(pos)*sh) ;
129  }
130 
131  }
132  return res ;
133 }

136 Term_eq Domain_spheric_periodic_shell::fitwaves (const Term_eq& so, const Array<double>& phases, int dim, double factor) const {
137 
138  // Check it is a tensor
139  if (so.get_type_data() != TERM_T) {
140  cerr << "fitwaves only defined with respect for a tensor" << endl ;
141  abort() ;
142  }
143 
144  // Right domain ?
145  int dom = so.get_dom() ;
146  if (this != so.get_val_t().get_space().get_domain(dom)) {
147  cerr << "Domain mismatch in Domain_spheric_periodic_shell::fitwaves (Term_eq version)" << endl ;
148  abort() ;
149  }
150 
151  Tensor resval (so.get_val_t(), false) ;
152  for (int i=0 ; i<so.get_val_t().get_n_comp() ; i++) {
153  Array<int> ind (so.get_val_t().indices(i)) ;
154  Val_domain value (so.get_val_t()(ind)(dom)) ;
155  if (value.check_if_zero())
156  resval.set(ind).set_domain(dom).set_zero() ;
157  else {
158  resval.set(ind).set_domain(dom) = fitwaves(value, phases, dim, factor) ;
159  }
160  }
161 
162  if (so.get_p_der_t() !=0x0) {
163  Tensor resder (so.get_val_t(), false) ;
164  for (int i=0 ; i<so.get_der_t().get_n_comp() ; i++) {
165  Array<int> ind (so.get_der_t().indices(i)) ;
166  Val_domain valder (so.get_der_t()(ind)(dom)) ;
167  if (valder.check_if_zero())
168  resder.set(ind).set_domain(dom).set_zero() ;
169  else {
170  resder.set(ind).set_domain(dom) = fitwaves(valder, phases, dim, factor) ;
171  }
172  }
173  return Term_eq (dom, resval, resder) ;
174  }
175  else return Term_eq (dom, resval) ;
176 }
