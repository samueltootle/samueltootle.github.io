47 template <typename ParamC>
48 int kadath_config_boost<ParamC>::open_config() {
49  read_config();
50  int status = -1;
51  auto key = MBCO.find(container.get_type());
52  if (container.get_type() == "binary" || key != MBCO.end()) {
53  branch_in = read_branch(tree, container.get_type());
54  } else {
55  for (auto& e : MBCO) {
56  branch_in = read_branch(tree, e.first);
57  if (!branch_in.empty()) {
58  status = e.second;
59  break;
60  }
61  }
62  }
63  if (branch_in.empty()) {
64  std::cout << "No Node found. \n";
65  return status;
66  }
67  container.read_params(branch_in);
68 
69  if (tree.find("fields") != tree.not_found())
70  read_keys(MBCO_FIELDS, fields, read_branch(tree, "fields"));
71  if (tree.find("stages") != tree.not_found())
72  read_keys(MSTAGE, stages, read_branch(tree, "stages"));
73  if (tree.find("sequence_controls") != tree.not_found())
74  read_keys(MCONTROLS, controls, read_branch(tree, "sequence_controls"));
75  if (tree.find("sequence_settings") != tree.not_found())
76  read_keys(MSEQ_SETTINGS, seq_settings,
77  read_branch(tree, "sequence_settings"));
78  return status;
79 }

81 template <typename ParamC>
82 void kadath_config_boost<ParamC>::write_config(std::string ofile) {
83 
84  Tree new_tree;
85  Tree branch = container.return_branch();
86 
87  std::string s{"initial"};
88  if (tree.find(s) != tree.not_found()) {
89  if (this->control(UPDATE_INIT))
90  new_tree.put_child(s, branch_in);
91  else
92  new_tree.push_back(std::make_pair(s, tree.get_child(s)));
93  } else {
94  if (this->control(UPDATE_INIT))
95  new_tree.put_child(s, branch);
96  }
97 
98  // get branch from parameter container
99  s = container.get_type();
100  for (auto& key : branch) {
101  if (key.first == s) {
102  new_tree.push_back(std::make_pair(s, key.second));
103  } else {
104  std::string news = s + "." + key.first;
105  if (key.second.empty())
106  new_tree.put(news.c_str(), key.second.data());
107  else
108  add_branch_data(new_tree, key.second, news);
109  }
110  }
111 
112  // add branches for fields, stages, and controls
113  s = "fields";
114  new_tree.push_back(
115  std::make_pair(s, build_branch<Tree>(MBCO_FIELDS, fields)));
116  s = "stages";
117  auto stage_map = append_map(MSTAGE, container.get_stage_map(), stages);
118  new_tree.push_back(
119  std::make_pair(s, build_branch<Tree>(stage_map, stages, true)));
120  s = "sequence_controls";
121  new_tree.push_back(
122  std::make_pair(s, build_branch<Tree>(MCONTROLS, controls, true)));
123  s = "sequence_settings";
124  new_tree.push_back(
125  std::make_pair(s, build_branch<Tree>(MSEQ_SETTINGS, seq_settings)));
126 
127  if (ofile != "null") {
128  //this will update outputdir if ofile includes a path
129  set_filename(ofile);
130  }
131 
132  s = config_filename_abs();
133  pt::write_info(s, new_tree);
134 }
