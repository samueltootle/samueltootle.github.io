101 T get_excision_r (space_t const & space,
102  T r, T const & theta_, T const & phi_, int const dom_, T const xshift_) {
103  // Generate cartesian point based on (r,t,p)
104  auto p = point_spherical(r, theta_, phi_, xshift_);
105 
106  // bool to ascertain if the point is inside the excision region
107  auto is_in_excision = [&] (auto start, auto stop) -> bool {
108  bool result = false;
109  for(auto d = start; d < stop; ++d)
110  result = result || space.get_domain(d)->is_in(p);
111  return result;
112  };
113 
114  size_t cnt{0};
115  constexpr size_t max_iter = 1000;
116  // Excision region consists of two domains, hence, dom_ - 2
117  while(is_in_excision(dom_ - 2, dom_)) {
118  r *= 1.01;
119  p = point_spherical(r, theta_, phi_, xshift_);
120  cnt++;
121 
122  // FIXME in some cases this is not very efficient - fails for max_iter = 100
123  // usually at the pole
124  if(cnt > max_iter){
125  cerr<< p << "not found by radius increase. inc failed at " << r << ".\n";
126  std::_Exit(EXIT_FAILURE);
127  }
128  };
129 
130  // the point should lay in the domain just outside the excision surface
131  // if it's not, there must be a problem.
132  if(!space.get_domain(dom_)->is_in(p)){
133  cerr << p << " not in dom " << dom_ << endl;
134  std::_Exit(EXIT_FAILURE);
135  }
136 
137  return r;
138 }
