111 template <class T, class space_t>
112 T get_excision_r(space_t const& space,
113  T r,
114  T const& theta_,
115  T const& phi_,
116  int const dom_,
117  T const xshift_) {
118  // Generate cartesian point based on (r,t,p)
119  auto p = point_spherical(r, theta_, phi_, xshift_);
120 
121  // bool to ascertain if the point is inside the excision region
122  auto is_in_excision = [&](auto start, auto stop) -> bool {
123  bool result = false;
124  for (auto d = start; d < stop; ++d)
125  result = result || space.get_domain(d)->is_in(p);
126  return result;
127  };
128 
129  size_t cnt{0};
130  constexpr size_t max_iter = 1000;
131  // Excision region consists of two domains, hence, dom_ - 2
132  while (is_in_excision(dom_ - 2, dom_)) {
133  r *= 1.01;
134  p = point_spherical(r, theta_, phi_, xshift_);
135  cnt++;
136 
137  // FIXME in some cases this is not very efficient - fails for max_iter = 100
138  // usually at the pole
139  if (cnt > max_iter) {
140  cerr << p << "not found by radius increase. inc failed at " << r << ".\n";
141  std::_Exit(EXIT_FAILURE);
142  }
143  };
144 
145  // the point should lay in the domain just outside the excision surface
146  // if it's not, there must be a problem.
147  if (!space.get_domain(dom_)->is_in(p)) {
148  cerr << p << " not in dom " << dom_ << endl;
149  std::_Exit(EXIT_FAILURE);
150  }
151 
152  return r;
153 }

537  const int dom_to_fill,
538  const int src_dom) {
539  using namespace Kadath;
540 
541  // We set this to false to ensure that bh_exporter.interpolate_pointwise will not fit to origin
542  bh_exporter.set__export_ready(false);
543 
544  auto dom = bh_exporter.get_space()->get_domain(dom_to_fill);
545  auto npts = dom->get_nbr_points();
546 
547  Index pos(npts);
548  Val_domain xx = dom->get_cart(1);
549  Val_domain yy = dom->get_cart(2);
550  Val_domain zz = dom->get_cart(3);
551 
552  auto& lapse = bh_exporter.get_lapse();
553  auto& conf = bh_exporter.get_conformal_factor();
554  auto& shift = bh_exporter.get_shift();
555 
556  do {
557  if (pos(0) < npts(0) - 1) {
558  double x = xx(pos);
559  double y = yy(pos);
560  double z = zz(pos);
561  bh_exporter.interpolate_pointwise__solution_gfs(x, y, z);
562  auto qv = bh_exporter.get__quant_vals();
563  conf->set_domain(dom_to_fill).set(pos) =
564  qv[bh_exporter_t::XCTS_VARS::XCTS_PSI];
565  lapse->set_domain(dom_to_fill).set(pos) =
566  qv[bh_exporter_t::XCTS_VARS::XCTS_ALPHA];
567 
568  shift->set(1).set_domain(dom_to_fill).set(pos) =
569  qv[bh_exporter_t::XCTS_VARS::XCTS_BETA1];
570  shift->set(2).set_domain(dom_to_fill).set(pos) =
571  qv[bh_exporter_t::XCTS_VARS::XCTS_BETA2];
572  shift->set(3).set_domain(dom_to_fill).set(pos) =
573  qv[bh_exporter_t::XCTS_VARS::XCTS_BETA3];
574  } else {

578  lapse->set_domain(dom_to_fill).set(pos) =
579  lapse->set_domain(src_dom)(bcpos);

607  void set__Jacobians() {
608  using std::cos;
609  using std::sin;
610  const double sint = sin(theta_);
611  const double cost = cos(theta_);
612  const double sinp = sin(phi_);
613  const double cosp = cos(phi_);
614 
615  const double rsq = r_ * r_;
616  const double rsint = r_ * sint;
617  const double rsint_sq = rsint * rsint;
618 
619  const double div_r = 1. / r_;
620  const double div_rsint = 1. / rsint;
621 
622  // dr^i/dx^i'
623  Jac_dSph_dCart[0][0] = sint * cosp;
624  Jac_dSph_dCart[0][1] = sint * sinp;
625  Jac_dSph_dCart[0][2] = cost;
626 
627  // dtheta^i/dx^i'
628  Jac_dSph_dCart[1][0] = cost * cosp * div_r;
629  Jac_dSph_dCart[1][1] = cost * sinp * div_r;
630  Jac_dSph_dCart[1][2] = -sint * div_r;
631 
632  // dphi^i/dx^i'
633  Jac_dSph_dCart[2][0] = -sinp * div_rsint;
634  Jac_dSph_dCart[2][1] = cosp * div_rsint;
635  Jac_dSph_dCart[2][2] = 0.0;
636 
637  // dx^i/dx^i'
638  Jac_dCart_dSph[0][0] = Jac_dSph_dCart[0][0];
639  Jac_dCart_dSph[0][1] = Jac_dSph_dCart[1][0] * rsq;
640  Jac_dCart_dSph[0][2] = Jac_dSph_dCart[2][0] * rsint * rsint;
641 
642  // dy^i/dx^i'
643  Jac_dCart_dSph[1][0] = Jac_dSph_dCart[0][1];
644  Jac_dCart_dSph[1][1] = Jac_dSph_dCart[1][1] * rsq;
645  Jac_dCart_dSph[1][2] = Jac_dSph_dCart[2][1] * rsint_sq;
646 
647  // dz^i/dx^i'
648  Jac_dCart_dSph[2][0] = Jac_dSph_dCart[0][2];
649  Jac_dCart_dSph[2][1] = Jac_dSph_dCart[1][2] * rsq;
650  Jac_dCart_dSph[2][2] = 0.0;
651  }
