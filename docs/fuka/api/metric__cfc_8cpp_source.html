48 void Metric_cfc::compute_cov (int dd) const {
49 
50  int dim = espace.get_ndim() ;
51  if (dim!=3) {
52  cerr << "Function only implemented for dimension 3" << endl ;
53  abort() ;
54  }
55 
56  int place = place_syst + (dd-syst->dom_min) ;
57  //Need to work component by components...
58  bool doder = (syst->term[place]->der_t==0x0) ? false : true ;
59 
60  Scalar val (espace) ;
61  Scalar der (espace) ;
62 
63  // CFC
64  val.set_domain(dd) = pow((*syst->term[place]->val_t)()(dd), 4) ;
65  val.std_base() ;
66  if (doder) {
67  der.set_domain(dd) = 4*pow((*syst->term[place]->val_t)()(dd), 3)* (*syst->term[place]->der_t)()(dd);
68  der.std_base() ;
69  }
70 
71 
72  // Value field :
73  Metric_tensor resval (espace, COV, basis) ;
74  resval.set(1,1) = val ;
75  resval.set(1,2) = 0 ;
76  resval.set(1,3) = 0 ;
77  resval.set(2,2) = val ;
78  resval.set(2,3) = 0 ;
79  resval.set(3,3) = val ;
80 
81  if (!doder) {
82  if (p_met_cov[dd]==0x0)
83  p_met_cov[dd] = new Term_eq(dd, resval) ;
84  else
85  *p_met_cov[dd] = Term_eq(dd, resval) ;
86  }
87  else {
88  // Der field :
89  Metric_tensor resder (espace, COV, basis) ;
90  resder.set(1,1) = der ;
91  resder.set(1,2) = 0 ;
92  resder.set(1,3) = 0 ;
93  resder.set(2,2) = der ;
94  resder.set(2,3) = 0 ;
95  resder.set(3,3) = der ;
96 
97  if (p_met_cov[dd]==0x0)
98  p_met_cov[dd] = new Term_eq(dd, resval, resder) ;
99  else
100  *p_met_cov[dd] = Term_eq(dd, resval, resder) ;
101  }
102 }

162 void Metric_cfc::compute_christo (int dd) const {
163  // Need both representation of the metric)
164  if (type_tensor == CON) {
165  if (p_met_con[dd]==0x0)
166  compute_con(dd) ;
167  if (p_met_cov[dd]==0x0)
168  compute_cov(dd) ;
169  }
170  else {
171  if (p_met_cov[dd]==0x0)
172  compute_cov(dd) ;
173  if (p_met_con[dd]==0x0)
174  compute_con(dd) ;
175  }
176 
177  Array<int> type_ind (3) ;
178  type_ind.set(0) = COV ; type_ind.set(1) = COV ; type_ind.set(2) = CON ;
179  Tensor res_val (espace, 3, type_ind, basis) ;
180  Tensor res_der (espace, 3, type_ind, basis) ;
181 
182  Term_eq flat_der (fmet.derive(COV, ' ', (*p_met_cov[dd]))) ;
183  bool doder = (flat_der.der_t==0x0) ? false : true ;
184 
185  Index pos (res_val) ;
186  do {
187 
188  Val_domain cmpval (espace.get_domain(dd)) ;
189  cmpval = 0 ;
190  for (int l=1 ; l<=espace.get_ndim() ; l++)
191  cmpval += 0.5*(*p_met_con[dd]->val_t)(pos(2)+1,l)(dd)*((*flat_der.val_t)(pos(0)+1, pos(1)+1, l)(dd) +
192  (*flat_der.val_t)(pos(1)+1, pos(0)+1, l)(dd) - (*flat_der.val_t)(l, pos(0)+1, pos(1)+1)(dd)) ;
193 
194  res_val.set(pos).set_domain(dd) = cmpval ;
195 
196  if (doder) {
197  Val_domain cmpder (espace.get_domain(dd)) ;
198  cmpder = 0 ;
199  for (int l=1 ; l<=espace.get_ndim() ; l++)
200  cmpder += 0.5*(*p_met_con[dd]->der_t)(pos(2)+1,l)(dd)*((*flat_der.val_t)(pos(0)+1, pos(1)+1, l)(dd) +
201  (*flat_der.val_t)(pos(1)+1, pos(0)+1, l)(dd) - (*flat_der.val_t)(l, pos(0)+1, pos(1)+1)(dd))
202  + 0.5*(*p_met_con[dd]->val_t)(pos(2)+1,l)(dd)*((*flat_der.der_t)(pos(0)+1, pos(1)+1, l)(dd) +
203  (*flat_der.der_t)(pos(1)+1, pos(0)+1, l)(dd) - (*flat_der.der_t)(l, pos(0)+1, pos(1)+1)(dd)) ;
204 
205  res_der.set(pos).set_domain(dd) = cmpder ;
206  }
207  }
208  while (pos.inc()) ;
209 
210  if (!doder) {
211  if (p_christo[dd]==0x0)
212  p_christo[dd] = new Term_eq(dd, res_val) ;
213  else
214  *p_christo[dd] = Term_eq (dd, res_val) ;
215  }
216  else {
217  if (p_christo[dd]==0x0)
218  p_christo[dd] = new Term_eq(dd, res_val, res_der) ;
219  else
220  *p_christo[dd] = Term_eq(dd, res_val, res_der) ;
221  }

227 void Metric_cfc::compute_riemann (int dd) const {
228 
229  // Need christoffels
230  if (p_christo[dd]==0x0)
231  compute_christo(dd) ;
232 
233 
234  Array<int> indices (4) ;
235  indices.set(0) = CON ; indices.set(1) = COV ; indices.set(2) = COV ; indices.set(3) = COV ;
236  Tensor res_val (espace, 4, indices, basis) ;
237  Tensor res_der (espace, 4, indices, basis) ;
238 
239  Term_eq flat_der (fmet.derive(COV, ' ', (*p_christo[dd]))) ;
240  bool doder = (flat_der.der_t==0x0) ? false : true ;
241 
242  Index pos (res_val) ;
243  do {
244 
245  Val_domain cmpval ((*flat_der.val_t)(pos(2)+1, pos(1)+1,pos(3)+1,pos(0)+1)(dd)
246  - (*flat_der.val_t)(pos(3)+1, pos(1)+1, pos(2)+1, pos(0)+1)(dd)) ;
247  for (int m=1 ; m<=espace.get_ndim() ; m++) {
248  cmpval += (*p_christo[dd]->val_t)(pos(2)+1,m, pos(0)+1)(dd)*(*p_christo[dd]->val_t)(pos(1)+1,pos(3)+1,m)(dd)
249  - (*p_christo[dd]->val_t)(pos(3)+1,m,pos(0)+1)(dd)*(*p_christo[dd]->val_t)(pos(1)+1,pos(2)+1,m)(dd) ;
250  }
251  res_val.set(pos).set_domain(dd) = cmpval;
252  if (doder) {
253  Val_domain cmpder ((*flat_der.der_t)(pos(2)+1, pos(1)+1,pos(3)+1,pos(0)+1)(dd)
254  - (*flat_der.der_t)(pos(3)+1, pos(1)+1, pos(2)+1, pos(0)+1)(dd)) ;
255  for (int m=1 ; m<=espace.get_ndim() ; m++) {
256  cmpder += (*p_christo[dd]->der_t)(pos(2)+1,m, pos(0)+1)(dd)*(*p_christo[dd]->val_t)(pos(1)+1,pos(3)+1,m)(dd)
257  + (*p_christo[dd]->val_t)(pos(2)+1,m, pos(0)+1)(dd)*(*p_christo[dd]->der_t)(pos(1)+1,pos(3)+1,m)(dd)
258  - (*p_christo[dd]->der_t)(pos(3)+1,m,pos(0)+1)(dd)*(*p_christo[dd]->val_t)(pos(1)+1,pos(2)+1,m)(dd)
259  - (*p_christo[dd]->val_t)(pos(3)+1,m,pos(0)+1)(dd)*(*p_christo[dd]->der_t)(pos(1)+1,pos(2)+1,m)(dd);
260  }
261  res_der.set(pos).set_domain(dd) = cmpder ;
262  }
263  }
264  while (pos.inc()) ;
265 
266  if (!doder) {
267  if (p_riemann[dd]==0x0)
268  p_riemann[dd] = new Term_eq(dd, res_val) ;
269  else
270  *p_riemann[dd] = Term_eq (dd, res_val) ;
271  }
272  else {
273  if (p_riemann[dd]==0x0)
274  p_riemann[dd] = new Term_eq(dd, res_val, res_der) ;
275  else
276  *p_riemann[dd] = Term_eq(dd, res_val, res_der) ;
277  }

283 void Metric_cfc::compute_ricci_tensor (int dd) const {
284  // Need christoffels
285  if (p_christo[dd]==0x0)
286  compute_christo(dd) ;
287 
288 
289  Array<int> indices (2) ;
290  indices.set(0) = COV ; indices.set(1) = COV ;
291  Tensor res_val (espace, 2, indices, basis) ;
292  Tensor res_der (espace, 2, indices, basis) ;
293 
294  Term_eq flat_der (fmet.derive(COV, ' ', (*p_christo[dd]))) ;
295  bool doder = (flat_der.der_t==0x0) ? false : true ;
296 
297  Index pos (res_val) ;
298  do {
299 
300  Val_domain cmpval (espace.get_domain(dd)) ;
301  cmpval = 0 ;
302 
303  for (int l=1 ; l<=espace.get_ndim() ; l++) {
304  cmpval += (*flat_der.val_t)(l, pos(1)+1,pos(0)+1,l)(dd)
305  - (*flat_der.val_t)(pos(1)+1, pos(0)+1,l ,l)(dd) ;
306  for (int m=1 ; m<=espace.get_ndim() ; m++) {
307  cmpval += (*p_christo[dd]->val_t)(l,m, l)(dd)*(*p_christo[dd]->val_t)(pos(0)+1,pos(1)+1,m)(dd)
308  - (*p_christo[dd]->val_t)(pos(0)+1,l, m)(dd)*(*p_christo[dd]->val_t)(pos(1)+1,m,l)(dd) ;
309  }
310  }
311 
312  res_val.set(pos).set_domain(dd) = cmpval;
313  if (doder) {
314  Val_domain cmpder(espace.get_domain(dd)) ;
315  cmpder = 0 ;
316 
317  for (int l=1 ; l<=espace.get_ndim() ; l++) {
318  cmpder += (*flat_der.der_t)(l, pos(1)+1,pos(0)+1,l)(dd)
319  - (*flat_der.der_t)(pos(1)+1, pos(0)+1,l ,l)(dd) ;
320  for (int m=1 ; m<=espace.get_ndim() ; m++) {
321  cmpder += (*p_christo[dd]->der_t)(l,m, l)(dd)*(*p_christo[dd]->val_t)(pos(0)+1,pos(1)+1,m)(dd)
322  + (*p_christo[dd]->val_t)(l,m, l)(dd)*(*p_christo[dd]->der_t)(pos(0)+1,pos(1)+1,m)(dd)
323  - (*p_christo[dd]->der_t)(pos(0)+1,l, m)(dd)*(*p_christo[dd]->val_t)(pos(1)+1,m,l)(dd)
324  - (*p_christo[dd]->val_t)(pos(0)+1,l, m)(dd)*(*p_christo[dd]->der_t)(pos(1)+1,m,l)(dd);
325  }
326  }
327  res_der.set(pos).set_domain(dd) = cmpder ;
328  }
329  }
330  while (pos.inc()) ;
331 
332  if (!doder) {
333  if (p_ricci_tensor[dd]==0x0)
334  p_ricci_tensor[dd] = new Term_eq(dd, res_val) ;
335  else
336  *p_ricci_tensor[dd] = Term_eq (dd, res_val) ;
337  }
338  else {
339  if (p_ricci_tensor[dd]==0x0)
340  p_ricci_tensor[dd] = new Term_eq(dd, res_val, res_der) ;
341  else
342  *p_ricci_tensor[dd] = Term_eq(dd, res_val, res_der) ;
343  }

366 Term_eq Metric_cfc::derive (int type_der, char ind_der, const Term_eq& so) const {
367 
368  int dd = so.get_dom() ;
369 
370  if (p_christo[dd]==0x0)
371  compute_christo(dd) ;
372  // The partial derivative part
373  Term_eq res (fmet.derive_with_other (type_der, ind_der, so, this)) ;
374 
375  // Add the part containing the Christoffel :
376  //Must find a name for summation on Christofel :
377  bool found = false ;
378  int start = 97 ;
379  do {
380  bool same = false ;
381  if (ind_der==char(start))
382  same = true ;
383  for (int i=0 ; i<so.val_t->get_valence() ; i++)
384  if (so.val_t->get_name_ind()[i]==char(start))
385  same = true ;
386  if (!same)
387  found = true ;
388  else
389  start ++ ;
390  }
391  while ((!found) && (start<123)) ;
392  if (!found) {
393  cerr << "Trouble with indices in derive (you are not using tensors of order > 24, are you ?)" << endl ;
394  abort() ;
395  }
396  char name_sum = char(start) ;
397 
398  bool doder = ((so.der_t==0x0) || (p_christo[dd]->der_t==0x0)) ? false : true ;
399 
400  //loop on the components of so :
401  for (int cmp=0 ; cmp<so.val_t->get_valence() ; cmp ++) {
402 
403  int genre_indice = so.val_t->get_index_type(cmp) ;
404 
405  //Affecte names on the Christoffels :
406  p_christo[dd]->val_t->set_name_affected() ;
407  p_christo[dd]->val_t->set_name_ind(0, ind_der) ;
408  if (genre_indice==COV) {
409  p_christo[dd]->val_t->set_name_ind(1, so.val_t->get_name_ind()[cmp]) ;
410  p_christo[dd]->val_t->set_name_ind(2, name_sum) ;
411  }
412  else {
413  p_christo[dd]->val_t->set_name_ind(2, so.val_t->get_name_ind()[cmp]) ;
414  p_christo[dd]->val_t->set_name_ind(1, name_sum) ;
415  }
416 
417  if (doder) {
418  p_christo[dd]->der_t->set_name_affected() ;
419  p_christo[dd]->der_t->set_name_ind(0, ind_der) ;
420  if (genre_indice==COV) {
421  p_christo[dd]->der_t->set_name_ind(1, so.der_t->get_name_ind()[cmp]) ;
422  p_christo[dd]->der_t->set_name_ind(2, name_sum) ;
423  }
424  else {
425  p_christo[dd]->der_t->set_name_ind(2, so.der_t->get_name_ind()[cmp]) ;
426  p_christo[dd]->der_t->set_name_ind(1, name_sum) ;
427  }
428  }
429 
430  Term_eq auxi_christ (*p_christo[dd]) ;
431 
432  if (type_der==CON)
433  manipulate_ind(auxi_christ, 0) ;
434 
435  // Check if one inner summation is needed :
436  bool need_sum = false ;
437  char* ind = p_christo[dd]->val_t->get_name_ind() ;
438  if ((ind[0]==ind[2]) || (ind[1]==ind[2]) || (ind[0]==ind[1]))
439  need_sum = true ;
440 
441  Term_eq* christ ;
442  if (need_sum)
443  if (!doder)
444  christ = new Term_eq (dd, auxi_christ.val_t->do_summation_one_dom(dd)) ;
445  else
446  christ = new Term_eq (dd, auxi_christ.val_t->do_summation_one_dom(dd),
447  auxi_christ.der_t->do_summation_one_dom(dd)) ;
448  else
449  christ = new Term_eq (auxi_christ) ;
450 
451  // Affecte names on the field :
452  Term_eq copie (so) ;
453  copie.val_t->set_name_affected() ;
454  for (int i=0 ; i<so.val_t->get_valence() ; i++)
455  if (i!=cmp)
456  copie.val_t->set_name_ind(i, so.val_t->get_name_ind()[i]) ;
457  else
458  copie.val_t->set_name_ind(i, name_sum) ;
459  if (doder) {
460  copie.der_t->set_name_affected() ;
461  for (int i=0 ; i<so.der_t->get_valence() ; i++)
462  if (i!=cmp)
463  copie.der_t->set_name_ind(i, so.der_t->get_name_ind()[i]) ;
464  else
465  copie.der_t->set_name_ind(i, name_sum) ;
466  }
467 
468  Term_eq part_christo ((*christ)*copie) ;
469  delete christ ;
470 
471  if (genre_indice==CON)
472  res = res + part_christo ;
473  else
474  res = res - part_christo ;
475  }
476  return res ;
477 }
