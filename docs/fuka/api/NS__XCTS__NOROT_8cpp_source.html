164 template <class eos_t>
165 void NS_XCTS_NOROT<eos_t>::syst_init() {
166  using namespace ::Kadath::Margherita;
167 
168  // call the (flat) conformal metric "f"
169  fmet->set_system(*syst, "f");
170 
171  // define numerical constants
172  syst->add_cst("4piG", (*bconfig)(BCO_QPIG));
173 
174  // include the coordinate fields
175  syst->add_cst("einf", *(*coord_vectors)[S_INF]);
176 
177  // the basic fields, conformal factor, lapse and (log) enthalpy
178  syst->add_var("P", *conformal_factor);
179  syst->add_var("N", *lapse);
180  syst->add_var("H", *logh);
181 
182  // define common combinations of conformal factor and lapse
183  syst->add_def("NP = P*N");
184  syst->add_def("Ntilde = N / P^6");
185 
186  // define quantity to be integrated at infinity
187  // two (in this case) equivalent definitions of ADM mass
188  // as well as the Komar mass
189  syst->add_def(ndom - 1, "intMadm = - einf^i * D_i P / 4piG * 2");
190  syst->add_def(ndom - 1, "intMk = einf^i * D_i N / 4piG");
191  syst->add_def(ndom - 1, "intMadmalt = -dr(P) * 2 / 4piG");
192 
193  // enthalpy from the logarithmic enthalpy, the latter is the actual variable
194  // in this system
195  syst->add_def("h = exp(H)");
196 
197  // define the EOS operators
198  Param p;
199  set_eos_ope<eos_t>(*syst, p);
200 
201  // define rest-mass density, internal energy and pressure through the
202  // enthalpy
203  syst->add_def("rho = rho(h)");
204  syst->add_def("eps = eps(h)");
205  syst->add_def("press = press(h)");
206  syst->add_def("dHdlnrho = dHdlnrho(h)");
207 
208  // definition to rescale the equations
209  // delta = p / rho
210  syst->add_def("delta = h - eps - 1.");
211 }
