33 Term_eq Domain_polar_shell_inner_adapted::derive_r (const Term_eq& so) const {
34 
35  assert (so.get_dom() == num_dom) ;
36 
37  Tensor val_res(*so.val_t, false) ;
38 
39  for (int cmp = 0 ; cmp<so.val_t->get_n_comp() ; cmp++) {
40  Array<int>ind (val_res.indices(cmp)) ;
41  val_res.set(ind).set_domain(num_dom) = (*so.val_t)(ind)(num_dom).der_var(1) / (*der_rad_term_eq->val_t)()(num_dom) ;
42  }
43 
44  bool doder = ((so.der_t==0x0) || (der_rad_term_eq->der_t==0x0)) ? false : true ;
45  if (doder) {
46  Tensor der_res (*so.der_t, false) ;
47  for (int cmp = 0 ; cmp<so.val_t->get_n_comp() ; cmp++) {
48  Array<int>ind (val_res.indices(cmp)) ;
49  der_res.set(ind).set_domain(num_dom) = ((*so.der_t)(ind)(num_dom).der_var(1)*(*der_rad_term_eq->val_t)()(num_dom)
50  - (*so.val_t)(ind)(num_dom).der_var(1)*(*der_rad_term_eq->der_t)()(num_dom)) /
51  ((*der_rad_term_eq->val_t)()(num_dom) * (*der_rad_term_eq->val_t)()(num_dom)) ;
52  }
53  return Term_eq (num_dom, val_res, der_res) ;
54  }
55  else
56  return Term_eq (num_dom, val_res) ;
57 }

59 Term_eq Domain_polar_shell_inner_adapted::derive_t (const Term_eq& so) const {
60 
61  assert (so.get_dom() == num_dom) ;
62  Term_eq* dtprime ;
63  Tensor val_res(*so.val_t, false) ;
64 
65  for (int cmp = 0 ; cmp<so.val_t->get_n_comp() ; cmp++) {
66  Array<int>ind (val_res.indices(cmp)) ;
67  val_res.set(ind).set_domain(num_dom) = (*so.val_t)(ind)(num_dom).der_var(2) ;
68  }
69  bool doder = (so.der_t==0x0) ? false : true ;
70 
71  if (doder) {
72  Tensor der_res (*so.der_t, false) ;
73  for (int cmp = 0 ; cmp<so.val_t->get_n_comp() ; cmp++) {
74  Array<int>ind (val_res.indices(cmp)) ;
75  der_res.set(ind).set_domain(num_dom) = (*so.der_t)(ind)(num_dom).der_var(2) ;
76  }
77  dtprime = new Term_eq (num_dom, val_res, der_res) ;
78  }
79  else
80  dtprime = new Term_eq (num_dom, val_res) ;
81 
82 
83  Term_eq res (*dtprime - (*dt_rad_term_eq) * derive_r(so)) ;
84  delete dtprime ;
85  return res ;

89 Term_eq Domain_polar_shell_inner_adapted::flat_grad_spher (const Term_eq& so) const {
90 
91  assert (so.get_dom() == num_dom) ;
92 
93  int valso = so.get_val_t().get_valence() ;
94  Array<int> type_ind (valso+1) ;
95  type_ind.set(0) = COV ;
96  for (int i=0 ; i<valso ; i++)
97  type_ind.set(i+1) = so.get_val_t().get_index_type(i) ;
98 
99  Base_tensor basis (sp) ;
100  basis.set_basis(num_dom) = SPHERICAL_BASIS ;
101  Tensor res (sp, valso+1 , type_ind, basis) ;
102 
103  Term_eq comp_r (derive_r(so)) ;
104  Term_eq comp_t (derive_t(so)/(*rad_term_eq)) ;
105 
106  // Loop on cmp :
107  for (int nc=0 ; nc<so.get_val_t().get_n_comp() ; nc++) {
108 
109  Array<int> ind (so.get_val_t().indices(nc)) ;
110  Array<int> indtarget (valso+1) ;
111  for (int i=0 ; i<valso ; i++)
112  indtarget.set(i+1) = ind(i) ;
113 
114  // R comp :
115  indtarget.set(0) = 1 ;
116  res.set(indtarget).set_domain(num_dom) = comp_r.get_val_t()(ind)(num_dom);
117  // theta comp :
118  indtarget.set(0) = 2 ;
119  res.set(indtarget).set_domain(num_dom) = comp_t.get_val_t()(ind)(num_dom) ;
120  }
121  bool doder = ((so.der_t==0x0) || (inner_radius_term_eq->der_t==0x0)) ? false : true ;
122 
123  if (!doder) {
124  return Term_eq (num_dom, res) ;
125  }
126  else {
127 
128  Tensor resder (sp, valso+1 , type_ind, basis) ;
129  // Loop on cmp :
130  for (int nc=0 ; nc<so.get_val_t().get_n_comp() ; nc++) {
131 
132  Array<int> ind (so.get_val_t().indices(nc)) ;
133  Array<int> indtarget (valso+1) ;
134  for (int i=0 ; i<valso ; i++)
135  indtarget.set(i+1) = ind(i) ;
136 
137  // R comp :
138  indtarget.set(0) = 1 ;
139  resder.set(indtarget).set_domain(num_dom) = comp_r.get_der_t()(ind)(num_dom);
140  // theta comp :
141  indtarget.set(0) = 2 ;
142  resder.set(indtarget).set_domain(num_dom) = comp_t.get_der_t()(ind)(num_dom) ;
143  }
144  return Term_eq (num_dom, res, resder) ;
145  }

148 void Domain_polar_shell_inner_adapted::do_normal_spher () const {
149 
150  Term_eq grad (flat_grad_spher(*rad_term_eq - *inner_radius_term_eq)) ;
151 
152  Scalar val_norme (sp) ;
153  val_norme.set_domain(num_dom) = sqrt((*grad.val_t)(1)(num_dom)*(*grad.val_t)(1)(num_dom)
154  + (*grad.val_t)(2)(num_dom)*(*grad.val_t)(2)(num_dom)) ;
155  bool doder = ((grad.der_t==0x0) || (inner_radius_term_eq->der_t==0x0)) ? false : true ;
156  if (doder) {
157  Scalar der_norme (sp) ;
158  der_norme.set_domain(num_dom) = ((*grad.der_t)(1)(num_dom)*(*grad.val_t)(1)(num_dom)
159  + (*grad.der_t)(2)(num_dom)*(*grad.val_t)(2)(num_dom)) / val_norme(num_dom) ;
160  Term_eq norme (num_dom, val_norme, der_norme) ;
161  if (normal_spher==0x0)
162  normal_spher = new Term_eq (grad / norme) ;
163  else
164  *normal_spher = Term_eq(grad/norme) ;
165  }
166  else {
167  Term_eq norme (num_dom, val_norme) ;
168  if (normal_spher==0x0)
169  normal_spher = new Term_eq (grad / norme) ;
170  else
171  *normal_spher = Term_eq(grad/norme) ;
172  }

175 void Domain_polar_shell_inner_adapted::do_normal_cart () const {
176 
177 
178  do_normal_spher() ;
179 
180  Base_tensor basis (sp) ;
181  basis.set_basis(num_dom) = CARTESIAN_BASIS ;
182  Vector val (sp, CON, basis) ;
183 
184  val.set(1).set_domain(num_dom) = mult_sin_theta((*normal_spher->val_t)(1)(num_dom)) + mult_cos_theta((*normal_spher->val_t)(2)(num_dom)) ;
185  val.set(2).set_domain(num_dom) = mult_cos_theta((*normal_spher->val_t)(1)(num_dom)) - mult_sin_theta((*normal_spher->val_t)(2)(num_dom)) ;
186 
187  bool doder = (normal_spher->der_t==0x0) ? false : true ;
188  if (doder) {
189  Vector der (sp, CON, basis) ;
190 
191  der.set(1).set_domain(num_dom) = mult_sin_theta((*normal_spher->der_t)(1)(num_dom)) + mult_cos_theta((*normal_spher->der_t)(2)(num_dom)) ;
192  der.set(2).set_domain(num_dom) = mult_cos_theta ((*normal_spher->der_t)(1)(num_dom)) - mult_sin_theta((*normal_spher->der_t)(2)(num_dom)) ;
193 
194  if (normal_cart==0x0)
195  normal_cart = new Term_eq (num_dom, val, der) ;
196  else
197  *normal_cart = Term_eq(num_dom, val,der) ;
198  }
199  else {
200  if (normal_cart==0x0)
201  normal_cart = new Term_eq (num_dom, val) ;
202  else
203  *normal_cart = Term_eq(num_dom, val) ;
204  }

207 Term_eq Domain_polar_shell_inner_adapted::der_normal_term_eq (const Term_eq& so, int bound) const {
208 
209 
210  switch (bound) {
211  case INNER_BC : {
212  // Deformed surface
213  if (normal_spher == 0x0)
214  do_normal_spher() ;
215 
216  int valso = so.get_val_t().get_valence() ;
217 
218  Term_eq grad (flat_grad_spher(so)) ;
219 
220  Tensor res (so.get_val_t(), false) ;
221 
222  // Loop on cmp :
223  for (int nc=0 ; nc<so.get_val_t().get_n_comp() ; nc++) {
224 
225  Array<int> ind (so.get_val_t().indices(nc)) ;
226  Array<int> indgrad (valso+1) ;
227  for (int i=0 ; i<valso ; i++)
228  indgrad.set(i+1) = ind(i) ;
229 
230  indgrad.set(0) = 1 ;
231  res.set(ind).set_domain(num_dom) = (*grad.val_t)(indgrad)(num_dom) * (*normal_spher->val_t)(1)(num_dom) ;
232  indgrad.set(0) = 2 ;
233  res.set(ind).set_domain(num_dom) += (*grad.val_t)(indgrad)(num_dom) * (*normal_spher->val_t)(2)(num_dom) ;
234  }
235 
236  bool doder = ((grad.der_t == 0x0) || (normal_spher->der_t == 0x0)) ? false : true ;
237  if (doder) {
238 
239  Tensor der (so.get_val_t(), false) ;
240 
241  // Loop on cmp :
242  for (int nc=0 ; nc<so.get_val_t().get_n_comp() ; nc++) {
243 
244  Array<int> ind (so.get_val_t().indices(nc)) ;
245  Array<int> indgrad (valso+1) ;
246  for (int i=0 ; i<valso ; i++)
247  indgrad.set(i+1) = ind(i) ;
248 
249  indgrad.set(0) = 1 ;
250  der.set(ind).set_domain(num_dom) = (*grad.der_t)(indgrad)(num_dom) * (*normal_spher->val_t)(1)(num_dom)
251  + (*grad.val_t)(indgrad)(num_dom) * (*normal_spher->der_t)(1)(num_dom) ;
252  indgrad.set(0) = 2 ;
253  der.set(ind).set_domain(num_dom) += (*grad.der_t)(indgrad)(num_dom) * (*normal_spher->val_t)(2)(num_dom)
254  + (*grad.val_t)(indgrad)(num_dom) * (*normal_spher->der_t)(2)(num_dom) ;
255  }
256 
257  return Term_eq (num_dom, res, der) ;
258  }
259  else
260  return Term_eq (num_dom, res) ;
261  }
262  case OUTER_BC : {
263  return derive_r(so) ;
264  }
265  default :
266  cerr << "Unknown boundary in Domain_polar_shell_inner_adapted::der_normal" << endl ;
267  abort() ;
268  }

272 Term_eq Domain_polar_shell_inner_adapted::lap_term_eq (const Term_eq& so, int mm) const {
273 
274  if (so.get_val_t().get_valence() != 0) {
275  cerr << "Domain_polar_shell_inner_adapted::lap_term_eq only defined for scalars" << endl ;
276  abort() ;
277  }
278 
279  // Angular part :
280 
281  Term_eq auxi (do_comp_by_comp(so, &Domain::div_sin_theta)) ;
282 
283  Term_eq dert (derive_t(so));
284  Term_eq p1 (derive_t(dert)) ;
285  Term_eq p2 (do_comp_by_comp(do_comp_by_comp(dert, &Domain::mult_cos_theta) - mm * auxi , &Domain::div_sin_theta)) ;
286 
287  Term_eq dr(derive_r(so)) ;
288 
289  Term_eq res (derive_r(dr) + 2 *dr / (*rad_term_eq) + (p1+p2)/(*rad_term_eq)/(*rad_term_eq));
290  return res ;
291 }

328 void Domain_polar_shell_inner_adapted::update_term_eq (Term_eq* so) const {
329 
330  Tensor der (*so->val_t, false) ;
331  for (int cmp=0 ; cmp<so->val_t->get_n_comp() ; cmp ++) {
332 
333 
334 
335  Val_domain derr ((*(*so->val_t).cmp[cmp])(num_dom).der_var(1) / (*der_rad_term_eq->val_t)()(num_dom)) ;
336 
337  if (!derr.check_if_zero()) {
338  Val_domain res(this) ;
339  res.allocate_conf() ;
340  Index pos (nbr_points) ;
341  do {
342  res.set(pos) = (*(*inner_radius_term_eq->der_t).cmp[0])(num_dom)(pos) / 2. * (1-((*coloc[0])(pos(0)))) * derr(pos) ;
343  }
344  while (pos.inc()) ;
345  res.set_base() = (*(*so->val_t).cmp[cmp])(num_dom).get_base() ;
346  der.cmp[cmp]->set_domain(num_dom) = res ;
347  }
348  else
349  der.cmp[cmp]->set_domain(num_dom).set_zero() ;
350  }
351 
352  so->set_der_t (der) ;
353 }

355 Term_eq Domain_polar_shell_inner_adapted::partial_spher (const Term_eq& so) const {
356  int dom = so.get_dom() ;
357  assert (dom == num_dom) ;
358 
359  int valence = so.val_t->get_valence() ;
360 
361  Array<int> type_ind (valence+1) ;
362  type_ind.set(0) = COV ;
363  for (int i=1 ; i<valence+1 ; i++)
364  type_ind.set(i) = so.val_t->get_index_type(i-1) ;
365 
366  Base_tensor basis (so.get_val_t().get_space()) ;
367  basis.set_basis(dom) = SPHERICAL_BASIS ;
368 
369  Term_eq comp_r (derive_r(so)) ;
370  Term_eq comp_t (derive_t(so) / (*rad_term_eq)) ;
371 
372  Tensor val_res (so.get_val_t().get_space(), valence+1, type_ind, basis) ;
373  {
374  Index pos_so (*so.val_t) ;
375  Index pos_res (val_res) ;
376  do {
377  for (int i=1 ; i<valence+1 ; i++)
378  pos_res.set(i) = pos_so(i-1) ;
379  // R part
380  pos_res.set(0) = 0 ;
381  val_res.set(pos_res).set_domain(num_dom) = (*comp_r.val_t)(pos_so)(num_dom) ;
382  // Theta part
383  pos_res.set(0) = 1 ;
384  val_res.set(pos_res).set_domain(num_dom) = (*comp_t.val_t)(pos_so)(num_dom) ;
385  }
386  while (pos_so.inc()) ;
387 }
388 
389  bool doder = ((so.der_t==0x0) || (inner_radius_term_eq->der_t==0x0)) ? false : true ;
390  if (doder) {
391  Tensor der_res (so.get_val_t().get_space(), valence+1, type_ind, basis) ;
392  Index pos_so (*so.der_t) ;
393  Index pos_res (val_res) ;
394  do {
395  for (int i=1 ; i<valence+1 ; i++)
396  pos_res.set(i) = pos_so(i-1) ;
397  // R part
398  pos_res.set(0) = 0 ;
399  der_res.set(pos_res).set_domain(num_dom) = (*comp_r.der_t)(pos_so)(num_dom) ;
400  // Theta part
401  pos_res.set(0) = 1 ;
402  der_res.set(pos_res).set_domain(num_dom) = (*comp_t.der_t)(pos_so)(num_dom) ;
403  }
404  while (pos_so.inc()) ;
405 
406  return Term_eq (num_dom, val_res, der_res) ;
407  }
408  else
409  return Term_eq (num_dom, val_res) ;

413 Term_eq Domain_polar_shell_inner_adapted::partial_cart (const Term_eq& so) const {
414  int dom = so.get_dom() ;
415  assert (dom == num_dom) ;
416 
417  int valence = so.val_t->get_valence() ;
418 
419  Array<int> type_ind (valence+1) ;
420  type_ind.set(0) = COV ;
421  for (int i=1 ; i<valence+1 ; i++)
422  type_ind.set(i) = so.val_t->get_index_type(i-1) ;
423 
424  Base_tensor basis (so.get_val_t().get_space()) ;
425  basis.set_basis(dom) = CARTESIAN_BASIS ;
426 
427  Term_eq comp_r (derive_r(so)) ;
428  Term_eq comp_t (derive_t(so) / (*rad_term_eq)) ;
429 
430  Tensor val_res (so.get_val_t().get_space(), valence+1, type_ind, basis) ;
431  {
432  Index pos_so (*so.val_t) ;
433  Index pos_res (val_res) ;
434  do {
435  for (int i=1 ; i<valence+1 ; i++)
436  pos_res.set(i) = pos_so(i-1) ;
437 
438  // X part
439  pos_res.set(0) = 0 ;
440  val_res.set(pos_res).set_domain(num_dom) = mult_sin_theta((*comp_r.val_t)(pos_so)(num_dom)) + mult_cos_theta((*comp_t.val_t)(pos_so)(num_dom)) ;
441  // Z part
442  pos_res.set(0) = 1 ;
443  val_res.set(pos_res).set_domain(num_dom) =
444  mult_cos_theta((*comp_r.val_t)(pos_so)(num_dom)) - mult_sin_theta((*comp_t.val_t)(pos_so)(num_dom)) ;
445  }
446  while (pos_so.inc()) ;
447 }
448 
449  bool doder = ((so.der_t==0x0) || (rad_term_eq->der_t==0x0)) ? false : true ;
450  if (doder) {
451  Tensor der_res (so.get_val_t().get_space(), valence+1, type_ind, basis) ;
452  Index pos_so (*so.der_t) ;
453  Index pos_res (val_res) ;
454  do {
455  for (int i=1 ; i<valence+1 ; i++)
456  pos_res.set(i) = pos_so(i-1) ;
457 
458  Val_domain auxi (mult_sin_theta((*comp_r.der_t)(pos_so)(num_dom)) + mult_cos_theta((*comp_t.der_t)(pos_so)(num_dom))) ;
459  // X part
460  pos_res.set(0) = 0 ;
461  der_res.set(pos_res).set_domain(num_dom) = mult_sin_theta((*comp_r.der_t)(pos_so)(num_dom)) + mult_cos_theta((*comp_t.der_t)(pos_so)(num_dom)) ;
462  // Z part
463  pos_res.set(0) = 1 ;
464  der_res.set(pos_res).set_domain(num_dom) =
465  mult_cos_theta((*comp_r.der_t)(pos_so)(num_dom)) - mult_sin_theta((*comp_t.der_t)(pos_so)(num_dom)) ;
466  }
467  while (pos_so.inc()) ;
468 
469  return Term_eq (num_dom, val_res, der_res) ;
470  }
471  else
472  return Term_eq (num_dom, val_res) ;
