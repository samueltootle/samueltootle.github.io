347  template <class eos_t>
348  struct export_pointwise_imp {
349  friend CFMS_NS_ISO_Exporter;
350 
351  output_ary_t operator()(CFMS_NS_ISO_Exporter& base,
352  double const& x,
353  double const& y,
354  double const& z) {
355  // Reset to NAN
356  base.quant_vals.fill(NAN);
357  base.quant_vals = base.interpolate_pointwise(x, y, z);
358 
359  export_utils::basis_transform_spherical_tofrom_cart basis_transform;
360  basis_transform.set__cart(x, y, z);
361  double const theta = basis_transform.get_theta();
362  double const rsq = basis_transform.get_r_sq();
363 
364  double const N = base.quant_vals[ISO_VARS::ISO_ALPHA];
365  double const U_factor = base.quant_vals[ISO_VARS::ISO_U];
366  double const domega_dr = base.quant_vals[ISO_VARS::ISO_DOMEGA_DR];
367  double const domega_dt = base.quant_vals[ISO_VARS::ISO_DOMEGA_DTHETA];
368 
369  double const A = base.quant_vals[ISO_VARS::ISO_METRIC_A];
370  double const B = base.quant_vals[ISO_VARS::ISO_METRIC_B];
371 
372  double const Asq = A * A;
373  double const Bsq = B * B;
374 
375  // clang-format off
376  export_utils::basis_transform_spherical_tofrom_cart::vector_t FluidVelU_Sph = {
377  0.0,
378  0.0,
379  U_factor
380  };
381 
382  export_utils::basis_transform_spherical_tofrom_cart::vector_t betaU_Sph = {
383  0.0,
384  0.0,
385  -base.quant_vals[ISO_VARS::ISO_METRIC_OMEGA]
386  };
387 
388  export_utils::basis_transform_spherical_tofrom_cart::matrix_t gammaDD_Sph = {{
389  {{ Asq, 0.0 , 0.0 }},
390  {{ 0.0, Asq * rsq, 0.0 }},
391  {{ 0.0, 0.0 , Bsq * rsq * sin(theta) * sin(theta) }}
392  }};
393  export_utils::basis_transform_spherical_tofrom_cart::matrix_t KDD_Sph = {{
394  {{ 0.0, 0.0, -gammaDD_Sph[2][2] / 2.0 / N * domega_dr }},
395  {{ 0.0, 0.0, -gammaDD_Sph[2][2] / 2.0 / N * domega_dt }},
396  {{ 0.0, 0.0, 0.0}}
397  }};
398  // clang-format on
399 
400  auto const gammaDD_Cart =
401  basis_transform.matrixDD__sph_to_cart(gammaDD_Sph);
402  auto const KDD_Cart = basis_transform.matrixDD__sph_to_cart(KDD_Sph);
403  auto const betaU_Cart = basis_transform.vectorU__sph_to_cart(betaU_Sph);
404  auto const FluidVelU_Cart =
405  basis_transform.vectorU__sph_to_cart(FluidVelU_Sph);
406  base.out_pw[OUTPUT_VARS::BETA1] = betaU_Cart[0];
407  base.out_pw[OUTPUT_VARS::BETA2] = betaU_Cart[1];
408  base.out_pw[OUTPUT_VARS::BETA3] = betaU_Cart[2];
409  base.out_pw[OUTPUT_VARS::G11] = gammaDD_Cart[0][0];
410  base.out_pw[OUTPUT_VARS::G12] = gammaDD_Cart[0][1];
411  base.out_pw[OUTPUT_VARS::G13] = gammaDD_Cart[0][2];
412  base.out_pw[OUTPUT_VARS::G22] = gammaDD_Cart[1][1];
413  base.out_pw[OUTPUT_VARS::G23] = gammaDD_Cart[1][2];
414  base.out_pw[OUTPUT_VARS::G33] = gammaDD_Cart[2][2];
415  base.out_pw[OUTPUT_VARS::K11] = KDD_Cart[0][0];
416  base.out_pw[OUTPUT_VARS::K12] = KDD_Cart[0][1];
417  base.out_pw[OUTPUT_VARS::K13] = KDD_Cart[0][2];
418  base.out_pw[OUTPUT_VARS::K22] = KDD_Cart[1][1];
419  base.out_pw[OUTPUT_VARS::K23] = KDD_Cart[1][2];
420  base.out_pw[OUTPUT_VARS::K33] = KDD_Cart[2][2];
421  base.out_pw[OUTPUT_VARS::VEL1] = FluidVelU_Cart[0];
422  base.out_pw[OUTPUT_VARS::VEL2] = FluidVelU_Cart[1];
423  base.out_pw[OUTPUT_VARS::VEL3] = FluidVelU_Cart[2];
424 
425  double const H = base.quant_vals[ISO_VARS::ISO_H];
426  double h = std::exp(H);
427  double rho, eps, press;
428 
429  // get quantities point-wise, since h is smoothest, and cut data at H=0
430  if (std::fabs(H) <= 1e-14) {
431  rho = 0.;
432  eps = 0.;
433  press = 0.;
434  base.out_pw[OUTPUT_VARS::VEL1] = 0.;
435  base.out_pw[OUTPUT_VARS::VEL2] = 0.;
436  base.out_pw[OUTPUT_VARS::VEL3] = 0.;
437  } else {
438  rho = EOS<eos_t, DENSITY>::get(h);
439  eps = EOS<eos_t, EPSILON>::get(h);
440  press = EOS<eos_t, PRESSURE>::get(h);
441  }
442  base.out_pw[OUTPUT_VARS::ALPHA] = base.quant_vals[ISO_VARS::ISO_ALPHA];
443  base.out_pw[OUTPUT_VARS::RHO] = rho;
444  base.out_pw[OUTPUT_VARS::EPS] = eps;
445  base.out_pw[OUTPUT_VARS::PRESS] = press;
446  base.output_base = OUTPUT_BASIS::CARTESIAN;
447  return base.out_pw;
448  }
449  };

491  template <class eos_t>
492  struct export_pointwise__spherical_imp {
493  friend CFMS_NS_ISO_Exporter;
494 
495  output_ary_t operator()(CFMS_NS_ISO_Exporter& base,
496  double const& x,
497  double const& y,
498  double const& z) {
499  // Reset to NAN
500  base.quant_vals.fill(NAN);
501  base.quant_vals = base.interpolate_pointwise(x, y, z);
502 
503  const double r2 = x * x + y * y + z * z;
504  const double r = std::sqrt(r2);
505 
506  auto get_theta = [&]() {
507  double theta;
508  if (r2 < 1e-12) {
509  theta = acos(z / 1e-10); // theta (angle from Z to xy plane)
510  } else {
511  theta = acos(z / r); // theta (angle from Z to xy plane)
512  }
513  return theta;
514  };
515  const double theta = get_theta();
516  const double sint = std::sin(theta);
517 
518  const double domega_dr = base.quant_vals[ISO_VARS::ISO_DOMEGA_DR];
519  const double domega_dt = base.quant_vals[ISO_VARS::ISO_DOMEGA_DTHETA];
520 
521  const double N = base.quant_vals[ISO_VARS::ISO_ALPHA];
522  const double A = base.quant_vals[ISO_VARS::ISO_METRIC_A];
523  const double B = base.quant_vals[ISO_VARS::ISO_METRIC_B];
524 
525  base.out_pw[OUTPUT_VARS::ALPHA] = N;
526 
527  base.out_pw[OUTPUT_VARS::BETA1] = 0.0;
528  base.out_pw[OUTPUT_VARS::BETA2] = 0.0;
529  base.out_pw[OUTPUT_VARS::BETA3] =
530  -base.quant_vals[ISO_VARS::ISO_METRIC_OMEGA];
531 
532  base.out_pw[OUTPUT_VARS::G11] = A * A;
533  base.out_pw[OUTPUT_VARS::G12] = 0.0;
534  base.out_pw[OUTPUT_VARS::G13] = 0.0;
535  base.out_pw[OUTPUT_VARS::G22] = A * A * r2;
536  base.out_pw[OUTPUT_VARS::G23] = 0.0;
537  base.out_pw[OUTPUT_VARS::G33] = B * B * r2 * sint * sint;
538  base.out_pw[OUTPUT_VARS::K11] = 0.0;
539  base.out_pw[OUTPUT_VARS::K12] = 0.0;
540  base.out_pw[OUTPUT_VARS::K13] =
541  -base.out_pw[OUTPUT_VARS::G33] / 2.0 / N * domega_dr;
542  base.out_pw[OUTPUT_VARS::K22] = 0.0;
543  base.out_pw[OUTPUT_VARS::K23] =
544  -base.out_pw[OUTPUT_VARS::G33] / 2.0 / N * domega_dt;
545  base.out_pw[OUTPUT_VARS::K33] = 0.0;
546 
547  double const H = base.quant_vals[ISO_VARS::ISO_H];
548  double h = std::exp(H);
549  double rho, eps, press, vphiU;
550 
551  // get quantities point-wise, since h is smoothest, and cut data at H=0
552  if (std::fabs(H) <= 1e-14) {
553  rho = 0.;
554  eps = 0.;
555  press = 0.;
556  vphiU = 0.;
557  } else {
558  rho = EOS<eos_t, DENSITY>::get(h);
559  eps = EOS<eos_t, EPSILON>::get(h);
560  press = EOS<eos_t, PRESSURE>::get(h);
561  vphiU = base.quant_vals[ISO_VARS::ISO_U];
562  }
563  base.out_pw[OUTPUT_VARS::RHO] = rho;
564  base.out_pw[OUTPUT_VARS::EPS] = eps;
565  base.out_pw[OUTPUT_VARS::PRESS] = press;
566  base.out_pw[OUTPUT_VARS::VEL1] = 0.0;
567  base.out_pw[OUTPUT_VARS::VEL2] = 0.0;
568  base.out_pw[OUTPUT_VARS::VEL3] = vphiU;
569  base.output_base = OUTPUT_BASIS::SPHERICAL;
570  return base.out_pw;
571  }
572  };
