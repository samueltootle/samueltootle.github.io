40 void Metric_dirac_const::set_system (System_of_eqs& ss, const char* name_met) {
41 
42  syst = &ss ;
43 
44  if (ss.met!=0x0) {
45  cerr << "Metric already set for the system" << endl ;
46  abort() ;
47  }
48 
49  // Position in the system :
50  place_syst = ss.ndom*ss.ncst ;
51 
52  ss.add_cst (0x0, *p_met) ;
53 
54  ss.met = this ;
55  ss.name_met = new char[LMAX] ;

61 void Metric_dirac_const::compute_cov (int dd) const {
62 
63  int dim = espace.get_ndim() ;
64  if (dim!=3) {
65  cerr << "Function only implemented for dimension 3" << endl ;
66  abort() ;
67  }
68 
69  int place = place_syst + (dd-syst->dom_min) ;
70 
71  // Right storage : simple copy.
72  if (type_tensor==COV) {
73 
74  if (p_met_cov[dd]==0x0)
75  p_met_cov[dd] = new Term_eq(*syst->cst[place]) ;
76  else
77  *p_met_cov[dd] = Term_eq(*syst->cst[place]) ;
78  }
79  else {
80  Term_eq** res = new Term_eq* [p_met->get_n_comp()] ;
81 
82 
83  Scalar val (espace) ;
84  Val_domain cmpval(espace.get_domain(dd)) ;
85 
86  // Compo 1 1
87  cmpval = (*syst->cst[place]->val_t)(2,2)(dd)*(*syst->cst[place]->val_t)(3,3)(dd)
88  -(*syst->cst[place]->val_t)(2,3)(dd)*(*syst->cst[place]->val_t)(2,3)(dd) ;
89  val.set_domain(dd) = cmpval ;
90  res[0] = new Term_eq (dd, val) ;
91 
92 
93  // Compo 1 2
94  cmpval = (*syst->cst[place]->val_t)(1,3)(dd)*(*syst->cst[place]->val_t)(2,3)(dd)
95  -(*syst->cst[place]->val_t)(1,2)(dd)*(*syst->cst[place]->val_t)(3,3)(dd) ;
96  val.set_domain(dd) = cmpval ;
97  res[1] = new Term_eq (dd, val) ;
98 
99  // Compo 1 3
100  cmpval = (*syst->cst[place]->val_t)(1,2)(dd)*(*syst->cst[place]->val_t)(2,3)(dd)
101  -(*syst->cst[place]->val_t)(1,3)(dd)*(*syst->cst[place]->val_t)(2,2)(dd) ;
102  val.set_domain(dd) = cmpval ;
103  res[2] = new Term_eq (dd, val) ;
104 
105 
106  // Compo 2 2
107  cmpval = (*syst->cst[place]->val_t)(1,1)(dd)*(*syst->cst[place]->val_t)(3,3)(dd)
108  -(*syst->cst[place]->val_t)(1,3)(dd)*(*syst->cst[place]->val_t)(1,3)(dd) ;
109  val.set_domain(dd) = cmpval ;
110  res[3] = new Term_eq (dd, val) ;
111 
112  // Compo 2 3
113  cmpval = (*syst->cst[place]->val_t)(1,3)(dd)*(*syst->cst[place]->val_t)(1,2)(dd)
114  -(*syst->cst[place]->val_t)(1,1)(dd)*(*syst->cst[place]->val_t)(2,3)(dd) ;
115  val.set_domain(dd) = cmpval ;
116  res[4] = new Term_eq (dd, val) ;
117 
118 
119  // Compo 3 3
120  cmpval = (*syst->cst[place]->val_t)(1,1)(dd)*(*syst->cst[place]->val_t)(2,2)(dd)
121  -(*syst->cst[place]->val_t)(1,2)(dd)*(*syst->cst[place]->val_t)(1,2)(dd) ;
122  val.set_domain(dd) = cmpval ;
123  res[5] = new Term_eq (dd, val) ;
124 
125 
126  // Value field :
127  Metric_tensor resval (espace, COV, basis) ;
128  resval.set(1,1) = res[0]->get_val_t() ;
129  resval.set(1,2) = res[1]->get_val_t() ;
130  resval.set(1,3) = res[2]->get_val_t() ;
131  resval.set(2,2) = res[3]->get_val_t() ;
132  resval.set(2,3) = res[4]->get_val_t() ;
133  resval.set(3,3) = res[5]->get_val_t() ;
134 
135  Metric_tensor zero (espace, COV, basis) ;
136  for (int i=1 ; i<=3 ; i++)
137  for (int j=i ; j<=3 ; j++)
138  zero.set(i,j) = 0 ;
139 
140  if (p_met_cov[dd]==0x0)
141  p_met_cov[dd] = new Term_eq(dd, resval, zero) ;
142  else
143  *p_met_cov[dd] = Term_eq(dd, resval, zero) ;
144  for (int i=0 ; i<p_met->get_n_comp() ; i++)
145  delete res [i] ;
146  delete [] res ;
147  }
148 }
