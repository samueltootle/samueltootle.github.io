111  void set(double _val, double _init, double _final) {
112  parameter_val = _val;
113  seqinit = _init;
114  seqfinal = _final;
115 
118  if(is_set()) {
119  if (seqinit < seqfinal)
120  conditional = std::less_equal<double>{};
121  else
122  conditional = std::greater_equal<double>{};
123  }
124  }
125 
127  void set_N(int _N) {
128  assert(N > 0);
129  N = _N;
130  }
131 
133  double step_size() const {
134  if(is_set())
135  return (seqfinal-seqinit) / N;
136  return std::nan("1");
137  }
138 
140  int const & iterations() const { return N; }
141  std::tuple<Ts...> const & get_indices() const { return parameter_indices; }
143  bool loop_condition(double const & val) const { return conditional(val, seqfinal); }
145  friend std::ostream &operator<<(std::ostream &, const BIN_INFO &);
146 };
