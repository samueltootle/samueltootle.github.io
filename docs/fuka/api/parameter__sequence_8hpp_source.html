131  void set(const double _val, const double _init, const double _final) {
132  parameter_val = _val;
133  seqinit = _init;
134  seqfinal = _final;
135 
138  if (is_set()) {
139  parameter_val = seqinit;
140  if (seqinit < seqfinal)
141  conditional = std::less_equal<double>{};
142  else
143  conditional = std::greater_equal<double>{};
144  }
145  }
146 
148  void set_N(int _N) {
149  assert(N > 0);
150  N = _N;
151  }
152 
154  double step_size() const {
155  if (is_set())
156  return (seqfinal - seqinit) / N;
157  return std::nan("1");
158  }
159 
161  int const& iterations() const { return N; }
162 
163  std::tuple<Ts...> const& get_indices() const { return parameter_indices; }
164 
166  bool loop_condition(double const& val) const {
167  return conditional(val, seqfinal);
168  }
169 
171  friend std::ostream& operator<< <Ts...>(std::ostream&,
172  const Parameter_sequence<Ts...>&);
173 };
