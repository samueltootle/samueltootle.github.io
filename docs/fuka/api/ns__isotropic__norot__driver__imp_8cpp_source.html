94 template <class config_t, class Res_t>
95 inline int ns_isotropic_norot_driver(config_t& bconfig,
96  Res_t& resolution,
97  std::string outputdir,
98  ns_sequence const* seq) {
99  int exit_status = RELOAD_FILE;
100  int rank = 0;
101  MPI_Comm_rank(MPI_COMM_WORLD, &rank);
102 
103  bool res_inc = (resolution.final() > resolution.init() ||
104  bconfig.control(CONTROLS::REGRID));
105  auto resolution_indices = resolution.get_indices();
106  auto const& final_res = resolution.final();
107 
108  std::array<bool, NUM_STAGES>& stage_enabled = bconfig.return_stages();
109  auto [last_stage, last_stage_idx] = get_last_enabled(MSTAGE, stage_enabled);
110 
111  exit_status = ns_isotropic_norot_stationary_driver(bconfig, outputdir, seq);
112  // We now have a "low" resolution solution for the NS of interest
113  // Set this to false to avoid iterative M and CHI
114  bconfig.control(CONTROLS::SEQUENCES) = false;
115 
116  auto regrid = [&]() {
117  std::string fname{"ns_regrid"};
118 
119  if (rank == 0)
120  exit_status = ns_isotropic_norot_regrid(bconfig, fname);
121  MPI_Barrier(MPI_COMM_WORLD);
122  bconfig.set_filename(fname);
123  bconfig.open_config();
124 
125  stage_enabled[STAGES::NOROT_BC] = true;
126  };
127  // Since the 2D code focuses on sequences, we always regrid to make sure
128  // we start/end on an optimal grid structure.
129  regrid();
130  exit_status = ns_isotropic_norot_stationary_driver(bconfig, outputdir, seq);
131 
132  while (res_inc) {
133  int next_res = bco_utils::next_resolution(bconfig(resolution_indices));
134  // iterative res increase
135  if (bconfig.control(CONTROLS::REGRID)) {
136  bconfig.control(CONTROLS::REGRID) = false;
137  res_inc = (resolution.final() > resolution.init());
138  } else if (next_res >= final_res) {
139  bconfig.set(resolution_indices) = next_res;
140  res_inc = false;
141  } else {
142  bconfig.set(resolution_indices) = next_res;
143  }
144  regrid();
145 
146  exit_status = ns_isotropic_norot_stationary_driver(bconfig, outputdir, seq);
147  }
148  return exit_status;
149 }
