71 // Increment
72 inline bool Index::inc (int increm, int var) {
73  bool res = ((var >=0) && (var<get_ndim())) ? true : false ;
74  if (res) {
75  coord[var] += increm ;
76  for (int i=var ; i<get_ndim()-1 ; i++) {
77  div_t division = div(coord[i], sizes(i)) ;
78  coord[i] = division.rem ;
79  coord[i+1] += division.quot ;
80  }
81  res = (coord[get_ndim()-1] >= sizes(get_ndim()-1)) ? false : true ;
82  }
83  return res ;
84 }
