52 template <class bconfig_t, typename... idx_t>
53 boost::python::dict build_config_dict(bconfig_t& bconfig, idx_t... bco) {
54  auto map = bconfig.get_map(bco...);
55  boost::python::dict config;
56  for (auto& tup : map) {
57  auto name{tup.first};
58  auto index{tup.second};
59  if (!std::isnan(bconfig.set(index, bco...)))
60  config[name] = bconfig(index, bco...);
61  }
62  return config;
63 }

74 template <class bconfig_t, typename... idx_t>
75 void add_eos_todict(bconfig_t& bconfig,
76  boost::python::dict& config,
77  idx_t... bco) {
78  // map of <std::string, enum>
79  auto map = bconfig.get_eos_map(bco...);
80  for (auto& tup : map) {
81  auto name{tup.first};
82  auto index{tup.second};
83 
84  auto assign_val = [&](auto&& v) {
85  using v_t = std::decay_t<decltype(v)>;
86  if constexpr (std::is_same_v<v_t, double>) {
87  if (!std::isnan(v))
88  config[name] = v;
89  } else {
90  config[name] = v;
91  }
92  };
93 
94  // EOS Params are stored in a std::variant
95  std::visit(assign_val, bconfig.set_eos(index, bco...));
96  }
97  return;
98 }

109 template <class bconfig_t, typename... idx_t>
110 void add_diffrot_todict(bconfig_t& bconfig,
111  boost::python::dict& config,
112  idx_t... bco) {
113  // map of <std::string, enum>
114  boost::python::dict diffrot;
115  auto map = bconfig.get_diffrot_map(bco...);
116  for (auto& tup : map) {
117  auto name{tup.first};
118  auto index{tup.second};
119 
120  auto assign_val = [&](auto&& v) {
121  using v_t = std::decay_t<decltype(v)>;
122  if constexpr (std::is_same_v<v_t, double>) {
123  if (!std::isnan(v))
124  diffrot[name] = v;
125  } else {
126  diffrot[name] = v;
127  }
128  };
129 
130  // Differential Rotation Params are stored in a std::variant
131  std::visit(assign_val, bconfig.set_diffrot(index, bco...));
132  }
133  if (boost::python::len(diffrot) > 0)
134  config["differential_rotation"] = diffrot;
135  return;
136 }

143 template <class config_t>
144 struct Configurator_reader_t {
145  Configurator_reader_t(std::string const filename_)
146  : filename(filename_), bconfig(config_t(filename)) {
147  add_dict();
148  }
149 
150  // Keep this public to make it easy to access
151  boost::python::dict config;
152 
153  protected:
161  template <bool NS = false, typename... idx_t>
162  void add_dict(idx_t... bco) {
163  std::string dict_name{bconfig.get_name_string(bco...)};
164  boost::python::dict tmp = build_config_dict(bconfig, bco...);
165  if constexpr (NS)
166  add_eos_todict(bconfig, tmp, bco...);
167  config[dict_name] = tmp;
168  };
169 
170  std::string filename{};
171  config_t bconfig;
172 };

203 struct ns_Configurator_reader_t : Configurator_reader_t<ns_config_t> {
204  ns_Configurator_reader_t(std::string const filename_)
205  : Configurator_reader_t(filename_) {
206  // The base class only extracts the object parameters
207  // Here we extract the dictionary and add the EOS information

209  boost::python::dict dict =
210  boost::python::extract<boost::python::dict>(config[dict_name]);
211 
212  add_eos_todict(bconfig, dict);
213  add_diffrot_todict(bconfig, dict);
214  config[dict_name] = dict;
215  }
