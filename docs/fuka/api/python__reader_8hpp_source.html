57 // general reader class to be used within python
58 // fills the internal dictionaries using a quantities type
59 template<typename space_t, typename vars_t>
60 class python_reader_t {
61  protected:
62  // the file from which gets read
63  FILE * file;
64 
65  // the space on which everything is defined
66  space_t space;
67 
68  public:
69  boost::python::dict vars;
70  boost::python::dict config;
71 
72  // general construcor, filling the dicts
73  python_reader_t(std::string const & filename)
74  : file(fopen(filename.c_str(), "r")), space(file) {
75  for(auto q : vars_t::vars) {
76  switch(q.second) {
77  case CONSTANT:
78  // read in constant, i.e. single scalars
79  double tmp;
80  fread_be(&tmp, sizeof(double), 1, file);
81  vars[q.first] = tmp;
82  break;
83  case SCALAR:
84  // read in scalar field
85  vars[q.first] = Scalar(space, file);
86  break;
87  case VECTOR:
88  // read in vector field
89  vars[q.first] = Vector(space, file);
90  break;
91  }
92  }
93  // close the file again, we don't need it anymore
94  fclose(file);
95  }

107  // point-wise tensor field exporter interface for python
108  boost::python::list getFieldValues(std::string const & fieldname, boost::python::list const & coord_list, int bound = -1) {
109  // list of values to return
110  boost::python::list values;
111 
112  // extract field as a general tensor object
113  Tensor const & tensor = extractField<Tensor const &>(fieldname);
114 
115  // get number of dimensions and construct index
116  int ndim = tensor.get_ndim();
117  auto ind = Index(Dim_array(ndim));
118 
119  // loop through all given coords
120  for(int i = 0; i < boost::python::len(coord_list); ++i) {
121  // extract coords
122  boost::python::list coords = boost::python::extract<boost::python::list>(coord_list[i]);
123 
124  // check if we got the expected number of coords
125  if(boost::python::len(coords) != ndim) {
126  std::cerr << "ERROR: Got wrong number of coordinates, expecting exactly " << ndim << std::endl;
127  return values;
128  }
129 
130  // construct Kadath point
131  Point abs_coords(ndim);
132  for(int j = 0; j < ndim; ++j) {
133  abs_coords.set(j+1) = boost::python::extract<double>(coords[j]);
134  }
135 
136  // add values to output list
137  if(bound == -1) {
138  if(tensor.get_valence() == 0)
139  values.append(tensor(ind).val_point(abs_coords));
140  else {
141  boost::python::list comps;
142  do {
143  comps.append(tensor(ind).val_point(abs_coords));
144  } while(ind.inc());
145  values.append(comps);
146  }
147  }
148  else {
149  if(tensor.get_valence() == 0)
150  values.append(tensor(ind).val_point_bound(abs_coords,bound));
151  else {
152  boost::python::list comps;
153  do {
154  comps.append(tensor(ind).val_point_bound(abs_coords,bound));
155  } while(ind.inc());
156  values.append(comps);
157  }
158  }
159  }
160  return values;
161  }
162 
163 };

165 template<class eos_t>
166 struct PygetEOSValues {
167  // point-wise EOS exporter interface for python
168  // boost::python::list getEOSValues(boost::python::list const & coord_list) {
169  template<class reader_t>
170  boost::python::list operator()(boost::python::list const & coord_list, reader_t * this_reader) {
171  auto logh_values = this_reader->getFieldValues("logh", coord_list);
172 
173  // list of EOS values to return
174  boost::python::list values;
175 
176  for(int i = 0; i < len(coord_list); ++i) {
177  double h = std::exp(boost::python::extract<double>(logh_values[i]));
178 
179  boost::python::list eos_tuple;
180 
181  eos_tuple.append(EOS<eos_t, DENSITY>::get(h));
182  eos_tuple.append(EOS<eos_t, EPSILON>::get(h));
183  eos_tuple.append(EOS<eos_t, PRESSURE>::get(h));
184 
185  values.append(eos_tuple);
186  }
187 
188  return values;
189  }

192 template<typename space_t>
193 void initPythonBinding() {
194  using namespace boost::python;
195 
196  // this is needed, so that python lets us read in the fields
197  class_<Kadath::var_t>("KadathVarType");
198  // this is needed, so that python understands what the field types are
199  class_<Kadath::Tensor>("KadathTensorType", init<Kadath::Tensor const &, bool>());
200  class_<Kadath::Scalar, bases<Kadath::Tensor>>("KadathScalarType", init<space_t const &, FILE*>());
201  class_<Kadath::Vector, bases<Kadath::Tensor>>("KadathVectorType", init<space_t const &, FILE*>());
202 }

205 template<typename reader_t>
206 void constructPythonReader(std::string reader_name) {
207  using namespace boost::python;
208 
209  auto reader = class_<reader_t>(reader_name.c_str(), init<std::string>());
210  reader.def("getFieldValues", &reader_t::getFieldValues);
211  reader.def("getEOSValues", &reader_t::getEOSValues);
212  reader.def_readonly("vars", &reader_t::vars);
213  reader.def_readonly("config", &reader_t::config);
