54  base_config = seqconfig;
55  base_config.set(BCO_PARAMS::CHI) = 0;
56  }
57  auto mass_fixing = seq.mass_idx();
58  auto spin_fixing = seq.spin_idx();
59 
60  // Should be deprecated...
61  if (seq.is_set() && std::isnan(base_config.set(sequence_var_indices))) {
62  base_config.set(sequence_var_indices) = seq.init();
63  if (ns_seq_is_mass_fixing(seq))
64  mass_fixing = std::get<0>(sequence_var_indices);
65  }
66 
67  // Save this in case an invalid ADM mass is given for the EOS used
68  const double final_MADM = (seq.mass_idx() == BCO_PARAMS::MADM)
69  ? base_config(BCO_PARAMS::MADM)
70  : std::nan("1");
71  base_config.control(CONTROLS::ITERATIVE_M) = false;
72  config_t bconfig{base_config};
73 
74  const std::string eos_type =
75  bconfig.template eos<std::string>(EOS_PARAMS::EOSTYPE);
76  // Not tested...
77  if (bconfig.control(CONTROLS::SEQUENCES) ||
78  bconfig.control(CONTROLS::RESOLVE)) {
79  int n_shells = (std::isnan(bconfig.set(BCO_PARAMS::NSHELLS)))
80  ? 0
81  : bconfig.set(BCO_PARAMS::NSHELLS);
82  // the 2D code is very sensitive to the initial domain decomposition.
83  // Although polytropic laws are very stable, tabulated solution are more
84  // sensitive.
85  bconfig.set(BCO_PARAMS::NSHELLS) = 0.;
86  if (rank == 0) {

88  bconfig, eos_type, bconfig, mass_fixing);
89  }

91  // make sure all ranks have the same config
92  bconfig.open_config();
93  bconfig.set(BCO_PARAMS::NSHELLS) = n_shells;
94 
95  bconfig.control(CONTROLS::ITERATIVE_M) =
96  !std::isnan(final_MADM) &&
97  (std::fabs(1. - bconfig(BCO_PARAMS::MADM) / final_MADM) > 1e-3);
98 
99  if (bconfig.control(CONTROLS::ITERATIVE_M) &&
100  std::fabs(bconfig(spin_fixing)) < 1e-3) {
101  if (rank == 0)
102  std::cerr << "Cannot solve TOV for Madm = " << final_MADM
103  << " without spin. " << bconfig(BCO_PARAMS::MADM) << '\n';
104  std::_Exit(EXIT_FAILURE);
105  }
106  bconfig.control(CONTROLS::SEQUENCES) = true;
107  }
108 
109  // Need to get a rotating solution before increasing the

111  std::array<bool, NUM_STAGES>& stage_enabled = bconfig.return_stages();
112  auto [last_stage, last_stage_idx] = get_last_enabled(MSTAGE, stage_enabled);
113  if (bconfig.control(CONTROLS::ITERATIVE_M)) {
114  // Only obtain the iterative solution at the initial_resolution
115  auto const res_init{resolution.init()};
116  Parameter_sequence tmp_res("res", BCO_PARAMS::BCO_RES);
117  tmp_res.set(res_init, res_init, res_init);
118 
119  exit_status = ns_isotropic_base_solution_driver(bconfig, outputdir, &seq);
120 
121  // Update config such that the next solving round uses
122  // the final ADM mass and spin
123  bconfig(BCO_PARAMS::MADM) = final_MADM;
124  bconfig.control(CONTROLS::SEQUENCES) = false;
125  }
126  exit_status = ns_isotropic_base_solution_driver(bconfig, outputdir, &seq);
127 
128  // Ensure only the final stage is used
129  // e.g. avoid NOROT stage
130  stage_enabled.fill(false);
131  stage_enabled[last_stage_idx] = true;
132 
133  base_config = bconfig;
134  if (seq.is_set() && std::isnan(bconfig.set(sequence_var_indices)))
135  bconfig.set(sequence_var_indices) = seq.init();
136 #ifdef DEBUG
137  std::cout << seq << std::endl;
138  std::cout << resolution << std::endl;
139 #endif
140  auto single_seq = [&](auto val) {
141  stage_enabled[last_stage_idx] = true;
142  auto old_val = bconfig(sequence_var_indices);
143  if (seq.is_set())
144  bconfig.set(sequence_var_indices) = val;
145  if (std::get<0>(sequence_var_indices) == BCO_PARAMS::CHI) {
146  bconfig.seq_setting(SEQ_SETTINGS::FINAL_CHI) = bconfig(BCO_PARAMS::CHI);
147  if (std::fabs(old_val) < 0.1 && val > 0.2) {
148  bconfig.control(CONTROLS::SEQUENCES) = true;
149  exit_status = ns_isotropic_driver(bconfig, resolution, outputdir, &seq);
150  }
151  }
152  exit_status = ns_isotropic_driver(bconfig, resolution, outputdir, &seq);
153  resolution.set(resolution.final(), resolution.final(), resolution.final());
154  return exit_status;
155  };
156 
157  // Loop if a valid sequence is set
158  if (seq.is_set() && std::fabs(dx) > 0) {
159  for (auto val = seq.init(); seq.loop_condition(val); val += dx) {
160  exit_status = single_seq(val);
161  }
162  } else if (seq.is_set()) {
163  exit_status = single_seq(seq.init());
164  } else {
165  exit_status = single_seq(0);
166  }
167  return bconfig;
168 }

178 template <typename config_t>
179 int ns_isotropic_base_solution_driver(config_t& bconfig,
180  std::string outputdir,
181  ns_sequence const* seq = nullptr) {
182  int exit_status = RELOAD_FILE;
183  int rank = 0;
184  MPI_Comm_rank(MPI_COMM_WORLD, &rank);
185 
186  if (std::abs(bconfig(BCO_PARAMS::CHI)) > 0.7) {
187  std::cerr << "Unable to handle chi > 0.7\n";
188  return EXIT_FAILURE;
189  }
190  // We need to store the final desired CHI in case of iterative chi
191  bconfig.seq_setting(SEQ_SETTINGS::FINAL_CHI) = bconfig(BCO_PARAMS::CHI);
192 
193  // In the event we wish to solve for a highly spinning solution
194  // we need to do an initial slow rotating solution before going to
195  // faster rotations otherwise the solution will diverge.
196  bconfig.control(CONTROLS::ITERATIVE_CHI) =
197  bconfig.control(CONTROLS::SEQUENCES) &&
198  std::fabs(bconfig.seq_setting(SEQ_SETTINGS::FINAL_CHI)) > 0.2;
199 
200  // Lower chi in case of iterative chi
201  bconfig(BCO_PARAMS::CHI) =
202  (bconfig.control(CONTROLS::ITERATIVE_CHI))
203  ? std::copysign(0.1, bconfig.seq_setting(SEQ_SETTINGS::FINAL_CHI))
204  : bconfig(BCO_PARAMS::CHI);
205 
206  // make sure NS directory exists for outputs
207  if (outputdir == "./") {
208  fs::path cwd = fs::current_path();
209  outputdir = cwd.string();
210  }
211  if (rank == 0)
212  std::cout << "Solutions will be stored in: " << outputdir << "\n"
213  << "Directory will be created if it doesn't exist.\n";
214  fs::create_directory(outputdir);
215 
216  // if(std::isnan(bconfig.set(BCO_PARAMS::MADM)) &&
217  // std::isnan(bconfig.set(BCO_PARAMS::MB))){
218  // if(rank == 0)
219  // std::cout << "Config error. No madm nor mb found. \n\n";
220  // std::_Exit(EXIT_FAILURE);
221  // }
222 
223  std::string spacein = bconfig.space_filename();
224  if (!fs::exists(spacein)) {
225  // mainly for debugging MPI bugs
226  if (rank == 0) {
227  std::cerr << "File: " << spacein << " not found.\n\n";
228  } else {
229  std::cerr << "File: " << spacein << " not found for another rank.\n\n";
230  }
231  std::_Exit(EXIT_FAILURE);
232  }
233  std::array<bool, NUM_STAGES> stage_enabled = bconfig.return_stages();
234  auto [last_stage, last_stage_idx] = get_last_enabled(MSTAGE, stage_enabled);
235  while (exit_status == RELOAD_FILE) {
236  exit_status = EXIT_FAILURE;
237  if (stage_enabled[STAGES::NOROT_BC]) {
238  double const omega = bconfig.set(BCO_PARAMS::OMEGA);
239  double const chi = bconfig.set(BCO_PARAMS::CHI);
240  bconfig.set(BCO_PARAMS::OMEGA) = 0.;
241  bconfig.set(BCO_PARAMS::CHI) = 0.;
242 
243  exit_status =
244  ns_isotropic_norot_stationary_driver(bconfig, outputdir, seq);
245  stage_enabled[STAGES::NOROT_BC] = false;
246  bconfig.set(BCO_PARAMS::OMEGA) = omega;
247  bconfig.set(BCO_PARAMS::CHI) = chi;
248  if (last_stage_idx != STAGES::NOROT_BC) {
249  stage_enabled[STAGES::NOROT_BC] = false;
250  exit_status = RELOAD_FILE;
251  }
252  bconfig.return_stages() = stage_enabled;
253  } else if (stage_enabled[STAGES::UNIFORM_ROT]) {
254  exit_status =
255  ns_isotropic_uniform_rot_stationary_driver(bconfig, outputdir, seq);
256  if (last_stage_idx != STAGES::UNIFORM_ROT) {
257  stage_enabled[STAGES::UNIFORM_ROT] = false;
258  exit_status = RELOAD_FILE;
259  }
260  bconfig.return_stages() = stage_enabled;
261  } else if (stage_enabled[STAGES::DIFF_ROT]) {
262  exit_status =
263  ns_isotropic_diff_rot_stationary_driver(bconfig, outputdir, seq);
264  if (last_stage_idx != STAGES::DIFF_ROT) {
265  stage_enabled[STAGES::DIFF_ROT] = false;
266  exit_status = RELOAD_FILE;
267  }
268  bconfig.return_stages() = stage_enabled;
269  }
270  MPI_Barrier(MPI_COMM_WORLD);
271  }
272  return exit_status;
273 }
