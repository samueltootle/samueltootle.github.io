28 Val_domain Domain_shell::fitschwarz (const Val_domain& so, int dim) const {
29 
30  so.coef() ;
31  Val_domain dso (der_normal(so, OUTER_BC)) ;
32 
33  Val_domain res(this) ;
34  res.base = so.base ;
35  res.allocate_coef() ;
36 
37  double rmax = alpha + beta ;
38 
39  Index pos (nbr_coefs) ;
40 
41  // Loop on harmonic
42  for (int nn=0 ; nn<nbr_coefs(1) ; nn++) {
43 
44  pos.set(1) = nn ;
45 
46  double sh = (nn==0) ? 1./pow(rmax, dim-2) : 0 ;
47  double dsh = (nn==0) ? -double(dim-2)/pow(rmax, dim-1) : 1 ;
48 
49  // Affecte to res
50  for (int i=0 ; i<nbr_coefs(0) ; i++) {
51  pos.set(0) = i ;
52  res.set_coef(pos) = (nn==0) ? so.get_coef(pos)*dsh - dso.get_coef(pos)*sh : so.get_coef(pos);
53  }
54 
55  }
56  return res ;
57 }

60 // Term_eq version
61 Term_eq Domain_shell::fitschwarz (const Term_eq& so, int dim) const {
62 
63  // Check it is a tensor
64  if (so.get_type_data() != TERM_T) {
65  cerr << "fitschwarz only defined with respect for a tensor" << endl ;
66  abort() ;
67  }
68 
69  // Right domain ?
70  int dom = so.get_dom() ;
71  if (this != so.get_val_t().get_space().get_domain(dom)) {
72  cerr << "Domain mismatch in Domain_shell::fitschwarz (Term_eq version)" << endl ;
73  abort() ;
74  }
75 
76  Tensor resval (so.get_val_t(), false) ;
77  for (int i=0 ; i<so.get_val_t().get_n_comp() ; i++) {
78  Array<int> ind (so.get_val_t().indices(i)) ;
79  Val_domain value (so.get_val_t()(ind)(dom)) ;
80  if (value.check_if_zero())
81  resval.set(ind).set_domain(dom).set_zero() ;
82  else {
83  resval.set(ind).set_domain(dom) = fitschwarz(value, dim) ;
84  }
85  }
86 
87  if (so.get_p_der_t() !=0x0) {
88  Tensor resder (so.get_val_t(), false) ;
89  for (int i=0 ; i<so.get_der_t().get_n_comp() ; i++) {
90  Array<int> ind (so.get_der_t().indices(i)) ;
91  Val_domain valder (so.get_der_t()(ind)(dom)) ;
92  if (valder.check_if_zero())
93  resder.set(ind).set_domain(dom).set_zero() ;
94  else {
95  resder.set(ind).set_domain(dom) = fitschwarz(valder, dim) ;
96  }
97  }
98  return Term_eq (dom, resval, resder) ;
99  }
100  else return Term_eq (dom, resval) ;
101 }}
