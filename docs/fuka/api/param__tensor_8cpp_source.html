60 int add_m_quant (const Param_tensor* aa, const Param_tensor* bb) {
61  int res = 0 ;
62  if ((aa==0x0) || (bb==0x0))
63  res = 0 ;
64  else {
65  int val_a = (aa->m_quant_affected) ? aa->m_quant : 0 ;
66  int val_b = (bb->m_quant_affected) ? bb->m_quant : 0 ;
67  res = (val_a<val_b) ? val_a : val_b ;
68  }
69  return res ;
70 }

72 int mult_m_quant (const Param_tensor* aa, const Param_tensor* bb) {
73  int res = 0 ;
74  if ((aa==0x0) && (bb==0x0))
75  return 0 ;
76  if (aa==0x0) {
77  res = (bb->m_quant_affected) ? bb->m_quant : 0 ;
78  return res ;
79  }
80  if (bb==0x0) {
81  res = (aa->m_quant_affected) ? aa->m_quant : 0 ;
82  return res ;
83  }
84  int val_a = (aa->m_quant_affected) ? aa->m_quant : 0 ;
85  int val_b = (bb->m_quant_affected) ? bb->m_quant : 0 ;
86  return (val_a+val_b) ;
