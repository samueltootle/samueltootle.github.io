46 void Eq_first_integral::export_val (int& conte, Term_eq** residus, Array<double>& sec, int& pos_res) const {
47 
48  // Standard parts
49  assert (residus[conte]->get_type_data()==TERM_T) ;
50  // Recover pointer on the space
51  const Space& space = residus[conte]->get_val_t().get_space() ;
52 
53  // Get the constant part
54  Index pori (space.get_domain(dom_min)->get_nbr_points()) ;
55  double val_cst_part = residus[conte]->get_val_t()()(dom_min)(pori) ;
56  conte ++ ;
57 
58  // Recover origin value of first integral
59  double val_ori = residus[conte]->get_val_t()()(dom_min)(pori) ;
60 
61  // Inner domain
62  // Remove value at the origin
63  Scalar auximin (space) ;
64  auximin.set_domain(dom_min) = residus[conte]->get_val_t()()(dom_min) - val_ori ;
65  auximin.set_domain(dom_min).coef_i() ;
66 
67  // At the origin put the cst_part (no continuity but does it matter ?)
68  Index pos (space.get_domain(dom_min)->get_nbr_points()) ;
69  do {
70  if (pos(0)==0)
71  auximin.set_domain(dom_min).set(pos) = val_cst_part ;
72  }
73  while (pos.inc()) ;
74 
75  space.get_domain(dom_min)->export_tau (auximin, dom_min, 0, sec, pos_res, *n_cond) ;
76  conte ++ ;
77 
78  // Export all the other domains
79  for (int d=dom_min+1 ; d<=dom_max ; d++) {
80  Scalar auxi (space) ;
81  auxi.set_domain(d) = residus[conte]->get_val_t()()(d) - val_ori ;
82  space.get_domain(d)->export_tau (auxi, d, 0, sec, pos_res, *n_cond) ;
83  conte ++ ;
84  }
85 }
