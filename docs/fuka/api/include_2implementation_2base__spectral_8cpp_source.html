92 void Base_spectral::allocate(const Dim_array& nbr_coef) {
93 
94  for (int i=0 ; i<ndim ; i++)
95  if (bases_1d[i] != 0x0) delete bases_1d[i] ;
96 
97  bases_1d[ndim-1] = new Array<int>(1) ;
98  for (int i=ndim-2 ; i>=0 ; i--) {
99  Dim_array taille (ndim-1-i) ;
100  for (int k=0 ; k<ndim-1-i ; k++)
101  taille.set(k) = nbr_coef(i+k+1) ;
102  bases_1d[i] = new Array<int>(taille) ;
103  }
104 }

106 bool operator== (const Base_spectral& a, const Base_spectral& b) {
107  bool res = true ;
108  if ((!a.def) || (!b.def) || (a.ndim!= b.ndim))
109  res = false ;
110  for (int i=0 ; i<a.ndim ; i++) {
111  Index index (a.bases_1d[i]->get_dimensions()) ;
112  do if ((*a.bases_1d[i])(index) != (*b.bases_1d[i])(index))
113  res = false ;
114  while (index.inc()) ;
115  }
116  return res ;
117 }

134 void Base_spectral::set(Dim_array const& nbr_coefs, int BASEPHI, int BASETHETA, int BASER)
135 {
136

139  allocate(nbr_coefs);
