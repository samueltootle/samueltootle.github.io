27 Tensor Domain_shell_symphi::change_basis_spher_to_cart (int dd, const Tensor& so) const {
28 
29  // Lust start from spherical tensorial basis
30  if (so.get_basis().get_basis(dd) != SPHERICAL_BASIS) {
31  cerr << "The input tensorial basis must be spherical in Domain_shell_symphi::change_basis_spher_to_cart" << endl ;
32  abort() ;
33  }
34 
35  // Need to remove the symetry :
36  int val = so.get_valence() ;
37  Array<int> type_ind (so.get_index_type()) ;
38  Tensor res (so.get_space(), val, type_ind, so.get_basis()) ;
39 
40  if (so.is_name_affected()) {
41  res.set_name_affected() ;
42  for (int i=0 ; i<val ; i++)
43  res.set_name_ind (i, so.get_name_ind()[i]) ;
44  }
45  for (int i=0 ; i<res.get_n_comp() ; i++)
46  res.set(res.indices(i)).set_domain(dd) = so(res.indices(i))(dd) ;
47 
48  // Loop on the number of indices :
49  Dim_array dimother (so.get_valence()-1) ;
50  for (int i=0 ; i<so.get_valence()-1 ; i++)
51  dimother.set(i) = 3 ;
52 
53  for (int ind=0 ; ind<so.get_valence() ; ind++) {
54 
55 
56  Index posother (dimother) ;
57  do {
58 
59 
60  Index posr (so) ;
61  Index post (so) ;
62  Index posp (so) ;
63  int pos_inother= 0 ;
64  for (int conte=0 ; conte<so.get_valence() ; conte++) {
65  if (conte==ind) {
66  posr.set(conte) = 0 ;
67  post.set(conte) = 1 ;
68  posp.set(conte) = 2 ;
69  }
70  else {
71  posr.set(conte) = posother(pos_inother) ;
72  post.set(conte) = posother(pos_inother) ;
73  posp.set(conte) = posother(pos_inother) ;
74  pos_inother ++ ;
75  }
76  }
77 
78  Val_domain tmp (res(posr)(dd).mult_sin_theta() + res(post)(dd).mult_cos_theta()) ;
79  Val_domain vx (tmp.mult_cos_phi() - res(posp)(dd).mult_sin_phi()) ;
80  Val_domain vy (tmp.mult_sin_phi() + res(posp)(dd).mult_cos_phi()) ;
81  Val_domain vz (res(posr)(dd).mult_cos_theta() - res(post)(dd).mult_sin_theta()) ;
82  res.set(posr).set_domain(dd) = vx ;
83  res.set(post).set_domain(dd) = vy ;
84  res.set(posp).set_domain(dd) = vz ;
85  }
86  while (posother.inc()) ;
87  }
88 
89  res.set_basis(dd) = CARTESIAN_BASIS ;
90  return res ;
91 }
