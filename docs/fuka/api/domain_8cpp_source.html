194 Term_eq Domain::import (int numdom, int bound, int n_ope, Term_eq** parts) const {
195 
196  // get indices of domains :
197  Array<int> zedoms(n_ope) ;
198  for (int i=0 ; i<n_ope ; i++)
199  zedoms.set(i) = parts[i]->get_dom() ;
200 
201  // Get val part :
202  Tensor** parts_val = new Tensor* [n_ope] ;
203  for (int i=0 ; i<n_ope ; i++)
204  parts_val[i] = parts[i]->val_t ;
205 
206  Tensor res_val (import(numdom, bound, n_ope, zedoms, parts_val)) ;
207  delete [] parts_val ;
208 
209  // Do the derivative parts ?
210  bool doder = true ;
211  for (int i=0 ; i<n_ope ; i++)
212  if (parts[i]->der_t==0x0)
213  doder = false ;
214 
215  if (doder) {
216  Tensor** parts_der = new Tensor* [n_ope] ;
217  for (int i=0 ; i<n_ope ; i++)
218  parts_der[i] = parts[i]->der_t ;
219  Tensor res_der (import(numdom, bound, n_ope, zedoms, parts_der)) ;
220  delete [] parts_der ;
221  return Term_eq (numdom, res_val, res_der) ;
222  }
223  else
224  return Term_eq (numdom, res_val) ;
225 }

276 Term_eq Domain::integ_term_eq (const Term_eq& so, int bound) const {
277 
278  // Check it is a tensor
279  if (so.type_data != TERM_T) {
280  cerr << "integ_term_eq only defined with respect for a tensor" << endl ;
281  abort() ;
282  }
283 
284  if (so.val_t->get_n_comp() != 1) {
285  cerr << "integ_term_eq only defined with respect to a scalar" << endl ;
286  abort() ;
287  }
288 
289  assert (so.dom==num_dom) ;
290 
291 
292  // The value
293  Array<int> ind (so.val_t->indices(0)) ;
294  Val_domain value ((*so.val_t)(ind)(so.dom)) ;
295  double resval ;
296  if (value.check_if_zero())
297  resval= 0. ;
298  else
299  resval = integ(value, bound) ;
300 
301  if (so.der_t!=0x0) {
302  Val_domain value_der ((*so.der_t)(ind)(so.dom)) ;
303  double resder ;
304  if (value_der.check_if_zero())
305  resder = 0. ;
306  else
307  resder = integ(value_der, bound) ;
308  Term_eq res (so.dom, resval, resder) ;
309  return res ;
310  }
311  else {
312  Term_eq res (so.dom, resval) ;
313  return res ;

318 
319  // The value
320  Tensor resval (*target.val_t, false) ;
321 
322  for (int i=0 ; i<target.val_t->get_n_comp() ; i++) {
323  Array<int> ind (target.val_t->indices(i)) ;
324  Val_domain value ((*target.val_t)(ind)(num_dom)) ;
325  if (value.check_if_zero())
326  resval.set(ind).set_domain(num_dom).set_zero() ;
327  else {
328  Val_domain auxi ((this->*pfunc)(value, val)) ;
329  resval.set(ind).set_domain(num_dom) = auxi ;
330  }
331  if (target.get_val_t().get_parameters()!=0x0)
332  resval.set_parameters() = new Param_tensor(*target.get_val_t().get_parameters()) ;
333 
334  // affecte parameters

336  resval.set_name_affected() ;
337  for (int cmp=0 ; cmp<resval.get_valence() ; cmp++)
338  resval.name_indice[cmp] = target.get_val_t().name_indice[cmp] ;
339  }
340  }
341 
342  if (target.der_t!=0x0) {
343  Tensor resder (*target.der_t, false) ;
344  for (int i=0 ; i<target.der_t->get_n_comp() ; i++) {
345  Array<int> ind (target.der_t->indices(i)) ;

427 Term_eq Domain::partial_cart (const Term_eq& so) const {
428  int dom = so.get_dom() ;
429  int val_res = so.val_t->get_valence() + 1 ;
430  Array<int> type_ind (val_res) ;
431  type_ind.set(0) = COV ;
432  for (int i=1 ; i<val_res ; i++)
433  type_ind.set(i) = so.val_t->get_index_type(i-1) ;
434 
435  Base_tensor basis (so.get_val_t().get_space()) ;
436  basis.set_basis(dom) = CARTESIAN_BASIS ;
437 
438  // Tensor for val
439  Tensor auxi_val (so.get_val_t().get_space(), val_res, type_ind, basis) ;
440 
441  //Loop on the components :
442  Index pos_auxi(auxi_val) ;
443  Index pos_so (*so.val_t) ;
444  do {
445  for (int i=0 ; i<val_res-1 ; i++)
446  pos_so.set(i) = pos_auxi(i+1) ;
447  auxi_val.set(pos_auxi).set_domain(dom) = (*so.val_t)(pos_so)(dom).der_abs(pos_auxi(0)+1) ;
448  }
449  while (pos_auxi.inc()) ;
450 
451  if (so.der_t==0x0) {
452  return Term_eq (dom, auxi_val) ;
453  }
454  else {
455  // Need to compute the derivative :
456  // Tensor for der
457  Tensor auxi_der (so.get_val_t().get_space(), val_res, type_ind, basis) ;
458 
459  //Loop on the components :
460  Index pos_auxi_der(auxi_der) ;
461  do {
462  for (int i=0 ; i<val_res-1 ; i++)
463  pos_so.set(i) = pos_auxi_der(i+1) ;
464  auxi_der.set(pos_auxi_der).set_domain(dom) = (*so.der_t)(pos_so)(dom).der_abs(pos_auxi_der(0)+1) ;
465  }
466  while (pos_auxi_der.inc()) ;
467 
468  return Term_eq (dom, auxi_val, auxi_der) ;
469  }

472 Term_eq Domain::partial_spher (const Term_eq& so) const {
473 
474 
475 
476  int dom = so.get_dom() ;
477  int val_res = so.val_t->get_valence() + 1 ;
478  Array<int> type_ind (val_res) ;
479  type_ind.set(0) = COV ;
480  for (int i=1 ; i<val_res ; i++)
481  type_ind.set(i) = so.val_t->get_index_type(i-1) ;
482 
483  Base_tensor basis (so.get_val_t().get_space()) ;
484  basis.set_basis(dom) = SPHERICAL_BASIS ;
485 
486  // Tensor for val
487  Tensor auxi_val (so.get_val_t().get_space(), val_res, type_ind, basis) ;
488 
489  //Loop on the components :
490  Index pos_auxi(auxi_val) ;
491  Index pos_so (*so.val_t) ;
492  do {
493  for (int i=0 ; i<val_res-1 ; i++)
494  pos_so.set(i) = pos_auxi(i+1) ;
495  switch (pos_auxi(0)) {
496  case 0 :
497  // d/dr :
498  auxi_val.set(pos_auxi).set_domain(dom) = (*so.val_t)(pos_so)(dom).der_r() ;
499  break ;
500  case 1 :
501  // 1/r dtheta
502  auxi_val.set(pos_auxi).set_domain(dom) = (*so.val_t)(pos_so)(dom).der_var(2).div_r() ;
503  break ;
504  case 2 :
505  // 1/r sint d/dphi
506  auxi_val.set(pos_auxi).set_domain(dom) = (*so.val_t)(pos_so)(dom).der_var(3).div_r().div_sin_theta() ;
507  break ;
508  default :
509  cerr << "Bad indice in Domain::derive_partial_spher" << endl ;
510  abort() ;
511  }
512  }
513  while (pos_auxi.inc()) ;
514 
515  if (so.der_t==0x0) {
516  return Term_eq (dom, auxi_val) ;
517  }
518  else {
519  // Need to compute the derivative :
520  // Tensor for der
521  Tensor auxi_der (so.get_val_t().get_space(), val_res, type_ind, basis) ;
522 
523  //Loop on the components :
524  Index pos_auxi_der(auxi_der) ;
525  do {
526  for (int i=0 ; i<val_res-1 ; i++)
527  pos_so.set(i) = pos_auxi_der(i+1) ;
528  switch (pos_auxi_der(0)) {
529  case 0 :
530  // d/dr :
531  auxi_der.set(pos_auxi_der).set_domain(dom) = (*so.der_t)(pos_so)(dom).der_r() ;
532  break ;
533  case 1 :
534  // 1/r dtheta
535  auxi_der.set(pos_auxi_der).set_domain(dom) = (*so.der_t)(pos_so)(dom).der_var(2).div_r() ;
536  break ;
537  case 2 :
538  // 1/r sint d/dphi
539  auxi_der.set(pos_auxi_der).set_domain(dom) = (*so.der_t)(pos_so)(dom).der_var(3).div_r().div_sin_theta() ;
540  break ;
541  default :
542  cerr << "Bad indice in Domain::derive_partial_spher" << endl ;
543  abort() ;
544  }
545  }
546  while (pos_auxi_der.inc()) ;
547 
548  return Term_eq (dom, auxi_val, auxi_der) ;
549  }
