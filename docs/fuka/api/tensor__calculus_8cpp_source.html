23 Tensor Tensor::do_summation() const {
24  if (!name_affected) {
25  cerr << "Names of indices must be affected in Tensor::do_summation" << endl ;
26  abort() ;
27  }
28 
29  Array<int> sum_ind (valence) ;
30  Array<int> nsame (valence) ;
31  sum_ind = -1 ;
32  nsame = 0 ;
33  for (int i=0 ; i<valence ; i++)
34  for (int j=i+1 ; j<valence ; j++) {
35  if (name_indice[i]==name_indice[j]) {
36  nsame.set(i) ++ ;
37  nsame.set(j) ++ ;
38  if ((nsame(i)>1) || (nsame(j)>1)) {
39  cerr << "Too many identical indices in Tensor::do_summation" << endl ;
40  abort() ;
41  }
42  sum_ind.set(i) = j ;
43  sum_ind.set(j) = i ;
44  if (type_indice(i)==type_indice(j)) {
45  cerr << "Can not sum on indices of the same type in Tensor::do_summation" << endl ;
46  abort() ;
47  }
48  }
49  }
50 
51  int valence_res = 0 ;
52  for (int i=0 ; i<valence ; i++)
53  if (nsame(i)==0)
54  valence_res ++ ;
55 
56  // difference Scalar or not :
57  if (valence_res > 0) {
58  Array<int> type_ind_res (valence_res) ;
59  char* name_ind_res = new char[valence_res] ;
60  int conte = 0 ;
61  for (int i=0 ; i<valence ; i++)
62  if (nsame(i)==0) {
63  type_ind_res.set(conte) = type_indice(i) ;
64  name_ind_res[conte] = name_indice[i] ;
65  conte ++ ;
66  }
67 
68  Tensor res (espace, valence_res, type_ind_res, basis) ;
69  res.name_affected = true ;
70  for (int i=0 ; i<valence_res ; i++)
71  res.name_indice[i] = name_ind_res[i] ;
72  delete name_ind_res ;
73 
74  res = 0 ;
75  Array<bool> first (res.get_n_comp()) ;
76  first = true ;
77 
78  // Loop on the various components :
79  Index pos (*this) ;
80  Index pos_res (res) ;
81  do {
82  bool take = true ;
83  for (int i=0 ; i<valence ; i++)
84  if (pos(sum_ind(i))!=pos(i))
85  take = false ;
86 
87  if (take) {
88  conte = 0 ;
89  for (int i=0 ; i<valence ; i++)
90  if (nsame(i)==0) {
91  pos_res.set(conte)=pos(i) ;
92  conte ++ ;
93  }
94  int ind (res.position(pos_res)) ;
95  if (first(ind)) {
96  for (int dd=0 ; dd<espace.get_nbr_domains() ; dd++)
97  res.set(pos_res).set_domain(dd).set_base() = (*this)(pos)(dd).get_base() ;
98  first.set(ind) = false ;
99  }
100  res.set(pos_res) += (*this)(pos) ;
101  }
102  }
103  while (pos.inc()) ;
104  return res ;
105  }
106  else {
107  // Scalar case :
108  Scalar res(espace) ;
109  res = 0 ;
110  bool first = true ;
111  Index pos (*this) ;
112  do {
113  bool take = true ;
114  for (int i=0 ; i<valence ; i++)
115  if (pos(sum_ind(i))!=pos(i))
116  take = false ;
117 
118  if (take) {
119  if (first) {
120  for (int dd=0 ; dd<espace.get_nbr_domains() ; dd++)
121  res.set_domain(dd).set_base() = (*this)(pos)(dd).get_base() ;
122  first = false ;
123  }
124  res += (*this)(pos) ;
125  }
126  }
127  while (pos.inc()) ;
128  return res ;
129  }
130 }

132 Tensor Tensor::grad() const {
133 
134  // Verif triad (only implemented for cartesian coordinates so far)
135  for (int d=0 ; d<ndom ; d++)
136  assert (basis.get_basis(d)==CARTESIAN_BASIS) ;
137 
138  Array<int> indices_res (valence+1) ;
139  indices_res.set(0) = COV ;
140  for (int i=0 ; i<valence ; i++)
141  indices_res.set(i+1) = type_indice(i) ;
142  Tensor res (espace, valence+1, indices_res, basis) ;
143 
144  for (int i=0 ; i<ndim ; i++) {
145  indices_res.set(0) = i+1 ;
146  for (int j=0 ; j<n_comp ; j++) {
147  Array<int> cible(indices(j)) ;
148  for (int k=0 ; k<valence ; k++)
149  indices_res.set(k+1) = cible(k) ;
150  res.set(indices_res) = operator()(cible).der_abs(i+1) ;
151  }
152  }
153  return res ;
154 }}
