80 template <class config_t>
81 void Initialize_Solver<config_t>::init_solver(int argc, char** argv) {
82  // Necessary to get default construction
83  Initialize_Solver<config_t>::bconfig = config_t();
84  if (argc < 2) {
85  if (Initialize_Solver::rank == 0) {
86  std::cout << "INFO config file missing - generating a minimal setup"
87  << std::endl
88  << "Modify as needed before rerunning `solve "
89  << Initialize_Solver::input_configname << " <outputdir>`\n"
90  << "For a full setup, rerun with `solve full`\n";
91  }
92  Initialize_Solver::example_setup = Initialize_Solver::setup_first = true;
93  }
94 
95  // if the input config is called something other than the default filename
96  // check if we have a dat file associated with it. If not, create a dat
97  // file based on the info file before solving.
98  else if (Initialize_Solver::input_configname != std::string{argv[1]}) {
99  std::string arg1 = std::string{argv[1]};
100  if (arg1 == "full") {
101  Initialize_Solver<config_t>::minimal_config = false;
102  if (Initialize_Solver::rank == 0) {
103  std::cout << "INFO config file missing - generating a full setup"
104  << std::endl
105  << "Modify as needed before rerunning `solve "
106  << Initialize_Solver::input_configname << " <outputdir>`\n";
107  }
108  Initialize_Solver::example_setup = Initialize_Solver::setup_first = true;
109  } else {
110  Initialize_Solver::input_configname = arg1;
111  Initialize_Solver::bconfig.set_filename(
112  Initialize_Solver::input_configname);
113 
114  std::string input_setupname = Initialize_Solver::bconfig.space_filename();
115  std::string file_path = Initialize_Solver::bconfig.config_outputdir();
116 
117  if (fs::exists(Initialize_Solver::input_configname) &&
118  fs::exists(input_setupname)) {
119  if (Initialize_Solver::rank == 0)
120  std::cout << "Solving based on previous solution: "
121  << Initialize_Solver::input_configname << std::endl;
122  } else if (fs::exists(Initialize_Solver::input_configname) &&
123  !fs::exists(input_setupname)) {
124  if (Initialize_Solver::rank == 0) {
125  std::cout << "No dat file associated with : "
126  << Initialize_Solver::input_configname << std::endl;
127  std::cout << "Creating a setup based on config file..." << std::endl;
128  }
129  Initialize_Solver::setup_first = true;
130  } else {
131  if (Initialize_Solver::rank == 0)
132  std::cerr << "Config file input is invalid. Check for typos."
133  << std::endl;
134  std::_Exit(EXIT_FAILURE);
135  }
136  }
137  } else {
138  // if the default <example name>.info file is used, we always start over.
139  Initialize_Solver::setup_first = true;
140  }
141 
142  if (argc > 2)
143  Initialize_Solver::outputdir = std::string{argv[2]};
144 
145  Initialize_Solver::bconfig.set_filename(Initialize_Solver::input_configname);
146 
147  // Redirect rank > 0 stdout and stderr to /dev/null
148  if (Initialize_Solver::rank > 0) {
149  fflush(stderr);
150  freopen("/dev/null", "w", stderr);
151 
152  fflush(stdout);
153  freopen("/dev/null", "w", stdout);
154  }
155 }
