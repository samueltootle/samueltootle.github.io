122  BCO_PARAMS const idx,

129  if (!std::isnan(_val)) {

137  std::string msg{

141  seq.set_mass_idx(idx);
142  seq.set_mass_val(_val);

150 inline ns_sequence find_ns_sequence(Tree const& tree) {
151  auto const& map = MBCO_PARAMS;
152  std::string const branch_name = "ns";
153 
154  // Sets the Parameter Sequence up

156 
157  // Need to check for fixing values
158  for (const auto& [key, index] : map) {
159  // Ignore resolution since this is treated separately
