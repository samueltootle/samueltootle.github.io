224  std::string const branch_name{"binary"};

226  auto const& bco_param_map{MBCO_PARAMS};
227 
228  Tree branch = read_branch(tree, branch_name);
229 
230  auto const& node_map{MBCO};
231 
232  std::array<NODES, 2> bcos{NODES::BCO1, NODES::BCO2};
233  for (const auto& node : branch) {
234  if (!node.second.empty()) {
235  std::string const node_str = node.first;
236 
237  // remove suffix (e,g, bh1 -> bh, ns1 -> ns)
238  auto const tstr = node_str.substr(0, 2);
239 
240  const auto& it = node_map.find(tstr);
241 
242  if (it != node_map.end()) {
243  // save suffix {1, 2}
244  int const tstr_suffix = std::atoi(&node_str.back());
245  if (tstr_suffix > 2 || tstr_suffix < 1) {
246  std::cerr << node_str << " not recognized in Config\n";
247  std::_Exit(EXIT_FAILURE);
248  }
249  int const idx = tstr_suffix - 1;
250  auto res = find_sequence(branch, MBCO_PARAMS, node_str, bcos[idx]);
251  return res;
252  }
253  }
254  }
255  return parse_seq_tree(tree, branch_name, "", (*bco_param_map.begin()).second,
256  BCO1);

301  if (resolution.is_set())
302  return;
303 
304  // Determine lowest resolution
305  auto init_res = (std::isnan(bconfig.seq_setting(SEQ_SETTINGS::INIT_RES)))
306  ? 9
307  : bconfig.seq_setting(SEQ_SETTINGS::INIT_RES);
308 
309  // Determine highest resolution
310  auto final_res =

312  if (final_res < init_res)

314 
315  resolution.set(final_res, init_res, final_res);
