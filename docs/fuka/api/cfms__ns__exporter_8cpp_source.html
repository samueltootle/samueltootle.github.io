90 void CFMS_NS_Exporter::extract_computed_grid_functions() {
91  using namespace Kadath::FUKA_EOS;
92 
93  // fields depending on the coords
94  CoordFields<Space_spheric_adapted> cf_generator(*space);
95  vec_ary_t coord_vectors{default_co_vector_ary(*space)};
96 
97  // get origin of the system and initialize coordinate fields
98  double xo = Kadath::bco_utils::get_center(*space, 0);
99 
100  update_fields_co(cf_generator, coord_vectors, {}, xo);
101 
102  // Initialize Flat Metric
103  Base_tensor basis(shift->get_basis());
104  Metric_flat fmet(*space, basis);
105 
106  // Start - Setup System of equations
107  System_of_eqs syst(*space);
108  fmet.set_system(syst, "f");
109 
110  Param p;
111  EOS_Function_Dispatcher::dispatch<set_eos_ope_struct>(*bconfig, eos_type,
112  syst, p);
113 
114  // Fields - must be initialized before common setup
115  syst.add_cst("N", *lapse);
116  syst.add_cst("bet", *shift);
117 
118  if (bconfig->field(Kadath::FUKA_Config::BCO_FIELDS::DIFF_OMEGA)) {
119 #ifdef DEBUG
120  std::cout << "**** Importing differential rotation profile ****\n";
121 #endif
122  syst.add_cst("Omega", *diff_omega);
123  } else {
124 #ifdef DEBUG
125  std::cout << "**** Importing uniform rotation profile ****\n";
126 #endif
127  syst.add_cst("Omega", (*bconfig)(Kadath::FUKA_Config::BCO_PARAMS::OMEGA));
128  }
129 
130  syst.add_cst("mg", *coord_vectors[GLOBAL_ROT]);
131  syst.add_def("omega^i = bet^i + Omega * mg^i");
132 
133  syst.add_def(
134  "A_ij = (D_i bet_j + D_j bet_i - 2. / 3.* D^k bet_k * f_ij) /2. / N");
135  A.reset(new Tensor(syst.give_val_def("A")));
136  A->coef();
137 
138  // definitions for the fluid 3-velocity
139  syst.add_def("U^i = omega^i / N");
140  fluidvel.reset(new Vector(syst.give_val_def("U")));
141  fluidvel->coef();

144 void CFMS_NS_Exporter::populate_quants() {
145  if (quants.capacity() != XCTS_VARS::NUM_XCTS_VARS) {
146  for (size_t i = 0; i < XCTS_VARS::NUM_XCTS_VARS; ++i)
147  quants.push_back(std::cref(*conformal_factor));
148  }
149  quants[XCTS_VARS::XCTS_PSI] = std::cref(*conformal_factor);
150  quants[XCTS_VARS::XCTS_ALPHA] = std::cref(*lapse);
151  quants[XCTS_VARS::XCTS_BETA1] = std::cref((*shift)(1));
152  quants[XCTS_VARS::XCTS_BETA2] = std::cref((*shift)(2));
153  quants[XCTS_VARS::XCTS_BETA3] = std::cref((*shift)(3));
154 
155  export_utils::add_tensor_refs(quants,
156  {XCTS_VARS::XCTS_A11, XCTS_VARS::XCTS_A12,
157  XCTS_VARS::XCTS_A13, XCTS_VARS::XCTS_A22,
158  XCTS_VARS::XCTS_A23, XCTS_VARS::XCTS_A33},
159  *A);
160 
161  // Fluid related quantities
162  quants[XCTS_VARS::XCTS_H] = std::cref(*logh);
163  quants[XCTS_VARS::XCTS_UX] = std::cref((*fluidvel)(1));
164  quants[XCTS_VARS::XCTS_UY] = std::cref((*fluidvel)(2));
165  quants[XCTS_VARS::XCTS_UZ] = std::cref((*fluidvel)(3));
166  export_ready = true;
167 }
