40 void Space_Kerr_bbh::add_eq (System_of_eqs& sys, const char* eq, const char* rac, const char* rac_der, int nused, Array<int>** pused) {
41  // First bh
42  sys.add_eq_inside (BH1+2, eq, nused, pused) ;
43  for(int i = 0; i < n_shells1; ++i) {
44  sys.add_eq_matching (BH1+2+i, OUTER_BC, rac, nused, pused) ;
45  sys.add_eq_matching (BH1+2+i, OUTER_BC, rac_der, nused, pused) ;
46  sys.add_eq_inside (BH1+3+i, eq, nused, pused) ;
47  }
48 
49  // Matching with bispheric :
50  sys.add_eq_matching_import (BH1 + n_shells1 + 2, OUTER_BC, rac, nused, pused) ;
51  sys.add_eq_matching_import (OUTER, INNER_BC, rac_der, nused, pused) ;
52  sys.add_eq_matching_import (OUTER+1, INNER_BC, rac_der, nused, pused) ;
53 
54  // Second BH :
55  sys.add_eq_inside (BH2+2, eq, nused, pused) ;
56  for(int i = 0; i < n_shells2; ++i) {
57  sys.add_eq_matching (BH2+2+i, OUTER_BC, rac, nused, pused) ;
58  sys.add_eq_matching (BH2+2+i, OUTER_BC, rac_der, nused, pused) ;
59  sys.add_eq_inside (BH2+3+i, eq, nused, pused) ;
60  }
61 
62  // Matching with bispheric :
63  sys.add_eq_matching_import (BH2 + n_shells2 + 2, OUTER_BC, rac, nused, pused) ;
64  sys.add_eq_matching_import (OUTER+3, INNER_BC, rac_der, nused, pused) ;
65  sys.add_eq_matching_import (OUTER+4, INNER_BC, rac_der, nused, pused) ;
66 
67  // Chi first
68  sys.add_eq_inside (OUTER, eq, nused, pused) ;
69  sys.add_eq_matching (OUTER, CHI_ONE_BC, rac, nused, pused) ;
70  sys.add_eq_matching (OUTER, CHI_ONE_BC, rac_der, nused, pused) ;
71 
72  // Rect :
73  sys.add_eq_inside (OUTER+1, eq, nused, pused) ;
74  sys.add_eq_matching (OUTER+1, ETA_PLUS_BC, rac, nused, pused) ;
75  sys.add_eq_matching (OUTER+1, ETA_PLUS_BC, rac_der, nused, pused) ;
76 
77  // Eta first
78  sys.add_eq_inside (OUTER+2, eq, nused, pused) ;
79  sys.add_eq_matching (OUTER+2, ETA_PLUS_BC, rac, nused, pused) ;
80  sys.add_eq_matching (OUTER+2, ETA_PLUS_BC, rac_der, nused, pused) ;
81 
82  // Rect
83  sys.add_eq_inside (OUTER+3, eq, nused, pused) ;
84  sys.add_eq_matching (OUTER+3, CHI_ONE_BC, rac, nused, pused) ;
85  sys.add_eq_matching (OUTER+3, CHI_ONE_BC, rac_der, nused, pused) ;
86 
87 
88  // chi first :
89  sys.add_eq_inside (OUTER+4, eq, nused, pused) ;
90 
91  // Matching outer domain :
92  for (int d=OUTER ; d<=OUTER+4 ; d++)
93  sys.add_eq_matching_import (d, OUTER_BC, rac, nused, pused) ;
94 
95  //Matching for first shell or compactified domain
96  sys.add_eq_matching_import (OUTER+5, INNER_BC, rac_der, nused, pused) ;
97 
98  // Optional spherical shells between bi-spherical and compactified domain
99  for (int d=0 ; d<n_shells_outer ; d++) {
100  sys.add_eq_inside (OUTER+5+d, eq, nused, pused) ;
101  sys.add_eq_matching (OUTER+5+d, OUTER_BC, rac, nused, pused) ;
102  sys.add_eq_matching (OUTER+5+d, OUTER_BC, rac_der, nused, pused) ;
103  }
104 
105  //Compactified domain
106  sys.add_eq_inside (OUTER+5+n_shells_outer, eq, nused, pused) ;
107 
108 }

110 void Space_Kerr_bbh::add_eq_int_inf (System_of_eqs& sys, const char* nom) {
111 
112  // Check the last domain is of the right type :
113  const Domain_compact* pcomp = dynamic_cast <const Domain_compact*> (domains[nbr_domains-1]) ;
114  if (pcomp==0x0) {
115  cerr << "add_eq_int_inf requires a compactified domain" << endl ;
116  abort() ;
117  }
118  int dom = nbr_domains-1 ;
119 
120  // Get the lhs and rhs
121  char p1[LMAX] ;
122  char p2[LMAX] ;
123  bool indic = sys.is_ope_bin(nom, p1, p2, '=') ;
124  if (!indic) {
125  cerr << "= needed for equations" << endl ;
126  abort() ;
127  }
128  else {
129  sys.eq_int_list.push_back(std::make_tuple(nom, nbr_domains-1, OUTER_BC));
130  // Verif lhs = 0 ?
131  indic = ((p2[0]=='0') && (p2[1]==' ') && (p2[2]=='\0')) ?
132  true : false ;
133 
134  // Construction of the equation
135  sys.eq_int[sys.neq_int] = new Eq_int(1) ;
136 
137  // Affectation :
138  // no lhs :
139  if (indic)
140  sys.eq_int[sys.neq_int]->set_part(0, sys.give_ope(dom, p1, OUTER_BC)) ;
141 
142  else
143  sys.eq_int[sys.neq_int]->set_part(0, new Ope_sub(&sys, sys.give_ope(dom, p1, OUTER_BC), sys.give_ope(dom, p2, OUTER_BC))) ;
144  sys.neq_int ++ ;
145  }
146  sys.nbr_conditions = -1 ;
147 }
