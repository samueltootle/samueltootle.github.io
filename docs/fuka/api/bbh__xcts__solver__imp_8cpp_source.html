63 // standardized filename for each converged dataset at the end of each stage.
64 template <typename config_t, typename space_t>
65 std::string bbh_xcts_solver<config_t, space_t>::converged_filename(
66  const std::string stage) const {
67  auto res = space.get_domain(0)->get_nbr_points()(0);
68  auto M1 = bconfig(MCH, BCO1);
69  auto M2 = bconfig(MCH, BCO2);
70  if (M2 > M1)
71  std::swap(M1, M2);
72  bconfig.set(Q) = M2 / M1;
73  auto Mtot = M1 + M2;
74  std::stringstream ss;
75  ss << "BBH";
76  if (stage != "")
77  ss << "_" << stage << ".";
78  else
79  ss << ".";
80  ss << bconfig(DIST) << "." << bconfig(CHI, BCO1) << "." << bconfig(CHI, BCO2)
81  << "." << Mtot << ".q" << bconfig(Q) << "." << std::setfill('0')
82  << std::setw(1) << bconfig(NSHELLS, BCO1) << "." << std::setfill('0')
83  << std::setw(1) << bconfig(NSHELLS, BCO2) << "." << std::setfill('0')
84  << std::setw(2) << res;
85  return ss.str();
86 }

88 template <typename config_t, typename space_t>
89 int bbh_xcts_solver<config_t, space_t>::solve() {
90  int rank = 0;
91  MPI_Comm_rank(MPI_COMM_WORLD, &rank);
92  int exit_status = EXIT_SUCCESS;
93 
94  std::array<bool, NUM_STAGES> stage_enabled = bconfig.return_stages();
95  auto [last_stage, last_stage_idx] = get_last_enabled(MSTAGE, stage_enabled);
96 
97  // check if we have solved this before til the last stage
98  if (solution_exists(last_stage)) {
99  if (rank == 0)
100  std::cout << "Solved previously: " << bconfig.config_filename()
101  << std::endl;
102  return EXIT_SUCCESS;
103  }
104 
105  if (stage_enabled[TOTAL_BC]) {
106  this->solver_stage = TOTAL_BC;
107  exit_status = solve_stage("TOTAL_BC");
108 
109  if (bconfig.control(FIXED_GOMEGA)) {
110  bconfig.control(FIXED_GOMEGA) = false;
111  exit_status = solve_stage("TOTAL_BC");
112  }
113  }
114 
115  if (stage_enabled[ECC_RED]) {
116  this->solver_stage = ECC_RED;
117  exit_status = solve_stage("ECC_RED");
118  }
119 
120  // Barrier needed in case we need to read from the previous output
121  MPI_Barrier(MPI_COMM_WORLD);
122 
123  return exit_status;
124 }

126 template <typename config_t, typename space_t>
127 void bbh_xcts_solver<config_t, space_t>::syst_init(System_of_eqs& syst) {
128  const int ndom = space.get_nbr_domains();
129  // call the (flat) conformal metric "f"
130  fmet.set_system(syst, "f");
131 
132  // define numerical constants
133  syst.add_cst("4piG", bconfig(QPIG));
134  syst.add_cst("PI", M_PI);
135 
136  // BH1 fixing parameters
137  syst.add_cst("Mm", bconfig(MIRR, BCO1));
138  syst.add_cst("chim", bconfig(CHI, BCO1));
139  syst.add_cst("CMm", bconfig(MCH, BCO1));
140 
141  // BH2 fixing parameters
142  syst.add_cst("Mp", bconfig(MIRR, BCO2));
143  syst.add_cst("chip", bconfig(CHI, BCO2));
144  syst.add_cst("CMp", bconfig(MCH, BCO2));
145 
146  // constant vector fields
147  syst.add_cst("mg", *coord_vectors[GLOBAL_ROT]);
148 
149  syst.add_cst("mm", *coord_vectors[BCO1_ROT]);
150  syst.add_cst("sm", *coord_vectors[S_BCO1]);
151  syst.add_cst("mp", *coord_vectors[BCO2_ROT]);
152  syst.add_cst("sp", *coord_vectors[S_BCO2]);
153 
154  syst.add_cst("ex", *coord_vectors[EX]);
155  syst.add_cst("ey", *coord_vectors[EY]);
156  syst.add_cst("ez", *coord_vectors[EZ]);
157  syst.add_cst("einf", *coord_vectors[S_INF]);
158 
159  // the basic fields, conformal factor, lapse and (log) enthalpy
160  syst.add_var("P", conf);
161  syst.add_var("N", lapse);
162  syst.add_var("bet", shift);
163 
164  // setup parameters accordingly for a corotating binary
165  if (bconfig.control(COROT_BIN)) {
166  bconfig.set(OMEGA, BCO1) = 0.;
167  bconfig.set(OMEGA, BCO2) = 0.;
168  } else {
169  // otherwise we setup the system for arbitrarily rotating BHs
170  // based on either fixed rotation frequency or variable
171  // rotation frequency fixed by a dimensionless
172  // spin parameter CHI.
173  if (!std::isnan(bconfig.set(FIXED_BCOMEGA, BCO1))) {
174  syst.add_cst("omesm", bconfig(FIXED_BCOMEGA, BCO1));
175  bconfig.set(OMEGA, BCO1) = bconfig(FIXED_BCOMEGA, BCO1);
176  } else
177  syst.add_var("omesm", bconfig(OMEGA, BCO1));
178 
179  syst.add_def(space.BH1 + 2, "ExOme^i = omesm * mm^i");
180 
181  if (!std::isnan(bconfig.set(FIXED_BCOMEGA, BCO2))) {
182  syst.add_cst("omesp", bconfig(FIXED_BCOMEGA, BCO2));
183  bconfig.set(OMEGA, BCO2) = bconfig(FIXED_BCOMEGA, BCO2);
184  } else
185  syst.add_var("omesp", bconfig(OMEGA, BCO2));
186  syst.add_def(space.BH2 + 2, "ExOme^i = omesp * mp^i");
187  }
188 
189  // define common combinations of conformal factor and lapse
190  syst.add_def("NP = P*N");
191  syst.add_def("Ntilde = N / P^6");
192 
193  // definition describing the binary rotation
194  syst.add_def("Morb^i = mg^i + xaxis * ey^i + yaxis * ex^i");
195 
196  syst.add_def(
197  "A^ij = (D^i bet^j + D^j bet^i - 2. / 3.* D_k bet^k * f^ij) / 2. / "
198  "Ntilde");
199 
200  // ADM linear momenta
201  syst.add_def("intPx = A_ij * ex^j * einf^i / 8 / PI");
202  syst.add_def("intPy = A_ij * ey^j * einf^i / 8 / PI");
203  syst.add_def("intPz = A_ij * ez^j * einf^i / 8 / PI");
204 
205  // spin definition on the apparent horizons
206  syst.add_def("intSm = A_ij * mm^i * sm^j / 8 / PI");
207  syst.add_def("intSp = A_ij * mp^i * sp^j / 8 / PI");
208 
209  // diagnostic from https://arxiv.org/abs/1506.01689
210  syst.add_def(ndom - 1, "COMx = -3 * P^4 * einf^i * ex_i / 8. / PI");
211  syst.add_def(ndom - 1, "COMy = 3 * P^4 * einf^i * ey_i / 8. / PI");
212  syst.add_def(ndom - 1, "COMz = 3 * P^4 * einf^i * ez_i / 8. / PI");
213 
214  // BH irreducible mass
215  syst.add_def("intMsq = P^4 / 16. / PI");
216 
217  // define quantity to be integrated at infinity
218  // two (in this case) equivalent definitions of ADM mass
219  // as well as the Komar mass
220  syst.add_def(ndom - 1, "intMadm = - einf^i * D_i P / 4piG * 2");
221  syst.add_def(ndom - 1, "intMk = einf^i * D_i N / 4piG");
222 }

224 template <typename config_t, typename space_t>
225 void bbh_xcts_solver<config_t, space_t>::print_diagnostics(
226  const System_of_eqs& syst,
227  const int ite,
228  const double conv) const {
229  std::ios_base::fmtflags f(std::cout.flags());
230  int ndom = space.get_nbr_domains();
231 
232  // local spin angular momentum BH1
233  Val_domain integSm(syst.give_val_def("intSm")()(space.BH1 + 2));
234  double Sm = space.get_domain(space.BH1 + 2)->integ(integSm, INNER_BC);
235 
236  // local spin angular momentum BH2
237  Val_domain integSp(syst.give_val_def("intSp")()(space.BH2 + 2));
238  double Sp = space.get_domain(space.BH2 + 2)->integ(integSp, INNER_BC);
239 
240  // irreducible mass BH1
241  Val_domain integMmsq(syst.give_val_def("intMsq")()(space.BH1 + 2));
242  double Mirrmsq = space.get_domain(space.BH1 + 2)->integ(integMmsq, INNER_BC);
243  double Mirrm = std::sqrt(Mirrmsq);
244  // MCH
245  double Mchm = std::sqrt(Mirrmsq + Sm * Sm / 4. / Mirrmsq);
246 
247  // irreducible mass BH2
248  Val_domain integMpsq(syst.give_val_def("intMsq")()(space.BH2 + 2));
249  double Mirrpsq = space.get_domain(space.BH2 + 2)->integ(integMpsq, INNER_BC);
250  double Mirrp = std::sqrt(Mirrpsq);
251  // MCH
252  double Mchp = std::sqrt(Mirrpsq + Sp * Sp / 4. / Mirrpsq);
253 
254  // ADM Linear Momenta
255  Val_domain integPx(syst.give_val_def("intPx")()(ndom - 1));
256  double Px = space.get_domain(ndom - 1)->integ(integPx, OUTER_BC);
257 
258  Val_domain integPy(syst.give_val_def("intPy")()(ndom - 1));
259  double Py = space.get_domain(ndom - 1)->integ(integPy, OUTER_BC);
260 
261  Val_domain integPz(syst.give_val_def("intPz")()(ndom - 1));
262  double Pz = space.get_domain(ndom - 1)->integ(integPz, OUTER_BC);
263 
264  cout << std::string(42, '=') << endl;
265  std::cout << FORMAT << "Iter: " << ite << std::endl
266  << FORMAT << "Error: " << conv << std::endl;
267  std::cout << FORMAT << "Omega: " << bconfig(GOMEGA) << std::endl;
268  if (!std::isnan(bconfig.set(ADOT)))
269  std::cout << FORMAT << "Adot: " << bconfig(ADOT) << std::endl;
270  std::cout << FORMAT << "Axis: " << bconfig(COM) << std::endl;
271  std::cout << FORMAT << "P: " << "[" << Px << ", " << Py << ", " << Pz
272  << "]\n\n";
273 
274  std::cout << FORMAT << "BH1-Mirr: " << Mirrm << std::endl;
275  std::cout << FORMAT << "BH1-Mch: " << Mchm << std::endl;
276  std::cout << FORMAT << "BH1-R: "
277  << bco_u::get_radius(space.get_domain(space.BH1 + 1), EQUI)
278  << std::endl;
279  std::cout << FORMAT << "BH1-S: " << Sm << std::endl;
280  std::cout << FORMAT << "BH1-Chi: " << Sm / Mchm / Mchm << std::endl;
281  std::cout << FORMAT << "BH1-Omega: " << bconfig(OMEGA, BCO1) << "\n\n";
282 
283  std::cout << FORMAT << "BH2-Mirr: " << Mirrp << std::endl;
284  std::cout << FORMAT << "BH2-Mch: " << Mchp << std::endl;
285  std::cout << FORMAT << "BH2-R: "
286  << bco_u::get_radius(space.get_domain(space.BH2 + 1), EQUI)
287  << std::endl;
288  std::cout << FORMAT << "BH2-S: " << Sp << std::endl;
289  std::cout << FORMAT << "BH2-Chi: " << Sp / Mchp / Mchp << std::endl;
290  std::cout << FORMAT << "BH2-Omega: " << bconfig(OMEGA, BCO2) << "\n\n";
291  std::cout.flags(f);
292 
293  cout << std::string(42, '=') << endl;
294 }
