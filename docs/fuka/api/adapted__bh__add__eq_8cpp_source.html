48 void Space_adapted_bh::add_eq_int_inf (System_of_eqs& sys, const char* nom) {
49 
50  // Check the last domain is of the right type :
51  const Domain_compact* pcomp = dynamic_cast <const Domain_compact*> (domains[nbr_domains-1]) ;
52  if (pcomp==0x0) {
53  cerr << "add_eq_int_inf requires a compactified domain" << endl ;
54  abort() ;
55  }
56  int dom = nbr_domains-1 ;
57  sys.eq_int_list.push_back(std::make_tuple(nom,dom,OUTER_BC));
58 
59  // Get the lhs and rhs
60  char p1[LMAX] ;
61  char p2[LMAX] ;
62  bool indic = sys.is_ope_bin(nom, p1, p2, '=') ;
63  if (!indic) {
64  cerr << "= needed for equations" << endl ;
65  abort() ;
66  }
67  else {
68  // Verif lhs = 0 ?
69  indic = ((p2[0]=='0') && (p2[1]==' ') && (p2[2]=='\0')) ?
70  true : false ;
71 
72  // Construction of the equation
73  sys.eq_int[sys.neq_int] = new Eq_int(1) ;
74 
75  // Affectation :
76  // no lhs :
77  if (indic)
78  sys.eq_int[sys.neq_int]->set_part(0, sys.give_ope(dom, p1, OUTER_BC)) ;
79 
80  else
81  sys.eq_int[sys.neq_int]->set_part(0, new Ope_sub(&sys, sys.give_ope(dom, p1, OUTER_BC), sys.give_ope(dom, p2, OUTER_BC))) ;
82  sys.neq_int ++ ;
83  }
84  sys.nbr_conditions = -1 ;
85 }
