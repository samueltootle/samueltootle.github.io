272 void Domain_polar_shell_outer_adapted::update_variable (const Val_domain& cor_outer_radius, const Scalar& old, Scalar& res) const {
273 
274  Val_domain dr (old(num_dom).der_r()) ;
275  if (dr.check_if_zero())
276  res.set_domain(num_dom) = 0 ;
277  else {
278 
279  Index pos(nbr_points) ;
280  res.set_domain(num_dom).allocate_conf() ;
281  do {
282  res.set_domain(num_dom).set(pos) = dr(pos) * (1+(*coloc[0])(pos(0))) / 2. ;
283  }
284  while (pos.inc()) ;
285 
286  res.set_domain(num_dom) = cor_outer_radius * res(num_dom) + old(num_dom) ;
287  res.set_domain(num_dom).set_base() = old(num_dom).get_base() ;

316 void Domain_polar_shell_outer_adapted::update() const {
317 
318  if (rad_term_eq != 0x0)
319  delete rad_term_eq ;
320  if (der_rad_term_eq != 0x0)
321  delete der_rad_term_eq ;
322  if (dt_rad_term_eq != 0x0)
323  delete dt_rad_term_eq ;
324 
325  // Computation of rad_term_eq
326  Scalar val_res (sp) ;
327  val_res.set_domain(num_dom).allocate_conf() ;
328  Index index (nbr_points) ;
329  do {
330  val_res.set_domain(num_dom).set(index) = (((*outer_radius_term_eq->val_t)()(num_dom))(index) - inner_radius )/2. * ((*coloc[0])(index(0))) +
331  (((*outer_radius_term_eq->val_t)()(num_dom))(index) + inner_radius )/2. ;
332  }
333  while (index.inc()) ;
334  val_res.set_domain(num_dom).std_r_base() ;
335 
336  bool doder = (outer_radius_term_eq->der_t ==0x0) ? false : true ;
337  if (doder) {
338  Scalar der_res (sp) ;
339  if ((*outer_radius_term_eq->der_t)()(num_dom).check_if_zero()) {
340  der_res.set_domain(num_dom).set_zero() ;
341  }
342  else {
343  der_res.set_domain(num_dom).allocate_conf() ;
344  index.set_start() ;
345  do {
346  der_res.set_domain(num_dom).set(index) = ((*outer_radius_term_eq->der_t)()(num_dom))(index)/2. * ((*coloc[0])(index(0))) +
347  ((*outer_radius_term_eq->der_t)()(num_dom))(index)/2. ;
348  }
349  while (index.inc()) ;
350  der_res.set_domain(num_dom).std_r_base() ;
351  }
352  rad_term_eq = new Term_eq (num_dom, val_res, der_res) ;
353  }
354  else
355  rad_term_eq = new Term_eq (num_dom, val_res) ;
356 
357  // Computation of der_rad_term_eq which is dr / dxi
358  val_res.set_domain(num_dom) = ((*outer_radius_term_eq->val_t)()(num_dom)-inner_radius) / 2. ;
359  if (doder) {
360  Scalar der_res (sp) ;
361  der_res.set_domain(num_dom) = (*outer_radius_term_eq->der_t)()(num_dom) / 2. ;
362  der_rad_term_eq = new Term_eq (num_dom, val_res, der_res) ;
363  }
364  else
365  der_rad_term_eq = new Term_eq (num_dom, val_res) ;
366 
367  // Computation of dt_rad_term_eq which is dr / d theta
368  val_res.set_domain(num_dom) = (*rad_term_eq->val_t)()(num_dom).der_var(2) ;
369  if (doder) {
370  Scalar der_res (sp) ;
371  der_res.set_domain(num_dom) = (*rad_term_eq->der_t)()(num_dom).der_var(2) ;
372  dt_rad_term_eq = new Term_eq (num_dom, val_res, der_res) ;
373  }
374  else
375  dt_rad_term_eq = new Term_eq (num_dom, val_res) ;
376 
377  do_normal_cart() ;
378 }

410  for (int i=0 ; i<2 ; i++)
411  assert (coloc[i] != 0x0) ;
412  for (int i=0 ; i<2 ; i++)
413  assert (absol[i] == 0x0) ;

415  absol[i] = new Val_domain(this) ;
416  absol[i]->allocate_conf() ;
417  }
418  Index index (nbr_points) ;
419  do {
420 
421  double rr = ((*outer_radius)(index) - inner_radius)/2. * ((*coloc[0])(index(0))) +
422  ((*outer_radius)(index) + inner_radius)/2. ;
423  absol[0]->set(index) = rr *
424  sin((*coloc[1])(index(1))) + center(1);
425  absol[1]->set(index) = rr * cos((*coloc[1])(index(1))) + center(2) ;
426  }
427  while (index.inc()) ;

484 // Convert absolute coordinates to numerical ones
485 const Point Domain_polar_shell_outer_adapted::absol_to_num(const Point& abs) const {
486  Point num(2) ;
487 
488  double rho_loc = fabs(abs(1) - center(1)) ;
489  double z_loc = abs(2) - center(2) ;
490  double air = sqrt(rho_loc*rho_loc+z_loc*z_loc) ;
491 
492  if (rho_loc==0) {
493  // On the axis ?
494  num.set(2) = (z_loc>=0) ? 0 : M_PI ;
495  }
496  else {
497  num.set(2) = atan(rho_loc/z_loc) ;
498  }
499 
500  if (num(2) <0)
501  num.set(2) = M_PI + num(2) ;
502 
503  // Get the boundary for those angles
504  num.set(1) = 1 ;
505  outer_radius->coef() ;
506  double outer = outer_radius->get_base().summation(num, outer_radius->get_coef()) ;
507  num.set(1) = (2./(outer-inner_radius)) * (air - (outer + inner_radius)/2.) ;
508 
509  return num ;
510 }

513 double coloc_leg(int, int) ;
514 void Domain_polar_shell_outer_adapted::do_coloc () {
515 
516  switch (type_base) {
517  case CHEB_TYPE:
518  nbr_coefs = nbr_points ;
519  del_deriv() ;
520  for (int i=0 ; i<ndim ; i++)
521  coloc[i] = new Array<double> (nbr_points(i)) ;
522  for (int i=0 ; i<nbr_points(0) ; i++)
523  coloc[0]->set(i) = -cos(M_PI*i/(nbr_points(0)-1)) ;
524  for (int j=0 ; j<nbr_points(1) ; j++)
525  coloc[1]->set(j) = M_PI/2.*j/(nbr_points(1)-1) ;
526  break ;
527  case LEG_TYPE:
528  nbr_coefs = nbr_points ;
529  del_deriv() ;
530  for (int i=0 ; i<ndim ; i++)
531  coloc[i] = new Array<double> (nbr_points(i)) ;
532  for (int i=0 ; i<nbr_points(0) ; i++)
533  coloc[0]->set(i) = coloc_leg(i, nbr_points(0)) ;
534  for (int j=0 ; j<nbr_points(1) ; j++)
535  coloc[1]->set(j) = M_PI/2.*j/(nbr_points(1)-1) ;
536  break ;
537  default :
538  cerr << "Unknown type of basis in Domain_polar_shell_outer_adapted::do_coloc" << endl ;
539  abort() ;
540  }

563 // standard base for a symetric function in z, using Chebyshev
564 void Domain_polar_shell_outer_adapted::set_cheb_base_with_m(Base_spectral& base, int m) const {
565 
566  assert (type_base == CHEB_TYPE) ;
567 
568  base.allocate(nbr_coefs) ;
569 
570  base.def =true ;
571 
572  base.bases_1d[1]->set(0) = (m%2==0) ? COS_EVEN : SIN_ODD ;
