115 template<typename tov_t, typename config_t>
116 void write_ns_init_setup_tofile_XCTS(Space_spheric_adapted& space, config_t& bconfig, tov_t& tov) {
117  using eos_t = typename tov_t::eos_t;
118  enum ltpQ { LAPSE=0, RHO, CONF };
119  const int ndom = space.get_nbr_domains();
120  Base_tensor basis(space, CARTESIAN_BASIS);
121  // setup fields
122  Scalar lapse(space);
123  lapse = 1.;
124 
125  Scalar conf(lapse);
126 
127  // H = log(h), the logarithm of the specific enthalpy
128  Scalar logh(space);
129  logh.annule_hard();
130 
131  // interpolate TOV solution
132  auto lintp = setup_interpolator_from_TOV(tov);
133 
134  // generate a radius field for use with the 1D interpolators
135  CoordFields<Space_spheric_adapted> cfg(space);
136  Scalar r_field(cfg.radius());
137 
138  // update the fields based on TOV solution for a given domain
139  auto update_fields= [&](const size_t dom) {
140  Index pos(space.get_domain(dom)->get_nbr_points());
141  do {
142  double rval = r_field(dom)(pos);
143  auto all_ltp = lintp.interpolate_all(rval);
144  auto rho = (all_ltp[ltpQ::RHO] <= 0) ? 1e-15 : all_ltp[ltpQ::RHO];
145 
146  auto h = EOS<eos_t, eos_var_t::DENSITY>::h_cold__rho(rho);
147  if(pos(0) == 0 && pos(1) == 0 && pos(2) == 0)
148  bconfig.set(HC) = h;
149  logh.set_domain(dom).set(pos) = (h < 1) ? 0. : std::log(h);
150  lapse.set_domain(dom).set(pos) = all_ltp[ltpQ::LAPSE];
151  conf.set_domain(dom).set(pos) = all_ltp[ltpQ::CONF];
152  }while(pos.inc());
153  };
154  for(int i = 0; i < 3; ++i)
155  update_fields(i);
156 
157  for (int d = 2; d < ndom; ++d)
158  logh.set_domain(d).annule_hard();
159 
160  Vector shift(space, CON, basis);
161  for (int i = 1; i <= 3; i++)
162  shift.set(i).annule_hard();
163  shift.std_base();
164  logh.std_base();
165  conf.std_base();
166  lapse.std_base();
167  // end setup fields
168 
169  Kadath::bco_utils::save_to_file(space, bconfig, conf, lapse, shift, logh);
170 }

173 auto setup_ns_config_from_TOV(config_t& bconfig) {
174  using namespace Kadath::Margherita;

181  bconfig.set(BCO_PARAMS::NC) = tov->rhoc;

183  bconfig.set(BCO_PARAMS::MB) = tov->baryon_mass;

193 template<typename tov_t>
194 auto setup_interpolator_from_TOV(tov_t& tov) {
195  const size_t max_iter = tov.state.size();
196 
197  std::unique_ptr<double[]> radius_lin_ptr{new double[max_iter]};
198  std::unique_ptr<double[]> conf_lin_ptr{new double[max_iter]};
199  std::unique_ptr<double[]> lapse_lin_ptr{new double[max_iter]};
200  std::unique_ptr<double[]> rho_lin_ptr{new double[max_iter]};
201 
202  for(auto j=0; j < max_iter; ++j) {
203  //lapse est.
204  auto phi = tov.state[j][tov.PHI];
205 
206  lapse_lin_ptr[j] = std::exp(phi);
207 
208  conf_lin_ptr[j] = tov.state[j][tov.CONF];
209 
210  radius_lin_ptr[j] = tov.state[j][tov.RISO];
211 
212  rho_lin_ptr[j] = tov.state[j][tov.RHOB];
213  }
214 
215  // linear interpolation of conf(r_isotropic), lapse(r_isotropic), and rho(r_isotropic)
216  // The order here dictates the enum ltpQ enum
217  linear_interp_t<double,3> ltp(max_iter, std::move(radius_lin_ptr), std::move(lapse_lin_ptr), std::move(rho_lin_ptr), std::move(conf_lin_ptr));
218  return ltp;
219 }
