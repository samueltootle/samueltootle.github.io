152 template <typename tov_t, typename config_t>
153 void write_ns_init_setup_tofile_XCTS(Space_spheric_adapted& space,
154  config_t& bconfig,
155  tov_t& tov) {
156  using eos_t = typename tov_t::eos_t;
157 
158  enum ltpQ { LAPSE = 0, RHO, CONF };
159 
160  const int ndom = space.get_nbr_domains();
161  Base_tensor basis(space, CARTESIAN_BASIS);
162  // setup fields
163  Scalar lapse(space);
164  lapse = 1.;
165 
166  Scalar conf(lapse);
167 
168  // H = log(h), the logarithm of the specific enthalpy
169  Scalar logh(space);
170  logh.annule_hard();
171 
172  // interpolate TOV solution
173  auto lintp = setup_interpolator_from_TOV(tov);
174 
175  // generate a radius field for use with the 1D interpolators
176  CoordFields<Space_spheric_adapted> cfg(space);
177  Scalar r_field(cfg.radius());
178 
179  // update the fields based on TOV solution for a given domain
180  auto update_fields = [&](const size_t dom) {
181  Index pos(space.get_domain(dom)->get_nbr_points());
182  do {
183  double rval = r_field(dom)(pos);
184  auto all_ltp = lintp.interpolate_all(rval);
185  auto rho = (all_ltp[ltpQ::RHO] <= 0) ? 1e-15 : all_ltp[ltpQ::RHO];
186 
187  auto h = EOS<eos_t, eos_var_t::DENSITY>::h_cold__rho(rho);
188  logh.set_domain(dom).set(pos) = (h <= 1) ? 0. : std::log(h);
189  lapse.set_domain(dom).set(pos) = all_ltp[ltpQ::LAPSE];
190  conf.set_domain(dom).set(pos) = all_ltp[ltpQ::CONF];
191  } while (pos.inc());
192  };
193  for (int i = 0; i < 3; ++i)
194  update_fields(i);
195 
196  for (int d = 2; d < ndom; ++d)
197  logh.set_domain(d).annule_hard();
198 
199  Vector shift(space, CON, basis);
200  for (int i = 1; i <= 3; i++)
201  shift.set(i).annule_hard();
202  shift.std_base();
203  logh.std_base();
204  conf.std_base();
205  lapse.std_base();
206  // end setup fields
207 
208  Kadath::bco_utils::save_to_file(space, bconfig, conf, lapse, shift, logh);
209 }

211 template <typename tov_t, typename config_t>
212 void write_ns2d_isotropic_init_setup_tofile(Space_polar_adapted& space,
213  config_t& bconfig,
214  tov_t& tov) {
215  using eos_t = typename tov_t::eos_t;
216 
217  enum ltpQ { LAPSE = 0, RHO, CONF };
218 
219  const int ndom = space.get_nbr_domains();
220 
221  // setup fields
222  Scalar lapse(space);
223  lapse = 1.;
224 
225  Scalar conf(lapse);
226 
227  // H = log(h), the logarithm of the specific enthalpy
228  Scalar logh(space);
229  logh.annule_hard();
230 
231  // interpolate TOV solution
232  auto lintp = setup_interpolator_from_TOV(tov);
233 
234  Scalar r_field(space);
235  r_field.annule_hard();
236  for (auto i = 0; i < ndom; ++i)
237  r_field.set_domain(i) = space.get_domain(i)->get_radius();
238  auto npts = space.get_domain(ndom - 1)->get_nbr_points();
239  Index pos_c(npts);
240  pos_c.set(0) = npts(0) - 1; // outer_bc
241 
242  for (auto i = 0; i < npts(1); ++i) {
243  pos_c.set(1) = i;
244  r_field.set_domain(ndom - 1).set(pos_c) = 1e10;
245  }
246  pos_c.set_start();
247 
248  // update the fields based on TOV solution for a given domain
249  auto update_fields = [&](const size_t dom) {
250  Index pos(space.get_domain(dom)->get_nbr_points());
251  do {
252  double rval = r_field(dom)(pos);
253  auto all_ltp = lintp.interpolate_all(rval);
254  auto rho = (all_ltp[ltpQ::RHO] <= 0) ? 1e-15 : all_ltp[ltpQ::RHO];
255  auto h = EOS<eos_t, eos_var_t::DENSITY>::h_cold__rho(rho);
256 
257  if (dom == 0 && pos(0) == 0 && pos(1) == 0)
258  bconfig.set(BCO_PARAMS::HC) = h;
259 
260  logh.set_domain(dom).set(pos) = (h <= 1) ? 0. : std::log(h);
261  lapse.set_domain(dom).set(pos) = all_ltp[ltpQ::LAPSE];
262  conf.set_domain(dom).set(pos) = all_ltp[ltpQ::CONF];
263  } while (pos.inc());
264  };
265  for (int i = 0; i < ndom; ++i)
266  update_fields(i);
267 
268  for (int d = space.ADAPTED_INNER; d < ndom; ++d)
269  logh.set_domain(d).annule_hard();
270 
271  // Fix compactified domain and additional shells
272  for (auto d = 3; d < ndom; ++d) {
273  pos_c.set_start();
274  auto decay_factor = r_field(d)(pos_c) / r_field(d);
275  if (d > 3) {
276  pos_c.set(0) = npts(0) - 1;
277  conf.set_domain(d) = 1 + (conf(d - 1)(pos_c) - 1) * decay_factor;
278  lapse.set_domain(d) = 1 + (lapse(d - 1)(pos_c) - 1) * decay_factor;
279  } else {
280  conf.set_domain(d) = 1 + (conf(d)(pos_c) - 1) * decay_factor;
281  lapse.set_domain(d) = 1 + (lapse(d)(pos_c) - 1) * decay_factor;
282  }
283  }
284 
285  Scalar A(conf * conf);
286  A.std_base();
287  lapse.std_base();
288 
289  // We store the quantities that appear in the lapace terms
290  // for solver stability. These will need to be transformed
291  // to obtain the full metric.
292  Scalar nu(log(lapse));
293  Scalar lap_aterm(nu + log(A));
294 
295  logh.std_base();
296  nu.std_base();
297  lap_aterm.std_base();
298 
299  Scalar tmp(lapse * A - 1);
300  Scalar lap_Bterm = Scalar(tmp.mult_r().mult_sin_theta());
301  // end setup fields
302 
303  bco_utils::save_to_file(space, bconfig, lap_aterm, nu, logh, lap_Bterm);
304 }

306 template <typename eos_t, typename config_t>
307 auto setup_ns_config_from_TOV(config_t& bconfig, size_t mass_fixing_idx) {
308  using namespace Kadath::Margherita;
309  auto tov = std::make_unique<MargheritaTOV<eos_t>>();
310 
311  bool adm_mass_fixing = (mass_fixing_idx == BCO_PARAMS::MADM);
312  bool nc_mass_fixing = (mass_fixing_idx == BCO_PARAMS::NC);
313  bool hc_mass_fixing = (mass_fixing_idx == BCO_PARAMS::HC);
314 
315  if (adm_mass_fixing) {
316  bool use_Mmax = tov->solve_for_MADM(bconfig(mass_fixing_idx));
317 
318  if (use_Mmax)
319  bconfig.set(BCO_PARAMS::MADM) = tov->mass;
320  } else if (nc_mass_fixing) {
321  tov->solve(bconfig(mass_fixing_idx));
322  } else if (hc_mass_fixing) {
323  bconfig.set(BCO_PARAMS::NC) =
324  EOS<eos_t, DENSITY>::get(bconfig(BCO_PARAMS::HC));
325  tov->solve(bconfig(BCO_PARAMS::NC));
326  } else {
327  std::string msg{"1D TOV solver not implemented for sequences using idx " +
328  std::to_string(mass_fixing_idx)};
329  throw std::runtime_error(msg.c_str());
330  }
331 
332  bconfig.set(BCO_PARAMS::NC) =
333  (nc_mass_fixing) ? bconfig(BCO_PARAMS::NC) : tov->rhoc;
334  bconfig.set(BCO_PARAMS::HC) =
335  (hc_mass_fixing) ? bconfig(BCO_PARAMS::HC)
336  : EOS<eos_t, eos_var_t::PRESSURE>::h_cold__rho(
337  bconfig(BCO_PARAMS::NC));
338  bconfig.set(BCO_PARAMS::MB) = tov->baryon_mass;
339  bconfig.set(BCO_PARAMS::MADM) =
340  (adm_mass_fixing) ? bconfig(BCO_PARAMS::MADM) : tov->mass;
341 
342  // update surface radius estimate
343  bconfig.set(BCO_PARAMS::RMID) = tov->radius;
344  bconfig.set(BCO_PARAMS::RIN) = 0.5 * bconfig(BCO_PARAMS::RMID);
345  bconfig.set(BCO_PARAMS::ROUT) =
346  bco_utils::gold_ratio * bconfig(BCO_PARAMS::RMID);
347 
348  return tov;
349 }

351 template <typename tov_t>
352 auto setup_interpolator_from_TOV(tov_t& tov) {
353  const size_t max_iter = tov.state.size();
354 
355  std::unique_ptr<double[]> radius_lin_ptr{new double[max_iter]};
356  std::unique_ptr<double[]> conf_lin_ptr{new double[max_iter]};
357  std::unique_ptr<double[]> lapse_lin_ptr{new double[max_iter]};
358  std::unique_ptr<double[]> rho_lin_ptr{new double[max_iter]};
359 
360  for (size_t j = 0; j < max_iter; ++j) {
361  // lapse est.
362  auto phi = tov.state[j][tov.PHI];
363 
364  lapse_lin_ptr[j] = std::exp(phi);
365 
366  conf_lin_ptr[j] = tov.state[j][tov.CONF];
367 
368  radius_lin_ptr[j] = tov.state[j][tov.RISO];
369 
370  rho_lin_ptr[j] = tov.state[j][tov.RHOB];
371  }
372 
373  // linear interpolation of conf(r_isotropic), lapse(r_isotropic), and
374  // rho(r_isotropic) The order here dictates the enum ltpQ enum
375  linear_interp_t<double, 3> ltp(max_iter, std::move(radius_lin_ptr),
376  std::move(lapse_lin_ptr),
377  std::move(rho_lin_ptr),
378  std::move(conf_lin_ptr));
379  return ltp;
380 }
