88 inline double ghl_tabulated_rho__h_cold(
89  std::unique_ptr<ghl_eos_parameters>& ghl_eos_params,
90  double& h_in) {
91  // GRHayL will throw an error rather than enforce table bounds
92  // so we enforce them here instead.
93  size_t const N = ghl_eos_params->N_rho;
94  const double h_min = std::exp(ghl_eos_params->lh_of_lr[0]);
95  const double h_max = std::exp(ghl_eos_params->lh_of_lr[N - 1]);
96  if (h_in < h_min) {
97  h_in = h_min;
98  return std::exp(ghl_eos_params->table_logrho[0]);
99  } else if (h_in > h_max) {
100  h_in = h_max;
101  return std::exp(ghl_eos_params->table_logrho[N - 1]);
102  }
103 
104  // It's only safe to linear interpolate in logp vs logrho space
105  // We rootfind the correct specific enthalpy rather than
106  // attempting a table look-up using the same method as used
107  // in Margherita::Cold_Table
108  const auto func = [&](const double& lrho) {
109  double const rho = std::exp(lrho);
110  double const eps =
111  ghl_tabulated_compute_eps_from_rho(ghl_eos_params.get(), rho);
112  double const press =
113  ghl_tabulated_compute_P_from_rho(ghl_eos_params.get(), rho);
114 
115  return h_in - (1. + eps + press / std::exp(lrho));
116  };
117 
118  // Note the root bounds are quite sensitive given GRHaYL will simply error out
119  // rather than enforcing table bounds.
120  auto lrho = zero_brent<>(log(ghl_eos_params->rho_min * 1.001),
121  log(ghl_eos_params->rho_max), 1.0e-13, func);
122  return std::exp(lrho);
123 }
