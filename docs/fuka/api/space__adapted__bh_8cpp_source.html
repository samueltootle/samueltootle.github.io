29 Space_adapted_bh::Space_adapted_bh (int ttype, const Point& center, const Dim_array& res, const std::vector<double>& BH_bounds) {
30  enum bound_index {RIN=0, RMID=1, NUM_IND};
31 
32  ndim = 3 ;
33 
34  nbr_domains = BH_bounds.size()+1 ;
35  type_base = ttype ;
36  domains = new Domain* [nbr_domains] ;
37  auto gen_bh_domains = [&](const int nuc_i, auto& bounds) {
38  int shells = bounds.size()-3;
39  Point center (ndim) ;
40  domains[nuc_i] = new Domain_nucleus (nuc_i, ttype, bounds[RIN], center, res) ;
41  domains[nuc_i+1] =
42  new Domain_shell_outer_homothetic (*this, nuc_i+1, ttype, bounds[RIN], bounds[RMID], center, res) ;
43  domains[nuc_i+2] =
44  new Domain_shell_inner_homothetic (*this, nuc_i+2, ttype, bounds[RMID], bounds[RMID+1], center, res) ;
45 
46  for(int i = 0; i < shells; ++i)
47  domains[nuc_i+3+i] = new Domain_shell(nuc_i+3+i, ttype, bounds[RMID+1+i], bounds[RMID+1+i+1], center, res);
48  };
49  gen_bh_domains(0, BH_bounds);
50 
51  domains[nbr_domains-1] = new Domain_compact (nbr_domains-1, ttype, BH_bounds.back(), center, res) ;
52 
53  const Domain_shell_outer_homothetic* pouter_1 = dynamic_cast<const Domain_shell_outer_homothetic*> (domains[1]) ;
54  pouter_1->vars_to_terms() ;
55  pouter_1->update() ;
56  const Domain_shell_inner_homothetic* pinner_1 = dynamic_cast<const Domain_shell_inner_homothetic*> (domains[2]) ;
57  pinner_1->vars_to_terms() ;
58  pinner_1->update() ;
59 }

99  domains[1] = new Domain_shell_outer_homothetic(*this, *pouter) ;

101  domains[2] = new Domain_shell_inner_homothetic(*this, *pinner) ;
102

104  const Domain_shell* d_shell = dynamic_cast<const Domain_shell*> (sp.get_domain(i)) ;
105  domains[i] = new Domain_shell(*d_shell, true) ;
106  }
107  // Compactified
108  const Domain_compact* d_compact = dynamic_cast<const Domain_compact*> (sp.get_domain(nbr_domains-1)) ;
109  domains[nbr_domains-1] = new Domain_compact(*d_compact, true) ;
110 
111  const Domain_shell_outer_homothetic* pouter_1 = dynamic_cast<const Domain_shell_outer_homothetic*> (domains[1]) ;

163 void Space_adapted_bh::xx_to_vars_variable_domains (System_of_eqs* sys, const Array<double>& xx, int& pos) const {
164 
165  // First get the corrections :
166  int old_pos = pos ;
167  Val_domain cor_outer (domains[1]) ;
168  domains[1]->xx_to_vars_from_adapted(cor_outer, xx, pos) ;
169  pos = old_pos ;
170  Val_domain cor_inner (domains[2]) ;
171  domains[2]->xx_to_vars_from_adapted(cor_inner, xx, pos) ;
172 
173  // Now update the variables :
174  for (int i=0 ; i<sys->nvar ; i++) {
175  for (int n=0 ; n<sys->var[i]->get_n_comp() ; n++) {
176  Scalar res(*this) ;
177  domains[1]->update_variable(cor_outer, *sys->var[i]->cmp[n], res) ;
178  domains[2]->update_variable(cor_inner, *sys->var[i]->cmp[n], res) ;
179 
180  sys->var[i]->cmp[n]->set_domain(1) = res(1) ;
181  sys->var[i]->cmp[n]->set_domain(2) = res(2) ;
182  }
183  }
184 
185  // Now the constants :
186  for (int i=0 ; i<sys->ncst ; i++)
187  if (sys->cst[i*(sys->dom_max-sys->dom_min+1)]->get_type_data()==TERM_T)
188  for (int n=0 ; n<sys->cst[i*(sys->dom_max-sys->dom_min+1)]->val_t->get_n_comp() ; n++) {
189 
190  Scalar so (*this) ;
191  so = 0 ;
192  so.set_domain(1) = (*sys->cst[i*(sys->dom_max-sys->dom_min+1) + 1]->val_t->cmp[n])(1) ;
193  so.set_domain(2) = (*sys->cst[i*(sys->dom_max-sys->dom_min+1) + 2]->val_t->cmp[n])(2) ;
194 
195  Scalar res(*this) ;
196  res = 0 ;
197  domains[1]->update_constante(cor_outer, so, res) ;
198  domains[2]->update_constante(cor_inner, so, res) ;
199 
200  sys->cst[i*(sys->dom_max-sys->dom_min+1) + 1]->val_t->cmp[n]->set_domain(1) = res(1) ;
201  sys->cst[i*(sys->dom_max-sys->dom_min+1) + 2]->val_t->cmp[n]->set_domain(2) = res(2) ;
202  }
203 
204  // Update the mapping :
205  domains[1]->update_mapping(cor_outer) ;
206  domains[2]->update_mapping(cor_inner) ;
207 }
208 Array<int> Space_adapted_bh::get_indices_matching_non_std(int dom, int bound) const {
209 
210  assert ((dom>=0) && (dom<nbr_domains)) ;
211  Array<int> res (2, 1) ;
212  switch (bound) {
213  case OUTER_BC :
214  res.set(0,0) = dom+1 ;
215  res.set(1,0) = INNER_BC ;
216  break ;
217  case INNER_BC :
218  res.set(0,0) = dom-1 ;
219  res.set(1,0) = OUTER_BC ;
220  break ;
221  default :
222  cerr << "Unknown boundary in " << endl ;
223  cerr << *this << endl ;
224  abort() ;
225  }
226  return res ;
227 }
