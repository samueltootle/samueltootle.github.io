284  output_ary_t operator()(CFMS_NS_Exporter& base,
285  double const& x,
286  double const& y,
287  double const& z) {
288  // Reset to NAN
289  base.quant_vals.fill(NAN);
290  base.quant_vals =
291  base.interpolate_pointwise_subset(x, y, z, base.xcts_fluid_indicies);
292 
293  double const H = base.quant_vals[XCTS_VARS::XCTS_H];
294  double h = std::exp(H);
295  double rho, eps, press;
296 
297  // get quantities point-wise, since h is smoothest, and cut data at H=0
298  if (std::fabs(H) <= 1e-12) {
299  rho = 0.;
300  eps = 0.;
301  press = 0.;
302  } else {
303  rho = EOS<eos_t, DENSITY>::get(h);
304  eps = EOS<eos_t, EPSILON>::get(h);
305  press = EOS<eos_t, PRESSURE>::get(h);
306  }
307  base.out_pw[OUTPUT_VARS::RHO] = rho;
308  base.out_pw[OUTPUT_VARS::EPS] = eps;
309  base.out_pw[OUTPUT_VARS::PRESS] = press;
310  base.out_pw[OUTPUT_VARS::VEL1] = base.quant_vals[XCTS_VARS::XCTS_UX];
311  base.out_pw[OUTPUT_VARS::VEL2] = base.quant_vals[XCTS_VARS::XCTS_UY];
312  base.out_pw[OUTPUT_VARS::VEL3] = base.quant_vals[XCTS_VARS::XCTS_UZ];
313  return base.out_pw;
314  }
