27 int Domain_compact::nbr_conditions_val_domain_boundary (const Val_domain& so, int mlim) const {
28 
29  int res = 0 ;
30  int kmin = 2*mlim + 2 ;
31 
32  for (int k=0 ; k<nbr_coefs(2) ; k++)
33  for (int j=0 ; j<nbr_coefs(1) ; j++) {
34  bool indic = true ;
35  // True coef in phi ?
36  if ((k==1) || (k==nbr_coefs(2)-1))
37  indic = false ;
38  // Get base in theta :
39  int baset = (*so.get_base().bases_1d[1])(k) ;
40  switch (baset) {
41  case COS_EVEN:
42  if ((j==0) && (k>=kmin))
43  indic = false ;
44  break ;
45  case COS_ODD:
46  if ((j==nbr_coefs(1)-1) || ((j==0) && (k>=kmin)))
47  indic = false ;
48  break ;
49  case SIN_EVEN:
50  if (((j==1)&&(k>=kmin+2)) || (j==0) || (j==nbr_coefs(1)-1))
51  indic = false ;
52  break ;
53  case SIN_ODD:
54  if (((j==0)&&(k>=kmin+2))||(j==nbr_coefs(1)-1))
55  indic = false ;
56  break ;
57  default:
58  cerr << "Unknow theta basis in Domain_compact::nbr_conditions_val_boundary" << endl ;
59  abort() ;
60  }
61 
62  if (indic)
63  res ++ ;
64  }
65  return res ;
66 }
