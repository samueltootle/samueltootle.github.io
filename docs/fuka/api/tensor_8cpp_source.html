35 int std_position_array (const Array<int>& idx, int ndim) {
36  assert (idx.get_ndim() == 1) ;
37  int valence =idx.get_size(0) ;
38 
39  for (int i=0 ; i<valence ; i++)
40  assert ((idx(i)>=1) && (idx(i)<=ndim)) ;
41  int res = 0 ;
42  for (int i=0 ; i<valence ; i++)
43  res = ndim*res+(idx(i)-1) ;
44 
45  return res;
46 }
47 
48 int std_position_index (const Index& idx, int ndim) {
49  int valence =idx.get_ndim() ;
50  for (int i=0 ; i<valence ; i++)
51  assert ((idx(i)>=0) && (idx(i)<ndim)) ;
52  int res = 0 ;
53  for (int i=0 ; i<valence ; i++)
54  res = ndim*res+(idx(i)) ;
55 
56  return res;
57 }
58 
59 Array<int> std_indices (int place, int valence, int ndim) {
60  Array<int> res(valence) ;
61  for (int i=valence-1 ; i>=0 ; i--) {
62  res.set(i) = div(place, ndim).rem ;
63  place = int((place-res(i))/ndim) ;
64  res.set(i)++ ;
65  }
66  return res ;
67 }

638 // Sets the standard spectal bases of decomposition for each component
639 void Tensor::std_base() {
640  // Loop on the domains
641  for (int d=0 ; d<ndom ; d++) {
642 
643  switch (valence) {
644 
645  case 0 : {
646  if (!is_m_quant_affected())
647  cmp[0]->std_base_domain(d) ;
648  else
649  cmp[0]->std_base_domain(d, parameters->get_m_quant()) ;
650  break ;
651  }
652  case 1 : {
653  bool done = false ;
654  if (basis.get_basis(d) ==CARTESIAN_BASIS) {
655  cmp[0]->std_base_x_cart_domain(d) ;
656  cmp[1]->std_base_y_cart_domain(d) ;
657  cmp[2]->std_base_z_cart_domain(d) ;
658  done = true ;
659  }
660  if (basis.get_basis(d) == SPHERICAL_BASIS) {
661  cmp[0]->std_base_r_spher_domain(d) ;
662  cmp[1]->std_base_t_spher_domain(d) ;
663  cmp[2]->std_base_p_spher_domain(d) ;
664  done = true ;
665  }
