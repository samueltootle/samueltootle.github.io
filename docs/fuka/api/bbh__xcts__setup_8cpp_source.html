16 inline void bbh_xcts_setup_headon_config(config_t& bconfig) {
17  using namespace ::Kadath::bco_utils;
18  check_dist(bconfig(BIN_PARAMS::DIST), bconfig(BCO_PARAMS::MCH, NODES::BCO1),
19  bconfig(BCO_PARAMS::MCH, NODES::BCO2));
20 
21  // Binary Parameters
22  bconfig.set(BIN_PARAMS::REXT) = 2 * bconfig(BIN_PARAMS::DIST);
23 
24  bconfig.set(BIN_PARAMS::Q) = bconfig(BCO_PARAMS::MCH, NODES::BCO2) /
25  bconfig(BCO_PARAMS::MCH, NODES::BCO1);
26  bconfig.set(BIN_PARAMS::Q) = (bconfig.set(BIN_PARAMS::Q) > 1.0)
27  ? 1.0 / bconfig.set(BIN_PARAMS::Q)
28  : bconfig.set(BIN_PARAMS::Q);
29  // classical Newtonian estimate
30  bconfig.set(BIN_PARAMS::COM) =
31  com_estimate(bconfig(BIN_PARAMS::DIST),
32  bconfig(BCO_PARAMS::MCH, NODES::BCO1),
33  bconfig(BCO_PARAMS::MCH, NODES::BCO2));
34 }

109  // update binary parameters - but save shell input
110  const int nshells1 = bconfig(BCO_PARAMS::NSHELLS, NODES::BCO1);
111  const int nshells2 = bconfig(BCO_PARAMS::NSHELLS, NODES::BCO2);
112  for (int i = 0; i < BCO_PARAMS::NUM_BCO_PARAMS; ++i) {
113  bconfig.set(i, NODES::BCO1) = BH1config.set(i);
114  bconfig.set(i, NODES::BCO2) = BH2config.set(i);
115  }
116  bconfig.set(BCO_PARAMS::NSHELLS, NODES::BCO1) = nshells1;
117  bconfig.set(BCO_PARAMS::NSHELLS, NODES::BCO2) = nshells2;
118 
119  const double r_max_tot = std::max(bconfig(BCO_PARAMS::RMID, NODES::BCO1),
120  bconfig(RMID, NODES::BCO2));
121 
122  const double rout_sep_est =
123  (bconfig(BIN_PARAMS::DIST) / 2. - r_max_tot) / 3. + r_max_tot;
124  // const double rout_min_est = bco_u::gold_ratio * r_max_tot;
125 
126  bconfig.set(BCO_PARAMS::ROUT, NODES::BCO1) = rout_sep_est;
127  // (rout_sep_est < rout_min_est) ? rout_min_est : rout_sep_est;
128  bconfig.set(BCO_PARAMS::ROUT, NODES::BCO2) =
129  bconfig(BCO_PARAMS::ROUT, NODES::BCO1);
130 
131  const double q = bconfig.set(BIN_PARAMS::Q);
132  bconfig.set(BCO_PARAMS::MIN_SHELL_DR, NODES::BCO1) =
133  1.7 * std::pow(2.0, bco_u::shell_factor / q);
134  bconfig.set(BCO_PARAMS::MIN_SHELL_DR, NODES::BCO2) =
135  bconfig(BCO_PARAMS::MIN_SHELL_DR, NODES::BCO1);
136 
137  std::vector<double> out_bounds(1 + bconfig(BIN_PARAMS::OUTER_SHELLS));
138  std::vector<double> BH1_bounds;
139  {
140  auto drPsi(compute_drPsi(spacein1, confin1,
141  Metric_flat(spacein1, shiftin1.get_basis()),
142  {0, 1}));
143  BH1_bounds = bco_u::set_arb_boundsv3(bconfig, drPsi, 2, NODES::BCO1);
144  }
145  std::vector<double> BH2_bounds;
146  {
147  auto drPsi(compute_drPsi(spacein2, confin2,
148  Metric_flat(spacein2, shiftin2.get_basis()),
149  {0, 1}));
150  BH2_bounds = bco_u::set_arb_boundsv3(bconfig, drPsi, 2, NODES::BCO2);
151  }
152 
153  // for out_bounds.size > 1 - add equi-distant shells
154  for (int e = 0; e < out_bounds.size(); ++e)
155  out_bounds[e] =
156  bconfig(BIN_PARAMS::REXT) + e * 0.25 * bconfig(BIN_PARAMS::REXT);
157  // end setup domain boundaries
158 
159  // create space containing the domain decomposition
160  int type_coloc = CHEB_TYPE;
161  Space_bin_bh space(type_coloc, bconfig(BIN_PARAMS::DIST), BH1_bounds,
162  BH2_bounds, out_bounds, bconfig(BIN_PARAMS::BIN_RES));
163  Base_tensor basis(space, CARTESIAN_BASIS);

165  auto interp_BH_fields = [&](auto& bhconf, auto& bhlapse, auto& bhshift,
166  auto& bhspacein) {
167  const Domain_shell_outer_homothetic* old_bh_outer =
168  dynamic_cast<const Domain_shell_outer_homothetic*>(
169  bhspacein.get_domain(1));
170  // Update BH fields based to help with interpolation later
171  bco_u::update_adapted_field(bhconf, 2, 1, old_bh_outer, OUTER_BC);
172  bco_u::update_adapted_field(bhlapse, 2, 1, old_bh_outer, OUTER_BC);
173  for (int i = 1; i <= 3; ++i)
174  bco_u::update_adapted_field(bhshift.set(i), 2, 1, old_bh_outer, OUTER_BC);
175  };

178  interp_BH_fields(confin2, lapsein2, shiftin2, spacein2);

181  double xc2 = bco_u::get_center(space, space.BH2);

183  Scalar conf(space);
184  conf.annule_hard();
185 
186  Scalar lapse(space);

191  shift.set(i).annule_hard();

193  const double bh1_invw4 = bco_u::set_decay(bconfig, NODES::BCO1);
194  const double bh2_invw4 = bco_u::set_decay(bconfig, NODES::BCO2);
195 
196  // start importing the fields from the single star
197  int ndom = space.get_nbr_domains();
198  for (int dom = 0; dom < ndom; dom++) {
199  // get an index in each domain to iterate over all colocation points
200  Index new_pos(space.get_domain(dom)->get_nbr_points());
201 
202  do {
203  if (dom <= space.BH1 + 1 || dom == space.BH2 || dom == space.BH2 + 1)
204  continue;
205  // get cartesian coordinates of the current colocation point
206  double x = space.get_domain(dom)->get_cart(1)(new_pos);
207  double y = space.get_domain(dom)->get_cart(2)(new_pos);
208  double z = space.get_domain(dom)->get_cart(3)(new_pos);
209 
210  // define a point shifted suitably to the stellar centers in the binary
211  Point absol1(3);
212  absol1.set(1) = (x - xc1);
213  absol1.set(2) = y;
214  absol1.set(3) = z;
215  double r2 = y * y + z * z;
216  double r2_1 = (x - xc1) * (x - xc1) + r2;
217  double r4_1 = r2_1 * r2_1;
218  double r4_invw4_1 = r4_1 * bh1_invw4;
219  double decay_1 = std::exp(-r4_invw4_1);
220 
221  Point absol2(3);
222  absol2.set(1) = (x - xc2);
223  absol2.set(2) = y;
224  absol2.set(3) = z;
225  double r2_2 = (x - xc2) * (x - xc2) + r2;
226  double r4_2 = r2_2 * r2_2;
227  double r4_invw4_2 = r4_2 * bh2_invw4;
228  double decay_2 = std::exp(-r4_invw4_2);
229 
230  if (dom < ndom - 1) {
231  conf.set_domain(dom).set(new_pos) =
232  1. + decay_1 * (confin1.val_point(absol1) - 1.) +
233  decay_2 * (confin2.val_point(absol2) - 1.);
234  lapse.set_domain(dom).set(new_pos) =
235  1. + decay_1 * (lapsein1.val_point(absol1) - 1.) +
236  decay_2 * (lapsein2.val_point(absol2) - 1.);
237  for (int i = 1; i <= 3; i++)
238  shift.set(i).set_domain(dom).set(new_pos) =
239  decay_1 * shiftin1(i).val_point(absol1) +
240  decay_2 * shiftin2(i).val_point(absol2);
241 
242  } else {
243  // We have to set the compactified domain manually since the outer
244  // collocation point is always at inf which is undefined numerically
245  conf.set_domain(dom).set(new_pos) = 1.;
246  lapse.set_domain(dom).set(new_pos) = 1.;
247  }
248  // loop over all colocation points
249  } while (new_pos.inc());
250  } // end importing fields
251 
252  // safety to ensure excision region is empty
253  auto clear_excision_region = [&](int nuc) {
254  for (int d = nuc; d < nuc + 2; ++d) {
255  conf.set_domain(d).annule_hard();
256  lapse.set_domain(d).annule_hard();
257  for (int i = 1; i <= 3; i++)
