28 Tensor Domain_bispheric_eta_first::import (int numdom, int bound, int n_ope, const Array<int>& zedoms, Tensor** parts) const {
29 
30  Tensor res (*parts[0], false);
31 
32  for (int nc=0 ; nc<res.get_n_comp() ; nc++)
33  res.cmp[nc]->set_domain(numdom).allocate_conf() ;
34 
35  // Need coefficients of parts :
36  for (int i=0 ; i<n_ope ; i++)
37  for (int nc=0 ; nc<parts[i]->get_n_comp() ; nc++)
38  parts[i]->cmp[nc]->set_domain(zedoms(i)).coef() ;
39 
40  // Loop on the points of the boundary :
41  Val_domain xx (get_cart(1)) ;
42  Val_domain yy (get_cart(2)) ;
43  Val_domain zz (get_cart(3)) ;
44 
45  int index_chi ;
46  switch (bound) {
47  case OUTER_BC :
48  index_chi = nbr_points(0)-1 ;
49  break ;
50  default :
51  cerr << "Unknown boundary in Domain_bispheric_eta_first::import" << endl ;
52  }
53 
54  Index pos (get_nbr_points()) ;
55  Index pos_bound (get_nbr_points()) ;
56 
57  for (int k=0 ; k<nbr_points(2) ; k++)
58  for (int j=0 ; j<nbr_points(1) ; j++) {
59 
60  // Indices in the shell
61  pos_bound.set(0) = index_chi ;
62  pos_bound.set(1) = j ;
63  pos_bound.set(2) = k ;
64 
65  // Absolute coordinates
66  Point MM (3) ;
67  MM.set(1) = xx(pos_bound) ;
68  MM.set(2) = yy(pos_bound) ;
69  MM.set(3) = zz(pos_bound) ;
70 
71  // In which other domain is it ?
72  bool found = false ;
73  int current = 0 ;
74  while ((current<n_ope) && (!found)) {
75  if (parts[0]->get_space().get_domain(zedoms(current))->is_in(MM, 1e-3))
76  found = true ;
77  else
78  current ++ ;
79  }
80  if (!found) {
81  cerr << "Point " << MM << " not found in other domains, for Domain_bispheric_eta_first::import" << endl ;
82  abort() ;
83  }
84 
85  // Convert to numerical coordinates of the other domain
86  Point num (parts[0]->get_space().get_domain(zedoms(current))->absol_to_num(MM)) ;
87 
88  // Now loop on the components :
89  for (int nc=0 ; nc<res.get_n_comp() ; nc++) {
90  double val = ((*parts[current]->cmp[nc])(zedoms(current)).check_if_zero()) ? 0 : (*parts[current]->cmp[nc])(zedoms(current)).get_base().summation (num, *(*parts[current]->cmp[nc])(zedoms(current)).cf) ;
91  // Loop on radius :
92  for (int i=0 ; i<nbr_points(0) ; i++) {
93  pos.set(0) = i ;
94  pos.set(1) = j ;
95  pos.set(2) = k ;
96 
97  res.cmp[nc]->set_domain(numdom).set(pos) = val ;
98  }
99  }
100  }
101 
102  // Assert a std_base :
103  res.std_base() ;
104 
105 
106  return res ;
107 }
108 }
