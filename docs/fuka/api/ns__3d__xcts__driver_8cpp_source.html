153  base_config.control(CONTROLS::RESOLVE))) {
154  base_config = seqconfig;
155  base_config.set(BCO_PARAMS::CHI) = 0;
156  }
157  auto mass_fixing = seq.mass_idx();
158  auto spin_fixing = seq.spin_idx();
159 
160  // in the event the user wants an MADM > MTOV
161  const double final_MADM = (seq.mass_idx() == BCO_PARAMS::MADM)
162  ? base_config(BCO_PARAMS::MADM)
163  : std::nan("1");
164  base_config.control(CONTROLS::ITERATIVE_M) = false;
165  config_t bconfig{base_config};
166 
167  const std::string eos_type =
168  bconfig.template eos<std::string>(EOS_PARAMS::EOSTYPE);
169 
170  if (bconfig.control(CONTROLS::SEQUENCES) ||
171  bconfig.control(CONTROLS::RESOLVE)) {
172  if (rank == 0) {
173  EOS_Function_Dispatcher::dispatch<setup_3dns_xcts_functor>(bconfig,
174  eos_type,
175  bconfig,
176  mass_fixing);
177  }
178  MPI_Barrier(MPI_COMM_WORLD);
179  // make sure all ranks have the same config
180  bconfig.open_config();
181  MPI_Barrier(MPI_COMM_WORLD);
182  bconfig.control(CONTROLS::ITERATIVE_M) =
183  !std::isnan(final_MADM) &&
184  (std::fabs(1. - bconfig(BCO_PARAMS::MADM) / final_MADM) > 1e-3);
185 
186  if (bconfig.control(CONTROLS::ITERATIVE_M) &&
187  std::fabs(bconfig(BCO_PARAMS::CHI)) < 1e-5) {
188  if (rank == 0)
189  std::cerr << "Cannot solve TOV for Madm = " << final_MADM
190  << " without spin.\n";
191  std::_Exit(EXIT_FAILURE);
192  }
193  }
194 
195  // Need to get a rotating solution before increasing the
196  // NS mass up to final_MADM
197  std::array<bool, NUM_STAGES>& stage_enabled = bconfig.return_stages();
198  auto [last_stage, last_stage_idx] = get_last_enabled(MSTAGE, stage_enabled);
199  if (bconfig.control(CONTROLS::ITERATIVE_M)) {
200  // Only obtain the iterative solution at the initial_resolution
201  auto const res_init{resolution.init()};
202  Parameter_sequence tmp_res("res", BCO_PARAMS::BCO_RES);
203  tmp_res.set(res_init, res_init, res_init);
204 
205  // exit_status = ns_3d_xcts_driver(bconfig, tmp_res, outputdir);
206  exit_status = ns_3d_xcts_base_solution_driver(bconfig, outputdir, &seq);
207 
208  // Update config such that the next solving round uses
209  // the final ADM mass and spin
210  bconfig(BCO_PARAMS::MADM) = final_MADM;
211  bconfig.control(CONTROLS::SEQUENCES) = false;
212 
213  // Ensure only the final stage is used
214  // e.g. avoid NOROT stage
215  stage_enabled.fill(false);
216  stage_enabled[last_stage_idx] = true;
217  }
218  exit_status = ns_3d_xcts_base_solution_driver(bconfig, outputdir, &seq);
219  // Ensure only the final stage is used
220  // e.g. avoid NOROT stage
221  stage_enabled.fill(false);
222  stage_enabled[last_stage_idx] = true;
223 
224  base_config = bconfig;
225  if (seq.is_set() && std::isnan(bconfig.set(sequence_var_indices)))
226  bconfig.set(sequence_var_indices) = seq.init();
227 #ifdef DEBUG
228  std::cout << seq << std::endl;
229  std::cout << resolution << std::endl;
230 #endif
231  auto single_seq = [&](auto val) {
232  // bconfig = base_config;
233  stage_enabled[last_stage_idx] = true;
234  auto old_val = bconfig(sequence_var_indices);
235  if (seq.is_set())
236  bconfig.set(sequence_var_indices) = val;
237  if (std::get<0>(sequence_var_indices) == BCO_PARAMS::CHI) {
238  bconfig.seq_setting(SEQ_SETTINGS::FINAL_CHI) = bconfig(BCO_PARAMS::CHI);
239  if (std::fabs(old_val) < 0.1 && val > 0.2) {
240  bconfig.control(CONTROLS::SEQUENCES) = true;
241  exit_status = ns_3d_xcts_base_solution_driver(bconfig, outputdir);
242  }
243  }
244  exit_status = ns_3d_xcts_driver(bconfig, resolution, outputdir, &seq);
245  resolution.set(resolution.final(), resolution.final(), resolution.final());
246  return exit_status;
247  };
248 
249  // Loop if a valid sequence is set
250  if (seq.is_set() && std::fabs(dx) > 0) {
251  for (auto val = seq.init(); seq.loop_condition(val); val += dx) {
252  exit_status = single_seq(val);
253  }
254  } else if (seq.is_set()) {
255  exit_status = single_seq(seq.init());
256  } else {
257  exit_status = single_seq(0);
258  }
259  return bconfig;
260 }

305 template <typename config_t>
306 int ns_3d_xcts_base_solution_driver(config_t& bconfig,
307  std::string outputdir,
308  ns_sequence const* seq) {
309  int exit_status = RELOAD_FILE;
310  int rank = 0;
311  MPI_Comm_rank(MPI_COMM_WORLD, &rank);
312 
313  if (std::abs(bconfig(BCO_PARAMS::CHI)) > 0.7) {
314  std::cerr << "Unable to handle chi > 0.7\n";
315  return EXIT_FAILURE;
316  }
317  // We need to store the final desired CHI in case of iterative chi
318  bconfig.seq_setting(SEQ_SETTINGS::FINAL_CHI) = bconfig(BCO_PARAMS::CHI);
319 
320  // In the event we wish to solve for a highly spinning solution
321  // we need to do an initial slow rotating solution before going to
322  // faster rotations otherwise the solution will diverge.
323  bconfig.control(CONTROLS::ITERATIVE_CHI) =
324  bconfig.control(CONTROLS::SEQUENCES) &&
325  std::fabs(bconfig.seq_setting(SEQ_SETTINGS::FINAL_CHI)) > 0.2;
326 
327  // Lower chi in case of iterative chi
328  bconfig(BCO_PARAMS::CHI) =
329  (bconfig.control(CONTROLS::ITERATIVE_CHI))
330  ? std::copysign(0.1, bconfig.seq_setting(SEQ_SETTINGS::FINAL_CHI))
331  : bconfig(BCO_PARAMS::CHI);
332 
333  // make sure NS directory exists for outputs
334  if (outputdir == "./") {
335  fs::path cwd = fs::current_path();
336  outputdir = cwd.string();
337  }
338  if (rank == 0)
339  std::cout << "Solutions will be stored in: " << outputdir << "\n"
340  << "Directory will be created if it doesn't exist.\n";
341  fs::create_directory(outputdir);
342 
343  // if(std::isnan(bconfig.set(BCO_PARAMS::MADM)) &&
344  // std::isnan(bconfig.set(BCO_PARAMS::MB))){
345  // if(rank == 0)
346  // std::cout << "Config error. No madm nor mb found. \n\n";
347  // std::_Exit(EXIT_FAILURE);
348  // }
349 
350  while (exit_status == RELOAD_FILE) {
351  exit_status = ns_3d_xcts_stationary_driver(bconfig, outputdir, seq);
352  MPI_Barrier(MPI_COMM_WORLD);
353  }
354  return exit_status;
355 }
