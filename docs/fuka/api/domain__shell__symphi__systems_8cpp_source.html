42 double Domain_shell_symphi::val_boundary (int bound, const Val_domain& so, const Index& pos_cf) const {
43 
44  if (so.check_if_zero())
45  return 0. ;
46 
47  else {
48  so.coef() ;
49  double res = 0 ;
50  Index copie_pos (pos_cf) ;
51  switch (bound) {
52  case OUTER_BC :
53  for (int i=0 ; i<nbr_coefs(0) ; i++) {
54  copie_pos.set(0) = i ;
55  res += so.get_coef(copie_pos) ;
56  }
57  break ;
58  case INNER_BC :
59  for (int i=0 ; i<nbr_coefs(0) ; i++) {
60  copie_pos.set(0) = i ;
61  if (i%2==0)

65  }
66  break ;
67  default :
68  cerr << "Unknown boundary type in Domain_shell_symphi::val_boundary" << endl ;
69  abort() ;
