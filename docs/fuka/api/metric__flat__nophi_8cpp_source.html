52 void Metric_flat_nophi::compute_cov (int dd) const {
53  Metric_tensor res (espace, COV, basis) ;
54 
55  for (int i=1 ; i<=3 ; i++)
56  for (int j=i ; j<=3 ; j++)
57  res.set(i,j).set_domain(dd) = (i==j) ? 1. : 0 ;
58  res.std_base() ;
59 
60  if (p_met_cov[dd]==0x0)
61  p_met_cov[dd] = new Term_eq(dd, res) ;
62  else
63  *p_met_cov[dd] = Term_eq (dd, res) ;
64  p_met_cov[dd]->set_der_zero() ;
65 }

98 Term_eq Metric_flat_nophi::derive_partial_spher (int type_der, char ind_der, const Term_eq& so) const {
99 
100  int dom = so.get_dom() ;
101  bool donames = ((so.val_t->is_name_affected()) || (so.val_t->get_valence()==0)) ? true : false ;
102 
103  // Computation of flat gradient :
104  Term_eq auxi (espace.get_domain(dom)->partial_spher (so)) ;
105 
106  int val_res = auxi.val_t->get_valence() ;
107 
108  if (donames) {
109  // Set the names of the indices :
110  auxi.val_t->set_name_affected() ;
111  auxi.val_t->set_name_ind(0, ind_der) ;
112  for (int i=1 ; i<val_res ; i++)
113  auxi.val_t->set_name_ind(i, so.val_t->get_name_ind()[i-1]) ;
114 
115  if (auxi.der_t!=0x0) {
116  auxi.der_t->set_name_affected() ;
117  auxi.der_t->set_name_ind(0, ind_der) ;
118  for (int i=1 ; i<val_res ; i++)
119  auxi.der_t->set_name_ind(i, so.val_t->get_name_ind()[i-1]) ;
120  }
121  }
122 
123  // Manipulate if Contravariant version
124  if (type_der==CON)
125  manipulate_ind (auxi, 0) ;
126 
127  bool need_sum = false ;
128  if (donames)
129  for (int i=1 ; i<val_res ; i++)
130  if (ind_der== so.val_t->get_name_ind()[i-1])
131  need_sum = true ;
132 
133  if (!need_sum)
134  return auxi ;
135  else {
136  if (auxi.der_t==0x0)
137  return Term_eq (dom, auxi.val_t->do_summation_one_dom(dom)) ;
138  else
139  return Term_eq (dom, auxi.val_t->do_summation_one_dom(dom), auxi.der_t->do_summation_one_dom(dom)) ;
140  }
141 }

144 Term_eq Metric_flat_nophi::derive_partial (int type_der, char ind_der, const Term_eq& so) const {

147 
148  if (p_met_con[dom]==0x0)
149  compute_con(dom) ;
150  if (p_met_cov[dom]==0x0)

152 
153  // so must be tensor :

155  cerr << "Metric_flat_nophi::derive partial only defined for tensor data" << endl ;

169 Term_eq Metric_flat_nophi::derive_spher (int type_der, char ind_der, const Term_eq& so) const {
170 
171  // Computation of flat gradient :
172  Term_eq part_der (derive_partial(type_der, ind_der, so)) ;
173 
174  int dom = so.get_dom() ;
175  bool donames = ((so.val_t->is_name_affected()) || (so.val_t->get_valence()==0)) ? true : false ;
176 
177  Term_eq auxi (espace.get_domain(dom)->connection_spher (so)) ;
178  int val_res = auxi.val_t->get_valence() ;
179 
180  if (donames) {
181  // Set the names of the indices :
182  auxi.val_t->set_name_affected() ;
183  auxi.val_t->set_name_ind(0, ind_der) ;
184  for (int i=1 ; i<val_res ; i++)
185  auxi.val_t->set_name_ind(i, so.val_t->get_name_ind()[i-1]) ;
186 
187  if (auxi.der_t!=0x0) {
188  auxi.der_t->set_name_affected() ;
189  auxi.der_t->set_name_ind(0, ind_der) ;
190  for (int i=1 ; i<val_res ; i++)
191  auxi.der_t->set_name_ind(i, so.val_t->get_name_ind()[i-1]) ;
192  }
193  }
194 
195  // Manipulate if Contravariant version
196  if (type_der==CON)
197  manipulate_ind (auxi, 0) ;
198 
199 
200  bool need_sum = false ;
201  if (donames)
202  for (int i=1 ; i<val_res ; i++)
203  if (ind_der== so.val_t->get_name_ind()[i-1])
204  need_sum = true ;
205 
206  if (!need_sum) {
207  return (part_der + auxi) ;
208  }
209  else {
210  if (auxi.der_t==0x0) {
211 
212 
213  return (part_der + Term_eq (dom, auxi.val_t->do_summation_one_dom(dom))) ;
214  }
215  else
216  return (part_der + Term_eq (dom, auxi.val_t->do_summation_one_dom(dom), auxi.der_t->do_summation_one_dom(dom))) ;
217  }
218 }
