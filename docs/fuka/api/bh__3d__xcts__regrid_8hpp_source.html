40 template <typename config_t>
41 int bh_3d_xcts_regrid(config_t& bconfig, std::string outputfile) {
42  int exit_status = EXIT_SUCCESS;
43 
44  std::string in_spacefile = bconfig.space_filename();
45 
46  if (!fs::exists(in_spacefile)) {
47  std::cerr << "File: " << in_spacefile << " not found.\n\n";
48  std::_Exit(EXIT_FAILURE);
49  }
50 
51  using space_t = Space_adapted_bh;
52  FILE* ff1 = fopen(in_spacefile.c_str(), "r");
53  space_t old_space(ff1);
54  Scalar old_conf(old_space, ff1);
55  Scalar old_lapse(old_space, ff1);
56  Vector old_shift(old_space, ff1);
57  fclose(ff1);
58 
59  // FIXME not sure if it's only about oddness...
60  if (((int)bconfig(BCO_RES) % 2) == 0) {
61  std::cout << "New Resolution is invalid. Must be odd (9,11,13,etc)"
62  << std::endl;
63  std::_Exit(EXIT_FAILURE);
64  }
65 
66  // Make sure the domains have adequate space between them based on
67  // the previous solution
68  bconfig.set(RMID) =
69  Kadath::bco_utils::get_radius(old_space.get_domain(1), OUTER_BC);
70 
71  // estimate how small the inner radius should be based on relation
72  // between conformal factor and numerical radius.
73  // see https://arxiv.org/pdf/0805.4192, eq(64)
74  double conf_inner =
75  Kadath::bco_utils::get_boundary_val(2, old_conf, INNER_BC);
76  double conf_i_sq = conf_inner * conf_inner;
77  double est_r_div2 = bconfig(MCH) / conf_i_sq;
78  bconfig.set(RIN) = est_r_div2;
79  bconfig.set(ROUT) = 4. * est_r_div2 * 2.;
80 
81  bconfig.set_filename(outputfile);
82 
83  // Set this to use radii in setup_co
84  bconfig.control(USE_CONFIG_VARS) = true;
85 
86  // Generate new space and fields
87  setup_3d_BH_xcts(bconfig);
88 
89  // Unset to prevent possible issues on ID reuse
90  bconfig.control(USE_CONFIG_VARS) = false;
91 
92  // Read in new fields
93  auto new_spacefile = bconfig.space_filename();
94  FILE* ff2 = fopen(new_spacefile.c_str(), "r");
95  space_t space(ff2);
96  Scalar conf(space, ff2);
97  Scalar lapse(space, ff2);
98  Vector shift(space, ff2);
99  fclose(ff2);
100 
101  std::cout << "Resolution of old space: ";
102  Kadath::bco_utils::print_constant_space_resolution(old_space);
103 
104  std::cout << "Resolution of new space: ";
105  Kadath::bco_utils::print_constant_space_resolution(space);
106 
107  std::cout << "\nold bounds:" << std::endl;
108  Kadath::bco_utils::print_bounds_from_space(old_space);
109 
110  std::cout << "New bounds:" << std::endl;
111  Kadath::bco_utils::print_bounds_from_space(space);
112  std::cout << endl;
113 
114  // needed in some cases, since the interpolation can go crazy
115  const Domain_shell_outer_homothetic* old_outer_homothetic =
116  dynamic_cast<const Domain_shell_outer_homothetic*>(
117  old_space.get_domain(1));
118 
119  // import fields
120  Kadath::bco_utils::update_adapted_field(old_conf, 2, 1, old_outer_homothetic,
121  OUTER_BC);
122  Kadath::bco_utils::update_adapted_field(old_lapse, 2, 1, old_outer_homothetic,
123  OUTER_BC);
124  for (int i = 1; i <= 3; ++i)
125  Kadath::bco_utils::update_adapted_field(old_shift.set(i), 2, 1,
126  old_outer_homothetic, OUTER_BC);
127 
128  conf.import(old_conf);
129  lapse.import(old_lapse);
130 
131  shift.set(1).import(old_shift.set(1));
132  shift.set(2).import(old_shift.set(2));
133  shift.set(3).import(old_shift.set(3));
134  // end import fields
135 
136  // reset fields to zero inside excised region
137  for (auto& d : {0, 1}) {
138  lapse.set_domain(d).annule_hard();
139  conf.set_domain(d).annule_hard();
140  for (int i = 1; i <= 3; ++i) {
141  shift.set(i).set_domain(d).annule_hard();
142  }
143  }
144  lapse.std_base();
145  conf.std_base();
146  shift.std_base();
147 
148  Kadath::bco_utils::save_to_file(space, bconfig, conf, lapse, shift);
149  return exit_status;
150 }
