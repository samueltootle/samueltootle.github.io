23 double sign(gsl_permutation* p) // compute permutation signature
24 {
25  int card(static_cast<int>(gsl_permutation_size(p)));
26  double signature(1.0);
27  for (int i(0 ); i < card - 1 ; ++i)
28  for (int j(i+1) ; j < card ; ++j)
29  {
30  double pj(static_cast<double>(gsl_permutation_get(p, j))); // convert to int just because I don't like compiler warnings...
31  double pi(static_cast<double>(gsl_permutation_get(p, i)));
32  signature *= (pj - pi) / (j - i); // compute signature
33  }
34  return signature;
35 }

37 vector<int> ind_com(int i, int j, int k, int l) // introduce matrix indices into the transposed comatrix components
38 {
39  vector<int> res(2);
40  if (k < j and l < i)
41  {
42  res[0] = k;
43  res[1] = l;
44  }
45  if (k >= j and l < i)
46  {
47  res[0] = k + 1;

54  }
55  if (k >= j and l >= i)
56  {
57  res[0] = k + 1;
58  res[1] = l + 1;
59  }
60  return res;

199 Metric_tensor Metric_tensor::inverse()
200 {
201  int dim(espace.get_ndim());
202  assert(dim == 3);
203  Metric_tensor res(*this);
204  res.set_index_type(0) = -this->get_index_type(0);
205  res.set_index_type(1) = -this->get_index_type(1);
206  for (int d(0) ; d < espace.get_nbr_domains() ; ++d) // sum over components, only the upper triangular part
207  {
208  Val_domain detval(espace.get_domain(d)); // let's compute the determinant...
209  detval = 0.0;
210  gsl_permutation* p(gsl_permutation_calloc(dim)); // initialise to identity 0, 1, 2
211  do
212  {
213  int p1(static_cast<int>(gsl_permutation_get(p, 0)) + 1); // for tensor indices, you need to add one to get a permutation of 1, 2, 3
214  int p2(static_cast<int>(gsl_permutation_get(p, 1)) + 1);
215  int p3(static_cast<int>(gsl_permutation_get(p, 2)) + 1);
216  detval += sign(p)*res(p1, 1)(d)*res(p2, 2)(d)*res(p3, 3)(d); // sum on permutations
217  }while(gsl_permutation_next(p) == GSL_SUCCESS);
218  gsl_permutation_free(p);
219 
220  // compute the transposed comatrix
221  Val_domain cmpval(espace.get_domain(d));
222  for (int i(1) ; i <= dim ; ++i) // sum over components, only the upper triangular part
223  {
224  for (int j(i) ; j <= dim ; ++j)
225  {
226  cmpval = 0.0; // initialise to 0 before each computation
227  p = gsl_permutation_calloc(dim - 1); // reinitialise to identity 0, 1, ready to be used for next component
228  do
229  {
230  int p1(static_cast<int>(gsl_permutation_get(p, 0)) + 1); // always add one for tensor indices, now we have a permutation of 1, 2
231  int p2(static_cast<int>(gsl_permutation_get(p, 1)) + 1);
232  vector<int> index1, index2; // manage the indices of the transposed comatrix
233  index1 = ind_com(i, j, p1, 1); // we need to play a bit with indices to get transpose(COM(metric))
234  index2 = ind_com(i, j, p2, 2);
235  cmpval += pow(-1.0, i + j)*sign(p)*(*this)(index1[0], index1[1])(d)*(*this)(index2[0],index2[1])(d); // sum on permutations
236  }while(gsl_permutation_next(p) == GSL_SUCCESS);
237  gsl_permutation_free(p);
238  res.set(i,j).set_domain(d) = cmpval / detval; // now we have the component of the inverse matrix
239  }
240  }
241  }
242  return res;
243 }
