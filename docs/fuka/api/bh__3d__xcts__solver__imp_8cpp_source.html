71 template <typename config_t, typename space_t>
72 int bh_3d_xcts_solver<config_t, space_t>::solve() {
73  int rank = 0;
74  MPI_Comm_rank(MPI_COMM_WORLD, &rank);
75  int exit_status = EXIT_SUCCESS;
76 
77  std::array<bool, NUM_STAGES>& stage_enabled = bconfig.return_stages();
78  auto [last_stage, last_stage_idx] =
79  get_last_enabled_no_throw(MSTAGE, stage_enabled);
80 
81  // Save in case of iterative solutions
82  auto const& final_chi = bconfig.seq_setting(FINAL_CHI);
83 
84  if (stage_enabled[TOTAL]) {
85  this->solver_stage = TOTAL;
86  exit_status = fixed_lapse_stage();
87  }
88 
89  if (stage_enabled[TOTAL_BC] && exit_status != RELOAD_FILE) {
90  this->solver_stage = TOTAL_BC;
91  if (bconfig.control(ITERATIVE_CHI)) {
92  while (bconfig.control(ITERATIVE_CHI)) {
93 #ifdef DEBUG
94  if (rank == 0)
95  cout << bconfig << endl;
96 #endif
97 
98  exit_status = von_Neumann_stage();
99  stage_enabled[STAGES::TOTAL_BC] = true;
100  if (exit_status == RELOAD_FILE)
101  break;
102 
103  if ((std::abs(final_chi) <= 0.8) || (std::abs(bconfig(CHI)) >= 0.8)) {
104  bconfig.control(ITERATIVE_CHI) = false;
105  } else if (std::abs(bconfig(CHI)) < 0.8)
106  bconfig(CHI) = std::copysign(0.8, final_chi);
107  }
108  if (exit_status != RELOAD_FILE)
109  bconfig(CHI) = final_chi;
110  }
111  if (exit_status != RELOAD_FILE)
112  exit_status = von_Neumann_stage();
113  }
114 
115  // Barrier needed in case we need to read from the previous output
116  MPI_Barrier(MPI_COMM_WORLD);
117  if (last_stage_idx == BIN_BOOST && exit_status != RELOAD_FILE)
118  exit_status = RUN_BOOST;
119  return exit_status;
120 }

122 template <typename config_t, typename space_t>
123 void bh_3d_xcts_solver<config_t, space_t>::syst_init(System_of_eqs& syst) {
124  const int ndom = space.get_nbr_domains();
125  // call the (flat) conformal metric "f"
126  fmet.set_system(syst, "f");
127 
128  // define numerical constants
129  syst.add_cst("4piG", bconfig(BCO_QPIG));
130  syst.add_cst("PI", M_PI);
131  syst.add_cst("M", bconfig(MIRR));
132  syst.add_cst("CM", bconfig(MCH));
133  syst.add_cst("chi", bconfig(CHI));
134  syst.add_var("ome", bconfig(OMEGA));
135 
136  // include the coordinate fields
137  syst.add_cst("mg", *coord_vectors[GLOBAL_ROT]);
138  syst.add_cst("sm", *coord_vectors[S_BCO1]);
139 
140  syst.add_cst("ex", *coord_vectors[EX]);
141  syst.add_cst("ey", *coord_vectors[EY]);
142  syst.add_cst("einf", *coord_vectors[S_INF]);
143 
144  // the basic fields, conformal factor, lapse and (log) enthalpy
145  syst.add_var("P", conf);
146  syst.add_var("N", lapse);
147  syst.add_var("bet", shift);
148 
149  // define common combinations of conformal factor and lapse
150  syst.add_def("NP = P*N");
151  syst.add_def("Ntilde = N / P^6");
152  syst.add_def(
153  "A^ij = (D^i bet^j + D^j bet^i - 2. / 3.* D_k bet^k * f^ij) / 2. / "
154  "Ntilde");
155 
156  // definitions of integrals on the excision surface
157  syst.add_def("intMsq= P^4 / 16. / PI");
158 
159  // define quantity to be integrated at infinity
160  // two (in this case) equivalent definitions of ADM mass
161  // as well as the Komar mass
162  syst.add_def(ndom - 1, "intMadm = - einf^i * D_i P / 4piG * 2");
163  syst.add_def(ndom - 1, "intMk = einf^i * D_i N / 4piG");
164 }

166 template <typename config_t, typename space_t>
167 void bh_3d_xcts_solver<config_t, space_t>::print_diagnostics_norot(
168  const System_of_eqs& syst,
169  const int ite,
170  const double conv) const {
171  int ndom = space.get_nbr_domains();
172  double r = Kadath::bco_utils::get_radius(space.get_domain(1), OUTER_BC);
173 
174  Val_domain integMsq(syst.give_val_def("intMsq")()(2));
175  double Mirrsq = space.get_domain(2)->integ(integMsq, INNER_BC);
176  double Mirr = std::sqrt(Mirrsq);
177 
178  // compute the ADM mass as surface integral at infinity
179  Val_domain integMadm(syst.give_val_def("intMadm")()(ndom - 1));
180  double Madm = space.get_domain(ndom - 1)->integ(integMadm, OUTER_BC);
181 
182  // compute the Komar mass as surface integral at infinity
183  Val_domain integMk(syst.give_val_def("intMk")()(ndom - 1));
184  double Mk = space.get_domain(ndom - 1)->integ(integMk, OUTER_BC);
185 
186  // output to standard output
187  std::ios_base::fmtflags f(std::cout.flags());
188  std::cout << "=======================================" << std::endl
189  << FORMAT << "Iter: " << ite << std::endl
190  << FORMAT << "Error: " << conv << std::endl
191  << FORMAT << "Madm: " << Madm << std::endl
192  << FORMAT << "Mk: " << Mk << " [" << std::abs(Madm - Mk) / Madm
193  << "]" << std::endl
194  << FORMAT << "Mirr: " << Mirr << std::endl;
195  std::cout << FORMAT << "R: " << r << std::endl;
196  std::cout.flags(f);
197 } // end print diagnostics norot

199 // runtime diagnostics specific for rotating solutions
200 template <typename config_t, typename space_t>
201 void bh_3d_xcts_solver<config_t, space_t>::print_diagnostics(
202  System_of_eqs const& syst,
203  const int ite,
204  const double conv) const {
205  // print all the diagnostics as in the non-rotating case first
206  print_diagnostics_norot(syst, ite, conv);
207 
208  int ndom = space.get_nbr_domains();
209 
210  Val_domain integS(syst.give_val_def("intS")()(2));
211  double S = space.get_domain(2)->integ(integS, INNER_BC);
212 
213  Val_domain integMsq(syst.give_val_def("intMsq")()(2));
214  double Mirrsq = space.get_domain(2)->integ(integMsq, INNER_BC);
215  double Mch = std::sqrt(Mirrsq + S * S / 4. / Mirrsq);
216 
217  // output the dimensionless spin and angular frequency parameter
218  std::ios_base::fmtflags f(std::cout.flags());
219  std::cout << FORMAT << "Omega: " << bconfig(OMEGA) << std::endl
220  << FORMAT << "S: " << S << std::endl
221  << FORMAT << "Mch: " << Mch << std::endl
222  << FORMAT << "Chi: " << S / Mch / Mch << std::endl;
223  std::cout.flags(f);
224  std::cout << "=======================================" << "\n\n";
225 } // end print diagnostics rot
