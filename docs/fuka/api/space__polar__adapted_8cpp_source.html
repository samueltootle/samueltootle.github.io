30 Space_polar_adapted::Space_polar_adapted (int ttype, const Point& center, const Dim_array& res, const Array<double>& bounds) {
31 
32  // Verif :
33  assert (bounds.get_ndim()== 1) ;
34 
35  ndim = 2 ;
36 
37  n_shells = bounds.get_size(0) - 3;
38  ADAPTED_OUTER = 1;
39  ADAPTED_INNER = 2;
40 
41  nbr_domains = bounds.get_size(0)+1 ;
42  assert (nbr_domains>=4) ;
43  type_base = ttype ;
44  domains = new Domain* [nbr_domains] ;
45 
46  // Nucleus
47  domains[0] = new Domain_polar_nucleus (0, ttype, bounds(0), center, res) ;
48  domains[ADAPTED_OUTER] = new Domain_polar_shell_outer_adapted (*this, ADAPTED_OUTER, ttype, bounds(ADAPTED_OUTER-1), bounds(ADAPTED_OUTER), center, res) ;
49  domains[ADAPTED_INNER] = new Domain_polar_shell_inner_adapted (*this, ADAPTED_INNER, ttype, bounds(ADAPTED_OUTER), bounds(ADAPTED_INNER), center, res) ;
50  for (int i=3 ; i<nbr_domains-1 ; i++)
51  domains[i] = new Domain_polar_shell(i, ttype, bounds(i-1), bounds(i), center, res) ;
52  domains[nbr_domains-1] = new Domain_polar_compact (nbr_domains-1, ttype, bounds(nbr_domains-2), center, res) ;
53 
54  const Domain_polar_shell_outer_adapted* pshell_outer = dynamic_cast<const Domain_polar_shell_outer_adapted*> (domains[ADAPTED_OUTER]) ;
55  pshell_outer->vars_to_terms() ;
56  const Domain_polar_shell_inner_adapted* pshell_inner = dynamic_cast<const Domain_polar_shell_inner_adapted*> (domains[ADAPTED_INNER]) ;
57  pshell_inner->vars_to_terms() ;
58 }

144 void Space_polar_adapted::affecte_coef_to_variable_domains (int& pos, int cc, Array<int>& zedoms) const {
145 
146  bool found_outer = false ;
147  int old_pos = pos ;
148  domains[ADAPTED_OUTER]->affecte_coef(pos, cc, found_outer) ;
149  pos = old_pos ;
150  bool found_inner = false ;
151  domains[ADAPTED_INNER]->affecte_coef(pos, cc, found_inner) ;
152  assert (found_outer == found_inner) ;
153  if (found_outer) {
154  zedoms.set(0) = ADAPTED_OUTER ;
155  zedoms.set(1) = ADAPTED_INNER ;
156  }
157  else {
158  zedoms.set(0) = -1 ;
159  zedoms.set(1) = -1 ;

171 void Space_polar_adapted::xx_to_vars_variable_domains (System_of_eqs* sys, const Array<double>& xx, int& pos) const {
172 
173  // First get the corrections :
174  int old_pos = pos ;
175  Val_domain cor_outer (domains[ADAPTED_OUTER]) ;
176  domains[ADAPTED_OUTER]->xx_to_vars_from_adapted(cor_outer, xx, pos) ;
177  pos = old_pos ;
178  Val_domain cor_inner (domains[ADAPTED_INNER]) ;
179  domains[ADAPTED_INNER]->xx_to_vars_from_adapted(cor_inner, xx, pos) ;
180 
181  // Now update the variables :
182  for (int i=0 ; i<sys->nvar ; i++) {
183  for (int n=0 ; n<sys->var[i]->get_n_comp() ; n++) {
184  Scalar res(*this) ;
185  domains[ADAPTED_OUTER]->update_variable(cor_outer, *sys->var[i]->cmp[n], res) ;
186  domains[ADAPTED_INNER]->update_variable(cor_inner, *sys->var[i]->cmp[n], res) ;
187 
188  sys->var[i]->cmp[n]->set_domain(ADAPTED_OUTER) = res(ADAPTED_OUTER) ;
189  sys->var[i]->cmp[n]->set_domain(ADAPTED_INNER) = res(ADAPTED_INNER) ;
190 
191  }
192  }
193 
194  // Now the constants :
195  for (int i=0 ; i<sys->ncst ; i++)
196  if (sys->cst[i*(sys->dom_max-sys->dom_min+1)]->get_type_data()==TERM_T)
197  for (int n=0 ; n<sys->cst[i*(sys->dom_max-sys->dom_min+1)]->val_t->get_n_comp() ; n++) {
198 
199  Scalar so (*this) ;
200  so = 0 ;
201  so.set_domain(ADAPTED_OUTER) = (*sys->cst[i*(sys->dom_max-sys->dom_min+1)+ADAPTED_OUTER-sys->dom_min]->val_t->cmp[n])(ADAPTED_OUTER) ;
202  so.set_domain(ADAPTED_INNER) = (*sys->cst[i*(sys->dom_max-sys->dom_min+1)+ADAPTED_INNER-sys->dom_min]->val_t->cmp[n])(ADAPTED_INNER) ;
203 
204  Scalar res(*this) ;
205  res = 0 ;
206  domains[ADAPTED_OUTER]->update_constante(cor_outer, so, res) ;
207  domains[ADAPTED_INNER]->update_constante(cor_inner, so, res) ;
208 
209  sys->cst[i*(sys->dom_max-sys->dom_min+1)+ADAPTED_OUTER-sys->dom_min]->val_t->cmp[n]->set_domain(ADAPTED_OUTER) = res(ADAPTED_OUTER) ;
210  sys->cst[i*(sys->dom_max-sys->dom_min+1)+ADAPTED_INNER-sys->dom_min]->val_t->cmp[n]->set_domain(ADAPTED_INNER) = res(ADAPTED_INNER) ;
211 
212  }
213 
214  // Update the mapping :
215  domains[ADAPTED_OUTER]->update_mapping(cor_outer) ;
216  domains[ADAPTED_INNER]->update_mapping(cor_inner) ;
217 }
