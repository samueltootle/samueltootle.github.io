103 Val_domain Domain_polar_shell_inner_adapted::laplacian (const Val_domain& so, int m) const {
104  Val_domain derr (so.der_r()) ;
105  Val_domain dert (so.der_var(2)) ;
106  Val_domain res (derr.der_r() + div_r(2*derr + div_r(dert.der_var(2) + dert.mult_cos_theta().div_sin_theta()))) ;
107  if (m!=0)
108  res -= m * m * div_r(div_r(so.div_sin_theta().div_sin_theta())) ;
109  return res ;

121 double Domain_polar_shell_inner_adapted::integrale (const Val_domain& so) const {
122  if (so.check_if_zero())
123  return 0 ;
124 
125  Val_domain r2 (get_radius()*get_radius()) ;
126  r2.std_base() ;
127  Val_domain integrant (r2*mult_sin_theta(so)*(*der_rad_term_eq->val_t)()(num_dom)) ;
128  integrant.coef() ;
129 
130  double res = 0 ;
131  Array<double> cf (integrant.get_coef()) ;
132 
133  int baset = (*integrant.get_base().bases_1d[1]) (0) ;
134  switch (baset) {
135  case COS_ODD :
136  break ;
137  case SIN_EVEN :
138  break ;
139  case COS_EVEN : {
140  // Only m=0 :
141  double facttheta = M_PI ;
142  int baser = (*integrant.get_base().bases_1d[0]) (0) ;
143  switch (baser) {
144  case CHEB : {
145  for (int i=0 ; i<nbr_coefs(0) ; i+=2)
146  res += facttheta * (1./double(i+1) - 1./double(i-1)) *cf(i,0) ;
147  break ;
148  }
149  case LEG : {
150  res += facttheta*2*cf(0,0) ;
151  break ;
152  }
153  default :
154  cerr << "Case not yet implemented in Domain_polar_shell_inner_adapted::integrale" << endl ;
155  abort() ;
156  }
157  break ;
158  }
159  case SIN_ODD : {
160  for (int j=0 ; j<nbr_coefs(1) ; j++) {
161  double facttheta = 2./double(2*j+1) ;
162  int baser = (*integrant.get_base().bases_1d[0]) (0) ;
163  switch (baser) {
164  case CHEB : {
165  for (int i=0 ; i<nbr_coefs(0) ; i+=2)
166  res += facttheta * (1./double(i+1) - 1./double(i-1)) *cf(i,j) ;
167  break ;
168  }
169  case LEG : {
170  res += facttheta*2*cf(0,j) ;
171  break ;
172  }
173  default :
174  cerr << "Case not yet implemented in Domain_polar_shell_inner_adapted::integrale" << endl ;
175  abort() ;
176  }
177  }
178  break ;
179  }
180  default :
181  cerr << "Case not yet implemented in Domain_polar_shell_inner_adapted::integrale" << endl ;
182  abort() ;
183  }
184  // Phi contribution :
185  // res *= 2*M_PI;
186  return res ;
187 }
