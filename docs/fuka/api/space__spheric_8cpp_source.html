67 Space_spheric::Space_spheric(int ttype, const Point& center, const Dim_array& res, const Array<double>& bounds, const Array<int>& type_shells) {
68 
69  // Verif :
70  assert (bounds.get_ndim()== 1) ;
71 
72  ndim = 3 ;
73 
74  nbr_domains = bounds.get_size(0)+1 ;
75  type_base = ttype ;
76  domains = new Domain* [nbr_domains] ;
77  // Nucleus
78  domains[0] = new Domain_nucleus(0, ttype, bounds(0), center, res) ;
79  for (int i=1 ; i<nbr_domains-1 ; i++) {
80  switch (type_shells(i-1)) {
81  case STD_TYPE :
82  domains[i] = new Domain_shell(i, ttype, bounds(i-1), bounds(i), center, res) ;
83  break ;
84  case LOG_TYPE :
85  domains[i] = new Domain_shell_log (i, ttype, bounds(i-1), bounds(i), center, res) ;
86  break ;
87  case SURR_TYPE :
88  domains[i] = new Domain_shell_surr (i, ttype, bounds(i-1), bounds(i), center, res) ;
89  break ;
90  default :
91  cerr << "Unknown type of shell" << endl ;
92  abort() ;
93  }
94  }
95 
96  domains[nbr_domains-1] = new Domain_compact (nbr_domains-1, ttype, bounds(nbr_domains-2), center, res) ;
97 }

132 Array<int> Space_spheric::get_indices_matching_non_std(int dom, int bound) const {
133 
134  assert ((dom>=0) && (dom<nbr_domains)) ;
135  Array<int> res (2, 1) ;
136  switch (bound) {
137  case OUTER_BC :
138  res.set(0,0) = dom+1 ;
139  res.set(1,0) = INNER_BC ;
140  break ;
141  case INNER_BC :
142  res.set(0,0) = dom-1 ;
143  res.set(1,0) = OUTER_BC ;
144  break ;
145  default :
146  cerr << "Unknown boundary in " << endl ;
147  cerr << *this << endl ;
148  abort() ;
149  }
150  return res ;
151 }
