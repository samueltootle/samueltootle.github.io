60 Array<int> Space_critic::get_indices_matching_non_std(int dom, int bound) const {
61 
62  switch (dom) {
63  case 0 : {
64  // Inner
65  Array<int> res (2,1) ;
66  switch (bound) {
67  case OUTER_BC :
68  res.set(0,0) = 1 ; // Outer domain
69  res.set(1,0) = INNER_BC ;
70  break ;
71  default :
72  cerr << "Bad bound in Space_critic::get_indices_matching_non_std" << endl ;
73  abort() ;
74  }
75  return res ;
76  }
77  case 1 :
78  {
79  // Outer domain
80  Array<int> res(2, 1) ;
81  switch (bound) {
82  case INNER_BC :
83  res.set(0,0) = 0 ; // Inner domain
84  res.set(1,0) = OUTER_BC ;
85  break ;
86  default :
87  cerr << "Bad bound in Space_critic::get_indices_matching_non_std" << endl ;
88  abort() ;
89  }
90  return res ;
91  }
92  default :
93  cerr << "Bad domain in Space_critic::get_indices_matching_non_std" << endl ;
94  abort() ;
95  }
