28 int Domain_polar_periodic_nucleus::nbr_conditions_val_domain_boundary (const Val_domain& so) const {
29 
30  int res = 0 ;
31 
32  for (int k=0 ; k<nbr_coefs(2) ; k++)
33  for (int j=0 ; j<nbr_coefs(1) ; j++) {
34  bool indic = true ;
35 
36  // Base in time
37  int basetime = (*so.get_base().bases_1d[2]) (0) ;
38  switch (basetime) {
39  case COS:
40  break ;
41  case SIN:
42  if ((k==0) || (k==nbr_coefs(2)-1))
43  indic = false ;
44  break ;
45  default:
46  cerr << "Unknow time basis in Domain_polar_polar_nucleus::nbr_conditions_val_domain_boundary" << endl ;
47  abort() ;
48  }
49 
50 
51  // Get base in theta :
52  int baset = (*so.get_base().bases_1d[1])(k) ;
53  switch (baset) {
54  case COS_EVEN:
55  break ;
56  case COS_ODD:
57  if (j==nbr_coefs(1)-1)
58  indic = false ;
59  break ;
60  case SIN_EVEN:
61  if ((j==0) || (j==nbr_coefs(1)-1))
62  indic = false ;
63  break ;
64  case SIN_ODD:
65  if (j==nbr_coefs(1)-1)
66  indic = false ;
67  break ;
68  default:
69  cerr << "Unknow theta basis in Domain_polar_periodic_nucleus::nbr_conditions_val_boundary" << endl ;
70  abort() ;
71  }
72 
73  if (indic)
74  res ++ ;
75  }
76  return res ;
77 }

82  if (bound!=OUTER_BC) {
83  cerr << "Unknown boundary in Domain_polar_periodic_nucleus::nbr_conditions_boundary" << endl ;

106  res.set(i) = nbr_conditions_val_domain_boundary (tt(3)(dom)) ;
107  }
108  break ;
109  case 2 :

118  }

121  res.set(i) = nbr_conditions_val_domain_boundary (tt(1, 1)(dom)) ;

132  }
133  }
134  // not symetric

148  if (((*p_cmp[i])(0)==1) && ((*p_cmp[i])(1)==1))
