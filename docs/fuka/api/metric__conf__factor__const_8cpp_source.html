31  Metric_conf_factor(met, ome) {
32  type_tensor = met.get_type() ;

34  // Compute the flat gradient of conformal factor (trick pass by Term_eq even if not needed)

52 void Metric_conf_factor_const::compute_cov (int dd) const {
53 
54 
55  int place = place_syst + (dd-syst->dom_min) ;
56  // Right storage : simple copy.
57  if (type_tensor==COV) {
58 
59  if (p_met_cov[dd]==0x0)
60  p_met_cov[dd] = new Term_eq(*syst->cst[place]) ;
61  else
62  *p_met_cov[dd] = Term_eq(*syst->cst[place]) ;
63  }
64  else {
65  Term_eq** res = new Term_eq* [p_met->get_n_comp()] ;
66 
67  Scalar val (espace) ;
68 
69  Val_domain detval (espace.get_domain(dd)) ;
70  detval = (*syst->cst[place]->val_t)(1,1)(dd)*(*syst->cst[place]->val_t)(2,2)(dd)*(*syst->cst[place]->val_t)(3,3)(dd)
71  + (*syst->cst[place]->val_t)(1,2)(dd)*(*syst->cst[place]->val_t)(2,3)(dd)*(*syst->cst[place]->val_t)(1,3)(dd)
72  + (*syst->cst[place]->val_t)(1,3)(dd)*(*syst->cst[place]->val_t)(1,2)(dd)*(*syst->cst[place]->val_t)(2,3)(dd)
73  - (*syst->cst[place]->val_t)(1,3)(dd)*(*syst->cst[place]->val_t)(1,3)(dd)*(*syst->cst[place]->val_t)(2,2)(dd)
74  - (*syst->cst[place]->val_t)(2,3)(dd)*(*syst->cst[place]->val_t)(2,3)(dd)*(*syst->cst[place]->val_t)(1,1)(dd)
75  - (*syst->cst[place]->val_t)(1,2)(dd)*(*syst->cst[place]->val_t)(1,2)(dd)*(*syst->cst[place]->val_t)(3,3)(dd) ;
76 
77  Val_domain cmpval(espace.get_domain(dd)) ;
78 
79  // Compo 1 1
80  cmpval = (*syst->cst[place]->val_t)(2,2)(dd)*(*syst->cst[place]->val_t)(3,3)(dd)
81  -(*syst->cst[place]->val_t)(2,3)(dd)*(*syst->cst[place]->val_t)(2,3)(dd) ;
82  val.set_domain(dd) = cmpval / detval;
83  res[0] = new Term_eq (dd, val) ;
84 
85  // Compo 1 2
86  cmpval = (*syst->cst[place]->val_t)(1,3)(dd)*(*syst->cst[place]->val_t)(2,3)(dd)
87  -(*syst->cst[place]->val_t)(1,2)(dd)*(*syst->cst[place]->val_t)(3,3)(dd) ;
88  val.set_domain(dd) = cmpval / detval ;
89  res[1] = new Term_eq (dd, val) ;
90 
91  // Compo 1 3
92  cmpval = (*syst->cst[place]->val_t)(1,2)(dd)*(*syst->cst[place]->val_t)(2,3)(dd)
93  -(*syst->cst[place]->val_t)(1,3)(dd)*(*syst->cst[place]->val_t)(2,2)(dd) ;
94  val.set_domain(dd) = cmpval / detval ;
95  res[2] = new Term_eq (dd, val) ;
96 
97  // Compo 2 2
98  cmpval = (*syst->cst[place]->val_t)(1,1)(dd)*(*syst->cst[place]->val_t)(3,3)(dd)
99  -(*syst->cst[place]->val_t)(1,3)(dd)*(*syst->cst[place]->val_t)(1,3)(dd) ;
100  val.set_domain(dd) = cmpval / detval ;
101  res[3] = new Term_eq (dd, val) ;
102 
103  // Compo 2 3
104  cmpval = (*syst->cst[place]->val_t)(1,3)(dd)*(*syst->cst[place]->val_t)(1,2)(dd)
105  -(*syst->cst[place]->val_t)(1,1)(dd)*(*syst->cst[place]->val_t)(2,3)(dd) ;
106  val.set_domain(dd) = cmpval / detval ;
107  res[4] = new Term_eq (dd, val) ;
108 
109  // Compo 3 3
110  cmpval = (*syst->cst[place]->val_t)(1,1)(dd)*(*syst->cst[place]->val_t)(2,2)(dd)
111  -(*syst->cst[place]->val_t)(1,2)(dd)*(*syst->cst[place]->val_t)(1,2)(dd) ;
112  val.set_domain(dd) = cmpval / detval ;
113  res[5] = new Term_eq (dd, val) ;
114 
115  // Value field :
116  Metric_tensor resval (espace, COV, basis) ;
117  resval.set(1,1) = res[0]->get_val_t() ;
118  resval.set(1,2) = res[1]->get_val_t() ;
119  resval.set(1,3) = res[2]->get_val_t() ;
120  resval.set(2,2) = res[3]->get_val_t() ;
121  resval.set(2,3) = res[4]->get_val_t() ;
122  resval.set(3,3) = res[5]->get_val_t() ;
123 
124 
125  Metric_tensor zero (espace, COV, basis) ;
126  for (int i=1 ; i<=3 ; i++)
127  for (int j=i ; j<=3 ; j++)
128  zero.set(i,j) = 0 ;
129 
130 
131  if (p_met_cov[dd]==0x0)
132  p_met_cov[dd] = new Term_eq(dd, resval, zero) ;
133  else
134  *p_met_cov[dd] = Term_eq(dd, resval, zero) ;
135 
136  for (int i=0 ; i<p_met->get_n_comp() ; i++)
137  delete res [i] ;
138  delete [] res ;
139  }
