30 Term_eq Ope_pow::action() const {
31  if (power==0)
32  return Term_eq (dom, 1., 0.) ;
33  else {
34  Term_eq res_p0 (parts[0]->action()) ;
35  // Check of type and valence :
36  //int valence = res_p0.get_val_t().get_valence() ;
37  //if (valence !=0) {
38  if (res_p0.get_type_data()==TERM_T and res_p0.get_p_val_t()->get_valence() !=0) {
39  cerr << "Ope_pow only defined for scalars" << endl ;
40  abort() ;
41  }
42  Term_eq auxi (dom, res_p0.get_type_data()) ;
43  auxi = (power>0) ? res_p0 : Term_eq(dom, 1., 0.) / res_p0 ;
44  Term_eq res (auxi) ;
45  for (int i=1 ; i<fabs(power) ; i++)
46  res = res * auxi ;
47  return res ;
48  }
