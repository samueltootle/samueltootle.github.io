73 Scalar Space::get_cart_field(int const cart) const {
74  assert(cart <= ndim);
75  Scalar cart_field(*this);
76  cart_field.annule_hard();
77  if(domains != 0x0) {
78  for(auto dom = 0; dom < nbr_domains; ++dom)
79  cart_field.set_domain(dom) = domains[dom]->get_cart(cart);

85 Scalar Space::get_cart_field_bound(int const cart, int const bound) const {
86  assert(cart <= ndim);
87  Scalar cart_field(*this);
88  cart_field = 1.;
89 
90  if(domains != 0x0) {
91  for(auto dom = 0; dom < nbr_domains; ++dom) {
92  auto npts = domains[dom]->get_nbr_points();
93  auto npts_r = npts(0);
94  int bound_idx{0};
95  switch(bound) {
96  case INNER_BC:
97  break;
98  case OUTER_BC:
99  bound_idx = npts_r - 1;
100  break;
101  default:
102  cerr << "Undefined bound " << bound << "\n";
103  std::_Exit(EXIT_FAILURE);
104  }
105 
106  Index cart_pos(npts);
107  do {
108  Index bound_pos(npts);
109  bound_pos.set(0) = bound_idx;
110  bound_pos.set(1) = cart_pos(1);
111  bound_pos.set(2) = cart_pos(2);
112  cart_field.set_domain(dom).set(cart_pos) = domains[dom]->get_cart(cart)(bound_pos);
113  }while(cart_pos.inc());
114  }
115  }
116  return cart_field;
117 }
