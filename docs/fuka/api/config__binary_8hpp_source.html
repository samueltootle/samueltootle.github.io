166  /* BIN_INFO::read_params
167  * A tree containing all the binary and bco parameters is received.
168  * The parameters related to the binary are stored locally.
169  * The BCO types are determined based on the branch nodes
170  *
171  * @param[input] bin_tree Tree containing all parameters of the binary and BCOs
172  */
173  void read_params(Tree& bin_tree) {
174  // Read-in strictly binary related parameters
175  read_keys(bin_map, bin_params, bin_tree);
176 
177  // Determine BCO types based on branch node names
178  std::array<std::string, 2> tmp_ary;
179  get_branch_nodes(MBCO, tmp_ary, bin_tree, true);
180 
181  // Temporary index used to iterate over BCO pointers
182  int tidx = 0;
183 
184  // Initialize BCO pointer array based on the BCO types from the Tree
185  // and send the branch to BCO_INFO child for reading in the parameters
186  for (auto& bco : tmp_ary) {
187  init_bco(tidx, bco.substr(0, 2));
188  BCOS[tidx]->read_params(bin_tree.get_child(bco.data()));
189  bco = bco.substr(0, 2);
190  ++tidx;
191  }
192 
193  initialize_bin_stage_map(tmp_ary);
194  }

349  template <typename config_t>
350  void set_defaults(config_t& bconfig) {
351  bool includes_matter = false;
352  std::array<double, 2> Ms{};
353  // copy Compact Object defaults
354  for (auto& bco : {NODES::BCO1, NODES::BCO2}) {
355  const auto bcotype = BCOS[bco]->get_type();
356  if (bcotype == "ns") {
357  kadath_config_boost<BCO_NS_INFO> nsconfig;
358  nsconfig.set_defaults();
359  for (auto i = 0; i < BCO_PARAMS::NUM_BCO_PARAMS; ++i)
360  bconfig.set(i, bco) = nsconfig.set(i);
361  for (auto i = 0; i < EOS_PARAMS::NUM_EOS_PARAMS; ++i)
362  bconfig.set_eos(i, bco) = nsconfig.set_eos(i);
363  includes_matter = true;
364  Ms[bco] = bconfig(BCO_PARAMS::MADM, bco);
365  } else if (bcotype == "bh") {
366  kadath_config_boost<BCO_BH_INFO> bhconfig;
367  bhconfig.set_defaults();
368  for (int i = 0; i < BCO_PARAMS::NUM_BCO_PARAMS; ++i)
369  bconfig.set(i, bco) = bhconfig.set(i);
370  Ms[bco] = bconfig(BCO_PARAMS::MCH, bco);
371  }
372  }
373  // Set binary defaults
374  bconfig.set(BIN_PARAMS::BIN_RES) = 9.;
375  bconfig.set(BIN_PARAMS::DIST) = 10. * Ms[0] + Ms[1];
376  bconfig.set(BIN_PARAMS::REXT) = 2. * bconfig(BIN_PARAMS::DIST);
377  bconfig.set(BIN_PARAMS::QPIG) = bconfig(BCO_PARAMS::BCO_QPIG, NODES::BCO1);
378  bconfig.set(BIN_PARAMS::Q) = Ms[1] / Ms[0];
379  bconfig.set(BIN_PARAMS::COM) = 0.;
380  bconfig.set(BIN_PARAMS::COMY) = 0.;
381  bconfig.set(BIN_PARAMS::OUTER_SHELLS) = 0;
382  bconfig.set(BIN_PARAMS::GOMEGA) = 0.;
383 
384  // Set default fields
385  bconfig.set_field(BCO_FIELDS::SHIFT) = true;
386  bconfig.set_field(BCO_FIELDS::LAPSE) = true;
387  bconfig.set_field(BCO_FIELDS::CONF) = true;
388  if (includes_matter) {
389  bconfig.set_field(BCO_FIELDS::LOGH) = true;
390  bconfig.set_field(BCO_FIELDS::PHI) = true;
391  }
392 
393  // The following controls are required for an
394  // initial solution using the v2 Solvers
395  bconfig.control(CONTROLS::SEQUENCES) = true;
396  bconfig.control(CONTROLS::USE_BOOSTED_CO) = true;
397  bconfig.control(CONTROLS::FIXED_GOMEGA) = true;
398  bconfig.control(CONTROLS::NEW_ID) = true;
399 
400  bconfig.set_stage(STAGES::TOTAL_BC) = true;
401  bconfig.set_stage(STAGES::ECC_RED) = true;
402 
403  bconfig.seq_setting(SEQ_SETTINGS::INIT_RES) = 9;
404  }
