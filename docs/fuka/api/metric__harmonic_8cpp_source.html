106 void Metric_harmonic::compute_ricci_tensor (int dd) const {
107 
108  // Need christoffels
109  if (p_det_cov[dd]==0x0)
110  compute_det_cov(dd) ;
111 
112  if (p_christo[dd]==0x0)
113  compute_christo(dd) ;
114  if (p_dirac[dd]==0x0)
115  compute_dirac(dd) ;
116 
117  // Need christoffels
118  if (p_christo[dd]==0x0)
119  compute_christo(dd) ;
120 
121  Tensor res_val (espace, 2, COV, basis) ;
122  Tensor res_der (espace, 2, COV, basis) ;
123 
124  Term_eq der_cov (fmet.derive (COV, ' ', (*p_met_cov[dd]))) ;
125  Term_eq der_con (fmet.derive (COV, ' ', (*p_met_con[dd]))) ;
126  Term_eq dder_cov (fmet.derive (COV, 'u', der_cov)) ;
127 
128  bool doder = ((der_cov.der_t==0x0) || (der_con.der_t==0x0) || (dder_cov.der_t==0x0)) ? false : true ;
129 
130  Index pos (res_val) ;
131  do {
132  Val_domain cmpval (espace.get_domain(dd)) ;
133  cmpval = 0 ;
134 
135  for (int k=1 ; k<=espace.get_ndim() ; k++)
136  for (int l=1 ; l<=espace.get_ndim() ; l++) {
137 
138  cmpval += -0.5 * (*p_met_con[dd]->val_t)(k,l)(dd)*(*dder_cov.val_t)(k,l, pos(0)+1, pos(1)+1)(dd)
139  - 0.5*(*der_con.val_t)(pos(1)+1, k,l)(dd)*(*der_cov.val_t)(k, pos(0)+1, l)(dd)
140  - 0.5*(*der_con.val_t)(pos(0)+1, k,l)(dd)*(*der_cov.val_t)(k, pos(1)+1, l)(dd)
141  + 0.5*(*der_con.val_t)(pos(0)+1, k,l)(dd)*(*der_cov.val_t)(pos(1)+1, k, l)(dd)
142  + (*p_christo[dd]->val_t)(pos(0)+1, pos(1)+1, k)(dd)*(*p_christo[dd]->val_t)(k, l, l)(dd)
143  - (*p_christo[dd]->val_t)(pos(0)+1, k, l )(dd)*(*p_christo[dd]->val_t)(pos(1)+1, l, k)(dd) ;
144 
145  for (int m=1 ; m<=espace.get_ndim() ; m++) {
146  cmpval += (*p_met_cov[dd]->val_t)(m,l)(dd)* (*der_con.val_t)(k, k,l)(dd) * (*p_christo[dd]->val_t)(pos(0)+1, pos(1)+1, m)(dd)
147  - (*p_met_cov[dd]->val_t)(m,l)(dd) * (*der_con.val_t)(pos(1)+1, k,l)(dd) * (*p_christo[dd]->val_t)(pos(0)+1, k, m)(dd) ;
148 
149 
150  }
151  }
152  res_val.set(pos).set_domain(dd) = cmpval ;
153 
154  if (doder) {
155  Val_domain cmpder (espace.get_domain(dd)) ;
156  cmpder = 0 ;
157 
158  for (int k=1 ; k<=espace.get_ndim() ; k++)
159  for (int l=1 ; l<=espace.get_ndim() ; l++) {
160 
161  cmpder += -0.5 * (*p_met_con[dd]->der_t)(k,l)(dd)*(*dder_cov.val_t)(k,l, pos(0)+1, pos(1)+1)(dd)
162  -0.5 * (*p_met_con[dd]->val_t)(k,l)(dd)*(*dder_cov.der_t)(k,l, pos(0)+1, pos(1)+1)(dd)
163  - 0.5*(*der_con.der_t)(pos(1)+1, k,l)(dd)*(*der_cov.val_t)(k, pos(0)+1, l)(dd)
164  - 0.5*(*der_con.val_t)(pos(1)+1, k,l)(dd)*(*der_cov.der_t)(k, pos(0)+1, l)(dd)
165  - 0.5*(*der_con.der_t)(pos(0)+1, k,l)(dd)*(*der_cov.val_t)(k, pos(1)+1, l)(dd)
166  - 0.5*(*der_con.val_t)(pos(0)+1, k,l)(dd)*(*der_cov.der_t)(k, pos(1)+1, l)(dd)
167  + 0.5*(*der_con.der_t)(pos(0)+1, k,l)(dd)*(*der_cov.val_t)(pos(1)+1, k, l)(dd)
168  + 0.5*(*der_con.val_t)(pos(0)+1, k,l)(dd)*(*der_cov.der_t)(pos(1)+1, k, l)(dd)
169  + (*p_christo[dd]->der_t)(pos(0)+1, pos(1)+1, k)(dd)*(*p_christo[dd]->val_t)(k, l, l)(dd)
170  + (*p_christo[dd]->val_t)(pos(0)+1, pos(1)+1, k)(dd)*(*p_christo[dd]->der_t)(k, l, l)(dd)
171  - (*p_christo[dd]->der_t)(pos(0)+1, k, l )(dd)*(*p_christo[dd]->val_t)(pos(1)+1, l, k)(dd) ;
172  - (*p_christo[dd]->val_t)(pos(0)+1, k, l )(dd)*(*p_christo[dd]->der_t)(pos(1)+1, l, k)(dd) ;
173 
174  for (int m=1 ; m<=espace.get_ndim() ; m++) {
175 
176  cmpder += (*p_met_cov[dd]->der_t)(m,l)(dd)* (*der_con.val_t)(k, k,l)(dd) * (*p_christo[dd]->val_t)(pos(0)+1, pos(1)+1, m)(dd)
177  + (*p_met_cov[dd]->val_t)(m,l)(dd)* (*der_con.der_t)(k, k,l)(dd) * (*p_christo[dd]->val_t)(pos(0)+1, pos(1)+1, m)(dd)
178  + (*p_met_cov[dd]->val_t)(m,l)(dd)* (*der_con.val_t)(k, k,l)(dd) * (*p_christo[dd]->der_t)(pos(0)+1, pos(1)+1, m)(dd)
179  - (*p_met_cov[dd]->der_t)(m,l)(dd) * (*der_con.val_t)(pos(1)+1, k,l)(dd) * (*p_christo[dd]->val_t)(pos(0)+1, k, m)(dd)
180  - (*p_met_cov[dd]->val_t)(m,l)(dd) * (*der_con.der_t)(pos(1)+1, k,l)(dd) * (*p_christo[dd]->val_t)(pos(0)+1, k, m)(dd)
181  - (*p_met_cov[dd]->val_t)(m,l)(dd) * (*der_con.val_t)(pos(1)+1, k,l)(dd) * (*p_christo[dd]->der_t)(pos(0)+1, k, m)(dd) ;
