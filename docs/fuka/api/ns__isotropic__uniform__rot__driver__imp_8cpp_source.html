149 template <class config_t, class Res_t>
150 inline int ns_isotropic_uniform_rot_driver(config_t& bconfig,
151  Res_t& resolution,
152  std::string outputdir,
153  ns_sequence const* seq) {
154  int exit_status = RELOAD_FILE;
155  int rank = 0;
156  MPI_Comm_rank(MPI_COMM_WORLD, &rank);
157 
158  bool res_inc = (resolution.final() > resolution.init());
159  auto resolution_indices = resolution.get_indices();
160  auto const& final_res = resolution.final();
161 
162  std::array<bool, NUM_STAGES>& stage_enabled = bconfig.return_stages();
163  auto [last_stage, last_stage_idx] = get_last_enabled(MSTAGE, stage_enabled);
164 
165  exit_status =
166  ns_isotropic_uniform_rot_stationary_driver(bconfig, outputdir, seq);
167  // We now have a "low" resolution solution for the NS of interest
168  // Set this to false to avoid iterative M and CHI
169  bconfig.control(CONTROLS::SEQUENCES) = false;
170 
171  auto regrid = [&]() {
172  std::string fname{"ns_regrid"};
173 
174  if (rank == 0)
175  exit_status = ns_isotropic_uniform_rot_regrid(bconfig, fname);
176  MPI_Barrier(MPI_COMM_WORLD);
177  bconfig.set_filename(fname);
178  bconfig.open_config();
179 
180  stage_enabled[STAGES::NOROT_BC] = false;
181  stage_enabled[STAGES::UNIFORM_ROT] = true;
182  };
183 
184  // Since the 2D code focuses on sequences, we always regrid to make sure
185  // we start/end on an optimal grid structure.
186  regrid();
187  exit_status =
188  ns_isotropic_uniform_rot_stationary_driver(bconfig, outputdir, seq);
189 
190  while (res_inc) {
191  int next_res = bco_utils::next_resolution(bconfig(resolution_indices));
192  // iterative res increase
193  if (next_res >= final_res) {
194  bconfig.set(resolution_indices) = next_res;
195  res_inc = false;
196  } else {
197  bconfig.set(resolution_indices) = next_res;
198  }
199  regrid();
200 
201  exit_status =
202  ns_isotropic_uniform_rot_stationary_driver(bconfig, outputdir, seq);
203  }
204  return exit_status;
205 }
