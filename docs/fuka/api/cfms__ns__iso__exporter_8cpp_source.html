114 void CFMS_NS_ISO_Exporter::extract_computed_grid_functions() {
115  using namespace Kadath::FUKA_EOS;
116 
117  System_of_eqs syst(*space);
118 
119  Param p;
120  EOS_Function_Dispatcher::dispatch<set_eos_ope_struct>(*bconfig, eos_type,
121  syst, p);
122 
123  // Fields - must be initialized before common setup
124  syst.add_cst("nu", *Nu);
125  syst.add_cst("lapAterm", *lap_Aterm);
126  syst.add_cst("lapBterm", *lap_Bterm);
127 
128  if (bconfig->field(Kadath::FUKA_Config::BCO_FIELDS::DIFF_OMEGA)) {
129 #ifdef DEBUG
130  std::cout << "**** Importing differential rotation profile ****\n";
131 #endif
132  syst.add_cst("Omega", *omega);
133  } else {
134 #ifdef DEBUG
135  std::cout << "**** Importing uniform rotation profile ****\n";
136 #endif
137  syst.add_cst("Omega", (*bconfig)(Kadath::FUKA_Config::BCO_PARAMS::OMEGA));
138  }
139  syst.add_def("N = exp(nu)");
140  syst.add_def("A = exp(lapAterm - nu)");
141  syst.add_def("B = (divrsint(lapBterm) + 1) / N");
142 
143  if (bconfig->set_field(Kadath::FUKA_Config::BCO_FIELDS::LAP_WTERM)) {
144  syst.add_cst("wrsint", *lap_omega_term);
145  syst.add_def("w = divrsint(wrsint)");
146  syst.add_def("drw = dr(w)");
147  syst.add_def("dtw = dt(w)");
148 
149  syst.add_def("UphiU = 1 / N * (Omega - w)");
150 
151  fluidvel.reset(new Scalar(syst.give_val_def("UphiU")));
152  fluidvel->coef();
153 
154  metric_omega.reset(new Scalar(syst.give_val_def("w")));
155  metric_omega->coef();
156  domega_dr.reset(new Scalar(syst.give_val_def("drw")));
157  domega_dr->coef();
158  domega_dt.reset(new Scalar(syst.give_val_def("dtw")));
159  domega_dt->coef();
160  } else {
161 #ifdef DEBUG
162  std::cout << "**** Importing non-rotation profile ****\n";
163 #endif
164  fluidvel.reset(new Scalar(*space));
165  fluidvel->annule_hard();
166  fluidvel->std_base();
167 
168  metric_omega.reset(new Scalar(*space));
169  metric_omega->annule_hard();
170  metric_omega->std_base();
171  domega_dr.reset(new Scalar(*space));
172  domega_dr->annule_hard();
173  domega_dr->std_base();
174  domega_dt.reset(new Scalar(*space));
175  domega_dt->annule_hard();
176  domega_dt->std_base();
177  }
178  metric_A.reset(new Scalar(syst.give_val_def("A")));
179  metric_A->coef();
180  metric_B.reset(new Scalar(syst.give_val_def("B")));
181  metric_B->coef();
182  lapse.reset(new Scalar(syst.give_val_def("N")));
183  lapse->coef();
184 }

186 void CFMS_NS_ISO_Exporter::populate_quants() {
187  if (quants.capacity() != ISO_VARS::NUM_ISO_VARS) {
188  for (size_t i = 0; i < ISO_VARS::NUM_ISO_VARS; ++i)
189  quants.push_back(std::cref(*Nu));
190  }
191  quants[ISO_VARS::ISO_METRIC_A] = std::cref(*metric_A);
192  quants[ISO_VARS::ISO_METRIC_B] = std::cref(*metric_B);
193  quants[ISO_VARS::ISO_ALPHA] = std::cref(*lapse);
194  quants[ISO_VARS::ISO_METRIC_OMEGA] = std::cref(*metric_omega);
195  quants[ISO_VARS::ISO_DOMEGA_DR] = std::cref(*domega_dr);
196  quants[ISO_VARS::ISO_DOMEGA_DTHETA] = std::cref(*domega_dt);
197 
198  // Fluid related quantities
199  quants[ISO_VARS::ISO_H] = std::cref(*logh);
200  quants[ISO_VARS::ISO_U] = std::cref(*fluidvel);
201  export_ready = true;
202 }

204 CFMS_NS_ISO_Exporter::interp_ary_t CFMS_NS_ISO_Exporter::interpolate_pointwise(
205  double const& x,
206  double const& y,
207  double const& z) {
208  double r2_xy = x * x + y * y;
209  double r_xy = std::sqrt(r2_xy);
210  Point abs_coords(ndim);
211  abs_coords.set(1) = r_xy;
212  abs_coords.set(2) = z;
213 
214  for (size_t k = 0; k < ISO_VARS::NUM_ISO_VARS; ++k) {
215  quant_vals[k] = quants[k].get().val_point(abs_coords);
216  }
217 
218  return quant_vals;
219 }
