25 Space_spheric_periodic::Space_spheric_periodic(int ttype, int typet, const Dim_array& res, const Array<double>& bounds, double ome) : omega(ome) {
26 
27  // Verif :
28  assert (bounds.get_ndim()== 1) ;
29 
30  ndim = 3 ;
31 
32  nbr_domains = bounds.get_size(0)+1 ;
33  type_base = ttype ;
34  domains = new Domain* [nbr_domains] ;
35  // Nucleus
36  domains[0] = new Domain_spheric_periodic_nucleus(0, ttype, typet, bounds(0), omega, res) ;
37  for (int i=1 ; i<nbr_domains-1 ; i++)
38  domains[i] = new Domain_spheric_periodic_shell(i, ttype, typet, bounds(i-1), bounds(i), omega, res) ;
39 
40  domains[nbr_domains-1] = new Domain_spheric_periodic_compact (nbr_domains-1, ttype, typet, bounds(nbr_domains-2), omega, res) ;
41 }
