244  output_ary_t operator()(CFMS_BHNS_Exporter& base,
245  double const& x,
246  double const& y,
247  double const& z,
248  double const interpolation_offset = 0.,
249  int const interp_order = 8,
250  double const delta_r_rel = 0.3) {
251  base.quant_vals =
252  base.interpolate_pointwise(x, y, z, interpolation_offset,
253  interp_order, delta_r_rel);
254 
255  // Fill output vector by storing non-conformal quantities
256  auto const psi = base.quant_vals[XCTS_VARS::XCTS_PSI];
257  auto const psi2 = psi * psi;
258  auto const psi4 = psi2 * psi2;
259 
260  base.out_pw[OUTPUT_VARS::ALPHA] = base.quant_vals[XCTS_VARS::XCTS_ALPHA];
261 
262  base.out_pw[OUTPUT_VARS::BETA1] = base.quant_vals[XCTS_VARS::XCTS_BETA1];
263  base.out_pw[OUTPUT_VARS::BETA2] = base.quant_vals[XCTS_VARS::XCTS_BETA2];
264  base.out_pw[OUTPUT_VARS::BETA3] = base.quant_vals[XCTS_VARS::XCTS_BETA3];
265 
266  double g[3][3];
267  g[0][0] = psi4;
268  g[0][1] = 0.0;
269  g[0][2] = 0.0;
270  g[1][1] = psi4;
271  g[1][2] = 0.0;
272  g[2][2] = psi4;
273  g[1][0] = g[0][1];
274  g[2][0] = g[0][2];
275  g[2][1] = g[1][2];
276 
277  base.out_pw[OUTPUT_VARS::G11] = g[0][0];
278  base.out_pw[OUTPUT_VARS::G12] = g[0][1];
279  base.out_pw[OUTPUT_VARS::G13] = g[0][2];
280  base.out_pw[OUTPUT_VARS::G22] = g[1][1];
281  base.out_pw[OUTPUT_VARS::G23] = g[1][2];
282  base.out_pw[OUTPUT_VARS::G33] = g[2][2];
283 
284  base.out_pw[OUTPUT_VARS::K11] =
285  base.quant_vals[XCTS_VARS::XCTS_A11] * psi4;
286  base.out_pw[OUTPUT_VARS::K12] =
287  base.quant_vals[XCTS_VARS::XCTS_A12] * psi4;
288  base.out_pw[OUTPUT_VARS::K13] =
289  base.quant_vals[XCTS_VARS::XCTS_A13] * psi4;
290  base.out_pw[OUTPUT_VARS::K22] =
291  base.quant_vals[XCTS_VARS::XCTS_A22] * psi4;
292  base.out_pw[OUTPUT_VARS::K23] =
293  base.quant_vals[XCTS_VARS::XCTS_A23] * psi4;
294  base.out_pw[OUTPUT_VARS::K33] =
295  base.quant_vals[XCTS_VARS::XCTS_A33] * psi4;
296 
297  double const H = base.quant_vals[XCTS_VARS::XCTS_H];
298  double h = std::exp(H);
299  double rho, eps, press;
300 
301  // get quantities point-wise, since h is smoothest, and cut data at H=0
302  if (std::fabs(H) <= 1e-12) {
303  rho = 0.;
304  eps = 0.;
305  press = 0.;
306  } else {
307  rho = EOS<eos_t, eos_var_t::DENSITY>::get(h);
308  eps = EOS<eos_t, eos_var_t::EPSILON>::get(h);
309  press = EOS<eos_t, eos_var_t::PRESSURE>::get(h);
310  }
311  base.out_pw[OUTPUT_VARS::RHO] = rho;
312  base.out_pw[OUTPUT_VARS::EPS] = eps;
313  base.out_pw[OUTPUT_VARS::PRESS] = press;
314  base.out_pw[OUTPUT_VARS::VEL1] = base.quant_vals[XCTS_VARS::XCTS_UX];
315  base.out_pw[OUTPUT_VARS::VEL2] = base.quant_vals[XCTS_VARS::XCTS_UY];
316  base.out_pw[OUTPUT_VARS::VEL3] = base.quant_vals[XCTS_VARS::XCTS_UZ];
317  return base.out_pw;
318  }
