82 void CFMS_BNS_Exporter::extract_computed_grid_functions() {
83  using namespace Kadath::FUKA_EOS;
84 
85  // fields depending on the coords
86  CoordFields<space_t> cfields(*space);
87  vec_ary_t coord_vectors{default_binary_vector_ary(*space)};
88 
89  // get origin of the system and initialize coordinate fields
90  double xc1 = Kadath::bco_utils::get_center(*space, space->NS1);
91  double xc2 = Kadath::bco_utils::get_center(*space, space->NS2);
92  double xo = Kadath::bco_utils::get_center(*space, ndom - 1);
93  update_fields(cfields, coord_vectors, {}, xo, xc1, xc2);
94 
95  // Initialize Flat Metric
96  Base_tensor basis(shift->get_basis());
97  Metric_flat fmet(*space, basis);
98 
99  // Start - Setup System of equations
100  System_of_eqs syst(*space);
101  fmet.set_system(syst, "f");
102 
103  Param p;
104  EOS_Function_Dispatcher::dispatch<set_eos_ope_struct>(*bconfig, eos_type,
105  syst, p);
106 
107  // Fields - must be initialized before common setup
108  syst.add_cst("N", *lapse);
109  syst.add_cst("bet", *shift);
110  syst.add_cst("P", *conformal_factor);
111  syst.add_cst("H", *logh);
112  syst.add_cst("phi", *velpotential);
113 
114  syst.add_cst("omes1", (*bconfig)(Kadath::FUKA_Config::BCO_PARAMS::OMEGA,
115  Kadath::FUKA_Config::NODES::BCO1));
116  syst.add_cst("omes2", (*bconfig)(Kadath::FUKA_Config::BCO_PARAMS::OMEGA,
117  Kadath::FUKA_Config::NODES::BCO2));
118 
119  syst.add_cst("mg", *coord_vectors[Kadath::coord_vector::GLOBAL_ROT]);
120  syst.add_cst("mm", *coord_vectors[Kadath::coord_vector::BCO1_ROT]);
121  syst.add_cst("mp", *coord_vectors[Kadath::coord_vector::BCO2_ROT]);
122 
123  syst.add_def("h = exp(H)");
124  for (int d = space->NS1; d <= space->ADAPTED1; ++d) {
125  syst.add_def(d, "s^i = omes1 * mm^i");
126  }
127  for (int d = space->NS2; d <= space->ADAPTED2; ++d) {
128  syst.add_def(d, "s^i = omes2 * mp^i");
129  }
130  for (int d = 0; d < ndom; ++d) {
131  if ((d <= space->ADAPTED1) || (d <= space->ADAPTED2 && d >= space->NS2))
132  syst.add_def(d, "eta_i = D_i phi + P^4 * s_i");
133  else
134  syst.add_def(d, "eta_i = D_i phi");
135  }
136 
137  syst.add_def(
138  "A_ij = (D_i bet_j + D_j bet_i - 2. / 3.* D^k bet_k * f_ij) /2. / N");
139  A.reset(new Tensor(syst.give_val_def("A")));
140  A->coef();
141 
142  // definitions for the fluid 3-velocity
143  syst.add_def("Wsquare = eta^i * eta_i / h^2 / P^4 + 1.");
144  syst.add_def("W = sqrt(Wsquare)");
145 
146  syst.add_def("U^i = eta^i / P^4 / h / W");
147  fluidvel.reset(new Vector(syst.give_val_def("U")));
148  fluidvel->coef();
149 }

151 void CFMS_BNS_Exporter::populate_quants() {
152  if (quants.capacity() != XCTS_VARS::NUM_XCTS_VARS) {
153  for (size_t i = 0; i < XCTS_VARS::NUM_XCTS_VARS; ++i)
154  quants.push_back(std::cref(*conformal_factor));
155  }
156  quants[XCTS_VARS::XCTS_PSI] = std::cref(*conformal_factor);
157  quants[XCTS_VARS::XCTS_ALPHA] = std::cref(*lapse);
158  quants[XCTS_VARS::XCTS_BETA1] = std::cref((*shift)(1));
159  quants[XCTS_VARS::XCTS_BETA2] = std::cref((*shift)(2));

161 
162  export_utils::add_tensor_refs(quants,
163  {XCTS_VARS::XCTS_A11, XCTS_VARS::XCTS_A12,
164  XCTS_VARS::XCTS_A13, XCTS_VARS::XCTS_A22,
165  XCTS_VARS::XCTS_A23, XCTS_VARS::XCTS_A33},
166  *A);
167 
168  // Fluid related quantities
169  quants[XCTS_VARS::XCTS_H] = std::cref(*logh);
170  quants[XCTS_VARS::XCTS_UX] = std::cref((*fluidvel)(1));
171  quants[XCTS_VARS::XCTS_UY] = std::cref((*fluidvel)(2));
172  quants[XCTS_VARS::XCTS_UZ] = std::cref((*fluidvel)(3));
