27 void Domain_critic_inner::export_tau_val_domain_boundary (const Val_domain& so, int bound, Array<double>& sec, int& pos_sec, int ncond) const {
28 
29  if (so.check_if_zero())
30  pos_sec += ncond ;
31  else {
32  so.coef() ;
33  int base_t = (*so.get_base().bases_1d[1])(0) ;
34  assert ((base_t==COSSIN_EVEN) || (base_t==COSSIN_ODD)) ;
35 
36  Index pos_cf (nbr_coefs) ;
37  // Loop on t
38  int max = (base_t==COSSIN_EVEN) ? nbr_coefs(1)-1 : nbr_coefs(1)-2 ;
39  for (int j=0 ; j<max ; j++)
40  if ((j!=1) || (base_t!=COSSIN_EVEN)) {
41  pos_cf.set(1) = j ;
42  sec.set(pos_sec) = val_boundary (bound, so, pos_cf) ;
43  pos_sec ++ ;
44  }
45  }
46 }
