35 inline double func_ab (double aa, const Param& par) {
36  double r1 = par.get_double(0) ;
37  double r2 = par.get_double(1) ;
38  double d = par.get_double(2) ;
39  return (sqrt(aa*aa+r1*r1)+sqrt(aa*aa+r2*r2)-d) ;
40 }

42 Space_bin_bh::Space_bin_bh (int ttype, double dist, double rbh1, double rbh2, double rext, int nr) {
43 
44  ndim = 3;
45  nbr_domains = 12;
46 
47  type_base = ttype ;
48  domains = new Domain* [nbr_domains] ;
49 
50  Dim_array res (ndim) ;
51  res.set(0) = nr ; res.set(1) = nr ; res.set(2) = nr-1 ;
52  Dim_array res_bi(ndim) ;
53  res_bi.set(0) = nr ; res_bi.set(1) = nr ; res_bi.set(2) = nr ;
54 
55  // Bispheric :
56  // Computation of aa
57  Param par_a ;
58  par_a.add_double(2*rbh1,0) ;
59  par_a.add_double(2*rbh2,1) ;
60  par_a.add_double(dist,2) ;
61  double a_min = 0 ;
62  double a_max = dist/2. ;
63  double precis = PRECISION ;
64  int nitermax = 500 ;
65  int niter ;
66  double aa = zerosec(func_ab, par_a, a_min, a_max, precis, nitermax, niter) ;
67  double eta_plus = asinh(aa/2/rbh2) ;
68  double eta_minus = -asinh(aa/2/rbh1) ;
69 
70  double chi_c = 2*atan(aa/rext) ;
71  double eta_c = log((1+rext/aa)/(rext/aa-1)) ;
72  double eta_lim = eta_c/2. ;
73  double chi_lim = chi_lim_eta (eta_lim, rext, aa, chi_c) ;
74 
75  // First BH
76  Point center_minus (ndim) ;
77  center_minus.set(1) = aa*cosh(eta_minus)/sinh(eta_minus) ;
78  domains[0] = new Domain_nucleus (0, ttype, rbh1/4., center_minus, res) ;
79  domains[1] = new Domain_shell_outer_homothetic (*this, 1, ttype, rbh1/4., rbh1, center_minus, res) ;
80  domains[2] = new Domain_shell_inner_homothetic (*this, 2, ttype, rbh1, rbh1*2., center_minus, res) ;
81 
82  // Second bh
83  Point center_plus (ndim) ;
84  center_plus.set(1) = aa*cosh(eta_plus)/sinh(eta_plus) ;
85  domains[3] = new Domain_nucleus (3, ttype, rbh2/4., center_plus, res) ;
86  domains[4] = new Domain_shell_outer_homothetic (*this, 4, ttype, rbh2/4., rbh2, center_plus, res) ;
87  domains[5] = new Domain_shell_inner_homothetic (*this, 5, ttype, rbh2, rbh2*2., center_plus, res) ;
88 
89  // Bispheric part
90  domains[6] = new Domain_bispheric_chi_first(6, ttype, aa, eta_minus, rext, chi_lim, res_bi) ;
91  domains[7] = new Domain_bispheric_rect(7, ttype, aa, rext, eta_minus, -eta_lim, chi_lim, res_bi) ;
92  domains[8] = new Domain_bispheric_eta_first(8, ttype, aa, rext, -eta_lim, eta_lim, res_bi) ;
93  domains[9] = new Domain_bispheric_rect(9, ttype, aa, rext, eta_plus, eta_lim, chi_lim, res_bi) ;
94  domains[10] = new Domain_bispheric_chi_first(10, ttype, aa, eta_plus, rext, chi_lim, res_bi) ;
95 
96  // Compactified domain
97  Point center(3) ;
98  domains[11] = new Domain_compact (11, ttype, rext, center, res) ;
99 
100  const Domain_shell_outer_homothetic* pouter_1 = dynamic_cast<const Domain_shell_outer_homothetic*> (domains[1]) ;
101  pouter_1->vars_to_terms() ;
102  pouter_1->update() ;
103  const Domain_shell_inner_homothetic* pinner_1 = dynamic_cast<const Domain_shell_inner_homothetic*> (domains[2]) ;
104  pinner_1->vars_to_terms() ;
105  pinner_1->update() ;
106  const Domain_shell_outer_homothetic* pouter_2 = dynamic_cast<const Domain_shell_outer_homothetic*> (domains[4]) ;
107  pouter_2->vars_to_terms() ;
108  pouter_2->update() ;
109  const Domain_shell_inner_homothetic* pinner_2 = dynamic_cast<const Domain_shell_inner_homothetic*> (domains[5]) ;
110  pinner_2->vars_to_terms() ;
111  pinner_2->update() ;
112 }

156

162 
163  for(int i = 0; i < n_shells1; ++i) {
164  const Domain_shell* d_shell = dynamic_cast<const Domain_shell*> (sp.get_domain(BH2+3+i)) ;
165  domains[BH2+3+i] = new Domain_shell(*d_shell, true);

167 
168  // Bispheric
169  const Domain_bispheric_chi_first* sp_chi_1 = dynamic_cast<const Domain_bispheric_chi_first*> (sp.get_domain(OUTER)) ;

173  domains[OUTER+1] = new Domain_bispheric_rect(*this, *sp_rec_1) ;

175  const Domain_bispheric_eta_first* sp_eta = dynamic_cast<const Domain_bispheric_eta_first*> (sp.get_domain(OUTER+2)) ;
176  domains[OUTER+2] = new Domain_bispheric_eta_first(*this, *sp_eta) ;

178  const Domain_bispheric_rect* sp_rec_2 = dynamic_cast<const Domain_bispheric_rect*> (sp.get_domain(OUTER+3)) ;

181  const Domain_bispheric_chi_first* sp_chi_2 = dynamic_cast<const Domain_bispheric_chi_first*> (sp.get_domain(OUTER+4)) ;

183 
184  for (int i=0 ; i<n_shells_outer ; i++) {
185  const Domain_shell* d_shell = dynamic_cast<const Domain_shell*> (sp.get_domain(OUTER+5+i)) ;
186  domains[OUTER+5+i] = new Domain_shell(*d_shell, true);

188 
189  // Compactified

193  const Domain_shell_outer_homothetic* pouter_1 = dynamic_cast<const Domain_shell_outer_homothetic*> (domains[BH1+1]) ;
194  pouter_1->vars_to_terms() ;
195  pouter_1->update() ;

197  pinner_1->vars_to_terms() ;
198  pinner_1->update() ;

294 void Space_bin_bh::affecte_coef_to_variable_domains (int& pos, int cc, Array<int>& zedoms) const {
295  // In BH 1 ?
296  bool found_outer_1 = false ;
297  int old_pos = pos ;
298  domains[BH1+1]->affecte_coef(pos, cc, found_outer_1) ;
299  pos = old_pos ;
300  bool found_inner_1 = false ;
301  domains[BH1+2]->affecte_coef(pos, cc, found_inner_1) ;
302  assert (found_outer_1 == found_inner_1) ;
303  if (found_outer_1) {
304  zedoms.set(0) = BH1+1 ;
305  zedoms.set(1) = BH1+2 ;
306  }
307  else {
308  zedoms.set(0) = -1 ;
309  zedoms.set(1) = -1 ;
310  }
311 
312  // In BH 2 ?
313  bool found_outer_2 = false ;
314  old_pos = pos ;
315  domains[BH2+1]->affecte_coef(pos, cc, found_outer_2) ;
316  pos = old_pos ;
317  bool found_inner_2 = false ;
318  domains[BH2+2]->affecte_coef(pos, cc, found_inner_2) ;
319  assert (found_outer_2 == found_inner_2) ;
320  if (found_outer_2) {
321  zedoms.set(0) = BH2+1 ;
322  zedoms.set(1) = BH2+2 ;
323  }

339 void Space_bin_bh::xx_to_vars_variable_domains (System_of_eqs* sys, const Array<double>& xx, int& pos) const {
340  // First get the corrections :
341  // BH 1
342  int old_pos = pos ;
343  Val_domain cor_outer_1 (domains[BH1+1]) ;
344  domains[BH1+1]->xx_to_vars_from_adapted(cor_outer_1, xx, pos) ;
345  pos = old_pos ;
346  Val_domain cor_inner_1 (domains[BH1+2]) ;
347  domains[BH1+2]->xx_to_vars_from_adapted(cor_inner_1, xx, pos) ;
348 
349  // BH 2
350  old_pos = pos ;
351  Val_domain cor_outer_2 (domains[BH2+1]) ;
352  domains[BH2+1]->xx_to_vars_from_adapted(cor_outer_2, xx, pos) ;
353  pos = old_pos ;
354  Val_domain cor_inner_2 (domains[BH2+2]) ;
355  domains[BH2+2]->xx_to_vars_from_adapted(cor_inner_2, xx, pos) ;
356 
357  // Now update the variables :
358  for (int i=0 ; i<sys->nvar ; i++) {
359  for (int n=0 ; n<sys->var[i]->get_n_comp() ; n++) {
360  Scalar res(*this) ;
361  domains[BH1+1]->update_variable(cor_outer_1, *sys->var[i]->cmp[n], res) ;
362  domains[BH1+2]->update_variable(cor_inner_1, *sys->var[i]->cmp[n], res) ;
363  domains[BH2+1]->update_variable(cor_outer_2, *sys->var[i]->cmp[n], res) ;
364  domains[BH2+2]->update_variable(cor_inner_2, *sys->var[i]->cmp[n], res) ;
365 
366 
367  sys->var[i]->cmp[n]->set_domain(BH1+1) = res(BH1+1) ;
368  sys->var[i]->cmp[n]->set_domain(BH1+2) = res(BH1+2) ;
369  sys->var[i]->cmp[n]->set_domain(BH2+1) = res(BH2+1) ;
370  sys->var[i]->cmp[n]->set_domain(BH2+2) = res(BH2+2) ;
371  }
372  }
373 
374  // Now the constants :
375  for (int i=0 ; i<sys->ncst ; i++)
376  if (sys->cst[i*(sys->dom_max-sys->dom_min+1)]->get_type_data()==TERM_T)
377  for (int n=0 ; n<sys->cst[i*(sys->dom_max-sys->dom_min+1)]->val_t->get_n_comp() ; n++) {
378 
379  Scalar so (*this) ;
380  so = 0 ;
381  so.set_domain(BH1+1) = (*sys->cst[i*(sys->dom_max-sys->dom_min+1) + BH1+1]->val_t->cmp[n])(BH1+1) ;
382  so.set_domain(BH1+2) = (*sys->cst[i*(sys->dom_max-sys->dom_min+1) + BH1+2]->val_t->cmp[n])(BH1+2) ;
383  so.set_domain(BH2+1) = (*sys->cst[i*(sys->dom_max-sys->dom_min+1) + BH2+1]->val_t->cmp[n])(BH2+1) ;
384  so.set_domain(BH2+2) = (*sys->cst[i*(sys->dom_max-sys->dom_min+1) + BH2+2]->val_t->cmp[n])(BH2+2) ;
385 
386  Scalar res(*this) ;
387  res = 0 ;
388  domains[BH1+1]->update_constante(cor_outer_1, so, res) ;
389  domains[BH1+2]->update_constante(cor_inner_1, so, res) ;
390  domains[BH2+1]->update_constante(cor_outer_2, so, res) ;
391  domains[BH2+2]->update_constante(cor_inner_2, so, res) ;
392 
393  sys->cst[i*(sys->dom_max-sys->dom_min+1)+BH1+1]->val_t->cmp[n]->set_domain(BH1+1) = res(BH1+1) ;
394  sys->cst[i*(sys->dom_max-sys->dom_min+1)+BH1+2]->val_t->cmp[n]->set_domain(BH1+2) = res(BH1+2) ;
395  sys->cst[i*(sys->dom_max-sys->dom_min+1)+BH2+1]->val_t->cmp[n]->set_domain(BH2+1) = res(BH2+1) ;
396  sys->cst[i*(sys->dom_max-sys->dom_min+1)+BH2+2]->val_t->cmp[n]->set_domain(BH2+2) = res(BH2+2) ;
397  }
398 
399  // Update the mapping :
400  domains[BH1+1]->update_mapping(cor_outer_1) ;
401  domains[BH1+2]->update_mapping(cor_inner_1) ;
402  domains[BH2+1]->update_mapping(cor_outer_2) ;
403  domains[BH2+2]->update_mapping(cor_inner_2) ;
404 
405 }

408 Array<int> Space_bin_bh::get_indices_matching_non_std(int dom, int bound) const {
409  if (dom==BH1 + n_shells1 + 2) {
410  // First BH ;
411  Array<int> res (2,2) ;
412  switch (bound) {
413  case OUTER_BC :
414  res.set(0,0) = OUTER ; // Matching with chi first
415  res.set(1,0) = INNER_BC ;
416  res.set(0,1) = OUTER+1 ; // Matching with rect
417  res.set(1,1) = INNER_BC ;
418  break ;
419  default :
420  cerr << "Bad bound in Space_bin_bh::get_indices_matching_non_std" << endl ;
421  std::_Exit(EXIT_FAILURE) ;
422  }
423  return res ;
424  }
425 
426  if (dom== BH2 + n_shells2 + 2) {
427  // Second BH ;
428  Array<int> res(2, 2) ;
429  switch (bound) {
430  case OUTER_BC :
431  res.set(0,0) = OUTER+3 ; // Matching with rect
432  res.set(1,0) = INNER_BC ;
433  res.set(0,1) = OUTER+4 ; // Matching with chi_first
434  res.set(1,1) = INNER_BC ;
435  break ;

571  auto router2 = BH2_bounds[n_shells2+2];

574  type_base = ttype ;

576 
577  Dim_array res (ndim) ;
578  res.set(0) = nr ; res.set(1) = nr ; res.set(2) = nr-1 ;
579  Dim_array res_bi(ndim) ;
580  res_bi.set(0) = nr ; res_bi.set(1) = nr ; res_bi.set(2) = nr ;
581 
582  // Bispheric :
583  // Computation of aa
584  Param par_a ;
585  par_a.add_double(router1,0) ;
586  par_a.add_double(router2,1) ;
587  par_a.add_double(dist,2) ;
588  double a_min = 0 ;
589  double a_max = dist/2. ;
590  double precis = PRECISION ;
591  int nitermax = 500 ;
592  int niter ;

595  double eta_minus = -asinh(aa/router1) ;
596

609  domains[nuc_i+2] =
610  new Domain_shell_inner_homothetic (*this, nuc_i+2, ttype, bounds[RMID], bounds[RMID+1], center, res) ;
611 
612  for(int i = 0; i < shells; ++i)
613  domains[nuc_i+3+i] = new Domain_shell(nuc_i+3+i, ttype, bounds[RMID+1+i], bounds[RMID+1+i+1], center, res);
614  };
615  gen_bh_domains(BH1, BH1_bounds, eta_minus);
616  gen_bh_domains(BH2, BH2_bounds, eta_plus);
617 
618  // Bispheric part
619  domains[OUTER ] = new Domain_bispheric_chi_first(OUTER, ttype, aa, eta_minus, rext, chi_lim, res_bi) ;
620  domains[OUTER+1] = new Domain_bispheric_rect(OUTER+1, ttype, aa, rext, eta_minus, -eta_lim, chi_lim, res_bi) ;
621  domains[OUTER+2] = new Domain_bispheric_eta_first(OUTER+2, ttype, aa, rext, -eta_lim, eta_lim, res_bi) ;
622  domains[OUTER+3] = new Domain_bispheric_rect(OUTER+3, ttype, aa, rext, eta_plus, eta_lim, chi_lim, res_bi) ;

627  domains[OUTER+5+i] = new Domain_shell (OUTER+5+i, ttype, outer_bounds[i], outer_bounds[i+1], center, res) ;
