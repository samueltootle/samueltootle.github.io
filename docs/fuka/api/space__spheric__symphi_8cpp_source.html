37 Space_spheric_symphi::Space_spheric_symphi(int ttype, const Point& center, const Dim_array& res, const Array<double>& bounds, bool withzec) {
38 
39 
40  ndim = 3 ;
41 
42  nbr_domains = bounds.get_size(0) ;
43  type_base = ttype ;
44  domains = (withzec) ? new Domain* [nbr_domains+1] : new Domain* [nbr_domains] ;
45  // Nucleus
46  domains[0] = new Domain_nucleus_symphi(0, ttype, bounds(0), center, res) ;
47  for (int i=1 ; i<=nbr_domains-1 ; i++)
48  domains[i] = new Domain_shell_symphi(i, ttype, bounds(i-1), bounds(i), center, res) ;
49  if (withzec) {
50  nbr_domains ++ ;
51  domains[nbr_domains-1] = new Domain_compact_symphi(nbr_domains-1, ttype, bounds(nbr_domains-2), center, res) ;
52  }
53 }
