59 void Domain_shell_outer_homothetic::affecte_coef(int& conte, int cc, bool& found) const {
60  Val_domain auxi (this) ;
61  auxi.std_base() ;
62  auxi.set_in_coef() ;
63  auxi.allocate_coef() ;
64  *auxi.cf = 0 ;
65 
66  found = false ;
67 
68  if (conte==cc) {
69  Index pos_cf (nbr_coefs) ;
70  auxi.cf->set(pos_cf) = 1 ;
71  found = true ;
72  }
73 
74  conte ++ ;
75 
76  if (found) {
77  Scalar auxi_scal (sp) ;
78  auxi_scal.set_domain(num_dom) = auxi ;
79  outer_radius_term_eq->set_der_t(auxi_scal) ;
80  }
81  else {
82  outer_radius_term_eq->set_der_zero() ;
83  }
84  update() ;
85 }
