60 template <class dict_t>
61 void syst_vars_BH(dict_t& vars,
62  System_of_eqs& syst,
63  int const adapt_d,
64  std::string iden = {}) {
65 #ifdef DEBUG
66  std::cout << "Loading BH vars into dictionary.\n";
67 #endif
68  auto& space = syst.get_space();
69 
70  // Local angular momentum on the AH
71  double S =
72  space.get_domain(adapt_d)->integ(syst.give_val_def("intS")()(adapt_d),
73  INNER_BC);
74  vars[std::string{iden + "S"}.c_str()] = S;
75 
76  // Irreducible mass
77  double Mirrsq = space.get_domain(adapt_d)->integ(syst.give_val_def(
78  "intMirrsq")()(adapt_d),
79  INNER_BC);
80  double Mirr = sqrt(Mirrsq);
81  vars[std::string{iden + "Mirr"}.c_str()] = Mirr;
82 
83  // Area radius
84  vars[std::string{iden + "ArealR"}.c_str()] = 2. * Mirr;
85 
86  double Mch = std::sqrt(Mirrsq + S * S / 4. / Mirrsq);
87  vars[std::string{iden + "Mch"}.c_str()] = Mch;
88 
89  double Chi = S / Mch / Mch;
90  vars[std::string{iden + "Chi"}.c_str()] = Chi;
91 }
