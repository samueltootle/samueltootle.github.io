36 // conversion from solar mass to km
37 constexpr double M2km = 1.4769994423016508;

39 template <class eos_t>
40 struct reader_3d {
41  template <class config_t>
42  void operator()(config_t& bconfig) {
43  std::string spacein = bconfig.space_filename();
44  // load domain decomposition and fields from binary file
45  FILE* ff1 = fopen(spacein.c_str(), "r");
46  Space_spheric_adapted space(ff1);
47  Scalar conf(space, ff1);
48  Scalar lapse(space, ff1);
49  Vector shift(space, ff1);
50  Scalar logh(space, ff1);
51 
52  Scalar diff_omega(space);
53  if (bconfig.field(Kadath::FUKA_Config::BCO_FIELDS::DIFF_OMEGA)) {
54  diff_omega = Scalar(space, ff1);
55  }
56  fclose(ff1);
57 
58  // number of dimensions, a cartesian type of basis and the flat bg metric
59  int ndom = space.get_nbr_domains();
60  Base_tensor basis(space, CARTESIAN_BASIS);
61  Metric_flat fmet(space, basis);
62 
63  // fields depending on the coords
64  CoordFields<Space_spheric_adapted> cf_generator(space);
65  vec_ary_t coord_vectors{default_co_vector_ary(space)};
66 
67  // get origin of the system and initialize coordinate fields
68  double xo = bco_utils::get_center(space, 0);
69  update_fields_co(cf_generator, coord_vectors, {}, xo);
70 
71  // central values of the matter fields
72  double loghc = bco_utils::get_boundary_val(0, logh, INNER_BC);
73  double hc = std::exp(loghc);
74  double nc = EOS<eos_t, DENSITY>::get(hc);
75  double pc = EOS<eos_t, PRESSURE>::get(hc);
76 
77  // minimal and maximal radius of the adapted surface domain
78  auto [rmin, rmax] = bco_utils::get_rmin_rmax(space, 1);
79  // inner radius of the nucleus
80  double rin1 = bco_utils::get_radius(space.get_domain(0), OUTER_BC);
81 
82  // setup the system of equations to define derived quantities
83  System_of_eqs syst(space, 0, ndom - 1);
84 
85  // flat background metric
86  fmet.set_system(syst, "f");
87 
88  // constants
89  syst.add_cst("4piG", bconfig(BCO_QPIG));
90  syst.add_cst("PI", M_PI);
91 
92  // baryonic mass, dimensionless spin, central log enthalpy, angular
93  // frequency parameter
94  syst.add_cst("Mb", bconfig(MB));
95  syst.add_cst("chi", bconfig(CHI));
96  syst.add_cst("Hc", loghc);
97  if (bconfig.field(Kadath::FUKA_Config::BCO_FIELDS::DIFF_OMEGA)) {
98  syst.add_cst("ome", diff_omega);
99  } else {
100  syst.add_cst("ome", bconfig(OMEGA));
101  }
102  syst.add_cst("Madm", bconfig(MADM));
103 
104  // coordinate dependent fields
105  syst.add_cst("mg", *coord_vectors[GLOBAL_ROT]);
106  syst.add_cst("ex", *coord_vectors[EX]);
107  syst.add_cst("ey", *coord_vectors[EX]);
108  syst.add_cst("ez", *coord_vectors[EX]);
109  syst.add_cst("einf", *coord_vectors[S_INF]);
110 
111  // gravitational fields
112  syst.add_cst("P", conf);
113  syst.add_cst("N", lapse);
114  syst.add_cst("bet", shift);
115 
116  // matter represented by log enthalpy
117  syst.add_cst("H", logh);
118 
119  // common definitions
120  syst.add_def("NP = P*N");
121  syst.add_def("Ntilde = N / P^6");
122 
123  // beta combined with the rotational part
124  if (std::isnan(bconfig.set(BVELY)))
125  bconfig.set(BVELY) = 0.;
126  syst.add_def("omega^i = bet^i + ome * mg^i");
127 
128  // extrinsic curvature
129  syst.add_def(
130  "A^ij = (D^i bet^j + D^j bet^i - 2. / 3.* D_k bet^k * f^ij) / 2. / "
131  "Ntilde");
132  syst.add_def(ndom - 1, "intPx = A_i^j * ex_j * einf^i");
133  syst.add_def(ndom - 1, "intPy = A_i^j * ey_j * einf^i");
134  syst.add_def(ndom - 1, "intPz = A_i^j * ez_j * einf^i");
135 
136  // integrants at infinity
137  syst.add_def(ndom - 1, "intJ = multr(A_ij * mg^j * einf^i) / 8. / PI");
138  syst.add_def(ndom - 1, "intMadmalt = -dr(P) / 2 / PI");
139  syst.add_def(ndom - 1, "intMadm = - einf^i * D_i P / 2 / PI");
140  syst.add_def(ndom - 1,
141  "intMk = (einf^i * D_i N - A_ij * einf^i * bet^j) / 4 / PI");
142 
143  syst.add_def("dH = (ex^i * D_i H) / H");
144 
145  // setup operators for the EOS
146  Param p;
147  set_eos_ope_struct<eos_t>()(syst, p);
148 
149  // setup derived matter quantities
150  for (int d = 0; d < ndom; d++) {
151  switch (d) {
152  case 0:
153  case 1:
154  syst.add_def(d, "h = exp(H)");
155 
156  syst.add_def(d, "U^i = omega^i / N");
157  syst.add_def(d, "Usquare = P^4 * U_i * U^i");
158  syst.add_def(d, "Wsquare = 1. / (1. - Usquare)");
159  syst.add_def(d, "W = sqrt(Wsquare)");
160 
161  syst.add_def(d, "intMb = P^6 * rho(h) * W");
162  syst.add_def(d, "firstint = H + log(N) - log(W)");
163  syst.add_def(d, "intH = P^6 * H * W");
164 
165  syst.add_def(d, "j = P^4 * Wsquare * f_ij * U^i * mg^j / N");
166  syst.add_def(d, "vintJ = N * rho(h) * j * P^6");
167  syst.add_def(d, "intT = ome * vintJ");
168  syst.add_def(d, "inteps = intMb * eps(h)");
169  break;
170  }
171  }
172  double Px = space.get_domain(ndom - 1)->integ(syst.give_val_def("intPx")()(
173  ndom - 1),
174  OUTER_BC);
175  double Py = space.get_domain(ndom - 1)->integ(syst.give_val_def("intPy")()(
176  ndom - 1),
177  OUTER_BC);
178  double Pz = space.get_domain(ndom - 1)->integ(syst.give_val_def("intPz")()(
179  ndom - 1),
180  OUTER_BC);
181 
182  // ADM angular momentum at infinity
183  Val_domain integJ(syst.give_val_def("intJ")()(ndom - 1));
184  double J = space.get_domain(ndom - 1)->integ(integJ, OUTER_BC);
185 
186  // two (equivalent) computations of the ADM mass at infinity
187  Val_domain integMadm(syst.give_val_def("intMadm")()(ndom - 1));
188  double Madm = space.get_domain(ndom - 1)->integ(integMadm, OUTER_BC);
189 
190  Val_domain integMadmalt(syst.give_val_def("intMadmalt")()(ndom - 1));
191  double Madmalt = space.get_domain(ndom - 1)->integ(integMadmalt, OUTER_BC);
192 
193  // Komar mass at infinity
194  Val_domain integMk(syst.give_val_def("intMk")()(ndom - 1));
195  double Mk = space.get_domain(ndom - 1)->integ(integMk, OUTER_BC);
196 
197  // baryonic mass as volume integral over the two stellar domains
198  double baryonic_mass = syst.give_val_def("intMb")()(0).integ_volume() +
199  syst.give_val_def("intMb")()(1).integ_volume();
200  double H_integral = syst.give_val_def("intH")()(0).integ_volume() +
201  syst.give_val_def("intH")()(1).integ_volume();
202 
203  double T_integral = syst.give_val_def("intT")()(0).integ_volume() +
204  syst.give_val_def("intT")()(1).integ_volume();
205  T_integral *= 0.5;
206 
207  double J_vintegral = syst.give_val_def("vintJ")()(0).integ_volume() +
208  syst.give_val_def("vintJ")()(1).integ_volume();
209 
210  double eps_integral = syst.give_val_def("inteps")()(0).integ_volume() +
211  syst.give_val_def("inteps")()(1).integ_volume();
212 
213  double W_be = T_integral + eps_integral + baryonic_mass - Madm;
214  double BETA = T_integral / W_be;
215 
216  syst.add_def("intMsq = P^4");
217  double A =
218  space.get_domain(2)->integ(syst.give_val_def("intMsq")()(2), INNER_BC);
219  double AR = sqrt(A / 4. / acos(-1.));
220 
221  auto npts = space.get_domain(1)->get_nbr_points();
222 
223  Index pos_eq(npts);
224  pos_eq.set(0) = npts(0) - 1;
225  pos_eq.set(1) = npts(1) - 1;
226 
227  Index pos_pole(npts);
228  pos_pole.set(0) = npts(0) - 1;
229 
230  double conf_eq = conf(1)(pos_eq);
231  double Circumferential_R =
232  conf_eq * conf_eq * space.get_domain(1)->get_radius()(pos_eq);
233  double& CR = Circumferential_R;
234 
235  Scalar logh_dr(logh.der_r());
236  double mass_shedding_parameter = logh_dr(1)(pos_eq) / logh_dr(1)(pos_pole);
237 
238  Point P(3);
239  double central_dHdx = syst.give_val_def("dH")().val_point(P);
240  double central_euler = syst.give_val_def("firstint")().val_point(P);
241 
242 #ifdef FORMAT
243 #undef FORMAT
244 #endif
245 #define FORMAT \
246  std::setw(25) << std::right << std::setprecision(5) << std::fixed \
247  << std::showpos
248  auto print_shells = [&](int dom_min, int dom_max) {
249  int cnt = 1;
250  for (int i = dom_min; i < dom_max; ++i) {
251  std::string shell{"SHELL" + std::to_string(cnt) + " = "};
252  std::cout << FORMAT << shell
253  << bco_utils::get_radius(space.get_domain(i), OUTER_BC)
254  << std::endl;
255  cnt++;
256  }
257  };
258 
259  auto res_r = space.get_domain(0)->get_nbr_points()(0);
260  auto res_t = space.get_domain(0)->get_nbr_points()(1);
261  auto res_p = space.get_domain(0)->get_nbr_points()(2);
262 
263  // output to stdout
264  std::cout << FORMAT << "RES = " << "[" << res_r << "," << res_t << ","
265  << res_p << "]\n"
266  << FORMAT << "Coord R_IN = " << rin1 << std::endl
267  << FORMAT << "Coord R = " << "[" << rmin << ", " << rmax << "]\n";
268  print_shells(2, ndom - 2);
269  std::cout << FORMAT << "Coord R_OUT = "
270  << bco_utils::get_radius(space.get_domain(ndom - 2), OUTER_BC)
271  << "\n\n";
272 
273  std::cout << FORMAT << "Areal R = " << AR << " [" << AR * M2km << "km]\n"
274  << FORMAT << "Circumferential R = " << CR << " [" << CR * M2km
275  << "km]\n"
276  << FORMAT << "Mass Shedding = " << mass_shedding_parameter << "\n"
277  << FORMAT << "Baryonic Mass = " << baryonic_mass << std::endl
278  << FORMAT << "ADM Mass = " << Madm << " [" << bconfig(MADM)
279  << "]\n"
280  << FORMAT << "ADM Momentum = " << J << std::endl
281  << FORMAT << "Chi = " << J / Madm / Madm << " [" << bconfig(CHI)
282  << "]\n"
283  << FORMAT << "Omega = " << bconfig(OMEGA) << std::endl
284  << FORMAT << std::scientific << "Central Density = " << nc
285  << std::endl
286  << FORMAT << std::scientific << "Central log(h) = " << loghc
287  << std::endl
288  << FORMAT << std::scientific << "Central Pressure = " << pc
289  << std::endl
290  << FORMAT << std::scientific
291  << "Central dlog(h)/dx = " << central_dHdx << std::endl
292  << FORMAT << std::scientific
293  << "Central Euler Constant = " << central_euler << std::endl
294  << FORMAT << "Integrated log(h) = " << H_integral << "\n"
295  << FORMAT << "Beta = " << BETA << "\n\n";
296 
297  std::cout << FORMAT << "Mk = " << Mk << std::scientific
298  << ", Diff: " << 2. * fabs(Madm - Mk) / (Madm + Mk) << std::endl
299  << FORMAT << "Px = " << Px << std::endl
300  << FORMAT << "Py = " << Py << std::endl
301  << FORMAT << "Pz = " << Pz << std::endl;
302  }
303 };

305 int main(int argc, char** argv) {
306  // initialize MPI
307  int rc = MPI_Init(&argc, &argv);
308 
309  if (rc != MPI_SUCCESS) {
310  cerr << "Error starting MPI" << endl;
311  MPI_Abort(MPI_COMM_WORLD, rc);
312  }
313 
314  // expecting a configuration file on execution
315  if (argc < 2) {
316  std::cerr << "Usage: ./reader /<path>/<ID base name>.info" << std::endl;
317  std::cerr << "e.g. ./reader converged.NS.9.info" << endl;
318  std::_Exit(EXIT_FAILURE);
319  }
320 
321  // load the configuration
322  std::string ifilename{argv[1]};
323  kadath_config_boost<BCO_NS_INFO> bconfig(ifilename);
324  // setup the EOS
325  EOS_initialize::init(bconfig);
326  const std::string eos_type = bconfig.eos<std::string>(EOSTYPE);
327  EOS_Function_Dispatcher::dispatch<reader_3d>(bconfig, eos_type, bconfig);
328 
329  MPI_Finalize();
330  return EXIT_SUCCESS;
331 }
