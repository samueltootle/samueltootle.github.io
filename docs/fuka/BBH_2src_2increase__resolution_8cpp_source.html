34 template<typename config_t, typename space_t>
35 void setup_bin_config(config_t& bconfig, const space_t& space, const Scalar& conf, const Scalar& lapse){
36  int res = bconfig(BIN_RES);
37 
38  // FIXME not sure if it's only about oddness...
39  if((res % 2) == 0){
40  std::cout << "New Resolution is invalid. Must be odd (9,11,13,etc)" << std::endl;
41  std::_Exit(EXIT_FAILURE);
42  }
43 
44  if(std::isnan(bconfig.set(OUTER_SHELLS)))
45  bconfig.set(OUTER_SHELLS) = 0;
46 
47  bconfig.set(BCO_RES, BCO1) = res;
48  bconfig.set(BCO_RES, BCO2) = res;
49 
50  std::cout << "Resolution of new space: "
51  << res << " (r), "
52  << res << " (theta), "
53  << res - 1 << " (phi)" << std::endl;
54 
55  double Mtot = bconfig(MCH,BCO1) + bconfig(MCH,BCO2);
56 
57  bconfig.set(MCH, BCO1) = Mtot / (bconfig(Q) + 1);
58  bconfig.set(MIRR,BCO1) = bco_utils::mirr_from_mch(bconfig(CHI, BCO1), bconfig(MCH, BCO1));
59  bconfig.set(MCH, BCO2) = bconfig(Q) * bconfig(MCH,BCO1);
60  bconfig.set(MIRR,BCO2) = bco_utils::mirr_from_mch(bconfig(CHI, BCO2), bconfig(MCH, BCO2));
61  double kep_omega = std::sqrt(
62  (bconfig(MCH,BCO1) + bconfig(MCH,BCO2))
63  / 2.
64  / (bconfig(DIST) * bconfig(DIST) * bconfig(DIST)));
65 
66  std::cout << "\nTotal Mass of the System: " << Mtot << std::endl;
67  std::cout << "Keplarian Omega (for reference): " << kep_omega << std::endl;
68 
69  if(!bconfig.control(USE_CONFIG_VARS)) {
70  double rmax = 0;
71  int i = BCO1;
72 
73  for(auto& d : {space.BH1+1, space.BH2+1}) {
74  auto [ rmin, trmax ] = bco_utils::get_rmin_rmax(space, d);
75 
76  // estimate how small the inner radius should be based on relation
77  // between conformal factor and numerical radius.
78  // see https://arxiv.org/pdf/0805.4192, eq(64)
79  double conf_inner = bco_utils::get_boundary_val(d+1, conf, INNER_BC);
80  double conf_i_sq = conf_inner * conf_inner;
81  double est_r_div2 = bconfig(MCH, i) / conf_i_sq;
82  bconfig.set(RIN, i) = est_r_div2;
83 
84  // set this here only for shell bounds to be calculated.
85  bconfig.set(RMID, i) = 2. * est_r_div2;
86 
87  auto [fmin, fmax] = bco_utils::get_field_min_max(lapse, 2, INNER_BC);
88  bconfig.set(FIXED_LAPSE, i) = fmin;
89 
90  i = BCO2;
91  if(trmax > rmax) rmax = trmax;
92  }
93 
94  bconfig.set(REXT) = 2. * bconfig(DIST);
95  }
96 }

98 int main(int argc, char **argv) {
99  if (argc < 4) {
100  std::cout << "Missing arguments..." << std::endl
101  << "Possible input options: ./increase_resol <filename.info> <base output filename> <resolution>\n"
102  << "NOTES:\n"
103  << "all remaining attributes are pulled from the input config file.\n"
104  << "- shells per BH are pulled from bh<1/2>.nshells\n"
105  << "- mass ratio is pull from binary.q\n"
106  << "- total mass is determined from bh1.mch and bh2.mch\n"
107  << "- enabled 'use_config_vars' for pulling numerical domain values from the config file\n";
108 
109  std::_Exit(EXIT_FAILURE);
110  }
111 
112  // Name of config.info file
113  std::string in_filename{argv[1]};
114  std::string of_basename{argv[2]};
115  kadath_config_boost<BIN_INFO> bconfig(in_filename);
116  bconfig.set(BIN_RES) = stod(argv[3]);
117 
118  std::string in_spacefile = bconfig.space_filename();
119  FILE *ff1 = fopen(in_spacefile.c_str(), "r");
120 
121  Space_bin_bh old_space(ff1) ;
122  Scalar old_conf (old_space, ff1) ;
123  Scalar old_lapse(old_space, ff1) ;
124  Vector old_shift(old_space, ff1) ;
125  fclose(ff1) ;
126  setup_bin_config(bconfig, old_space, old_conf, old_lapse);
127 
128  std::cout << "Resolution of old space: "
129  << old_space.get_domain(0)->get_nbr_points()(0) << " (r), "
130  << old_space.get_domain(0)->get_nbr_points()(1) << " (theta), "
131  << old_space.get_domain(0)->get_nbr_points()(2) << " (phi)" << std::endl;
132 
133  double Mtot = bconfig(MCH,BCO1) + bconfig(MCH,BCO2);
134 
135  std::cout << "\nTotal Mass of the System: " << Mtot << std::endl;
136 
137  int type_coloc = old_space.get_type_base();
138 
139  //setup bounds
140  std::vector<double> out_bounds(1 + bconfig(OUTER_SHELLS));
141  for(int e = 0; e < out_bounds.size(); ++e)
142  out_bounds[e] = bconfig(REXT) * (1. + e * 0.25);
143 
144  std::vector<double> BH1_bounds(3+bconfig(NSHELLS,BCO1));
145  std::vector<double> BH2_bounds(3+bconfig(NSHELLS,BCO2));
146  bco_utils::set_BH_bounds(BH1_bounds, bconfig, BCO1);
147  bco_utils::set_BH_bounds(BH2_bounds, bconfig, BCO2);
148 
149  // Set radius of the excision boundary to the current radius so that the solver
150  // starts from the originial solution
151  BH1_bounds[1] = bco_utils::get_radius(old_space.get_domain(old_space.BH1+1), OUTER_BC) ;
152  BH2_bounds[1] = bco_utils::get_radius(old_space.get_domain(old_space.BH2+1), OUTER_BC) ;
153  //end setup bounds
154 
155  Space_bin_bh space (type_coloc, bconfig(DIST), BH1_bounds, BH2_bounds, out_bounds, bconfig(BIN_RES));
156  Base_tensor basis (space , CARTESIAN_BASIS);
157 
158  std::cout << "New bounds:" << std::endl;
159  for(int d = space.BH1; d < space.BH2; ++d)
160  std::cout << bco_utils::get_radius(space.get_domain(d), OUTER_BC) << " ";
161  std::cout << std::endl;
162  for(int d = space.BH2; d < space.OUTER; ++d)
163  std::cout << bco_utils::get_radius(space.get_domain(d), OUTER_BC) << " ";
164  std::cout << std::endl;
165 
166  double xc1 = bco_utils::get_center(space, space.BH1);
167  double xc2 = bco_utils::get_center(space, space.BH2);
168 
169  std::cout << "xc1: " << xc1 << std::endl;
170  std::cout << "xc2: " << xc2 << std::endl;
171 
172  //setup new fields
173  Scalar conf(space);
174  conf = 1.;
175  conf.std_base();
176 
177  Scalar lapse(space);
178  lapse = 1.;
179  lapse.std_base();
180 
181  Vector shift(space, CON, basis);
182  for (int i = 1; i <= 3; i++)
183  shift.set(i).annule_hard();
184  shift.std_base();
185  // end setup new fields
186 
187  // needed in some cases, since there is no data in 0,1 and the interpolation can go crazy
188  std::array<const int, 2> old_nuc_doms{old_space.BH1, old_space.BH2};
189  std::array<const Domain_shell_outer_homothetic*, 2> old_outer_homothetic {dynamic_cast<const Domain_shell_outer_homothetic*>(old_space.get_domain(old_space.BH1+1)),
190  dynamic_cast<const Domain_shell_outer_homothetic*>(old_space.get_domain(old_space.BH2+1))};
191  for(auto& i : {0, 1}){
192  // update BH fields to help with interpolation later
193  bco_utils::update_adapted_field(old_conf , old_nuc_doms[i]+2, old_nuc_doms[i]+1, old_outer_homothetic[i], OUTER_BC);
194  bco_utils::update_adapted_field(old_lapse, old_nuc_doms[i]+2, old_nuc_doms[i]+1, old_outer_homothetic[i], OUTER_BC);
195  for(int j = 1; j < 4; ++j)
196  bco_utils::update_adapted_field(old_shift.set(j), old_nuc_doms[i]+2, old_nuc_doms[i]+1, old_outer_homothetic[i], OUTER_BC);
197  }
198 
199  // import fields
200  conf.import(old_conf);
201  lapse.import(old_lapse);
202  for(int c = 1; c <= 3; ++c)
203  shift.set(c).import(old_shift(c));
204  // end import fields
205 
206  // make sure excised domains are set to zero
207  for(auto& dom : {space.BH1, space.BH2}) {
208  lapse.set_domain(dom).annule_hard();
209  lapse.set_domain(dom+1).annule_hard();
210 
211  conf.set_domain(dom).annule_hard();
212  conf.set_domain(dom+1).annule_hard();
213 
214  for(int i = 1; i <= 3; ++i) {
215  shift.set(i).set_domain(dom).annule_hard();
216  shift.set(i).set_domain(dom+1).annule_hard();
217  }
218  }
219  // end set excised domains
220 
221  lapse.std_base();
222  conf.std_base();
223  shift.std_base();
224 
225  bconfig.set_filename(of_basename);
226  bco_utils::save_to_file(space, bconfig, conf, lapse, shift);
227 }
