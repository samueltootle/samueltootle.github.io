57  void compute_defs() {
58  Kadath::Scalar const & conf = extractField<Kadath::Scalar>("conf");
59  Kadath::Scalar const & lapse = extractField<Kadath::Scalar>("lapse");
60  Kadath::Vector const & shift = extractField<Kadath::Vector>("shift");
61 
62  Base_tensor basis(shift.get_basis());
63  Metric_flat fmet (space, basis) ;
64  int ndom = space.get_nbr_domains() ;
65 
66  double xo = bco_utils::get_center(space, ndom-1);
67  std::vector<int> vac_Domains{
68  FUKA_Syst_tools::vector_of_domains(0, ndom)
69  };
70 
71  // setup coordinate vector fields for System_of_eqs
72  CoordFields<space_t> cfields(space);
73  vec_ary_t coord_vectors {default_co_vector_ary(space)};
74  update_fields_co(cfields, coord_vectors, {}, xo);
75  // end coordinate field setup
76 
77  // Setup system of equations and definitions
78  System_of_eqs syst (space, 0, ndom-1) ;
79  fmet.set_system(syst, "f") ;
80 
81  auto init_if = [&](auto c, auto idx) {
82  if(std::isnan(bconfig(idx)))
83  bconfig(idx) = 0.;
84  syst.add_cst(c, bconfig(idx));
85  };
86 
87  // Fields - must be initialized before common setup
88  syst.add_cst("P" , conf) ;
89  syst.add_cst("N" , lapse) ;
90  syst.add_cst("bet", shift) ;
91 
92  syst.add_cst("ome", bconfig(OMEGA));
93 
94  // Avoid excision region (d=0,1) and compactified (d=ndom-1)
95  for(auto d = 2; d < ndom-1; ++d) {
96  syst.add_def(d, "drP = dr(P)");
97  syst.add_def(d, "ddrP = dr(drP)");
98  }
99 
100  // Initialize constants and constant fields
101  FUKA_Syst_tools::syst_init_co(syst, coord_vectors, bconfig);
102 
103  // Define equations in relevant domains
104  FUKA_Syst_tools::syst_init_quasi_local_defs(
105  syst, vac_Domains,"mg","sm"
106  );
107  FUKA_Syst_tools::syst_init_eqdefs_vac(syst, vac_Domains);
108  FUKA_Syst_tools::syst_init_contraction_defs_vac(syst);
109 
110  FUKA_Syst_tools::syst_vars(vars, syst);
111  FUKA_Syst_tools::syst_vars_BH(vars, syst, 2);
112  FUKA_Syst_tools::export_radii(space, vars, 0, ndom-1, "BH_R");
113 
114  FUKA_Syst_tools::dict_add_vector_cmp(
115  syst, vars, "shift", Tensor(shift)
116  );
