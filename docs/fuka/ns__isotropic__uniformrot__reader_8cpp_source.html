53  template <typename eos_t>
54  struct compute_defs {
55  template <typename T>
56  void operator()(kadath_config_boost<BCO_ISO_NS_INFO> bconfig,
57  T* this_reader,
58  space_t const& space) {
59  Kadath::Scalar const& lap_Aterm =
60  this_reader->template extractField<Kadath::Scalar>("lap_Aterm");
61  Kadath::Scalar const& nu =
62  this_reader->template extractField<Kadath::Scalar>("nu");
63  Kadath::Scalar const& lap_Bterm =
64  this_reader->template extractField<Kadath::Scalar>("lap_Bterm");
65  Kadath::Scalar const& lap_wterm_in =
66  this_reader->template extractField<Kadath::Scalar>("lap_wterm");
67  Kadath::Scalar const& logh =
68  this_reader->template extractField<Kadath::Scalar>("logh");
69 
70  Kadath::Scalar lap_wterm(lap_wterm_in, true);
71  lap_wterm.affect_parameters();
72  lap_wterm.set_parameters()->set_m_quant() = 1;
73  lap_wterm.std_base();
74 
75  int ndom = space.get_nbr_domains();
76 
77  // Setup system of equations and definitions
78  System_of_eqs syst(space, 0, ndom - 1);
79  // define numerical constants
80  syst.add_cst("4piG", bconfig(BCO_PARAMS::BCO_QPIG));
81 
82  // Fields - must be initialized before common setup
83  syst.add_cst("H", logh);
84  syst.add_cst("nu", nu);
85  syst.add_cst("lapAterm", lap_Aterm);
86  syst.add_cst("lapBterm", lap_Bterm);
87  syst.add_cst("wrsint", lap_wterm);
88  syst.add_cst("Omega", bconfig(BCO_PARAMS::OMEGA));
89 
90  syst.add_def("N = exp(nu)");
91  syst.add_def("A = exp(lapAterm - nu)");
92  syst.add_def("B = (divrsint(lapBterm) + 1) / N");
93  syst.add_def("w = divrsint(wrsint)");
94  syst.add_def("drw = dr(w)");
95  syst.add_def("dtw = dt(w)");
96  syst.add_def("dtN = dt(N)");
97 
98  // define quantity to be integrated at infinity
99  // two (in this case) equivalent definitions of ADM mass
100  // as well as the Komar mass
101  syst.add_def(ndom - 1, "intMadm = - dr(B) / 4piG ");
102  syst.add_def(
103  ndom - 1,
104  "intMk = B * (dr(N) - multrsint(multrsint(B^2) / 2 / N * w * "
105  "dr(w))) / 4piG");
106  syst.add_def(ndom - 1, "intJ = -multrsint(multrsint(dr(w))) / 4 / 4piG");
107 
108  // enthalpy from the logarithmic enthalpy, the latter is the actual variable in this system
109  syst.add_def("h = exp(H)");
110 
111  // define the EOS operators
112  Param p;
113  syst.add_ope("eps", &EOS<eos_t, EPSILON>::action, &p);
114  syst.add_ope("press", &EOS<eos_t, PRESSURE>::action, &p);
115  syst.add_ope("rho", &EOS<eos_t, DENSITY>::action, &p);
116 
117  // define rest-mass density, internal energy and pressure through the enthalpy
118  syst.add_def("rho = rho(h)");
119  syst.add_def("eps = eps(h)");
120  syst.add_def("press = press(h)");
121 
122  // definition to rescale the equations
123  // delta = p / rho
124  syst.add_def("delta = h - eps - 1.");
125 
126  // (3.31) Upper Phi component of U vector (r, theta = 0)
127  syst.add_def("UphiU = (Omega - w) / N");
128  // (3.32)
129  syst.add_def("U = multrsint(B * UphiU)");
130  syst.add_def("Usq = U*U");
131  syst.add_def("Wsq = 1 / (1 - Usq)");
132  syst.add_def("W = sqrt(Wsq)");
133 
134  for (int d = 0; d < ndom; d++) {
135 
136  switch (d) {
137  // in the star the constraint equations are sourced by the matter
138  case 0:
139  case 1:
140 
141  // sources
142  syst.add_def(d, "E = Wsq * press * h - press * delta");
143  syst.add_def(d, "Srrtt = press * delta");
144  syst.add_def(d, "pphi = (E + Srrtt) * U");
145  syst.add_def(d, "Spp = delta * press * (1 + Usq) + E * Usq");
146  syst.add_def(d, "S = 2 * Srrtt + Spp");
147 
148  // constraint equations
149  syst.add_def(d,
150  "eqnu = delta * lap(nu) + delta * scal(grad(nu), "
151  "grad(nu + log(B))) "
152  "- delta * multrsint(multrsint(B^2)) / 2 / N^2 * "
153  "scal(grad(w), grad(w)) "
154  "- 4piG * A^2 * (E + S)");
155  syst.add_def(d,
156  "eqwrsint = delta * lap(wrsint) - delta * "
157  "multrsint(scal(grad(w), grad(nu - 3 * log(B))))"
158  "+ 4 * 4piG * N * A^2 / B * pphi");
159  syst.add_def(
160  d,
161  "eqBterm = delta * lap2(lapBterm) - 2 * 4piG * N * A^2 "
162  "* multrsint(B) * (2 * Srrtt)");
163  syst.add_def(d,
164  "eqAterm = delta * lap2(lapAterm) + delta * "
165  "scal(grad(nu), grad(nu))"
166  "- 3 * delta * multrsint(multrsint(B^2)) / 4 / N^2 * "
167  "scal(grad(w), grad(w))"
168  "- 2 * 4piG * A^2 * Spp");
169 
170  // definition for the baryonic mass integral
171  syst.add_def(d, "intMb = W * rho * A^2 * B * 4piG / 2");
172  syst.add_def(d, "intH = W * H * A^2 * B * 4piG / 2");
173  break;
174  // outside the matter is absent and the sources are zero
175  default:
176 
177  syst.add_def(d,
178  "eqnu = lap(nu) + scal(grad(nu), grad(nu + log(B))) "
179  "- multrsint(multrsint(B^2)) / 2 / N^2 * "
180  "scal(grad(w), grad(w))");
181  syst.add_def(d,
182  "eqAterm = lap2(lapAterm) + scal(grad(nu), grad(nu))"
183  "- 3 * multrsint(multrsint(B^2)) / 4 / N^2 * "
184  "scal(grad(w), grad(w))");
185  syst.add_def(d, "eqBterm = lap2(lapBterm)");
186  syst.add_def(d,
187  "eqwrsint = lap(wrsint) - multrsint(scal(grad(w), "
188  "grad(nu - 3 * log(B))))");
189  break;
190  }
191  }
192  auto add_surf_integ = [&](auto varstr, auto defstr, auto dom, auto bc) {
193  this_reader->vars[varstr] = syst.get_space().get_domain(dom)->integ(
194  syst.give_val_def(defstr)()(dom), bc);
195  };
196  add_surf_integ("Jadm", "intJ", ndom - 1, OUTER_BC);
197  add_surf_integ("Madm", "intMadm", ndom - 1, OUTER_BC);
198  add_surf_integ("Mk", "intMk", ndom - 1, OUTER_BC);
199 
200  // Populate vars dictionary
201  // FUKA_Syst_tools::syst_vars(vars, syst);
202  // FUKA_Syst_tools::syst_vars_hydro(vars, syst);
203  // FUKA_Syst_tools::export_radii(space, vars, 0, ndom-1, "NS_R");
204  // FUKA_Syst_tools::dict_add_vector_cmp(
205  // syst, vars, "shift", Tensor(shift)
206  // );
207 
208  // double Madm = boost::python::extract<double>(vars["Madm"]);
209  FUKA_Syst_tools::syst_vars_NS_isotropic(this_reader->vars, syst, 2);
210  FUKA_Syst_tools::syst_add_resolution_list(space, this_reader->vars);
211  this_reader->vars["nc"] =
212  EOS<eos_t, DENSITY>::get(bconfig(BCO_PARAMS::HC));
213  this_reader->vars["hc"] = bconfig(BCO_PARAMS::HC);
214 
215  auto add_from_def = [&](std::string in_str, std::string out_str = "") {
216  if (out_str == "")
217  out_str = in_str;
218  Scalar tmp(syst.give_val_def(in_str.c_str()));
219  tmp.coef_i();
220  tmp.std_base();
221  this_reader->vars[out_str.c_str()] = tmp;
222  };
223  add_from_def("w", "metric_omega");
224  add_from_def("A", "metric_A");
225  add_from_def("B", "metric_B");
226  add_from_def("U");
227  add_from_def("W");
228  add_from_def("N", "lapse");
229  add_from_def("eqnu", "cLapse");
230  add_from_def("eqAterm", "cA");
231  add_from_def("eqBterm", "cB");
232  add_from_def("eqwrsint", "comega");
233  add_from_def("dtN", "dLapse_dt");
234 
235  Scalar omega(syst.give_val_def("w")());
236  Scalar domega_dr(omega.der_var(1));
237  Scalar domega_dt(omega.der_var(2));
238  this_reader->vars["domega_dr"] = domega_dr;
239  this_reader->vars["domega_dt"] = domega_dt;
240 
241  Scalar N(syst.give_val_def("N")());
242  // (divrsint(lapBterm) + 1) / N
243  Scalar mB((lap_Bterm.div_rsint() + 1) / N);
244  Scalar Brsint(mB.mult_sin_theta().mult_r());
245  Scalar Brsint2(Brsint * Brsint);
246  // Brsint.coef();
247  this_reader->vars["Brsint"] = Brsint;
248  this_reader->vars["Brsintsq"] = Brsint2;
249 
250  // Scalar one(space);
251  // Scalar rsint(one.mult_r().mult_sin_theta());
252  // rsint.coef();
253  // this_reader->vars["rsint"] = rsint;
254 
255  auto npts = space.get_domain(1)->get_nbr_points();
256  Index pos_eq(npts);
257  pos_eq.set(0) = npts(0) - 1;
258  pos_eq.set(1) = npts(1) - 1;
259  auto B(syst.give_val_def("B")()(1));
260  auto r(space.get_domain(1)->get_radius());
261  double CR = B(pos_eq) * r(pos_eq);
262  this_reader->vars["CR"] = CR;
263  }
264  };

283  boost::python::list getExporterFieldValues__cartesian(
284  std::string const& fieldname,
285  boost::python::list const& coord_list) {
286  // list of values to return
287  boost::python::list values;
288 
289  // From CPPReference
290  auto str_tolower = [](std::string s) {
291  std::transform(s.begin(), s.end(), s.begin(),
292  [](unsigned char c) {
293  return std::tolower(c);
294  } // correct
295  );
296  return s;
297  };
298  auto key{str_tolower(fieldname)};
299  size_t idx;
300  if (auto search = exporter.output_var_map.find(key);
301  search == exporter.output_var_map.end()) {
302  std::string msg{"Invalid output fieldname pass: " + fieldname};
303  throw std::invalid_argument(msg.c_str());
304  } else {
305  idx = exporter.output_var_map[key];
306  cout << key << ": " << idx << endl;
307  }
308 
309  // loop through all given coords
310  for (int i = 0; i < boost::python::len(coord_list); ++i) {
311  // extract coords
312  boost::python::list coords =
313  boost::python::extract<boost::python::list>(coord_list[i]);
314  auto output_vars =
315  exporter.export_pointwise(boost::python::extract<double>(coords[0]),
316  boost::python::extract<double>(coords[1]),
317  boost::python::extract<double>(coords[2]));
318  values.append(output_vars[idx]);
319  }
320  return values;
321  }
