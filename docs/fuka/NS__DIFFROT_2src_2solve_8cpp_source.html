22 #include "mpi.h"
23 #include "FUKA_Solvers/NS_XCTS_driver.cpp"
24 #include "Solvers/sequences/ns_sequence.hpp"
25 #include "Solvers/sequences/parameter_sequence.hpp"
26 #include "Solvers/sequences/sequence_utilities.hpp"
27 #include "Solvers/solver_startup.hpp"

29 using namespace Kadath::FUKA_Config;
30 using namespace Kadath::FUKA_Solvers;
31 
32 int main(int argc, char** argv) {
33  int rc = MPI_Init(&argc, &argv);
34  if (rc != MPI_SUCCESS) {
35  cerr << "Error starting MPI" << endl;
36  MPI_Abort(MPI_COMM_WORLD, rc);
37  }
38  int rank = 0;
39  MPI_Comm_rank(MPI_COMM_WORLD, &rank);
40 
41  using config_t = kadath_config_boost<BCO_NS_INFO>;
42  using InitSolver = Initialize_Solver<config_t>;
43 
44  // Initialize static member variables
45  InitSolver::input_configname = "initial_ns.info";
46  InitSolver::bconfig;
47  InitSolver::rank = rank;
48 
49  // Run initialize routine based on CLI arguments
50  InitSolver::init_solver(argc, argv);
51  config_t bconfig = InitSolver::bconfig;
52 
53  if (InitSolver::example_setup) {
54  if (rank == 0) {
55  // Generate <example name>.info and terminate
56  if (InitSolver::minimal_config) {
57  bconfig.set_minimal_defaults();
58  bconfig.set_stage(STAGES::TOTAL_BC) = false;
59  bconfig.set_stage(STAGES::UNIFORM_ROT) = true;
60  bconfig.set_stage(STAGES::DIFF_ROT) = true;
61  initialize_diffrot(bconfig);
62  bconfig.write_minimal_config();
63  } else {
64  bconfig.set_defaults();
65  bconfig.set_stage(STAGES::TOTAL_BC) = false;
66  bconfig.set_stage(STAGES::UNIFORM_ROT) = true;
67  bconfig.control(CONTROLS::SEQUENCES) = InitSolver::setup_first;
68  initialize_diffrot(bconfig);
69  bconfig.write_config();
70  }
71  }
72  } else {
73  // We now have to assume bconfig is a minimal config
74  // that contains sequences _init/_final
75  bconfig.open_config();
76  EOS_initialize::init(bconfig);
77  bconfig.control(CONTROLS::SEQUENCES) = InitSolver::setup_first;
78  if (bconfig.set_stage(STAGES::TOTAL_BC)) {
79  bconfig.set_stage(STAGES::TOTAL_BC) = false;
80  bconfig.set_stage(STAGES::UNIFORM_ROT) = true;
81  }
82 
83  Tree tree;
84  pt::read_info(bconfig.config_filename_abs(), tree);
85  auto N = number_of_sequences(tree, MBCO_PARAMS, "ns");
86  if (N > 1) {
87  if (rank == 0) {
88  std::cerr << "Only sequences along one component is allowed.\n";
89  std::_Exit(EXIT_FAILURE);
90  }
91  }
92 
93  Parameter_sequence<BCO_PARAMS> resolution =
94  parse_seq_tree(tree, "ns", "res", BCO_PARAMS::BCO_RES);
95  verify_resolution_sequence(bconfig, resolution);
96 
97  ns_sequence seq = find_ns_sequence(tree);
98  if (seq.is_set()) {
99  auto [branch_name, key, val] = find_leaf(tree, "N");
100  if (!key.empty())
101  seq.set_N(std::stoi(val));
102  }
103 
104  verify_ns_fixing_values(bconfig, seq);
105  initialize_config_from_fixing_values(bconfig, seq);
106  ns_xcts_driver(bconfig, seq, resolution, InitSolver::outputdir);
107  }
108  MPI_Finalize();
109  return EXIT_SUCCESS;
110 }
