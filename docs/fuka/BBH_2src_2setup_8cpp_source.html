35 auto set_bin_defaults() {
36  /******* Equal Mass, irrot Initial Data Setup *********************/
37 
38  kadath_config_boost<BCO_BH_INFO> bhconfig;
39  bhconfig.set_defaults();
40 
41  kadath_config_boost<BIN_INFO> bconfig;
42  bconfig.initialize_binary({"bh","bh"});
43 
44  // copy default BH configuration to both BHs in binary
45  for(int i = 0; i < NUM_BCO_PARAMS; ++i) {
46  bconfig.set(i,BCO1) = bhconfig.set(i);
47  bconfig.set(i,BCO2) = bhconfig.set(i);
48  }
49 
50  // setup binary parameters
51  bconfig.set(BIN_RES) = 9;
52  bconfig.set(DIST) = 10;
53  bconfig.set(REXT) = 2 * bconfig(DIST);
54  bconfig.set(QPIG) = 4 * M_PI;
55  bconfig.set(Q) = 1;
56  bconfig.set(ADOT) = 0;
57  bconfig.set(OUTER_SHELLS) = 0;
58  return bconfig;
59 }
60 
61 int main(int argc, char** argv) {
62  kadath_config_boost<BIN_INFO> bconfig;
63 
64  if (argc < 2) {
65  std::cout << "Using default setup.\n";
66  bconfig = set_bin_defaults();
67  }
68  else {
69  std::string input_filename = std::string{argv[1]};
70  std::cout << "Creating setup from: " << input_filename << "\n";
71  bconfig.set_filename(input_filename);
72  bconfig.open_config();
73  }
74 
75  std::stringstream ifilename;
76  ifilename << "./initbin.info";
77  bconfig.set_filename(ifilename.str());
78 
79  std::vector<double> out_bounds(1);
80  std::vector<double> BH1_bounds(3+bconfig(NSHELLS,BCO2));
81  std::vector<double> BH2_bounds(3+bconfig(NSHELLS,BCO2));
82 
83  // for out_bounds.size > 1 - add equi-distant shells
84  for(int e = 0; e < out_bounds.size(); ++e)
85  out_bounds[e] = bconfig(REXT) + e * 0.25 * bconfig(REXT);
86 
87  bco_utils::set_BH_bounds(BH1_bounds, bconfig, BCO1);
88  bco_utils::set_BH_bounds(BH2_bounds, bconfig, BCO2);
89 
90  // create space containing the domain decomposition
91  int type_coloc = CHEB_TYPE;
92  Space_bin_bh space(type_coloc, bconfig(DIST), BH1_bounds, BH2_bounds, out_bounds, bconfig(BIN_RES));
93  Base_tensor basis(space, CARTESIAN_BASIS);
94 
95  double xmin = bco_utils::get_center(space, space.BH1);
96  double xmax = bco_utils::get_center(space, space.BH2);
97 
98  std::cout << "xmin/max: " << xmin << " " << xmax << std::endl;
99 
100  // remaining vars based on basic Newtonian estimates
101  if(!bconfig.control(USE_CONFIG_VARS)) {
102  bconfig.set(COMY) = 0;
103  bconfig.set(COM) = 0;
104  bco_utils::KadathPNOrbitalParams(bconfig, bconfig(MCH,BCO1), bconfig(MCH,BCO2));
105  bconfig.set(OMEGA, BCO1) = bconfig(GOMEGA); //irrotational
106  bconfig.set(OMEGA, BCO2) = bconfig(GOMEGA);
107 
108  bconfig.set_field(CONF) = true;
109  bconfig.set_field(LAPSE) = true;
110  bconfig.set_field(SHIFT) = true;
111 
112  bconfig.set_stage(PRE) = true;
113  bconfig.set_stage(FIXED_OMEGA) = true;
114  bconfig.set_stage(COROT_EQUAL) = true;
115  bconfig.set_stage(TOTAL) = true;
116  bconfig.set_stage(TOTAL_BC) = true;
117  }
118  // END OF CONFIG FILE SETUP
119 
120  // setup fields to flat space values
121  Scalar lapse(space);
122  lapse = 1.;
123  for(auto& dom : {space.BH1, space.BH2}) {
124  lapse.set_domain(dom).annule_hard();
125  lapse.set_domain(dom+1).annule_hard();
126  }
127  Scalar conf(lapse);
128 
129  Vector shift(space, CON, basis);
130  for (int i = 1; i <= 3; i++)
131  shift.set(i).annule_hard();
132 
133  conf.std_base();
134  lapse.std_base();
135  shift.std_base();
136  // end setup of fields
137 
138  bco_utils::save_to_file(space, bconfig, conf, lapse, shift);
139  return EXIT_SUCCESS ;
140 }
